75void DatalinkInspector::append_uid(QStringList blocks)
76{
77 for (auto s : blocks) {
78 if (s.startsWith('$'))
79 append_uid_item(s.mid(1));
80 }

82void DatalinkInspector::append_uid_item(QString uid)
83{
84 auto items = m_uidModel->items();
85
86 for (int i = 0; i < items.size(); ++i) {
87 if (items.at(i).at(0) != uid)
88 continue;
89 uint cnt = ++_uid_cnt[i];
90 m_uidModel->updateItem(i, QStringList() << uid << QString("[%1]").arg(cnt));
91 return;
92 }
93 // create new
94 _uid_cnt.append(1);
95 m_uidModel->append(QStringList() << uid);
96}

98void DatalinkInspector::filter(QString uid, bool exclude)
99{
100 uid.prepend('$');
101 if (exclude) {
102 m_proxyModel->add_filter(uid);
103 } else {
104 m_proxyModel->remove_filter(uid);
105 }
