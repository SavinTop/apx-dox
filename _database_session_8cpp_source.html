31DatabaseSession::DatabaseSession(QObject *parent,
32 const QString &name,
33 const QString &sessionName,
34 QString version)
35 : Fact(Database::instance(), name, "", "", Group)
36 , sessionName(sessionName)
37 , inTransaction(false)
38 , sql(QSqlDatabase::addDatabase("QSQLITE", sessionName))
39 , database(Database::instance())
40 , m_worker(nullptr)
41 , m_capacity(0)
42{
43 Q_UNUSED(parent)
44 connect(App::instance(), &App::aboutToQuit, this, &Fact::deleteFact);
45
46 setIcon("database");
47
48 // file name
49 // fileName = AppDirs::db().absoluteFilePath(version + QDir::separator() + name + ".db");
50 fileName = AppDirs::db().absoluteFilePath(name);
51 if (!version.isEmpty())
52 fileName.append('.').append(version);
53 fileName.append(".db");
54
55 QDir dir(QFileInfo(fileName).absoluteDir());
56 if (!dir.exists())
57 dir.mkpath(".");
58
59 //QMutexLocker lock(&mutex);
60 sql.setDatabaseName(fileName);
61 QStringList opts;
62 opts.append("QSQLITE_ENABLE_SHARED_CACHE");
63 opts.append("QSQLITE_BUSY_TIMEOUT=10000000");
64 //if(readOnly) opts.append("QSQLITE_OPEN_READONLY");
65 sql.setConnectOptions(opts.join("; "));
66 if (!sql.open()) {
67 apxMsgW() << sql.lastError();
68 QSqlDatabase::removeDatabase(sessionName);
69 } else {
70 QSqlQuery query(sql);
71 query.exec("PRAGMA foreign_keys = ON;");
72 query.exec("PRAGMA page_size = 4096;");
73 query.exec("PRAGMA cache_size = 16384;");
74 query.exec("PRAGMA synchronous = OFF;");
75 query.exec("PRAGMA temp_store = DEFAULT;");
76 query.exec("PRAGMA locking_mode = NORMAL;");
77 query.exec("PRAGMA journal_mode = WAL;");
78 query.exec("PRAGMA auto_vacuum = NONE;");
79 connect(this, &DatabaseSession::destroyed, [sessionName]() {
80 QSqlDatabase::removeDatabase(sessionName);
81 //qDebug()<<"DB"<<sessionName<<"removed";
82 });
83 }
84 m_worker = new DatabaseWorker(this, nullptr);
85
86 modifiedTimer.setSingleShot(true);
87 modifiedTimer.setInterval(500);
88 connect(&modifiedTimer, &QTimer::timeout, this, &DatabaseSession::modified);
89
90 //info update
91 infoQueueSize = 0;
92 evtUpdateInfo.forceTimeout = true;
93 connect(&evtUpdateInfo, &DelayedEvent::triggered, this, &DatabaseSession::updateInfo);
94 connect(m_worker,
95 &DatabaseWorker::infoChanged,
96 &evtUpdateInfo,
97 &DelayedEvent::schedule,
98 Qt::QueuedConnection);
99
100 //tools
101 f_vacuum = new Fact(this, "vacuum", tr("Optimize"), tr("Compress size"));
102 f_vacuum->setIcon("arrow-collapse-vertical");
103 connect(f_vacuum, &Fact::triggered, this, &DatabaseSession::vacuumTriggered);
104 f_analyze = new Fact(this, "analyze", tr("Analyze"), tr("Check integrity"));
105 f_analyze->setIcon("check-all");
106 connect(f_analyze, &Fact::triggered, this, &DatabaseSession::analyzeTriggered);
107
108 //capacity
109 connect(this, &DatabaseSession::capacityChanged, this, &DatabaseSession::updateDescr);
110 connect(this, &Fact::triggered, this, &DatabaseSession::updateCapacity);
111 updateCapacity();
112
113 updateInfo();
114 updateDescr();
115}

187bool DatabaseSession::commit(QSqlQuery &query, bool forceError)
188{
189 if (!sql.isOpen())
190 return false;
191 //qDebug()<<"commit"<<inTransaction;
192 if (!inTransaction)
193 return true;
194 inTransaction = false;
195 if (forceError || query.lastError().type() != QSqlError::NoError) {
196 apxConsoleW() << "SQL error:" << query.lastError().text().append(":")
197 << query.executedQuery();
198 query.prepare("ROLLBACK");
199 } else {
200 query.prepare("COMMIT");
201 }
202 bool rv = query.exec();
203 return rv;
204}

267bool DBReqMakeTable::run(QSqlQuery &query)
268{
269 query.prepare(QString("PRAGMA table_info('%1')").arg(tableName));
270 if (!query.exec())
271 return false;
272
273 QStringList fieldsList;
274 for (auto i : fields) {
275 if (i.contains('('))
276 continue;
277 if (!i.contains(' '))
278 continue;
279 fieldsList.append(i.split(' ').first());
280 }
281 db->updateTableFields(tableName, fieldsList);
282
283 if (!query.next()) {
284 //not exists - create new table
285 const QString &s = QString("CREATE TABLE IF NOT EXISTS %1 (%2) %3")
286 .arg(tableName)
287 .arg(fields.join(','))
288 .arg(tail);
289 db->transaction(query);
290 query.prepare(s);
291 if (!query.exec())
292 return false;
293 db->commit(query);
294 return true;
295 }
296 //update existing table
297 do {
298 QString s = query.value("name").toString();
299 for (int i = 0; i < fields.size(); ++i) {
300 if (!fields.at(i).simplified().startsWith(s + " "))
301 continue;
302 fields.removeAt(i);
303 break;
304 }
305 } while (query.next());
306 for (int i = 0; i < fields.size(); ++i) {
307 if (!fields.at(i).simplified().startsWith("FOREIGN KEY"))
308 continue;
309 fields.removeAt(i);
310 i--;
311 }
312 if (!fields.isEmpty()) {
313 db->transaction(query);
314 for (int i = 0; i < fields.size(); ++i) {
315 QString s = QString("ALTER TABLE '%1' ADD %2").arg(tableName).arg(fields.at(i));
316 qDebug() << s;
317 query.prepare(s);
318 if (!query.exec())
319 return false;
320 }
321 apxMsg() << tr("Table %1 updated").arg(tableName);
322 db->commit(query);
323 }
324 return true;
325}
