95bool FactData::updateValue(const QVariant &v)
96{
97 //filter the same
98 const QVariant &v_prev = value();
99
100 if (v_prev == v)
101 return false;
102
103 switch (dataType()) {
104 default:
105 m_value = v;
106 break;
107 case Enum: {
108 //qDebug()<<"set"<<path()<<vx<<"->"<<ev;
109 int enumIndex = enumValue(v);
110 if (enumIndex < 0) {
111 if (m_enumStrings.size() != 2)
112 return false;
113 enumIndex = _string_to_bool(v.toString());
114 if (enumIndex < 0)
115 return false;
116 }
117 if (v_prev.toInt() == enumIndex)
118 return false;
119 //qDebug() << path() << m_value << enumIndex;
120 m_value = QVariant::fromValue(enumIndex);
121 } break;
122 case Bool: {
123 int enumIndex = enumValue(v);
124 if (enumIndex < 0) {
125 enumIndex = _string_to_bool(v.toString());
126 if (enumIndex < 0)
127 enumIndex = 0;
128 }
129 if (v_prev.toBool() == enumIndex)
130 return false;
131 m_value = QVariant::fromValue(enumIndex);
132 } break;
133 case Text: {
134 QString s;
135 if (m_units.isEmpty()) {
136 int enumIndex = enumValue(v);
137 s = enumIndex < 0 ? v.toString() : enumText(enumIndex);
138 } else {
139 s = v.toString();
140 }
141
142 if (v_prev.toString() == s)
143 return false;
144 m_value = QVariant::fromValue(s);
145 } break;
146 case Int: {
147 if (units() == "mandala") {
148 quint32 bind_raw;
149 do {
150 if (v.isNull()) {
151 bind_raw = 0;
152 break;
153 }
154 bool ok = false;
155 bind_raw = v.toUInt(&ok);
156 if (_check_int(v) || ok) {
157 if (mandalaToString(bind_raw).isEmpty()) {
158 bind_raw = 0;
159 }
160 break;
161 }
162 bind_raw = stringToMandala(v.toString().trimmed());
163 } while (0);
164 const QVariant &vx = bind_raw ? QVariant::fromValue(bind_raw) : QVariant();
165 if (v_prev == vx)
166 return false;
167 m_value = vx;
168 break;
169 }
170
171 QVariant vx = v;
172 QString s = vx.toString();
173 QString u = units();
174 if (!u.isEmpty() && s.endsWith(u)) {
175 s = s.left(s.size() - u.size()).trimmed();
176 vx = s;
177 }
178
179 int enumIndex = enumValue(vx);
180 if (enumIndex >= 0) {
181 m_value = enumIndex;
182 break;
183 }
184 if (m_enumStrings.size() > 1)
185 return false;
186
187 long long vi = 0;
188 bool ok = false;
189
190 if (u == "hex") {
191 vi = s.toLongLong(&ok, 16);
192 if (!ok)
193 return false;
194 } else if (u == "s") {
195 vi = static_cast<long long>(AppRoot::timeFromString(s, true));
196 ok = true;
197 } else if (u == "min") {
198 vi = static_cast<long long>(AppRoot::timeFromString(s, false));
199 ok = true;
200 } else if (!_check_int(vx)) {
201 vi = s.toLongLong(&ok);
202 if (!ok)
203 vi = static_cast<long long>(round(s.toDouble(&ok)));
204 if (!ok)
205 vi = s.toLongLong(&ok, 16);
206 if (!ok)
207 return false;
208 } else {
209 vi = vx.toLongLong();
210 }
211 if (!m_min.isNull()) {
212 long long m = m_min.toLongLong();
213 if (vi < m)
214 vi = m;
215 }
216 if (!m_max.isNull()) {
217 long long m = m_max.toLongLong();
218 if (vi > m)
219 vi = m;
220 }
221 if (v_prev.toLongLong() == vi)
222 return false;
223 m_value = QVariant::fromValue(vi);
224 } break;
225 case Float: {
226 qreal vf;
227 if (!(_check_type(v, QMetaType::Double) || _check_type(v, QMetaType::Float))) {
228 QString s = v.toString();
229 QString u = units();
230 if (!u.isEmpty() && s.endsWith(u)) {
231 s = s.left(s.size() - u.size()).trimmed();
232 }
233 if (u == "lat") {
234 vf = AppRoot::latFromString(s);
235 } else if (u == "lon") {
236 vf = AppRoot::lonFromString(s);
237 } else {
238 bool ok = false;
239 vf = s.toDouble(&ok);
240 if (!ok) {
241 int i = _string_to_int(s);
242 if (i < 0)
243 return false;
244 vf = i;
245 }
246 if ((!m_min.isNull()) && vf < m_min.toDouble())
247 vf = m_min.toDouble();
248 if ((!m_max.isNull()) && vf > m_max.toDouble())
249 vf = m_max.toDouble();
250 }
251 } else
252 vf = v.toDouble();
253
254 if (v_prev.toDouble() == vf)
255 return false;
256
257 m_value = QVariant::fromValue(vf);
258 } break;
259 }
260 return true;
261}

287int FactData::_string_to_bool(QString s)
288{
289 s = s.toLower();
290 if (s == "true" || s == "on" || s == "yes")
291 return 1;
292 else if (s == "false" || s == "off" || s == "no")
293 return 0;
294 int v = _string_to_int(s);
295 if (v < 0)
296 return -1;
297 return v > 0 ? 1 : 0;
298}

367static double cint(double x)
368{
369 double i;
370 if (std::modf(x, &i) >= .5)
371 return x >= 0 ? std::ceil(x) : std::floor(x);
372 else
373 return x < 0 ? std::ceil(x) : std::floor(x);
374}
375QString FactData::toText(const QVariant &v) const
376{
377 Flag t = dataType();
378 if ((treeType() != Group) && (!m_enumStrings.isEmpty())) {
379 int ev = enumValue(v);
380 if (ev >= 0)
381 return enumText(ev);
382 if (t == Enum)
383 return v.toString();
384 }
385 if (t == Int) {
386 if (units() == "mandala")
387 return mandalaToString(static_cast<quint32>(v.toUInt()));
388 return QString::number(v.toString().toLongLong());
389 }
390 if (t == Bool) {
391 return QVariant(v.toBool()).toString();
392 }
393 if (t == Float) {
394 if (units() == "lat") {
395 return AppRoot::latToString(v.toDouble());
396 }
397 if (units() == "lon") {
398 return AppRoot::lonToString(v.toDouble());
399 }
400 }
401 if (_check_type(v, QMetaType::QByteArray))
402 return v.toByteArray().toHex().toUpper();
403 if (_check_type(v, QMetaType::Double) || _check_type(v, QMetaType::Float)) {
404 double vf = v.toDouble();
405 if (m_precision > 0) {
406 double p = std::pow(10.0, m_precision);
407 vf = cint(vf * p) / p;
408 } else if (m_precision == 0) {
409 vf = cint(vf);
410 }
411 QString s;
412 if (vf == 0.0)
413 s = "0";
414 else if (m_precision == 0) {
415 s = QString::number((qint64) vf);
416 } else {
417 s = QString("%1").arg(vf, 0, 'f', m_precision >= 0 ? m_precision : 8);
418 if (s.contains('.')) {
419 if (m_precision >= 0)
420 s = s.left(s.indexOf('.') + 1 + m_precision);
421 while (s.endsWith('0'))
422 s.chop(1);
423 if (s.endsWith('.'))
424 s.chop(1);
425 }
426 }
427 return s;
428 }
429 return v.toString();
430}

432bool FactData::isZero() const
433{
434 if (treeType() == Group) {
435 for (auto f : facts()) {
436 if (!f->isZero())
437 return false;
438 }
439 return true;
440 }
441
442 const QVariant &v = value();
443 if (v.isNull())
444 return true;
445
446 Flag t = dataType();
447 if (t == Text)
448 return v.toString().isEmpty();
449 if (t == Float)
450 return v.toDouble() == 0.0;
451 const QString &s = v.toString().trimmed();
452 if (s.isEmpty())
453 return true;
454 if (s == "0")
455 return true;
456 if (v.toInt() == 0)
457 return true;
458
459 return false;
460}

806void FactData::savePresistentValue()
807{
808 const QVariant v = value();
809 QString s = toText(v);
810 bool rm = false;
811
812 if (!m_defaultValue.isNull()) {
813 rm = v == defaultValue() || s == defaultValue() || s == toText(defaultValue());
814 }
815
816 if (m_options & SystemSettings) {
817 if (rm)
818 QSettings().remove(path());
819 else
820 QSettings().setValue(path(), s);
821 } else {
822 if (rm)
823 AppPrefs::instance()->removeValue(name(), prefsGroup());
824 else
825 AppPrefs::instance()->saveValue(name(), s, prefsGroup());
826 }
