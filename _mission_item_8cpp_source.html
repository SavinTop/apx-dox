29MissionItem::MissionItem(MissionGroup *parent,
30 const QString &name,
31 const QString &title,
32 const QString &descr)
33 : Fact(parent, name, title, descr, Group | ModifiedGroup)
34 , group(parent)
35 , blockUpdateCoordinate(false)
36 , m_bearing(0)
37 , m_time(0)
38 , m_distance(0)
39 , m_totalDistance(0)
40 , m_totalTime(0)
41 , m_selected(false)
42{
43 setOpt("pos", QPointF(0.25, 0.5));
44
45 f_order = new MissionField(this, "order", tr("Order"), tr("Object sequence number"), Int);
46 f_order->setMin(1);
47 updateOrderState();
48 connect(f_order, &Fact::valueChanged, this, &MissionItem::updateOrder);
49 connect(this, &Fact::numChanged, this, &MissionItem::updateOrderState);
50 connect(group, &Fact::sizeChanged, this, &MissionItem::updateOrderState);
51
52 f_latitude = new MissionField(this,
53 "lat",
54 tr("Latitude"),
55 tr("Global postition latitude"),
56 Float);
57 f_latitude->setUnits("lat");
58 f_longitude = new MissionField(this,
59 "lon",
60 tr("Longitude"),
61 tr("Global postition longitude"),
62 Float);
63 f_longitude->setUnits("lon");
64
65 f_remove = new Fact(this, "remove", tr("Remove"), tr("Remove mission item"), Action | Remove);
66 connect(f_remove, &Fact::triggered, this, &Fact::deleteFact);
67
68 //active index
69 if (group->f_activeIndex) {
70 Fact *f_select = new Fact(this,
71 "select",
72 tr("Set as current"),
73 group->f_activeIndex->descr(),
74 Action | Apply,
75 "target");
76 connect(f_select, &Fact::triggered, this, &MissionItem::selectTriggered);
77
78 connect(group->f_activeIndex, &Fact::valueChanged, this, [this]() {
79 setActive(group->f_activeIndex->value().toInt() == num());
80 });
81 setActive(group->f_activeIndex->value().toInt() == num());
82 }
83
84 connect(f_latitude, &Fact::valueChanged, this, &MissionItem::updateCoordinate);
85 connect(f_longitude, &Fact::valueChanged, this, &MissionItem::updateCoordinate);
86
87 connect(this, &MissionItem::coordinateChanged, this, &MissionItem::updatePath);
88
89 connect(this, &Fact::numChanged, this, &MissionItem::updatePath, Qt::QueuedConnection);
90
91 //selection support
92 connect(group->mission,
93 &VehicleMission::selectedItemChanged,
94 this,
95 &MissionItem::updateSelected);
96
97 //title
98 connect(this, &Fact::numChanged, this, &MissionItem::updateTitle);
99 updateTitle();
100
101 //totals
102 connect(this, &MissionItem::timeChanged, group, &MissionGroup::updateTime);
103 connect(this, &MissionItem::distanceChanged, group, &MissionGroup::updateDistance);
104
105 connect(this, &MissionItem::totalTimeChanged, this, &MissionItem::updateStatus);
106 connect(this, &MissionItem::totalDistanceChanged, this, &MissionItem::updateStatus);
107 updateStatus();
108}

137void MissionItem::updateStatus()
138{
139 uint d = totalDistance();
140 uint t = totalTime();
141 if ((d | t) == 0)
142 setValue(QVariant());
143 else {
144 QStringList st;
145 st.append(AppRoot::distanceToString(d));
146 st.append(AppRoot::timeToString(t, false));
147 setValue(st.join(' '));
148 }

158void MissionItem::updatePath()
159{
160 setGeoPath(getPath());
161 //check to propagate updates to next items
162 MissionItem *next = nextItem();
163 if (next) {
164 QGeoCoordinate p1, p2;
165 if (m_geoPath.path().size() >= 2)
166 p1 = QGeoCoordinate(m_geoPath.path().last());
167 if (next->geoPath().path().size() >= 2)
168 p2 = QGeoCoordinate(next->geoPath().path().first());
169 if (next->geoPath().path().size() < 2 || m_geoPath.path().size() < 2
170 || p1.latitude() != p2.latitude() || p1.longitude() != p2.longitude()) {
171 next->updatePath();
172 }
173 }
174}

176void MissionItem::updateOrder()
177{
178 int n = f_order->value().toInt() - 1;
179 if (n < 0)
180 n = 0;
181 else if (n >= group->size())
182 n = group->size() - 1;
183 if (n == num() || group->size() < 2) {
184 f_order->setValue(num() + 1);
185 return;
186 }
187 move(n, true);
188}
