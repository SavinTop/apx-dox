27static QRegion qwtAlphaMask(const QImage &image, const QVector<QRect> &rects)
28{
29 const int w = image.width();
30 const int h = image.height();
31
32 QRegion region;
33 QRect rect;
34
35 for (int i = 0; i < rects.size(); i++) {
36 int x1, x2, y1, y2;
37 rects[i].getCoords(&x1, &y1, &x2, &y2);
38
39 x1 = qMax(x1, 0);
40 x2 = qMin(x2, w - 1);
41 y1 = qMax(y1, 0);
42 y2 = qMin(y2, h - 1);
43
44 for (int y = y1; y <= y2; ++y) {
45 bool inRect = false;
46 int rx0 = -1;
47
48 const uint *line = reinterpret_cast<const uint *>(image.scanLine(y)) + x1;
49 for (int x = x1; x <= x2; x++) {
50 const bool on = ((*line++ >> 24) != 0);
51 if (on != inRect) {
52 if (inRect) {
53 rect.setCoords(rx0, y, x - 1, y);
54 region += rect;
55 } else {
56 rx0 = x;
57 }
58
59 inRect = on;
60 }
61 }
62
63 if (inRect) {
64 rect.setCoords(rx0, y, x2, y);
65 region = region.united(rect);
66 }
67 }
68 }
69
70 return region;
71}

174void QwtWidgetOverlay::updateMask()
175{
176 d_data->resetRgbaBuffer();
177
178 QRegion mask;
179
180 if (d_data->maskMode == QwtWidgetOverlay::MaskHint) {
181 mask = maskHint();
182 } else if (d_data->maskMode == QwtWidgetOverlay::AlphaMask) {
183 // xTODO: the image doesn't need to be larger than
184 // the bounding rectangle of the hint !!
185
186 QRegion hint = maskHint();
187 if (hint.isEmpty())
188 hint += QRect(0, 0, width(), height());
189
190 // A fresh buffer from calloc() is usually faster
191 // than reinitializing an existing one with
192 // QImage::fill( 0 ) or memset()
193
194 d_data->rgbaBuffer = (uchar *) ::calloc(width() * height(), 4);
195
196 QImage image(d_data->rgbaBuffer, width(), height(), qwtMaskImageFormat());
197
198 QPainter painter(&image);
199 draw(&painter);
200 painter.end();
201
202 mask = qwtAlphaMask(image, hint.rects());
203
204 if (d_data->renderMode == QwtWidgetOverlay::DrawOverlay) {
205 // we don't need the buffer later
206 d_data->resetRgbaBuffer();
207 }
208 }
209
210 // A bug in Qt initiates a full repaint of the widget
211 // when we change the mask, while we are visible !
212
213 setVisible(false);
214
215 if (mask.isEmpty())
216 clearMask();
217 else
218 setMask(mask);
219
220 setVisible(true);
221}

229void QwtWidgetOverlay::paintEvent(QPaintEvent *event)
230{
231 const QRegion &clipRegion = event->region();
232
233 QPainter painter(this);
234
235 bool useRgbaBuffer = false;
236 if (d_data->renderMode == QwtWidgetOverlay::CopyAlphaMask) {
237 useRgbaBuffer = true;
238 } else if (d_data->renderMode == QwtWidgetOverlay::AutoRenderMode) {
239 if (painter.paintEngine()->type() == QPaintEngine::Raster)
240 useRgbaBuffer = true;
241 }
242
243 if (d_data->rgbaBuffer && useRgbaBuffer) {
244 const QImage image(d_data->rgbaBuffer, width(), height(), qwtMaskImageFormat());
245
246 QVector<QRect> rects;
247 if (clipRegion.rects().size() > 2000) {
248 // the region is to complex
249 painter.setClipRegion(clipRegion);
250 rects += clipRegion.boundingRect();
251 } else {
252 rects = clipRegion.rects();
253 }
254
255 for (int i = 0; i < rects.size(); i++) {
256 const QRect r = rects[i];
257 painter.drawImage(r.topLeft(), image, r);
258 }
259 } else {
260 painter.setClipRegion(clipRegion);
261 draw(&painter);
262 }
263}

276void QwtWidgetOverlay::draw(QPainter *painter) const
277{
278 QWidget *widget = const_cast<QWidget *>(parentWidget());
279 if (widget) {
280 painter->setClipRect(parentWidget()->contentsRect());
281
282 // something special for the plot canvas
283
284 const int idx = widget->metaObject()->indexOfMethod("borderPath(QRect)");
285 if (idx >= 0) {
286 QPainterPath clipPath;
287
288 (void) QMetaObject::invokeMethod(widget,
289 "borderPath",
290 Qt::DirectConnection,
291 Q_RETURN_ARG(QPainterPath, clipPath),
292 Q_ARG(QRect, rect()));
293
294 if (!clipPath.isEmpty())
295 painter->setClipPath(clipPath, Qt::IntersectClip);
296 }
297 }
298
299 drawOverlay(painter);
300}
