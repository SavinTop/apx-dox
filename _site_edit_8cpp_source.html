84void SiteEdit::loadFromModelData()
85{
86 blockUpdateItemData = true;
87 f_title->setValue(modelData.value("title").toString());
88 f_latitude->setValue(modelData.value("lat").toDouble());
89 f_longitude->setValue(modelData.value("lon").toDouble());
90 blockUpdateItemData = false;
91}
92void SiteEdit::saveToModelData()
93{
94 if (blockUpdateItemData)
95 return;
96 if (f_title->text().isEmpty()) {
97 loadFromModelData();
98 return;
99 }
100 double lat = f_latitude->value().toDouble();
101 double lon = f_longitude->value().toDouble();
102 if (std::isnan(lat) || std::isnan(lon) || lat == 0.0 || lon == 0.0) {
103 loadFromModelData();
104 return;
105 }
106 modelData["title"] = f_title->text();
107 modelData["lat"] = lat;
108 modelData["lon"] = lon;
109 emit siteEdited(modelData);
110}
111void SiteEdit::setModelData(QVariantMap v)
112{
113 if (modelData.value("key").toULongLong() != v.value("key").toULongLong())
114 return;
115 modelData = v;
116 loadFromModelData();
117}
118void SiteEdit::updateFromEditedModelData(int i, QVariantMap v)
119{
120 Q_UNUSED(i)
121 setModelData(v);
122 emit siteEdited(modelData);
123}
