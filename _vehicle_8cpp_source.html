256void Vehicle::updateInfo()
257{
258 if (isIdentified()) {
259 _lastSeenTime = QDateTime::currentDateTime().toMSecsSinceEpoch();
260 }
261 QStringList st;
262 //st<<callsign();
263 if (!isReplay()) {
264 QString s;
265 int alt = f_hmsl->value().toInt();
266 if (std::abs(alt) >= 50)
267 alt = (alt / 10) * 10;
268 else if (alt < 1)
269 alt = 0;
270 s = QString("MSL%1").arg(alt);
271
272 int vs = f_vspeed->value().toInt();
273 if (vs > 1)
274 s.append(QString("+%1").arg(vs));
275 else if (vs < -1)
276 s.append(QString("%1").arg(vs));
277 st << s;
278 s = f_mode->text();
279 uint stage = f_stage->value().toUInt();
280 if (stage > 1)
281 s.append(QString("/%1").arg(stage));
282 st << s;
283 }
284
285 QString v = st.join("\n");
286 if (m_info == v)
287 return;
288 m_info = v;
289 emit infoChanged();
290}

406QString Vehicle::confTitle() const
407{
408 //if (protocol()->nodes->size() <= 0)
409 //return QString();
410
411 QMap<QString, QString> byName;
412 QString shiva;
413 QString longest;
414 QString anyName;
415 foreach (NodeItem *node, f_nodes->nodes()) {
416 QString name = node->title();
417 QString s = node->value().toString().trimmed();
418 if (anyName.isEmpty())
419 anyName = name;
420 if (s.isEmpty())
421 continue;
422 int sz = s.size();
423 if (byName.value(name).size() < sz)
424 byName[name] = s;
425 if (name.endsWith(".shiva")) {
426 if (shiva.size() < sz)
427 shiva = s;
428 }
429 if (longest.size() < sz)
430 longest = s;
431 }
432 if (!shiva.isEmpty())
433 return shiva;
434 QString s;
435 s = byName.value("nav");
436 if (!s.isEmpty())
437 return s;
438 s = byName.value("com");
439 if (!s.isEmpty())
440 return s;
441 s = byName.value("ifc");
442 if (!s.isEmpty())
443 return s;
444 s = longest;
445 if (!s.isEmpty())
446 return s;
447 return anyName;
448}

450void Vehicle::message(QString msg, AppNotify::NotifyFlags flags, QString subsystem)
451{
452 if (subsystem.isEmpty())
453 subsystem = title();
454 else
455 subsystem = QString("%1/%2").arg(title()).arg(subsystem);
456
457 AppNotify::NotifyFlags fType = flags & AppNotify::NotifyTypeMask;
458
459 static const QStringList err(QStringList() << "err"
460 << "fail"
461 << "critical"
462 << "timeout");
463 static const QStringList warn(QStringList() << "warn"
464 << "nofix"
465 << "missing"
466 << "none"
467 << "not found");
468
469 if (fType != AppNotify::Error && fType != AppNotify::Warning) {
470 AppNotify::NotifyFlags t = fType;
471 for (auto i : err) {
472 if (!msg.contains(i, Qt::CaseInsensitive))
473 continue;
474 t = AppNotify::Error;
475 break;
476 }
477 if (t != AppNotify::Error) {
478 for (auto i : warn) {
479 if (!msg.contains(i, Qt::CaseInsensitive))
480 continue;
481 t = AppNotify::Warning;
482 break;
483 }
484 }
485 if (t != fType)
486 flags = (flags & ~AppNotify::NotifyTypeMask) | t;
487 fType = t;
488 }
489 AppNotify::instance()->report(msg, flags, subsystem);
490
491 if (fType == AppNotify::Error) {
492 f_warnings->error(msg);
493 } else if (fType == AppNotify::Warning) {
494 f_warnings->warning(msg);
495 }
496}
