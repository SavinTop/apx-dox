257QGeoRectangle VehicleMission::boundingGeoRectangle() const
258{
259 QGeoRectangle r;
260 foreach (MissionGroup *group, groups) {
261 for (int i = 0; i < group->size(); ++i) {
262 QGeoRectangle re = static_cast<MissionItem *>(group->child(i))->boundingGeoRectangle();
263 r = r.isValid() ? r.united(re) : re;
264 }
265 }
266 return r;

284QVariant VehicleMission::toVariant()
285{
286 auto m = Fact::toVariant().value<QVariantMap>();
287
288 m.insert("format", MISSION_FORMAT);
289 m.insert("exported", QDateTime::currentDateTime().toString(Qt::RFC2822Date));
290 m.insert("version", App::version());
291
292 m.remove(f_title->name());
293
294 if (!site().isEmpty())
295 m.insert("site", site());
296
297 QGeoCoordinate c = coordinate();
298 m.insert("lat", c.latitude());
299 m.insert("lon", c.longitude());
300
301 QString title = f_title->text().simplified();
302 if (vehicle->isIdentified()) {
303 QString s = vehicle->title();
304 m.insert("callsign", s);
305 title.remove(s, Qt::CaseInsensitive);
306 }
307 if (f_runways->size() > 0) {
308 QString s = f_runways->child(0)->text();
309 m.insert("runway", s);
310 title.remove(s, Qt::CaseInsensitive);
311 }
312 title.replace('-', ' ');
313 title.replace('_', ' ');
314 title = title.simplified();
315 if (!title.isEmpty())
316 m.insert("title", title);
317
318 //details
319 QGeoRectangle rect = boundingGeoRectangle();
320 m.insert("topLeftLat", rect.topLeft().latitude());
321 m.insert("topLeftLon", rect.topLeft().longitude());
322 m.insert("bottomRightLat", rect.bottomRight().latitude());
323 m.insert("bottomRightLon", rect.bottomRight().longitude());
324 m.insert("distance", f_waypoints->distance());
325
326 //generate hash
327 QCryptographicHash hash(QCryptographicHash::Sha1);
328 hash.addData(QJsonDocument::fromVariant(m.value("rw")).toJson(QJsonDocument::Compact));
329 hash.addData(QJsonDocument::fromVariant(m.value("wp")).toJson(QJsonDocument::Compact));
330 hash.addData(QJsonDocument::fromVariant(m.value("tw")).toJson(QJsonDocument::Compact));
331 hash.addData(QJsonDocument::fromVariant(m.value("pi")).toJson(QJsonDocument::Compact));
332 m.insert("hash", hash.result().toHex().toUpper());
333
334 m.insert("time", QDateTime::currentDateTime().toMSecsSinceEpoch());
335 return m;
336}
337void VehicleMission::fromVariant(const QVariant &var)
338{
339 clearMission();
340
341 auto m = var.value<QVariantMap>();
342 setSite(m.value("site").toString());
343
344 f_title->setValue(m.value("title").toString());
345
346 blockSizeUpdate = true;
347 for (auto i : groups) {
348 i->fromVariant(m.value(i->name()));
349 }
350 blockSizeUpdate = false;
351
352 backup();
353 updateSize();
354
355 App::jsync(this);
356}

365void VehicleMission::test(int n)
366{
367 if (f_waypoints->size() <= 0)
368 return;
369 Waypoint *w = static_cast<Waypoint *>(f_waypoints->facts().last());
370 QGeoCoordinate p(w->f_latitude->value().toDouble(), w->f_longitude->value().toDouble());
371 double hdg = QRandomGenerator::global()->bounded(360.0);
372 for (int i = 0; i < n; ++i) {
373 hdg += QRandomGenerator::global()->bounded(200.0) - 100.0;
374 p = p.atDistanceAndAzimuth(100 + QRandomGenerator::global()->bounded(10000.0), hdg);
375 f_waypoints->addObject(p);
376 }
377}

379void VehicleMission::missionReceived(QVariant var)
380{
381 clearMission();
382 //qDebug() << json;
383
384 fromVariant(var);
385
386 if (empty()) {
387 vehicle->message(tr("Empty mission received from vehicle"), AppNotify::Warning);
388 } else {
389 emit missionAvailable();
390 emit missionDownloaded();
391 storage->saveMission();
392 QString s = QString("%1: %2").arg(tr("Mission received")).arg(text());
393 auto node_uid = var.value<QVariantMap>().value("node_uid").toString();
394 if (!node_uid.isEmpty()) {
395 auto node = vehicle->f_nodes->node(node_uid);
396 if (node) {
397 s.append(QString(" (%1)").arg(node->title()));
398 }
399 }
400 vehicle->message(s, AppNotify::Important);
401 }
402 backup();
403}
