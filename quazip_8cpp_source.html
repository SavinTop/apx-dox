165void QuaZipPrivate::addCurrentFileToDirectoryMap(const QString &fileName)
166{
167 if (!hasCurrentFile_f || fileName.isEmpty()) {
168 return;
169 }
170 // Adds current file to filename map as fileName
171 unz64_file_pos fileDirectoryPos;
172 unzGetFilePos64(unzFile_f, &fileDirectoryPos);
173 directoryCaseSensitive.insert(fileName, fileDirectoryPos);
174 // Only add lowercase to directory map if not already there
175 // ensures only map the first one seen
176 QString lower = fileName.toLower();
177 if (!directoryCaseInsensitive.contains(lower))
178 directoryCaseInsensitive.insert(lower, fileDirectoryPos);
179 // Mark last one
180 if (fileDirectoryPos.pos_in_zip_directory > lastMappedDirectoryEntry.pos_in_zip_directory)
181 lastMappedDirectoryEntry = fileDirectoryPos;
182}

184bool QuaZipPrivate::goToFirstUnmappedFile()
185{
186 zipError = UNZ_OK;
187 if (mode != QuaZip::mdUnzip) {
188 qWarning("QuaZipPrivate::goToNextUnmappedFile(): ZIP is not open in mdUnzip mode");
189 return false;
190 }
191 // If not mapped anything, go to beginning
192 if (lastMappedDirectoryEntry.pos_in_zip_directory == 0) {
193 unzGoToFirstFile(unzFile_f);
194 } else {
195 // Goto the last one mapped, plus one
196 unzGoToFilePos64(unzFile_f, &lastMappedDirectoryEntry);
197 unzGoToNextFile(unzFile_f);
198 }
199 hasCurrentFile_f=zipError==UNZ_OK;
200 if(zipError==UNZ_END_OF_LIST_OF_FILE)
201 zipError=UNZ_OK;
202 return hasCurrentFile_f;
203}

227bool QuaZip::open(Mode mode, zlib_filefunc_def* ioApi)
228{
229 p->zipError=UNZ_OK;
230 if(isOpen()) {
231 qWarning("QuaZip::open(): ZIP already opened");
232 return false;
233 }
234 QIODevice *ioDevice = p->ioDevice;
235 if (ioDevice == NULL) {
236 if (p->zipName.isEmpty()) {
237 qWarning("QuaZip::open(): set either ZIP file name or IO device first");
238 return false;
239 } else {
240 ioDevice = new QFile(p->zipName);
241 }
242 }
243 unsigned flags = 0;
244 switch(mode) {
245 case mdUnzip:
246 if (ioApi == NULL) {
247 if (p->autoClose)
248 flags |= UNZ_AUTO_CLOSE;
249 p->unzFile_f=unzOpenInternal(ioDevice, NULL, 1, flags);
250 } else {
251 // QuaZIP pre-zip64 compatibility mode
252 p->unzFile_f=unzOpen2(ioDevice, ioApi);
253 if (p->unzFile_f != NULL) {
254 if (p->autoClose) {
255 unzSetFlags(p->unzFile_f, UNZ_AUTO_CLOSE);
256 } else {
257 unzClearFlags(p->unzFile_f, UNZ_AUTO_CLOSE);
258 }
259 }
260 }
261 if(p->unzFile_f!=NULL) {
262 if (ioDevice->isSequential()) {
263 unzClose(p->unzFile_f);
264 if (!p->zipName.isEmpty())
265 delete ioDevice;
266 qWarning("QuaZip::open(): "
267 "only mdCreate can be used with "
268 "sequential devices");
269 return false;
270 }
271 p->mode=mode;
272 p->ioDevice = ioDevice;
273 return true;
274 } else {
275 p->zipError=UNZ_OPENERROR;
276 if (!p->zipName.isEmpty())
277 delete ioDevice;
278 return false;
279 }
280 case mdCreate:
281 case mdAppend:
282 case mdAdd:
283 if (ioApi == NULL) {
284 if (p->autoClose)
285 flags |= ZIP_AUTO_CLOSE;
286 if (p->dataDescriptorWritingEnabled)
287 flags |= ZIP_WRITE_DATA_DESCRIPTOR;
288 if (p->utf8)
289 flags |= ZIP_ENCODING_UTF8;
290 p->zipFile_f=zipOpen3(ioDevice,
291 mode==mdCreate?APPEND_STATUS_CREATE:
292 mode==mdAppend?APPEND_STATUS_CREATEAFTER:
293 APPEND_STATUS_ADDINZIP,
294 NULL, NULL, flags);
295 } else {
296 // QuaZIP pre-zip64 compatibility mode
297 p->zipFile_f=zipOpen2(ioDevice,
298 mode==mdCreate?APPEND_STATUS_CREATE:
299 mode==mdAppend?APPEND_STATUS_CREATEAFTER:
300 APPEND_STATUS_ADDINZIP,
301 NULL,
302 ioApi);
303 if (p->zipFile_f != NULL) {
304 zipSetFlags(p->zipFile_f, flags);
305 }
306 }
307 if(p->zipFile_f!=NULL) {
308 if (ioDevice->isSequential()) {
309 if (mode != mdCreate) {
310 zipClose(p->zipFile_f, NULL);
311 qWarning("QuaZip::open(): "
312 "only mdCreate can be used with "
313 "sequential devices");
314 if (!p->zipName.isEmpty())
315 delete ioDevice;
316 return false;
317 }
318 zipSetFlags(p->zipFile_f, ZIP_SEQUENTIAL);
319 }
320 p->mode=mode;
321 p->ioDevice = ioDevice;
322 return true;
323 } else {
324 p->zipError=UNZ_OPENERROR;
325 if (!p->zipName.isEmpty())
326 delete ioDevice;
327 return false;
328 }
329 default:
330 qWarning("QuaZip::open(): unknown mode: %d", (int)mode);
331 if (!p->zipName.isEmpty())
332 delete ioDevice;
333 return false;
334 break;
335 }
336}

338void QuaZip::close()
339{
340 p->zipError=UNZ_OK;
341 switch(p->mode) {
342 case mdNotOpen:
343 qWarning("QuaZip::close(): ZIP is not open");
344 return;
345 case mdUnzip:
346 p->zipError=unzClose(p->unzFile_f);
347 break;
348 case mdCreate:
349 case mdAppend:
350 case mdAdd:
351 p->zipError=zipClose(p->zipFile_f, p->comment.isNull() ? NULL : isUtf8Enabled()
352 ? p->comment.toUtf8().constData()
353 : p->commentCodec->fromUnicode(p->comment).constData());
354 break;
355 default:
356 qWarning("QuaZip::close(): unknown mode: %d", (int)p->mode);
357 return;
358 }
359 // opened by name, need to delete the internal IO device
360 if (!p->zipName.isEmpty()) {
361 delete p->ioDevice;
362 p->ioDevice = NULL;
363 }
364 p->clearDirectoryMap();
365 if(p->zipError==UNZ_OK)
366 p->mode=mdNotOpen;
367}

424bool QuaZip::setCurrentFile(const QString& fileName, CaseSensitivity cs)
425{
426 p->zipError=UNZ_OK;
427 if(p->mode!=mdUnzip) {
428 qWarning("QuaZip::setCurrentFile(): ZIP is not open in mdUnzip mode");
429 return false;
430 }
431 if(fileName.isEmpty()) {
432 p->hasCurrentFile_f=false;
433 return true;
434 }
435 // Unicode-aware reimplementation of the unzLocateFile function
436 if(p->unzFile_f==NULL) {
437 p->zipError=UNZ_PARAMERROR;
438 return false;
439 }
440 if(fileName.length()>MAX_FILE_NAME_LENGTH) {
441 p->zipError=UNZ_PARAMERROR;
442 return false;
443 }
444 // Find the file by name
445 bool sens = convertCaseSensitivity(cs) == Qt::CaseSensitive;
446 QString lower, current;
447 if(!sens) lower=fileName.toLower();
448 p->hasCurrentFile_f=false;
449
450 // Check the appropriate Map
451 unz64_file_pos fileDirPos;
452 fileDirPos.pos_in_zip_directory = 0;
453 if (sens) {
454 if (p->directoryCaseSensitive.contains(fileName))
455 fileDirPos = p->directoryCaseSensitive.value(fileName);
456 } else {
457 if (p->directoryCaseInsensitive.contains(lower))
458 fileDirPos = p->directoryCaseInsensitive.value(lower);
459 }
460
461 if (fileDirPos.pos_in_zip_directory != 0) {
462 p->zipError = unzGoToFilePos64(p->unzFile_f, &fileDirPos);
463 p->hasCurrentFile_f = p->zipError == UNZ_OK;
464 }
465
466 if (p->hasCurrentFile_f)
467 return p->hasCurrentFile_f;
468
469 // Not mapped yet, start from where we have got to so far
470 for(bool more=p->goToFirstUnmappedFile(); more; more=goToNextFile()) {
471 current=getCurrentFileName();
472 if(current.isEmpty()) return false;
473 if(sens) {
474 if(current==fileName) break;
475 } else {
476 if(current.toLower()==lower) break;
477 }
478 }
479 return p->hasCurrentFile_f;
480}

522bool QuaZip::getCurrentFileInfo(QuaZipFileInfo64 *info)const
523{
524 QuaZip *fakeThis=(QuaZip*)this; // non-const
525 fakeThis->p->zipError=UNZ_OK;
526 if(p->mode!=mdUnzip) {
527 qWarning("QuaZip::getCurrentFileInfo(): ZIP is not open in mdUnzip mode");
528 return false;
529 }
530 unz_file_info64 info_z;
531 QByteArray fileName;
532 QByteArray extra;
533 QByteArray comment;
534 if(info==NULL) return false;
535 if(!isOpen()||!hasCurrentFile()) return false;
536 if((fakeThis->p->zipError=unzGetCurrentFileInfo64(p->unzFile_f, &info_z, NULL, 0, NULL, 0, NULL, 0))!=UNZ_OK)
537 return false;
538 fileName.resize(info_z.size_filename);
539 extra.resize(info_z.size_file_extra);
540 comment.resize(info_z.size_file_comment);
541 if((fakeThis->p->zipError=unzGetCurrentFileInfo64(p->unzFile_f, NULL,
542 fileName.data(), fileName.size(),
543 extra.data(), extra.size(),
544 comment.data(), comment.size()))!=UNZ_OK)
545 return false;
546 info->versionCreated=info_z.version;
547 info->versionNeeded=info_z.version_needed;
548 info->flags=info_z.flag;
549 info->method=info_z.compression_method;
550 info->crc=info_z.crc;
551 info->compressedSize=info_z.compressed_size;
552 info->uncompressedSize=info_z.uncompressed_size;
553 info->diskNumberStart=info_z.disk_num_start;
554 info->internalAttr=info_z.internal_fa;
555 info->externalAttr=info_z.external_fa;
556 info->name=(info->flags & UNZ_ENCODING_UTF8) ? QString::fromUtf8(fileName) : p->fileNameCodec->toUnicode(fileName);
557 info->comment=(info->flags & UNZ_ENCODING_UTF8) ? QString::fromUtf8(comment) : p->commentCodec->toUnicode(comment);
558 info->extra=extra;
559 info->dateTime=QDateTime(
560 QDate(info_z.tmu_date.tm_year, info_z.tmu_date.tm_mon+1, info_z.tmu_date.tm_mday),
561 QTime(info_z.tmu_date.tm_hour, info_z.tmu_date.tm_min, info_z.tmu_date.tm_sec));
562 // Add to directory map
563 p->addCurrentFileToDirectoryMap(info->name);
564 return true;
565}

567QString QuaZip::getCurrentFileName()const
568{
569 QuaZip *fakeThis=(QuaZip*)this; // non-const
570 fakeThis->p->zipError=UNZ_OK;
571 if(p->mode!=mdUnzip) {
572 qWarning("QuaZip::getCurrentFileName(): ZIP is not open in mdUnzip mode");
573 return QString();
574 }
575 if(!isOpen()||!hasCurrentFile()) return QString();
576 QByteArray fileName(MAX_FILE_NAME_LENGTH, 0);
577 unz_file_info64 file_info;
578 if((fakeThis->p->zipError=unzGetCurrentFileInfo64(p->unzFile_f, &file_info, fileName.data(), fileName.size(),
579 NULL, 0, NULL, 0))!=UNZ_OK)
580 return QString();
581 fileName.resize(file_info.size_filename);
582 QString result = (file_info.flag & UNZ_ENCODING_UTF8)
583 ? QString::fromUtf8(fileName) : p->fileNameCodec->toUnicode(fileName);
584 if (result.isEmpty())
585 return result;
586 // Add to directory map
587 p->addCurrentFileToDirectoryMap(result);
588 return result;
589}

705template<typename TFileInfo>
706bool QuaZipPrivate::getFileInfoList(QList<TFileInfo> *result) const
707{
708 QuaZipPrivate *fakeThis=const_cast<QuaZipPrivate*>(this);
709 fakeThis->zipError=UNZ_OK;
710 if (mode!=QuaZip::mdUnzip) {
711 qWarning("QuaZip::getFileNameList/getFileInfoList(): "
712 "ZIP is not open in mdUnzip mode");
713 return false;
714 }
715 QString currentFile;
716 if (q->hasCurrentFile()) {
717 currentFile = q->getCurrentFileName();
718 }
719 if (q->goToFirstFile()) {
720 do {
721 bool ok;
722 result->append(QuaZip_getFileInfo<TFileInfo>(q, &ok));
723 if (!ok)
724 return false;
725 } while (q->goToNextFile());
726 }
727 if (zipError != UNZ_OK)
728 return false;
729 if (currentFile.isEmpty()) {
730 if (!q->goToFirstFile())
731 return false;
732 } else {
733 if (!q->setCurrentFile(currentFile))
734 return false;
735 }
736 return true;
737}
