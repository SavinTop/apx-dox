51void Terminal::exec(QString cmd)
52{
53 QString s = cmd.simplified();
54 if (s.isEmpty())
55 return;
56
57 if (s == "cls") {
58 App::instance()->notifyModel()->clear();
59 return;
60 }
61
62 if (s.startsWith(';') || s.startsWith('!')) {
63 s.remove(0, 1);
64 } else if (!(s.contains("(") || s.contains("=") || s.contains("+") || s.contains("-"))) {
65 QStringList st = s.split(' ');
66 if (!st.size())
67 return;
68 QString sc = st.takeFirst();
69 if ((sc.startsWith("set") || sc.startsWith("req") || sc.startsWith("send")) && st.size()) {
70 st.insert(0, "'" + st.takeFirst() + "'"); //quote var name
71 } else if (sc == "sh") {
72 s = sc + "(['" + st.join("','") + "'])";
73 } else {
74 s = sc + "(" + st.join(",") + ")";
75 }
76 }
77 QJSValue v = App::jsexec(s);
78 //history
79 _history.removeAll(cmd);
80 //if(!v.isError())
81 _history.append(cmd);
82 while (_history.size() > MAX_HISTORY)
83 _history.removeFirst();
84 QSettings().setValue("consoleHistory", _history);
85 historyReset();
86 enterResult(!v.isError());
87}

94QString Terminal::historyNext(QString cmd)
95{
96 //qDebug()<<_history<<_historyIndex<<cmd;
97 if (_historyIndex < 0) {
98 _historyIndex = _history.size();
99 _replacedHistory = cmd;
100 }
101
102 if (_historyIndex > _history.size()) {
103 _historyIndex = 0;
104 return cmd;
105 }
106 if (_replacedHistory.isEmpty()) {
107 if (_historyIndex > 0)
108 _historyIndex--;
109 return _history.value(_historyIndex).trimmed();
110 }
111 while (_historyIndex > 0) {
112 _historyIndex--;
113 QString s = _history.value(_historyIndex).trimmed();
114 if (s.startsWith(_replacedHistory))
115 return s;
116 }
117 return cmd;
118}
119QString Terminal::historyPrev(QString cmd)
120{
121 //qDebug()<<_history<<_historyIndex<<cmd;
122 if (_historyIndex < 0)
123 return cmd;
124 if (_replacedHistory.isEmpty()) {
125 _historyIndex++;
126 } else {
127 while (_historyIndex < _history.size()) {
128 _historyIndex++;
129 if (_historyIndex == _history.size())
130 break;
131 QString s = _history[_historyIndex].trimmed();
132 if (s == cmd)
133 continue;
134 if (s.startsWith(_replacedHistory))
135 break;
136 }
137 }
138 if (_historyIndex >= _history.size()) {
139 _historyIndex = _history.size();
140 return _replacedHistory;
141 }
142 return _history[_historyIndex].trimmed();
143}

145QString Terminal::autocomplete(QString cmd)
146{
147 QString prefix = cmd;
148 QString c = cmd;
149
150 c.remove(0, c.lastIndexOf(';') + 1);
151 if (c.endsWith(' '))
152 c = c.trimmed().append(' ');
153 else
154 c = c.trimmed();
155
156 QString scope;
157 QMap<QString, QJSValue> map;
158
159 QRegExp del("[\\ \\,\\:\\t\\{\\}\\[\\]\\(\\)\\=]");
160 if (!(c.contains(del) || prefix.startsWith('!') || c.contains('.'))) {
161 //first word input (std command?)
162 map = _get_js_properties(scope, c);
163 for (auto i : map.keys()) {
164 QJSValue v = map.value(i);
165 if (v.isObject())
166 continue;
167 if (v.isCallable())
168 continue;
169 map.remove(i);
170 }
171
172 } else {
173 //parameter or not a command
174 c = c.remove(0, c.lastIndexOf(del) + 1).trimmed();
175 bool bDot = c.contains('.');
176 if (bDot) {
177 scope = c.left(c.lastIndexOf('.'));
178 c.remove(0, c.lastIndexOf('.') + 1);
179 }
180
181 map = _get_js_properties(scope, c);
182 }
183 //hints collected
184 if (map.isEmpty())
185 return cmd;
186 if (map.size() == 1) {
187 QJSValue v = map.first();
188 cmd = prefix.left(prefix.size() - c.size()) + map.keys().first();
189 if (v.isCallable())
190 return cmd + " ";
191 if (v.isObject())
192 return cmd + ".";
193 return cmd;
194 }
195
196 //partial autocompletion
197 if (c.isEmpty()) {
198 c = cmd;
199 } else {
200 //partial autocompletion
201 bool bMatch = true;
202 QString s = c;
203 QString shortest = map.keys().first();
204 while (bMatch && s.size() < shortest.size()) {
205 s += shortest.at(s.size());
206 for (auto hint : map.keys()) {
207 if (hint.startsWith(s))
208 continue;
209 s.chop(1);
210 bMatch = false;
211 break;
212 }
213 }
214 //qDebug()<<c<<prefix<<s;
215 if (s != c) {
216 c = prefix.left(prefix.size() - c.size()) + s;
217 } else
218 c = cmd;
219 }
220 //hints output formatting
221 QStringList hints;
222
223 for (auto it = map.cbegin(); it != map.cend(); ++it) {
224 auto k = it.key();
225 QJSValue v = it.value();
226
227 if (v.isCallable()) {
228 hints.append("<font color='white'><b>" + k + "</b></font>");
229 } else if (v.isObject() && v.toVariant().value<Fact *>()) {
230 hints.append("<font color='green'>" + k + "</font>");
231 } else if (QString(v.toVariant().typeName()) == "QVariantMap") {
232 hints.append("<font color='yellow'>" + k + "</font>");
233 } else if (v.isNumber()) {
234 hints.append("<font color='cyan'>" + k + "</font>");
235 } else {
236 hints.append("<font color='gray'>" + k + "</font>");
237 }
238 }
239 hints.removeDuplicates();
240 hints.sort();
241 enter(hints.join(" "));
242 return c;
243}

245QMap<QString, QJSValue> Terminal::_get_js_properties(QString scope, QString flt)
246{
247 QMap<QString, QJSValue> map;
248
249 QJSValue v = App::instance()->engine()->jsGetProperty(scope);
250 if (v.isError()) {
251 qWarning() << v.errorType() << v.toString();
252 return map;
253 }
254
255 QRegExp re("^" + flt);
256 QJSValueIterator it(v);
257 while (it.hasNext()) {
258 it.next();
259 if (!flt.isEmpty() && !it.name().contains(re))
260 continue;
261 map.insert(it.name(), it.value());
262 }
263
264 //qDebug() << scope << v.isError() << v.toString() << st;
265 return map;
266}
