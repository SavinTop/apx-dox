51void HttpService::httpRequest(QTextStream &stream, QString req, bool *ok)
52{
53 if (req.startsWith("/kml")) {
54 stream << "HTTP/1.0 200 Ok\r\n";
55 stream << (req.contains(".dae") ? "application/xml dae"
56 : "application/vnd.google-earth.kml+xml");
57 stream << "; charset=\"utf-8\"\r\n";
58 stream << "\r\n";
59 stream << reply_google(req.mid(4));
60 *ok = true;
61 return;
62 }
63 if (req.startsWith("/mandala")) {
64 stream << "HTTP/1.0 200 Ok\r\n";
65 stream << "Content-Type: application/xml; charset=\"utf-8\"\r\n";
66 stream << "\r\n";
67 stream << reply_mandala(req.contains('?') ? req.mid(req.indexOf("?") + 1) : "");
68 *ok = true;
69 return;
70 }
71}

105QString HttpService::reply_mandala(const QString &req)
106{
107 QString reply;
108 QXmlStreamWriter xml(&reply);
109 xml.setAutoFormatting(true);
110 xml.writeStartDocument();
111 xml.writeStartElement("mandala");
112 //xml.writeTextElement("name","UAVOS GCU Mandala");
113 //xml.writeTextElement("version",FactSystem::version());
114 //------------------------------
115 bool doDescr = false;
116 QStringList rlist = req.trimmed().split('&', Qt::SkipEmptyParts);
117
118 // select mandala
119 Mandala *mandala = Vehicles::instance()->current()->f_mandala;
120
121 // collect requested facts
122 QList<MandalaFact *> facts;
123 for (auto const &s : rlist) {
124 MandalaFact *f = nullptr;
125 if (s.startsWith("descr")) {
126 doDescr = true;
127 } else if (s.contains('=')) {
128 QString sname = s.left(s.indexOf('='));
129 if (sname == "scr") {
130 App::instance()->jsexec(s.mid(sname.size() + 1));
131 } else {
132 App::instance()->jsexec(s);
133 f = mandala->fact(sname);
134 }
135 } else
136 f = mandala->fact(s);
137 if (f)
138 facts.append(f);
139 }
140 bool bAllFacts = false;
141 if (facts.isEmpty()) {
142 facts = mandala->valueFacts();
143 bAllFacts = true;
144 }
145
146 //mandala->currents
147 for (auto f : facts) {
148 if (doDescr) {
149 xml.writeStartElement(f->mpath());
150 xml.writeAttribute("descr", f->title());
151 xml.writeAttribute("uid", QString::number(f->uid()));
152 xml.writeAttribute("uid_hex", QString::number(f->uid(), 16).toUpper());
153 xml.writeCharacters(f->valueText());
154 xml.writeEndElement();
155 continue;
156 }
157
158 if (bAllFacts && !f->everReceived())
159 continue;
160
161 xml.writeTextElement(f->mpath(), f->valueText());
162 }
163 //------------------------------
164 xml.writeEndElement(); //mandala
165 xml.writeEndDocument();
166 return reply;
167}

169QString HttpService::reply_google(const QString &req)
170{
171 if (req.isEmpty() || req == "/")
172 return reply_kml();
173 if (req == "/telemetry")
174 return reply_telemetry();
175 if (req == "/flightplan")
176 return reply_flightplan();
177 if (req == "/chase")
178 return reply_chase();
179 if (req == "/chase_upd")
180 return reply_chase_upd();
181
182 if (req.contains(".dae")) {
183 QFile f(AppDirs::res().filePath("bitmaps" + req));
184 if (f.open(QIODevice::ReadOnly | QIODevice::Text))
185 return f.readAll();
186 //apxMsgW()<<"Model not found: %s",req.toUtf8().data());
187 }
188 return QString();
189}
