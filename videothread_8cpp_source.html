136static GstPadProbeReturn draw_overlay(GstPad *pad, GstPadProbeInfo *info, StreamContext *context)
137{
138 GstBuffer *buffer = GST_PAD_PROBE_INFO_BUFFER(info);
139
140 buffer = gst_buffer_make_writable(buffer);
141
142 if (buffer == nullptr)
143 return GST_PAD_PROBE_OK;
144
145 GstMapInfo map;
146 if (gst_buffer_map(buffer, &map, GST_MAP_WRITE)) {
147 std::shared_ptr<GstCaps> caps(gst_pad_get_current_caps(pad), &gst_caps_unref);
148 int width = 0;
149 int height = 0;
150 if (VideoThread::getFrameSizeFromCaps(caps, width, height)) {
151 QImage rgb32Wrapper(map.data, width, height, QImage::Format_RGBX8888);
152 context->overlayCallback(rgb32Wrapper);
153 }
154 gst_buffer_unmap(buffer, &map);
155 } else
156 apxMsgW() << "Can't map buffer";
157
158 GST_PAD_PROBE_INFO_DATA(info) = buffer;
159
160 return GST_PAD_PROBE_OK;
161}

163void VideoThread::run()
164{
165 // QmlOverlay *qmlOverlay = new QmlOverlay;
166 // setOverlayCallback(std::bind(&QmlOverlay::drawOverlay, qmlOverlay, _1));
167
168 auto context = std::make_unique<StreamContext>();
169 context->reencoding = m_reencoding;
170
171 m_loop.reset(g_main_loop_new(nullptr, false), &g_main_loop_unref);
172
173 //common elements
174 context->pipeline.reset(gst_pipeline_new("client"), &gst_object_unref);
175
176 context->capsFilter = gst_element_factory_make("capsfilter", nullptr);
177 context->source = createSourceElement(context.get());
178 context->parsebin = gst_element_factory_make("parsebin", nullptr);
179
180 context->teeparse = gst_element_factory_make("tee", nullptr);
181 context->teeconvert = gst_element_factory_make("tee", nullptr);
182
183 //play elements
184 context->playDecodebin = gst_element_factory_make("decodebin", nullptr);
185 context->playConverter = gst_element_factory_make("videoconvert", nullptr);
186 context->playAppsink = gst_element_factory_make("appsink", nullptr);
187
188 gst_bin_add_many(GST_BIN(context->pipeline.get()),
189 context->source,
190 context->capsFilter,
191 context->parsebin,
192 context->teeparse,
193 context->playDecodebin,
194 context->playConverter,
195 context->teeconvert,
196 context->playAppsink,
197 nullptr);
198
199 context->onFrameReceived = std::bind(&VideoThread::onSampleReceived, this, _1, _2);
200 context->isRecordRequested = std::bind(&VideoThread::getRecording, this);
201
202 if (!context->pipeline || !context->source || !context->capsFilter || !context->parsebin
203 || !context->teeparse || !context->playDecodebin || !context->playConverter
204 || !context->playAppsink) {
205 emit errorOccured("Can't create pipeline");
206 return;
207 }
208
209 //appsink configuring
210 g_object_set(G_OBJECT(context->playAppsink),
211 "emit-signals",
212 true,
213 "sync",
214 !m_lowLatency,
215 nullptr);
216 g_signal_connect(context->playAppsink,
217 "new-sample",
218 G_CALLBACK(on_new_sample_from_sink),
219 context.get());
220 gst_app_sink_set_max_buffers(GST_APP_SINK(context->playAppsink), 0);
221 gst_app_sink_set_drop(GST_APP_SINK(context->playAppsink), true);
222 g_object_set(context->playAppsink, "caps", getCapsForAppSink().get(), nullptr);
223
224 //pad-cb
225 if (GST_IS_BIN(context->source)) {
226 g_signal_connect(context->source,
227 "pad-added",
228 G_CALLBACK(on_pad_added_urisourcebin),
229 context->capsFilter);
230 } else {
231 if (!gst_element_link(context->source, context->capsFilter)) {
232 emit errorOccured("Can't link source and capsfilter");
233 return;
234 }
235 }
236 g_signal_connect(context->parsebin, "pad-added", G_CALLBACK(on_pad_added), context->teeparse);
237 g_signal_connect(context->playDecodebin,
238 "pad-added",
239 G_CALLBACK(on_pad_added),
240 context->playConverter);
241
242 if (!gst_element_link_many(context->capsFilter, context->parsebin, nullptr)) {
243 emit errorOccured("Can't link source, caps filter and parsebin");
244 return;
245 }
246
247 if (!gst_element_link(context->teeparse, context->playDecodebin)) {
248 emit errorOccured("Can't link tee and decodebin");
249 return;
250 }
251
252 if (!gst_element_link_many(context->playConverter,
253 context->teeconvert,
254 context->playAppsink,
255 nullptr)) {
256 emit errorOccured("Can't link converter, tee and appsink");
257 return;
258 }
259
260 if (gst_element_set_state(context->pipeline.get(), GST_STATE_PLAYING)
261 == GST_STATE_CHANGE_FAILURE) {
262 emit errorOccured("Can't play pipeline");
263 return;
264 }
265 g_main_loop_run(m_loop.get());
266
267 if (context->recording)
268 closeWriter(context.get());
269
270 gst_element_set_state(context->pipeline.get(), GST_STATE_NULL);
271}

273GstElement *VideoThread::createSourceElement(StreamContext *context)
274{
275 GstElement *result = nullptr;
276 if (m_uri.contains("avf://")) {
277 QUrl u(m_uri);
278 result = gst_element_factory_make("avfvideosrc", nullptr);
279 bool ok;
280 int index = u.host().remove("index").toInt(&ok);
281 if (ok)
282 g_object_set(result, "device-index", index, nullptr);
283
284 g_object_set(context->capsFilter, "caps", gst_caps_from_string("video/x-raw"), nullptr);
285 } else if (m_uri.contains("tcp://")) {
286 QUrl u(m_uri);
287 result = gst_element_factory_make("tcpclientsrc", nullptr);
288 QString host = u.host();
289 int port = u.port();
290 g_object_set(result, "host", host.toStdString().c_str(), nullptr);
291 g_object_set(result, "port", port, nullptr);
292
293 g_object_set(context->capsFilter,
294 "caps",
295 gst_caps_from_string("video/mpegts,systemstream=true"),
296 nullptr);
297 } else if (m_uri.contains("udp://")) {
298 QUrl u(m_uri);
299 result = gst_element_factory_make("udpsrc", nullptr);
300 int port = u.port();
301 QString query = u.query();
302 g_object_set(result, "port", port, nullptr);
303
304 std::shared_ptr<GstCaps> caps;
305 if (query.contains("codec=h265"))
306 caps = getCapsForUdpSrc("H265");
307 else
308 caps = getCapsForUdpSrc("H264");
309 g_object_set(result, "caps", caps.get(), nullptr);
310 } else {
311 result = gst_element_factory_make("urisourcebin", nullptr);
312 g_object_set(result, "uri", m_uri.toStdString().c_str(), nullptr);
313 }
314
315 return result;
316}

318void VideoThread::openWriter(StreamContext *context)
319{
320 QString filename = GstPlayer::getMediaFileName(GstPlayer::mtVideo);
321
322 //create
323 context->recQueue = gst_element_factory_make("queue", nullptr);
324 if (context->reencoding) {
325 context->recConverter = gst_element_factory_make("videoconvert", nullptr);
326 context->recEncoder = gst_element_factory_make("x264enc", nullptr);
327 }
328 context->recMuxer = gst_element_factory_make("matroskamux", nullptr);
329 context->recSink = gst_element_factory_make("filesink", nullptr);
330
331 if (!context->recQueue || !context->recMuxer || !context->recSink
332 || (context->reencoding && (!context->recEncoder || !context->recConverter))) {
333 emit errorOccured("Can't create recording elements");
334 return;
335 }
336
337 //tune
338 g_object_set(G_OBJECT(context->recSink), "location", filename.toStdString().c_str(), nullptr);
339 if (context->reencoding) {
340 // g_object_set(G_OBJECT(context->recEncoder), "speed-preset", 1, nullptr);
341 // g_object_set(G_OBJECT(context->recEncoder), "tune", 4, nullptr);
342
343 g_object_set(G_OBJECT(context->recEncoder),
344 "quantizer",
345 18,
346 "pass",
347 5,
348 "speed-preset",
349 1,
350 "tune",
351 4,
352 nullptr);
353 }
354
355 //add
356 gst_bin_add_many(GST_BIN(context->pipeline.get()),
357 context->recQueue,
358 context->recMuxer,
359 context->recSink,
360 nullptr);
361 if (context->reencoding)
362 gst_bin_add_many(GST_BIN(context->pipeline.get()),
363 context->recConverter,
364 context->recEncoder,
365 nullptr);
366
367 //link
368 bool linkResult = false;
369 if (!context->reencoding)
370 linkResult = gst_element_link_many(context->teeparse,
371 context->recQueue,
372 context->recMuxer,
373 context->recSink,
374 nullptr);
375 else
376 linkResult = gst_element_link_many(context->teeconvert,
377 context->recQueue,
378 context->recConverter,
379 context->recEncoder,
380 context->recMuxer,
381 context->recSink,
382 nullptr);
383 if (!linkResult) {
384 emit errorOccured("Can't link elements from record pipeline");
385 return;
386 }
387
388 //sync
389 bool syncResult = gst_element_sync_state_with_parent(context->recQueue)
390 && gst_element_sync_state_with_parent(context->recMuxer)
391 && gst_element_sync_state_with_parent(context->recSink);
392 if (context->reencoding) {
393 syncResult = syncResult && gst_element_sync_state_with_parent(context->recConverter)
394 && gst_element_sync_state_with_parent(context->recEncoder);
395 }
396 if (!syncResult) {
397 emit errorOccured("Can't sync state of rec elements with parent");
398 return;
399 }
400
401 //overlay callback
402 if (context->reencoding) {
403 std::shared_ptr<GstPad> pad(gst_element_get_static_pad(context->recQueue, "src"),
404 &gst_object_unref);
405 gst_pad_add_probe(pad.get(),
406 GST_PAD_PROBE_TYPE_BUFFER,
407 (GstPadProbeCallback) draw_overlay,
408 context,
409 nullptr);
410 context->overlayCallback = m_overlayCallback;
411 } else
412 context->overlayCallback = [](auto) {};
413
414 context->recording = true;
415}

417void VideoThread::closeWriter(StreamContext *context)
418{
419 gst_element_send_event(context->recMuxer, gst_event_new_eos());
420
421 //remove
422 gst_bin_remove_many(GST_BIN(context->pipeline.get()),
423 context->recQueue,
424 context->recMuxer,
425 context->recSink,
426 nullptr);
427 if (context->reencoding)
428 gst_bin_remove_many(GST_BIN(context->pipeline.get()),
429 context->recConverter,
430 context->recEncoder,
431 nullptr);
432
433 gst_element_set_state(context->recQueue, GST_STATE_NULL);
434 gst_element_set_state(context->recMuxer, GST_STATE_NULL);
435 gst_element_set_state(context->recSink, GST_STATE_NULL);
436
437 if (context->reencoding) {
438 gst_element_set_state(context->recConverter, GST_STATE_NULL);
439 gst_element_set_state(context->recEncoder, GST_STATE_NULL);
440 }
441
442 context->recording = false;
443}

445void VideoThread::onSampleReceived(StreamContext *context, GstElement *appsink)
446{
447 std::shared_ptr<GstSample> sample(gst_app_sink_pull_sample(GST_APP_SINK(appsink)),
448 &gst_sample_unref);
449 QImage frame = sample2qimage(sample);
450
451 if (!context->recording && context->isRecordRequested())
452 openWriter(context);
453
454 if (context->recording && !context->isRecordRequested())
455 closeWriter(context);
456
457 emit frameReceived(frame);
458}

460QImage VideoThread::sample2qimage(const std::shared_ptr<GstSample> &sample)
461{
462 std::shared_ptr<GstCaps> caps(gst_sample_get_caps(sample.get()), [](auto) {});
463 int width = 0;
464 int height = 0;
465 if (getFrameSizeFromCaps(caps, width, height)) {
466 GstBuffer *buffer = gst_sample_get_buffer(sample.get());
467 GstMapInfo info;
468 gboolean success = gst_buffer_map(buffer, &info, (GstMapFlags) GST_MAP_READ);
469 if (success) {
470 QImage rgb32Wrapper(info.data, width, height, QImage::Format_RGBX8888);
471 QImage rgb32Copy = rgb32Wrapper.copy();
472 gst_buffer_unmap(buffer, &info);
473 return rgb32Copy.convertToFormat(QImage::Format_RGB32);
474 }
475 }
476 return QImage(width, height, QImage::Format_RGB32);
477}
