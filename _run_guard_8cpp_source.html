28QString generateKeyHash(const QString &key, const QString &salt)
29{
30 QByteArray data;
31
32 data.append(key.toUtf8());
33 data.append(salt.toUtf8());

41RunGuard::RunGuard(const QString &key)
42 : memLockKey(generateKeyHash(key, "_memLockKey"))
43 , sharedmemKey(generateKeyHash(key, "_sharedmemKey"))
44 , sharedMem(sharedmemKey)

60bool RunGuard::isAnotherRunning()
61{
62 if (sharedMem.isAttached())
63 return false;
64
65 memLock.acquire();
66 const bool isRunning = sharedMem.attach();
67 if (isRunning)
68 sharedMem.detach();
69 memLock.release();
70
71 return isRunning;
72}

74bool RunGuard::tryToRun()
75{
76 if (isAnotherRunning()) // Extra check
77 return false;
78
79 memLock.acquire();
80 const bool result = sharedMem.create(sizeof(quint64));
81 memLock.release();
82 if (!result) {
83 release();
84 return false;
85 }
86
87 return true;
88}
