51void NodeScript::_update_cc_args()
52{
53 // find compiler
54 cc = "wasmcc";
55 if (_plugin) {
56 cc.clear();
57 auto c = _plugin->control;
58 if (!c->property("enabled").toBool()) {
59 // connect(c, SIGNAL("available"), this, SLOT("_update_cc_args"));
60 return;
61 }
62 QString pcc = c->property("cc").toString();
63 if (!pcc.isEmpty() && QFile::exists(pcc))
64 cc = pcc;
65 }
66
67 if (cc.isEmpty())
68 return;
69
70 // parse compiler args - fill from json file
71 cc_args.clear();
72 QFile ftasks(AppDirs::res().filePath("scripts/.vscode/tasks.json"));
73 if (ftasks.open(QFile::ReadOnly | QFile::Text)) {
74 QJsonDocument json = QJsonDocument::fromJson(ftasks.readAll());
75 ftasks.close();
76 //qDebug() << json;
77 foreach (QJsonValue v, json["tasks"].toArray()) {
78 if (!v["group"]["isDefault"].toBool())
79 continue;
80 QHash<QString, QString> map;
81 map.insert("config:wasm.sysroot", AppDirs::scripts().absoluteFilePath("sysroot"));
82 map.insert("config:wasm.flags", "-I" + AppDirs::scripts().absoluteFilePath("include"));
83 map.insert("fileDirname", QFileInfo(srcFile.fileName()).absolutePath());
84 map.insert("fileBasenameNoExtension", QFileInfo(srcFile.fileName()).baseName());
85 map.insert("file", srcFile.fileName());
86 for (auto a : v["args"].toArray().toVariantList()) {
87 QString s = a.toString();
88 if (s.contains("${")) {
89 for (auto k : map.keys())
90 s.replace(QString("${%1}").arg(k), map.value(k));
91 }
92 cc_args.append(s);
93 }
94 break;
95 }
96 }
97 //qDebug() << cc << cc_args;
98}

100void NodeScript::factValueChanged()
101{
102 QString value = _fact->value().toString();
103 if (_value_s == value)
104 return;
105
106 _code.clear();
107 _source.clear();
108 _title.clear();
109 _log.clear();
110 _error = false;
111
112 if (cc.isEmpty()) {
113 _updateFactText();
114 apxMsgW() << tr("Script compiler is missing");
115 return;
116 }
117
118 //qDebug() << value;
119
120 QStringList st = value.split(',', Qt::KeepEmptyParts);
121 QString title = st.value(0);
122 QString src = st.value(1);
123 QString src_code = st.value(2);
124
125 _title = title;
126
127 if (!src.isEmpty()) {
128 QByteArray src_ba = QByteArray::fromHex(src.toLocal8Bit());
129 _source = qUncompress(src_ba);
130 if (src_code.isEmpty())
131 _compile(_source);
132 }
133
134 // compile if necessary
135
136 if (src_code.isEmpty()) {
137 QString code_string;
138 if (!code().isEmpty())
139 code_string = qCompress(code(), 9).toHex().toUpper();
140
141 st.clear();
142 st << _title;
143 if (!code_string.isEmpty()) {
144 st << src;
145 st << code_string;
146 }
147 _value_s = st.join(',');
148
149 //qDebug() << _value_s;
150
151 _fact->setValue(_value_s);
152 } else {
153 QByteArray src_ba = QByteArray::fromHex(src_code.toLocal8Bit());
154 _code = qUncompress(src_ba);
155 }
156
157 _updateFactText();
158}

190bool NodeScript::_compile(QString src)
191{
192 if (cc.isEmpty())
193 return true;
194
195 qDebug() << "compiling:" << _title << src.size();
196 _code.clear();
197 _error = false;
198 srcFile.resize(0);
199 srcFile.flush();
200 QFile::remove(outFileName);
201 if (src.trimmed().isEmpty()) {
202 emit compiled();
203 return true;
204 }
205 //qDebug()<<src;
206 QTextStream s(&srcFile);
207 s << src;
208 s.flush();
209 srcFile.flush();
210
211 //process
212 _compile_wasm();
213
214 bool rv = true;
215 if (!proc.waitForFinished()) {
216 apxMsgW() << "script compiler error:" << proc.errorString();
217 rv = false;
218 } else if (proc.exitCode() != 0)
219 rv = false;
220 _log = proc.isOpen() ? proc.readAll() : QString();
221 if (!rv)
222 _log.append("\n\n" + proc.errorString());
223
224 //log file
225 QTemporaryFile logFile;
226 logFile.setFileTemplate(QDir::tempPath() + "/script_log");
227 logFile.setAutoRemove(false);
228 if (logFile.open()) {
229 QTextStream s(&logFile);
230 s << proc.program() + " " + proc.arguments().join(' ');
231 s << "\n\n";
232 s << _log;
233 s.flush();
234 logFile.flush();
235 logFile.close();
236 }
237
238 if (rv) {
239 //read out data
240 qDebug() << "compiled:" << outFileName;
241 QFile file(outFileName);
242 if (file.open(QFile::ReadOnly)) {
243 _code = file.readAll();
244 }
245 }
246 _error = !rv;
247 emit compiled();
248 //qDebug() << "compile" << rv;
249 return rv;
250}

259bool NodeScript::saveToFile(QString fname)
260{
261 QFile file(fname);
262 if (!file.open(QFile::WriteOnly | QFile::Text)) {
263 apxMsgW() << tr("Cannot write file")
264 << QString("%1:\n%2.").arg(fname).arg(file.errorString());
265 return false;
266 }
267 if (title() != QFileInfo(fname).baseName())
268 setSource(QFileInfo(fname).baseName(), source());
269 QTextStream s(&file);
270 s << source();
271 s.flush();
272 file.close();
273 _updateWatcher(fname);
274 return true;
