60QVariant JSTreeModel::data(const QModelIndex &index, int role) const
61{
62 if (!index.isValid())
63 return QVariant();
64 JSTreeItem *i = item(index);
65 if (!i)
66 return QVariant();
67 QJSValue value = i->value();
68 int col = index.column();
69 switch (role) {
70 case ModelDataRole:
71 return QVariant::fromValue(const_cast<JSTreeItem *>(i));
72 case NameRole:
73 return i->name;
74 case ValueRole:
75 return value.toVariant();
76 case DescrRole:
77 return i->descr;
78 case TextRole:
79 return value.toString();
80
81 case Qt::ForegroundRole:
82 if (col == JS_MODEL_COLUMN_NAME) {
83 if (i->isFunc) {
84 if (i->isCmd)
85 return QColor(Qt::yellow).lighter();
86 return QColor(Qt::red).lighter();
87 }
88 if (i->isFact)
89 return QColor(Qt::green).lighter();
90 if (i->isQObj)
91 return QColor(Qt::yellow).lighter();
92 if (i->isMap)
93 return QColor(Qt::cyan).lighter();
94 return QColor(Qt::gray).lighter(); //QVariant();
95 }
96 if (col == JS_MODEL_COLUMN_VALUE) {
97 //if(!enabled())return QColor(Qt::darkGray);
98 //if(size()) return QColor(Qt::darkGray); //expandable
99 //if(modified())return QColor(Qt::yellow);
100 //if(isZero())return QColor(Qt::gray);
101 return QColor(Qt::cyan).lighter(180);
102 }
103 return QColor(Qt::darkCyan);
104 case Qt::BackgroundRole:
105 return QVariant();
106 case Qt::FontRole: {
107 // QFont font(QFontDatabase::systemFont(QFontDatabase::GeneralFont));
108 QFont font(QGuiApplication::font());
109 if (col == JS_MODEL_COLUMN_DESCR)
110 return QVariant();
111 if (value.isQObject() && col == JS_MODEL_COLUMN_NAME)
112 font.setBold(true);
113 //if(ftype>=ft_regPID) return QFont("Monospace",-1,column==tc_field?QFont::Bold:QFont::Normal);
114 //if(col==FACT_MODEL_COLUMN_NAME) return QFont("Monospace",-1,QFont::Normal,isModified());
115 //if(ftype==ft_string) return QFont("",-1,QFont::Normal,true);
116 return font;
117 }
118 }
119
120 //value roles
121 if (role != Qt::DisplayRole && role != Qt::EditRole)
122 return QVariant();
123
124 switch (index.column()) {
125 case JS_MODEL_COLUMN_NAME:
126 return data(index, NameRole);
127 case JS_MODEL_COLUMN_VALUE:
128 return data(index, TextRole);
129 case JS_MODEL_COLUMN_DESCR:
130 return data(index, DescrRole);
131 }
132 return QVariant();
133}

151{

320void JSTreeItem::sync()
321{
322 //qDebug()<<"sync"<<name;
323 qDeleteAll(items);
324 items.clear();
325 //filter childs for some items
326 //check recursive
327 QJSValue v = value();
328 foreach (JSTreeItem *i, pathList(false)) {
329 if (i->value().strictlyEquals(v)) {
330 descr = "recursive reference";
331 qDebug() << path() << descr;
332 return;
333 }
334 }
335 //populate items
336 QJSValueIterator it(v);
337 while (it.hasNext()) {
338 it.next();
339 JSTreeItem *i = new JSTreeItem(this, it.name(), it.value());
340 items.append(i);
341 }
342 std::sort(items.begin(), items.end(), lessThan);
343}
344void JSTreeItem::updateSize()
345{
346 sync();
347 int cnt = items.size();
348 if (m_size == cnt)
349 return;
350 m_size = cnt;
351 sync();
352}
