53void Protocols::updateProtocol()
54{
55 PBase *protocol = nullptr;
56 const QString s = f_current->text();
57 for (auto i : findFacts<PBase>()) {
58 if (i->name() != s)
59 continue;
60 protocol = i;
61 break;
62 }
63 if (!protocol) {
64 apxMsgW() << tr("Can't select protocol").append(':') << s;
65 return;
66 }
67 if (_protocol == protocol)
68 return;
69
70 bool chg = _protocol;
71 if (chg) {
72 _protocol->trace()->enable(false);
73 disconnect(_protocol, nullptr, this, nullptr);
74 _protocol = nullptr;
75 }
76 _protocol = protocol;
77 qDebug() << "Protocol:" << s << _protocol->title();
78 if (chg || s != f_current->defaultValue().toString()) {
79 apxMsg() << tr("Selected protocol").append(':') << s;
80 }
81
82 // connect protocol interface
83 connect(_protocol, &PBase::tx_data, this, &Protocols::tx_data);
84 connect(_protocol, &PBase::vehicle_available, this, &Protocols::vehicle_available);
85 connect(_protocol->trace(), &PTrace::packet, this, &Protocols::trace_packet);
86}
