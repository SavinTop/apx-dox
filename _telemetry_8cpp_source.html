35Telemetry::Telemetry(Vehicle *parent)
36 : Fact(parent,
37 "telemetry",
38 tr("Telemetry"),
39 tr("Telemetry data recorder"),
40 Group,
41 "inbox-arrow-down")
42 , vehicle(parent)
43 , f_recorder(nullptr)
44 , f_lookup(nullptr)
45 , f_reader(nullptr)
46 , f_share(nullptr)
47{
48 if (vehicle->isReplay()) {
49 setOpt("pos", QPointF(1, 1));
50
51 f_lookup = new LookupTelemetry(this);
52 f_lookup->f_latest->createAction(this);
53 f_lookup->f_prev->createAction(this);
54 f_lookup->f_next->createAction(this);
55 f_reader = new TelemetryReader(f_lookup, this);
56 connect(f_reader, &Fact::valueChanged, this, &Telemetry::updateStatus);
57 connect(f_reader, &Fact::progressChanged, this, &Telemetry::updateProgress);
58 connect(f_reader,
59 &TelemetryReader::recordFactTriggered,
60 this,
61 &Telemetry::recordFactTriggered);
62
63 connect(f_reader,
64 &TelemetryReader::dataAvailable,
65 this,
66 &Telemetry::recordLoaded,
67 Qt::QueuedConnection);
68
69 f_player = new TelemetryPlayer(this, this);
70 connect(f_player, &Fact::valueChanged, this, &Telemetry::updateStatus);
71 bindProperty(f_player, "active", true);
72
73 f_share = new TelemetryShare(this, this);
74 connect(f_share, &TelemetryShare::importJobDone, this, [this](quint64 id) {
75 f_lookup->jumpToRecord(id);
76 });
77 connect(f_share, &Fact::progressChanged, this, &Telemetry::updateProgress);
78
79 connect(App::instance(), &App::loadingFinished, this, [this]() {
80 connect(vehicle,
81 &Vehicle::selected,
82 f_reader,
83 &TelemetryReader::loadCurrent,
84 Qt::QueuedConnection);
85 });
86
87 } else {
88 f_recorder = new TelemetryRecorder(vehicle, this);
89 connect(f_recorder, &TelemetryRecorder::recordingChanged, this, [this]() {
90 setActive(f_recorder->recording());
91 });
92 connect(f_recorder, &Fact::valueChanged, this, &Telemetry::updateStatus);
93 }
94 descr_s = descr();
95 updateStatus();
96}

98void Telemetry::updateStatus()
99{
100 if (f_recorder)
101 setValue(f_recorder->value());

104 QString("%1/%2").arg(f_player->value().toString()).arg(f_reader->value().toString()));

106void Telemetry::updateProgress()
107{
108 if (!f_reader)
109 return;
110 int v = f_reader->progress();
111 if (v < 0 && f_share)
112 v = f_share->progress();
113 setProgress(v);
114 updateDescr();
115}
116void Telemetry::updateDescr()
117{
118 if (!f_reader)
119 return;
120 if (f_reader->progress() >= 0) {
121 setDescr(tr("Reading").append("..."));
122 } else if (f_share && f_share->progress() >= 0) {
123 setDescr(f_share->descr());
124 } else
125 setDescr(descr_s);
126}

128void Telemetry::recordFactTriggered(Fact *f)
129{
130 const QString &s = f->name();
131 const QString &uid = f->descr();
132 if (s.startsWith("nodes")) {
133 vehicle->storage()->loadVehicleConfig(uid);
134 } else if (s.startsWith("mission")) {
135 vehicle->f_mission->storage->loadMission(uid);
136 } else {
137 if (f_player)
138 f_player->f_time->setValue(f->property("time").toULongLong() - 1);
139 }
140}

142void Telemetry::recordLoaded()
143{
144 if (!Vehicles::instance()->current()->isIdentified())
145 vehicle->f_select->trigger();
146
147 vehicle->setGeoPath(f_reader->geoPath);
148
149 Fact *f_events = f_reader->child("events");
150 if (f_events) {
151 Fact *f = f_events->child("mission");
152 if (f && f->size() > 0) {
153 f = f->child(0);
154 if (f)
155 f->trigger();
156 }
157 f = f_events->child("nodes");
158 if (f && f->size() > 0) {
159 f = f->child(f->size() - 1);
160 if (f)
161 f->trigger();
162 }
163 }
