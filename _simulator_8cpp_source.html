108bool Simulator::extract_apxfw()
109{
110 ApxFw *apxfw = AppGcs::apxfw();
111
112 QByteArray data;
113 apxfw->loadFirmware("sim", "APX", "application-" + target_os, &data, nullptr);
114
115 if (data.size() <= 0)
116 return false;
117
118 QDir dir(AppDirs::firmware().absoluteFilePath("sim"));
119 dir.mkpath(".");
120 QFile fsim(dir.absoluteFilePath(sim_executable));
121 fsim.remove();
122 if (!fsim.open(QFile::WriteOnly)) {
123 qWarning() << "can't write" << fsim.fileName();
124 return false;
125 }
126 fsim.write(data);
127 fsim.close();
128 fsim.setPermissions(QFileDevice::ReadOwner | QFileDevice::ExeOwner);
129 return true;
130}

132void Simulator::detectXplane()
133{
134 QStringList st;
135 QFileInfoList files;
136 QStringList locations;
137 locations << QStandardPaths::writableLocation(QStandardPaths::GenericConfigLocation);
138 locations << QStandardPaths::writableLocation(QStandardPaths::HomeLocation) + "/.x-plane";
139
140 for (auto i : locations) {
141 files = QDir(i, "x-plane_install*.txt").entryInfoList();
142 if (!files.isEmpty())
143 break;
144 }
145
146 foreach (QFileInfo fi, files) {
147 QString s = fi.baseName();
148 int ver = 0;
149 QString sv = s.split('_').last();
150 if (!sv.isEmpty())
151 ver = sv.toUInt();
152 if (ver < 9)
153 ver = 9;
154 //read dir
155 QFile file(fi.absoluteFilePath());
156 if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
157 apxConsoleW() << "Can't open file for reading:" << file.fileName();
158 continue;
159 }
160 QString xpath;
161 while (!file.atEnd()) {
162 QString s = file.readLine().trimmed();
163 if (s.isEmpty())
164 continue;
165 QDir dir(s);
166 if (!dir.exists())
167 continue;
168 xpath = dir.absolutePath();
169 xplaneDirs.append(xpath);
170 s = QString("XPlane %1").arg(ver);
171 if (!st.contains(s)) {
172 st.append(s);
173 continue;
174 }
175 int n = 2;
176 QString sn = s;
177 while (st.contains(sn)) {
178 sn = QString("%1 (%2)").arg(s).arg(n++);
179 }
180 st.append(sn);
181 }
182 if (xpath.isEmpty()) {
183 apxConsoleW() << "Error reading X-Plane config:" << file.fileName() << xpath;
184 }
185 }
186 //qDebug() << st << xplaneDirs;
187 f_type->setEnumStrings(st);
188 if (st.isEmpty())
189 f_type->setEnabled(false);
190 else {
191 f_type->setEnabled(true);
192 }
193}

204void Simulator::launchXplane()
205{
206 apxMsg() << tr("Launching X-Plane").append("...");
207
208 QString xplaneDir;
209 xplaneDir = xplaneDirs.value(f_type->value().toInt());
210 while (!xplaneDir.isEmpty()) {
211 //install xpl plugins
212 QDir d(xplaneDir);
213 if (d.cd("Resources") && d.cd("plugins")) {
214 QDir dir = QDir(AppDirs::res().absoluteFilePath("xplane"), "*.xpl");
215 if (dir.isEmpty())
216 dir = QDir(AppDirs::libs().absolutePath(), "*.xpl");
217 if (dir.isEmpty())
218 apxMsgW() << tr("XPL Plugin not found");
219 for (auto const &fi : dir.entryInfoList()) {
220 QString dest = d.absoluteFilePath(fi.fileName());
221 QFileInfo fiDest(dest);
222 //qDebug()<<dest;
223 if (fiDest.isSymLink() && fiDest.symLinkTarget() == fi.absoluteFilePath())
224 continue;
225 //qDebug()<<fi.absoluteFilePath()<<dest<<QFile::exists(dest);
226 if (fiDest.isSymLink() || QFile::exists(dest))
227 QFile::remove(dest);
228 //qDebug()<<fi.absoluteFilePath()<<dest<<QFile::exists(dest);
229 if (QFile::link(fi.absoluteFilePath(), dest)) {
230 apxMsg() << tr("XPL Plugin installed").append(":") << dest;
231 } else {
232 apxMsgW() << tr("XPL Plugin error").append(":") << dest;
233 }
234 }
235 }
236 //install xpl aircrafts
237 if (d.cd(xplaneDir) && d.cd("Aircraft")) {
238 d.mkpath("APX");
239 d.cd("APX");
240 QFileInfoList fiSrcList(QDir(AppDirs::res().absoluteFilePath("xplane/models"),
241 "",
242 QDir::Unsorted,
243 QDir::AllDirs | QDir::NoDotAndDotDot)
244 .entryInfoList());
245 foreach (QFileInfo fi, fiSrcList) {
246 QFileInfo fiDest(d.absoluteFilePath(fi.fileName()));
247 //qDebug()<<fiDest.absoluteFilePath();
248 if (fiDest.exists())
249 continue;
250 AppDirs::copyPath(fi.absoluteFilePath(), fiDest.absoluteFilePath());
251 apxMsg() << tr("XPL Aircraft installed").append(":") << fiDest.absoluteFilePath();
252 }
253 }
254
255 break;
256 }
257 if (!xplaneDir.isEmpty())
258 _launchXplane(xplaneDir);
259}

289void Simulator::_launchXplane(QString xplaneDir)
290{
291 //mount image
292 QFileInfoList fiList;
293 fiList = QDir(xplaneDir, "*.img").entryInfoList();
294 if (fiList.isEmpty())
295 fiList = QDir(xplaneDir, "*.iso").entryInfoList();
296
297 while (!fiList.isEmpty()) {
298 QString imgFile = fiList.first().absoluteFilePath();
299#if defined Q_OS_MAC
300 if (QProcess::execute("hdiutil", QStringList() << "attach" << imgFile) == 0)
301 break;
302#elif defined Q_OS_LINUX
303 QDir("/media/cdrom").mkpath(".");
304 if (QProcess::execute("mount",
305 QStringList() << imgFile << "/media/cdrom"
306 << "-o loop,ro")
307 == 0)
308 break;
309#endif
310 apxMsgW() << tr("Failed to mount X-Plane image") << imgFile << "(/media/cdrom)";
311 break;
312 }
313
314 //launch xplane
315 QString xplaneApp("X-Plane");
316#if defined Q_OS_MAC
317 xplaneApp.append(".app");
318 xplaneApp = QDir(xplaneApp).filePath("Contents/MacOS/X-Plane");
319#elif defined Q_OS_WIN
320 xplaneApp.append(".exe");
321#else
322 xplaneApp.append("-x86_64");
323#endif
324 /*QUrl xplaneUrl(QUrl::fromLocalFile(xplaneDir + "/" + xplaneApp));
325 if (!QDesktopServices::openUrl(xplaneUrl)) {
326 apxMsgW() << tr("Failed to start X-Plane app") << xplaneUrl;
327 }*/
328 QStringList args = f_cmd->text().split(' ');
329 QString exe = QDir(xplaneDir).absoluteFilePath(xplaneApp);
330 if (!QProcess::startDetached(exe, args)) {
331 apxMsgW() << tr("Failed to start X-Plane app") << exe;
