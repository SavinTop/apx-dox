36BlackboxNode::BlackboxNode(Fact *parent, NodeItem *node)
37 : BlackboxItem(parent, node->name(), "", "", Group, node->sn())
38 , node(node)
39 , bb_read(0)
40 , rec_size(0)
41 , req_blk(0)
42 , block_size(512)
43 , op(op_idle)
44{
45 bind(node);
46 connect(node, &Fact::removed, this, &Fact::remove);
47
48 if (!node->nodes->vehicle->isLocal() && node->nodes->vehicle->vehicleClass() != Vehicle::GCU) {
49 f_callsign->setValue(node->nodes->vehicle->callsign());
50 }
51
52 connect(this, &Fact::progressChanged, node, [this]() { this->node->setProgress(progress()); });
53
54 f_begin = new Fact(this, "begin", tr("Start block"), "", Int);
55 f_begin->setMin(0);
56 f_begin->setValue(0);
57 connect(f_begin, &Fact::valueChanged, this, &BlackboxNode::updateTotals);
58
59 f_end = new Fact(this, "end", tr("End block"), "", Int);
60 f_end->setMin(0);
61 connect(f_end, &Fact::valueChanged, this, &BlackboxNode::updateTotals);
62
63 f_commands = new Fact(this, "commands", tr("Commands"), "", Action | IconOnly, "dots-horizontal");
64
65 connect(node, &NodeItem::serviceDataReceived, this, &BlackboxNode::serviceDataReceived);
66 connect(node, &NodeItem::requestTimeout, this, [this](quint16 cmd, QByteArray) {
67 if (cmd > 0 && cmd == bb_read) {
68 apxMsgW() << tr("Blackbox download timeout");
69 stop();
70 }
71 });
72
73 connect(this, &Fact::triggered, this, &BlackboxItem::getStats);
74
75 updateActions();
76}

79{
80 quint64 begin = f_begin->value().toUInt();
81 quint64 end = f_end->value().toUInt();

109void BlackboxNode::request(quint32 blk)
110{
111 req_blk = blk;
112 QByteArray data;
113 data.resize(sizeof(uint32_t));
114 XbusStreamWriter stream(reinterpret_cast<uint8_t *>(data.data()));
115 uint32_t a = blk;
116 stream << a;
117 request(op_read, data);
118 blk -= req_begin;
119 quint32 sz = req_end - req_begin;
120 setProgress(static_cast<int>(blk * 100 / sz));
121 f_stats->setValue(QString("%1/%2 MB")
122 .arg(blk * (block_size / 1024.0 / 1024.0), 0, 'f', 2)
123 .arg(sz * (block_size / 1024.0 / 1024.0), 0, 'f', 2));
124}

158void BlackboxNode::serviceDataReceived(quint16 cmd, QByteArray data)
159{
160 if (op == op_idle)
161 return;
162 if (cmd != bb_read)
163 return;
164 if (data.size() < 3)
165 return;
166 uint16_t psize = static_cast<uint16_t>(data.size());
167 uint8_t *pdata = reinterpret_cast<uint8_t *>(data.data());
168 XbusStreamReader stream(pdata, psize);
169
170 uint8_t sub_cmd;
171 stream >> sub_cmd;
172 switch (sub_cmd) {
173 default:
174 break;
175 case xbus::node::blackbox::bbr_hdr: {
176 if (op != op_stats)
177 return;
178 xbus::node::blackbox::Header dHdr;
179 if (stream.tail() != dHdr.psize())
180 break;
181 dHdr.read(&stream);
182
183 rec_size = dHdr.rec_size;
184 if (dHdr.block_size != 0xFFFF && dHdr.block_size != 0 && (dHdr.block_size & 0xFF) == 0)
185 block_size = dHdr.block_size;
186
187 f_end->setValue(rec_size);
188 QStringList st;
189 st << QString("on: %1 times").arg(dHdr.oncnt);
190 st << QString("up: %1").arg(AppRoot::timeToString(dHdr.uptime, true));
191 f_stats->setDescr(st.join(", "));
192 node->acknowledgeRequest(bb_read);
193 stop();
194 return;
195 }
196 case xbus::node::blackbox::bbr_data: {
197 if (op != op_read)
198 return;
199 if (stream.tail() != (4 + block_size))
200 break;
201 if (!reader)
202 break;
203 uint32_t b;
204 stream >> b;
205 if (req_blk != b) {
206 qDebug("Duplicate block");
207 return;
208 }
209 //block ok
210 req_blk++;
211 node->acknowledgeRequest(bb_read);
212 reader->processData(data.mid(stream.position()));
213 //continue
214 if (req_blk >= req_end) {
215 apxMsg() << tr("Blackbox download finished");
216 stop();
217 return;
218 }
219 request(req_blk);
220 return;
221 }
222 } //switch
223 //error
224 apxMsgW() << tr("Unknown blackbox response") << cmd << data.size();
225}
