31bool PApxData::process_downlink(const xbus::pid_s &pid, PStreamReader &stream)
32{
33 bool is_request = pid.pri == xbus::pri_request;
34 do {
35 if (pid.uid < mandala::cmd::env::uid) {
36 if (is_request)
37 return true;
38 // regular packed variable
39 if (stream.available() <= mandala::spec_s::psize()) {
40 qWarning() << "size" << stream.available();
41 break;
42 }
43
44 mandala::spec_s spec;
45 spec.read(&stream);
46 trace()->block(QString("T%1").arg(spec.type));
47
48 trace()->data(stream.payload());
49
50 PBase::Values values = unpack(pid, spec, stream);
51 if (values.isEmpty() || stream.available() > 0) {
52 qWarning() << "unpack data values error";
53 break;
54 }
55 emit valuesData(values);
56 return true;
57 }
58
59 switch (pid.uid) {
60 default:
61 return false;
62 case mandala::cmd::env::stream::vcp::uid:
63 if (stream.available() > 1) {
64 uint8_t port_id = stream.read<uint8_t>();
65 trace()->block(QString::number(port_id));
66 trace()->data(stream.payload());
67 emit serialData(port_id, stream.payload());
68 return true;
69 }
70 qWarning() << "Empty serial data received";
71 break;
72 case mandala::cmd::env::script::jsexec::uid:
73 if (stream.available() > 2) {
74 QString script = stream.payload().trimmed();
75 trace()->block(script);
76 if (!script.isEmpty()) {
77 emit jsexecData(script);
78 return true;
79 }
80 }
81 qWarning() << "Empty jsexec data received" << stream.dump_payload();
82 break;
83 case mandala::cmd::env::stream::calib::uid:
84 if (is_request)
85 return true;
86 if (stream.available() > sizeof(mandala::uid_t)) {
87 mandala::uid_t uid;
88 stream >> uid;
89 findParent<PApx>()->trace_uid(uid);
90 trace()->data(stream.payload());
91 emit calibrationData(uid, stream.payload());
92 return true;
93 }
94 qWarning() << "Empty calibration data received";
95 break;
96 }
97 } while (0);
98
99 //error
100 trace()->block("ERR:");
101 trace()->data(stream.payload());
102
103 _vehicle->incErrcnt();
104 return true;
105}

138void PApxData::pack(const QVariant &v, mandala::type_id_e type, PStreamWriter &stream)
139{
140 switch (type) {
141 case mandala::type_byte:
142 stream.write<mandala::byte_t>(v.value<mandala::byte_t>());
143 return;
144 case mandala::type_word:

146 return;
147 case mandala::type_dword:

149 return;
150 case mandala::type_real:
151 stream.write<mandala::real_t>(v.value<mandala::real_t>());
152 return;
153 }

156void PApxData::sendValue(mandala::uid_t uid, QVariant value)
157{
158 if (mandala::is_bundle(uid)) {
159 sendBundle(uid, value);
160 return;
161 }
162
163 _req.request(uid, value.isNull() ? xbus::pri_request : xbus::pri_final);
164 mandala::spec_s spec{};
165 spec.type = Mandala::meta(uid).type_id;
166 spec.write(&_req);
167 trace()->block(QString("T%1").arg(spec.type));
168
169 if (!value.isNull()) {
170 size_t spos = _req.pos();
171 pack(value, spec.type, _req);
172 trace()->data(_req.toByteArray(spos));
173 }
174 _req.send();
175}

242void PApxData::sendBundle(mandala::uid_t uid, QVariant value)
243{
244 switch (uid) {
245 default:
246 break;
247 case mandala::cmd::nav::pos::uid: // fly to
248 case mandala::est::nav::pos::uid: // pos fix
249 case mandala::est::nav::ref::uid: // set home
250 {
251 QVariantList v = value.value<QVariantList>();
252 if (v.size() != 2)
253 break;
254 sendBundleT<mandala::bundle::pos_ll_s>(uid,
255 {mandala::to_gps(v.at(0).toDouble()),
256 mandala::to_gps(v.at(1).toDouble())});
257 return;
258 }
259 }
