13/*
14 * Graphics routines for X-Plane and OpenGL.
15 *
16 * A few notes on coordinate systems:
17 *
18 * X-Plane uses three kinds of coordinates. Global coordinates are specified
19 * as latitude, longitude and elevation. This coordinate system never changes
20 * but is not very precise.
21 *
22 * OpenGL (or 'local') coordinates are cartesian and shift with the plane.
23 * They offer more precision and are used for 3-d OpenGL drawing. The X axis
24 * is aligned east-west with positive X meaning east. The Y axis is aligned
25 * straight up and down at the point 0,0,0 (but since the earth is round it is
26 * not truly straight up and down at other points). The Z axis is aligned
27 * north-south at 0, 0, 0 with positive Z pointing south (but since the earth
28 * is round it isn't exactly north-south as you move east or west of 0, 0, 0).
29 * One unit is one meter and the point 0,0,0 is on the surface of the earth
30 * at sea level for some latitude and longitude picked by the sim such that
31 * the user's aircraft is reasonably nearby.
32 *
33 * Cockpit coordinates are 2d, with the X axis horizontal and the Y axis
34 * vertical. The point 0,0 is the bottom left and 1024,768 is the upper right
35 * of the screen. This is true no matter what resolution the user's monitor is
36 * in; when running in higher resolution, graphics will be scaled.
37 *
38 * Use X-Plane's routines to convert between global and local coordinates. Do
39 * not attempt to do this conversion yourself; the precise 'roundness' of
40 * X-Plane's physics model may not match your own, and (to make things
41 * weirder) the user can potentially customize the physics of the current
42 * planet.
43 *
44 */
45

83/*
84 * XPLMSetGraphicsState
85 *
86 * XPLMSetGraphicsState changes OpenGL's graphics state in a number of ways:
87 *
88 * inEnableFog - enables or disables fog, equivalent to: glEnable(GL_FOG);
89 *
90 * inNumberTexUnits - enables or disables a number of multitexturing units. If
91 * the number is 0, 2d texturing is disabled entirely, as in
92 * glDisable(GL_TEXTURE_2D); Otherwise, 2d texturing is enabled, and a
93 * number of multitexturing units are enabled sequentially, starting with
94 * unit 0, e.g. glActiveTextureARB(GL_TEXTURE0_ARB); glEnable
95 * (GL_TEXTURE_2D);
96 *
97 * inEnableLighting - enables or disables OpenGL lighting, e.g.
98 * glEnable(GL_LIGHTING); glEnable(GL_LIGHT0);
99 *
100 * inEnableAlphaTesting - enables or disables the alpha test per pixel, e.g.
101 * glEnable(GL_ALPHA_TEST);
102 *
103 * inEnableAlphaBlending - enables or disables alpha blending per pixel, e.g.
104 * glEnable(GL_BLEND);
105 *
106 * inEnableDepthTesting - enables per pixel depth testing, as in
107 * glEnable(GL_DEPTH_TEST);
108 *
109 * inEnableDepthWriting - enables writing back of depth information to the
110 * depth bufffer, as in glDepthMask(GL_TRUE);
111 *
112 * The purpose of this function is to change OpenGL state while keeping
113 * X-Plane aware of the state changes; this keeps X-Plane from getting
114 * surprised by OGL state changes, and prevents X-Plane and plug-ins from
115 * having to set all state before all draws; XPLMSetGraphicsState internally
116 * skips calls to change state that is already properly enabled.
117 *
118 * X-Plane does not have a 'default' OGL state to plug-ins; plug-ins should
119 * totally set OGL state before drawing. Use XPLMSetGraphicsState instead of
120 * any of the above OpenGL calls.
121 *
122 * WARNING: Any routine that performs drawing (e.g. XPLMDrawString or widget
123 * code) may change X-Plane's state. Always set state before drawing after
124 * unknown code has executed.
125 *
126 */
127XPLM_API void XPLMSetGraphicsState(
128 int inEnableFog,
129 int inNumberTexUnits,
130 int inEnableLighting,
131 int inEnableAlphaTesting,
132 int inEnableAlphaBlending,
133 int inEnableDepthTesting,
134 int inEnableDepthWriting);

136/*
137 * XPLMBindTexture2d
138 *
139 * XPLMBindTexture2d changes what texture is bound to the 2d texturing target.
140 * This routine caches the current 2d texture across all texturing units in
141 * the sim and plug-ins, preventing extraneous binding. For example, consider
142 * several plug-ins running in series; if they all use the 'general interface'
143 * bitmap to do UI, calling this function will skip the rebinding of the
144 * general interface texture on all but the first plug-in, which can provide
145 * better frame rate son some graphics cards.
146 *
147 * inTextureID is the ID of the texture object to bind; inTextureUnit is a
148 * zero-based texture unit (e.g. 0 for the first one), up to a maximum of 4
149 * units. (This number may increase in future versions of x-plane.)
150 *
151 * Use this routine instead of glBindTexture(GL_TEXTURE_2D, ....);
152 *
153 */
154XPLM_API void XPLMBindTexture2d(
155 int inTextureNum,
156 int inTextureUnit);

158/*
159 * XPLMGenerateTextureNumbers
160 *
161 * This routine generates unused texture numbers that a plug-in can use to
162 * safely bind textures. Use this routine instead of glGenTextures;
163 * glGenTextures will allocate texture numbers in ranges that X-Plane reserves
164 * for its own use but does not always use; for example, it might provide an
165 * ID within the range of textures reserved for terrain...loading a new .env
166 * file as the plane flies might then cause X-Plane to use this texture ID.

168 * texture IDs that are out of X-Plane's usage range.
169 *
170 */
