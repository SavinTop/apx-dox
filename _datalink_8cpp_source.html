91void Datalink::addConnection(DatalinkConnection *c)
92{
93 connections.append(c);
94 connect(c, &DatalinkConnection::packetReceived, this, &Datalink::connectionPacketReceived);
95 connect(c, &Fact::removed, this, [this, c]() {
96 //qDebug()<<"rm"<<c;
97 connections.removeOne(c);
98 updateStatus();
99 });
100 connect(c, &Fact::activeChanged, this, &Datalink::updateStatus);
101 updateStatus();

104void Datalink::updateStatus()
105{
106 int cnt = connections.size();
107 if (cnt <= 0) {
108 setValue(QVariant());
109 } else {
110 int acnt = 0;
111 for (int i = 0; i < cnt; ++i) {
112 DatalinkConnection *c = connections.at(i);
113 if (!c) {
114 connections.removeAt(i);
115 updateStatus();
116 return;
117 }
118 if (c->active())
119 acnt++;
120 }
121 setValue(QString("%1/%2").arg(acnt).arg(cnt));
122 setActive(acnt > 0);
123 }
124}

149void Datalink::connectionPacketReceived(QByteArray packet, quint16 network)
150{
151 //qDebug()<<"R"<<packet.toHex().toUpper();
152 if (packet.isEmpty())
153 return;
154 if (!network)
155 return;
156 DatalinkConnection *src = qobject_cast<DatalinkConnection *>(sender());
157 if (!src)
158 return;
159 for (int i = 0; i < connections.size(); ++i) {
160 DatalinkConnection *c = connections.at(i);
161 if (!c)
162 continue;
163 if (c == src)
164 continue;
165
166 c->sendPacket(packet, network);
167 }
168
169 emit packetReceived(packet);
170}
171void Datalink::sendPacket(QByteArray packet)
172{
173 //qDebug()<<"S"<<packet.toHex().toUpper();
174 if (bReadOnly)
175 return;
176 if (packet.isEmpty())
177 return;
178
179 for (int i = 0; i < connections.size(); ++i) {
180 DatalinkConnection *c = connections.at(i);
181 if (!c)
182 continue;
183 c->sendPacket(packet, LOCAL);
184 }
185 emit packetTransmitted(packet);
186}
