121void TileLoader::download(quint64 uid)
122{
123 //qDebug()<<"download"<<uid;
124 if (downloads.contains(uid))
125 return;
126
127 if (f_offline->value().toBool())
128 return;
129
130 //download
131 QNetworkRequest request;
132 MapID type = (MapID) TileLoader::type(uid);
133 int x = TileLoader::x(uid);
134 int y = TileLoader::y(uid);
135 int zoom = TileLoader::level(uid);
136 bool bVersionRequest = false;
137 //qDebug()<<spec<<type;
138 switch (type) {
139 default:
140 break;
141 case GoogleSatellite: {
142 // http://mt1.google.com/vt/lyrs=s
143 if (!checkGoogleVersion(&request)) {
144 if (request.url().isEmpty())
145 return;
146 bVersionRequest = true;
147 } else {
148 QString server = "khm";
149 QString req = "kh";
150 QString sec1 = ""; // after &x=...
151 QString sec2 = ""; // after &zoom=...
152 getSecGoogleWords(x, y, sec1, sec2);
153 request.setUrl(
154 QUrl(QString("http://%1%2.google.com/%3/v=%4&hl=%5&x=%6%7&y=%8&z=%9&s=%10")
155 .arg(server)
156 .arg(getServerNum(x, y, 4))
157 .arg(req)
158 .arg(versionGoogleMaps)
159 .arg(language)
160 .arg(x)
161 .arg(sec1)
162 .arg(y)
163 .arg(zoom)
164 .arg(sec2)));
165 request.setRawHeader("Referrer", "https://www.google.com/maps/preview");
166 }
167 } break;
168 case GoogleHybrid: {
169 if (!checkGoogleVersion(&request)) {
170 if (request.url().isEmpty())
171 return;
172 bVersionRequest = true;
173 } else {
174 request.setUrl(QUrl(QString("https://khm.google.com/vt/lbw/lyrs=y&x=%1&y=%2&z=%3")
175 .arg(x)
176 .arg(y)
177 .arg(zoom)));
178 request.setRawHeader("Referrer", "http://maps.google.com/");
179 }
180 } break;
181 }
182 //begin download
183 QNetworkReply *reply = downloadRequest(&request);
184 //qDebug()<<request.url();
185 connect(reply,
186 static_cast<void (QNetworkReply::*)(QNetworkReply::NetworkError)>(
187 &QNetworkReply::errorOccurred),
188 this,
189 &TileLoader::networkReplyError);
190 if (bVersionRequest) {
191 connect(reply, &QNetworkReply::finished, this, &TileLoader::versionReplyFinished);
192 pendingDownloads.append(uid);
193 } else {
194 connect(reply, &QNetworkReply::finished, this, &TileLoader::networkReplyFinished);
195 reqMap.insert(reply, uid);
196 downloads.append(uid);
197 emit requestCountChanged();
198 }
199}

201QNetworkReply *TileLoader::downloadRequest(QNetworkRequest *request)
202{
203#if !defined(__mobile__)
204 QNetworkProxy proxy = net->proxy();
205 QNetworkProxy tProxy;
206 tProxy.setType(QNetworkProxy::DefaultProxy);
207 net->setProxy(tProxy);
208#endif
209
210 QSslConfiguration ssl = request->sslConfiguration();
211 ssl.setPeerVerifyMode(QSslSocket::VerifyNone);
212 request->setSslConfiguration(ssl);
213
214 request->setRawHeader("Accept", "*/*");
215 request->setRawHeader("User-Agent", userAgent);
216
217 QNetworkReply *reply = net->get(*request);
218
219#if !defined(__mobile__)
220 net->setProxy(proxy);
221#endif
222 return reply;
223}

253void TileLoader::networkReplyFinished()
254{
255 QNetworkReply *reply = qobject_cast<QNetworkReply *>(sender());
256 if (!reply)
257 return;
258 quint64 uid = reqMap.value(reply);
259 reqMap.remove(reply);
260 downloads.removeOne(uid);
261 emit requestCountChanged();
262
263 if (reply->error() != QNetworkReply::NoError) {
264 return;
265 }
266 QByteArray data = reply->readAll();
267 emit tileLoaded(uid, data);
268
269 if (!checkImage(data)) {
270 apxConsoleW() << "Error downloading map (not an image)";
271 } else {
272 //db->reqSaveTile(type(uid),dbHash(uid),data);
273 (new DBReqSaveTile(db, type(uid), dbHash(uid), versionGoogleMaps.toUInt(), data))->exec();
274 }
275
276 //qDebug()<<"downloaded"<<uid;
277}

297void TileLoader::versionReplyFinished()
298{
299 QNetworkReply *reply = qobject_cast<QNetworkReply *>(sender());
300 if (!reply)
301 return;
302 reply->deleteLater();
303 //emit requestCountChanged();
304 //decRequestCount();
305 if (reply->error() != QNetworkReply::NoError) {
306 if (reply->error() != QNetworkReply::HostNotFoundError) {
307 apxMsgW() << "Error collecting Google maps version info";
308 }
309 return;
310 }
311 QString html = QString(reply->readAll());
312
313#if defined(DEBUG_GOOGLE_MAPS)
314 QString filename = QStandardPaths::writableLocation(QStandardPaths::HomeLocation);
315 filename += "/google.output";
316 QFile file(filename);
317 if (file.open(QIODevice::ReadWrite)) {
318 QTextStream stream(&file);
319 stream << html << endl;
320 }
321#endif
322
323 bool ok = false;
324 QRegExp reg("\"*https?://mt\\D?\\d..*/vt\\?lyrs=m@(\\d*)", Qt::CaseInsensitive);
325 if (reg.indexIn(html) != -1) {
326 QStringList gc = reg.capturedTexts();
327 //versions[GoogleMap] = QString("m@%1").arg(gc[1]);
328 }
329 reg = QRegExp("\"*https?://khm\\D?\\d.googleapis.com/kh\\?v=(\\d*)", Qt::CaseInsensitive);
330 if (reg.indexIn(html) != -1) {
331 QStringList gc = reg.capturedTexts();
332 versionGoogleMaps = gc[1];
333 ok = true;
334 }
335 reg = QRegExp("\"*https?://mt\\D?\\d..*/vt\\?lyrs=t@(\\d*),r@(\\d*)", Qt::CaseInsensitive);
336 if (reg.indexIn(html) != -1) {
337 QStringList gc = reg.capturedTexts();
338 //_versionGoogleTerrain = QString("t@%1,r@%2").arg(gc[1]).arg(gc[2]);
339 }
340 if (!ok)
341 return;
342 if (versionGoogleMaps.isEmpty())
343 versionGoogleMaps = "748";
344 //qDebug()<<versionGoogleMaps;
345 foreach (quint64 uid, pendingDownloads) {
346 download(uid);
347 }
348 pendingDownloads.clear();
