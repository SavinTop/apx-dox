74QModelIndex FactTreeModel::index(int row, int column, const QModelIndex &parent) const
75{
76 if (!hasIndex(row, column, parent))
77 return QModelIndex();
78 Fact *parentFact;
79 if (!parent.isValid())
80 parentFact = _root;
81 else
82 parentFact = fact(parent);
83 if (!parentFact)
84 return QModelIndex();
85 Fact *childFact = parentFact->child(row);
86 if (!childFact)
87 return QModelIndex();
88 checkConnections(childFact);
89 return factIndex(childFact, column);
90}

142Qt::ItemFlags FactTreeModel::flags(const QModelIndex &index) const
143{
144 Qt::ItemFlags fx = Qt::NoItemFlags;
145 Fact *f = fact(index);
146 if (!f)
147 return fx;
148 if (f->enabled())
149 fx |= Qt::ItemIsEnabled | Qt::ItemIsSelectable;
150 else
151 return fx;
152 if (index.column() != Fact::FACT_MODEL_COLUMN_VALUE)
153 return fx;
154 if (f->treeType() == Fact::Group && f->size() > 1 && f->child(0)->treeType() == Fact::Group
155 && f->value().toString().startsWith('[') && f->value().toString().endsWith(']'))
156 return fx | Qt::ItemIsEditable; //array editor
157
158 if (f->treeType())
159 return fx;
160 if (f->dataType() && f->dataType() != Fact::Count) {
161 fx |= Qt::ItemIsEditable;
162 }
163 return fx;
164}

313void FactTreeModel::updateTimerTimeout()
314{
315 updateTimer.stop();
316 foreach (Fact *f, updateList) {
317 if (!(f && f->size()))
318 continue;
319 if (!expandedFacts.contains(f))
320 continue;
321 foreach (int col, updateHash.values(f)) {
322 QModelIndex index1 = createIndex(0, col, f->child(0));
323 QModelIndex index2 = createIndex(f->size() - 1, col, f->child(f->size() - 1));
324 emit dataChanged(index1, index2, QVector<int>() << Qt::DisplayRole);
325 //qDebug()<<f->path();
326 }
327 }
328 //emit dataChanged(QModelIndex(),QModelIndex());
329 updateList.clear();
330 updateHash.clear();
331}
