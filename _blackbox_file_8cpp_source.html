45void BlackboxFile::readNext()
46{
47 if (!file.isOpen())
48 return;
49 reader->processData(file.read(1024));
50
51 if (file.atEnd()) {
52 apxMsg() << tr("Blackbox import finished");
53 stop();
54 return;
55 }
56 qint64 p = file.pos() * 100 / file.size();
57 setProgress(p);
58
59 QTimer::singleShot(1, this, &BlackboxFile::readNext);
60}

62void BlackboxFile::getStats()
63{
64 fi = QFileInfo();
65 updateStats();
66
67 QDir defaultDir(QStandardPaths::writableLocation(QStandardPaths::DownloadLocation));
68 QDir dir = QDir(QSettings()
69 .value(QString("SharePath_%1").arg(name()), defaultDir.canonicalPath())
70 .toString());
71 if (!dir.exists())
72 dir = defaultDir;
73 QFileDialog dlg(nullptr, descr(), dir.canonicalPath());
74 dlg.setAcceptMode(QFileDialog::AcceptOpen);
75 dlg.setFileMode(QFileDialog::ExistingFile);
76 dlg.setViewMode(QFileDialog::Detail);
77 if (!(dlg.exec() && dlg.selectedFiles().size() == 1)) {
78 return;
79 }
80
81 QSettings().setValue(QString("SharePath_%1").arg(name()), dlg.directory().absolutePath());
82
83 QString fname = dlg.selectedFiles().first();
84 fi = QFileInfo(fname);
85 updateStats();
86}
87void BlackboxFile::download()
88{
89 if (!fi.exists())
90 return;
91 if (fi.size() <= 0)
92 return;
93 BlackboxItem::download();
94 file.setFileName(fi.filePath());
95 if (!file.open(QFile::ReadOnly)) {
96 apxMsgW() << tr("Can't read file") << file.fileName();
97 stop();
98 return;
99 }
100 readNext();
101}
