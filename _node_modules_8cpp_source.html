86void NodeModules::modReceived(PNode::mod_cmd_e cmd, QByteArray adr, QStringList data)
87{
88 // called in root module only
89 auto m = findModule(adr);
90 if (!m) {
91 qWarning() << "missing module" << adr.toHex().toUpper() << data;
92 return;
93 }
94
95 switch (cmd) {
96 default:
97 qWarning() << "unknown cmd" << cmd << m->path() << data;
98 return;
99 case PNode::ls:
100 m->updateFacts(data);
101 break;
102 case PNode::status: {
103 QString s = data.join('|').replace('\t', ' ').replace('\n', '|').simplified();
104 if (m->size() > 0) {
105 if (s.isEmpty())
106 return;
107 m->unbindProperties(m->child(0));
108 }
109 m->setValue(s);
110 break;
111 }
112 }
113}
114NodeModules *NodeModules::findModule(QByteArray adr)
115{
116 if (adr.isEmpty())
117 return this;
118 auto n = static_cast<uint8_t>(adr.at(0));
119 for (auto i : facts()) {
120 auto m = qobject_cast<NodeModules *>(i);
121 if (!m)
122 continue;
123 if (!n--)
124 return m->findModule(adr.mid(1));
125 }
126 return {};
127}

129void NodeModules::updateFacts(QStringList names)
130{
131 if (_done_ls)
132 return;
133 _done_ls = true;
134
135 if (names.isEmpty()) {
136 setTreeType(NoFlags);
137 } else {
138 setTreeType(Group);
139 for (auto s : names) {
140 new NodeModules(this, _node, s);
141 }
142 if (size() > 0) {
143 QStringList st;
144 for (auto i : facts())
145 st.append(i->name());
146 setDescr(st.join(','));
147
148 if (!_is_root) {
149 auto f = child(0);
150 bindProperty(f, "value", true);
151 }
152 }
153 }
154
155 // update values
156 update();
157}
