58void Sites::createEditor(QVariantMap item)
59{
60 //qDebug()<<item.value("title").toString();
61 if (f_edit)
62 f_edit->deleteFact();
63 f_edit = new SiteEdit(this, "edit", tr("Edit site"), tr("Edit area parameters"), item);
64 f_edit->setIcon("cog");
65 connect(f_edit, &SiteEdit::removed, this, [this]() { f_edit = nullptr; });
66 connect(f_edit, &SiteEdit::removeTriggered, this, &Sites::dbRemoveSite);
67 connect(f_edit, &SiteEdit::siteEdited, this, &Sites::dbUpdateSite);
68 connect(f_lookup->dbModel(),
69 &DatabaseLookupModel::itemEdited,
70 f_edit,
71 &SiteEdit::updateFromEditedModelData);
72 connect(f_lookup->dbModel(), &DatabaseLookupModel::synced, this, &Sites::syncEditorFromModel);
73}
74void Sites::destroyEditor(QVariantMap item)
75{
76 //qDebug()<<item.value("title").toString();
77 Q_UNUSED(item)
78 if (!f_edit)
79 return;
80 if (f_edit->modelData.value("key").toULongLong() != item.value("key").toULongLong())
81 return;
82 f_edit->deleteFact();
83 f_edit = nullptr;
84}
85void Sites::syncEditorFromModel()
86{
87 if (!f_edit)
88 return;
89 int i = f_lookup->dbModel()->indexOf("key", f_edit->modelData.value("key"));
90 if (i < 0) {
91 destroyEditor(f_edit->modelData);
92 return;
93 }
94 f_edit->setModelData(f_lookup->dbModel()->get(i));
95}
