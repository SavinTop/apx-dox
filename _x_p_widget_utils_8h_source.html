13/*
14 * XPWidgetUtils - USAGE NOTES
15 *
16 * The XPWidgetUtils library contains useful functions that make writing and
17 * using widgets less of a pain.
18 *
19 * One set of functions are the widget behavior functions. These functions
20 * each add specific useful behaviors to widgets. They can be used in two
21 * manners:
22 *
23 * 1. You can add a widget behavior function to a widget as a callback proc
24 * using the XPAddWidgetCallback function. The widget will gain that behavior.
25 * Remember that the last function you add has highest priority. You can use
26 * this to change or augment the behavior of an existing finished widget.
27 *
28 * 2. You can call a widget function from inside your own widget function.
29 * This allows you to include useful behaviors in custom-built widgets. A
30 * number of the standard widgets get their behavior from this library. To do
31 * this, call the behavior function from your function first. If it returns 1,
32 * that means it handled the event and you don't need to; simply return 1.
33 *
34 */

75/*
76 * XPWidgetCreate_t
77 *
78 * This structure contains all of the parameters needed to create a wiget. It
79 * is used with XPUCreateWidgets to create widgets in bulk from an array. All
80 * parameters correspond to those of XPCreateWidget except for the container
81 * index. If the container index is equal to the index of a widget in the
82 * array, the widget in the array passed to XPUCreateWidgets is used as the
83 * parent of this widget. Note that if you pass an index greater than your own
84 * position in the array, the parent you are requesting will not exist yet. If
85 * the container index is NO_PARENT, the parent widget is specified as NULL.
86 * If the container index is PARAM_PARENT, the widget passed into
87 * XPUCreateWidgets is used.
88 *
89 */

91 int left;
92 int top;
93 int right;
94 int bottom;
95 int visible;
96 const char * descriptor;
97 int isRoot;
98 int containerIndex;
99 XPWidgetClass widgetClass;
100} XPWidgetCreate_t;

108/*
109 * XPUCreateWidgets
110 *
111 * This function creates a series of widgets from a table...see
112 * XPCreateWidget_t above. Pass in an array of widget creation structures and
113 * an array of widget IDs that will receive each widget.
114 *
115 * Widget parents are specified by index into the created widget table,
116 * allowing you to create nested widget structures. You can create multiple
117 * widget trees in one table. Generally you should create widget trees from
118 * the top down.
119 *
120 * You can also pass in a widget ID that will be used when the widget's parent
121 * is listed as PARAM_PARENT; this allows you to embed widgets created with
122 * XPUCreateWidgets in a widget created previously.
123 *
124 */
