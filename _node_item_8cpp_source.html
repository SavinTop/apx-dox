268{
269 //return;
270 //hide grouped arrays (gpio, controls etc)
271 FactList groups;
272 for (auto f : m_fields) {
273 Fact *groupItem = f->parentFact();
274 if (!groupItem || groupItem == this || groups.contains(groupItem))

276

278 //qDebug() << groupItem->path();

285 if (!f) {
286 break;

289 break;

292 }

295 }
296 }
297 for (auto group : groups) {
298 //qDebug() << group->path();
299 groupArrays(group);

303void NodeItem::groupArrays(Fact *group)
304{
305 //create action with underlaying table structure to edit array rows
306 Fact *action = new Fact(group, group->name(), group->title(), group->descr(), Action);
307 group->setMenu(action);
308 //action->setActionsModel(group->actionsModel());
309 new NodeViewActions(action, _nodes);
310 action->bindProperty(group, "modified", true);
311
312 //hide group members
313 for (auto i : group->facts()) {
314 i->setVisible(false);
315 }
316
317 Fact *f1 = group->child(0);
318
319 group->bindProperty(f1, "value", true);
320
321 bool bControls = group->name() == "controls" || group->name() == "mixer";
322
323 int colCnt = group->size();
324 for (int row = 0; row < f1->size(); ++row) {
325 Fact *fi = f1->child(row);
326 Fact *fRow = new Fact(action, fi->name(), fi->title(), "", Group | ModifiedGroup);
327 new NodeViewActions(fRow, _nodes);
328
329 fRow->bindProperty(fi, "valueText", true);
330 fRow->bindProperty(fi, "modified", true);
331
332 connect(fi, &Fact::valueTextChanged, this, [this, fRow]() { updateArrayRowDescr(fRow); });
333
334 Fact *f_ch = nullptr;
335 int f_ch_max = 0;
336 for (int i = 0; i < colCnt; ++i) {
337 Fact *fArray = group->child(i);
338 if (!fArray)
339 continue;
340 Fact *fp;
341 bool bChParam = false;
342 if (bControls && f_ch && fArray->name().startsWith("ch_")) {
343 fp = fArray->child(0);
344 f_ch_max = fArray->size();
345 bChParam = true;
346 } else {
347 fp = fArray->child(row);
348 }
349 if (!fp)
350 continue;
351 Fact *f = new Fact(fRow,
352 fArray->name(),
353 fArray->title(),
354 fArray->descr(),
355 fp->treeType() | fp->dataType() | ModifiedTrack);
356 new NodeViewActions(f, _nodes);
357 connect(f, &Fact::valueTextChanged, fRow, [this, fRow]() { updateArrayRowDescr(fRow); });
358 if (bChParam) {
359 updateArrayChBinding(f, fArray, f_ch);
360 connect(f_ch, &Fact::valueChanged, f, [this, f, fArray, f_ch]() {
361 updateArrayChBinding(f, fArray, f_ch);
362 });
363 } else {
364 f->setBinding(fp);
365 }
366
367 if (bControls && f->name() == "ch") {
368 f_ch = f;
369 }
370 }
371 if (f_ch && f_ch_max > 0) {
372 f_ch->setMax(f_ch_max);
373 }
374 }

518NodeField *NodeItem::field(QString name) const
519{
520 QRegExp re("_(\\d+)$");
521 auto a = re.indexIn(name);
522 int aidx = -1;
523 if (a > 1) {
524 auto i = re.cap(1).toInt() - 1;
525 if (i < 0) {
526 qWarning() << "array" << name;
527 return nullptr;
528 }
529 name = name.left(a);
530 aidx = i;
531 }
532 NodeField *f = nullptr;
533 for (auto i : m_fields) {
534 if (i->fpath() != name)
535 continue;
536 f = i;
537 break;
538 }
539 if (!f)
540 return nullptr;
541 if (aidx < 0 && f->size() <= 0)
542 return f;
543
544 if (aidx >= f->size())
545 return nullptr;
546 return qobject_cast<NodeField *>(f->child(aidx));
547}

551void NodeItem::identReceived(QVariantMap ident)
552{
553 if (ident.isEmpty())
554 return;
555
556 if (!ident.contains("user")) {
557 QVariantMap user;
558 user.insert("machineUID", App::machineUID());
559 user.insert("username", App::username());
560 user.insert("hostname", App::hostname());
561 ident.insert("user", user);
562 }
563
564 if (_ident == ident && valid())
565 return;
566
567 qWarning() << "ident updated" << title() << _nodes->vehicle->title();

570 clear();

573 _lastSeenTime = QDateTime::currentDateTime().toMSecsSinceEpoch();
574 storage->saveNodeInfo();

578 do {
579 if (_nodes->vehicle->isLocal() && !_nodes->vehicle->active())
580 break;
581 if (_nodes->upgrading())
582 break;
583
584 _protocol->requestDict();
585 } while (0);

597void NodeItem::dictReceived(QVariantMap dict)
598{
599 if (dict.isEmpty())
600 return;
601
602 if (valid())
603 return;
604
605 clear();
606 _dict = dict;
607 _dict.remove("cached");
608
609 if (!_dict.value("time").toULongLong()) {
610 _dict.insert("time", QDateTime::currentDateTime().toMSecsSinceEpoch());
611 }
612
613 auto fields = dict.value("fields").value<QVariantList>();
614
615 xbus::node::usr::cmd_t cmd_cnt = 0;
616 for (auto const &i : fields) {
617 auto field = i.value<QVariantMap>();
618 QString name = field.value("name").toString();
619 QString title = field.value("title").toString();
620 QString type = field.value("type").toString();
621
622 // find group fact
623 Fact *g = this;
624 QStringList path = name.split('.');
625 while (path.size() > 1) {
626 g = g->child(path.takeFirst());
627 if (g)
628 continue;
629 qWarning() << "missing group" << name;
630 g = this;
631 break;
632 }
633 name = path.first();
634
635 if (type == "group") {
636 g = new Fact(g, name, title, "", Group | ModifiedGroup);
637 new NodeViewActions(g, _nodes);
638 } else if (type == "command") {
639 tools->addCommand(g, name, title, cmd_cnt++);
640 } else { // data field
641 NodeField *f = new NodeField(g, this, field, m_fields.size());
642 m_fields.append(f);
643 if (!_status_field) {
644 if (type == "string") {
645 _status_field = f;
646 connect(_status_field, &Fact::valueChanged, this, &NodeItem::updateStatus);
647 }
648 }
649 }
650 }
651 //qDebug() << m_fields.size();
652 removeEmptyGroups(this);
653 groupArrays();
654 linkGroupValues(this);
655
656 // update descr and help from meta DB cache
657 storage->loadNodeMeta();
658
659 backup();
660
661 if (_protocol) {
662 if (!_dict.value("cached").toBool())
663 storage->saveNodeDict();
664 if (!_nodes->vehicle->isLocal() || _nodes->vehicle->active())
665 _protocol->requestConf();
666 }
667}

697void NodeItem::confReceived(QVariantMap values)
698{
699 if (m_fields.isEmpty()) {
700 qWarning() << "missing dict:" << title();
701 return;
702 }
703
704 QStringList fields;
705 for (auto f : m_fields) {
706 QString fpath = f->fpath();
707 if (!values.contains(fpath)) {
708 //qWarning() << "missing data for:" << fpath;
709 continue;
710 }
711 fields.append(fpath);
712 f->fromVariant(values.value(fpath));
713 }
714
715 if (valid()) {
716 backup();
717 return;
718 }
719
720 // report missing fields
721 for (auto fpath : values.keys()) {
722 if (!fields.contains(fpath)) {
723 qWarning() << "missing field for:" << fpath;
724 }
725 }
726 if (fields.size() != m_fields.size()) {
727 apxMsgW() << tr("Inconsistent parameters");
728 return;
729 }
730
731 validateData();
732 setEnabled(true);
733
734 if (_protocol && !_ident.value("reconf").toBool())
735 storage->saveNodeConfig();
736}
