68void Poi::updateTitle()
69{
70 QStringList st;
71 st.append(QString::number(num() + 1));
72 int r = f_radius->value().toInt();
73 if (std::abs(r) > 0) {
74 st.append(AppRoot::distanceToString(std::abs(r)));
75 if (r < 0)
76 st.append(tr("CCW"));
77 } else
78 st.append("H");
79 setTitle(st.join(' '));

112void Poi::setRadiusPoint(const QGeoCoordinate &v)
113{
114 QGeoCoordinate p(f_latitude->value().toDouble(), f_longitude->value().toDouble());
115 double a = qDegreesToRadians(p.azimuthTo(v));
116 double d = p.distanceTo(v);
117 QPointF ne(d * std::cos(a), d * std::sin(a));
118 ne = AppRoot::rotate(ne, 90.0);
119 int rabs = std::abs(f_radius->value().toInt());
120 if (std::abs(ne.y()) > (rabs / 2.0)) {
121 //switch turn direction
122 f_radius->setValue(ne.y() > 0 ? rabs : -rabs);
123 }
124 int dist = ne.x();
125 if (dist < 20)
126 dist = 0;
127 else if (dist > 50000)
128 dist = 50000;
129 else if (dist > 500)
130 dist = (dist / 100) * 100;
131 else
132 dist = (dist / 10) * 10;
133 f_radius->setValue(f_radius->value().toInt() < 0 ? -dist : dist);
134}
