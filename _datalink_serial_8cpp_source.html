114void DatalinkSerial::openNext()
115{
116 if (dev->isOpen()) {
117 closePort();
118 return;
119 }
120 if (!activated())
121 return;
122
123 setStatus(tr("Searching"));
124
125 while (1) {
126 QSerialPortInfo spi;
127 if ((!m_devName.isEmpty()) && m_devName != "auto") {
128 //the port name is m_devName
129 spi = QSerialPortInfo(m_devName);
130 } else {
131 //[auto] - scan for any next available port
132 QList<QSerialPortInfo> list = QSerialPortInfo::availablePorts();
133 if (list.isEmpty())
134 break;
135 if (scanIdx >= list.size())
136 scanIdx = 0;
137 for (int i = 0; i < list.size(); i++) {
138 const QSerialPortInfo &lspi = list.at(scanIdx);
139 if (lspi.portName().contains("usb", Qt::CaseInsensitive)
140 && lspi.portName().contains("cu.", Qt::CaseInsensitive) && isAvailable(lspi)) {
141 spi = lspi;
142 break;
143 }
144 scanIdx++;
145 if (scanIdx >= list.size())
146 scanIdx = 0;
147 }
148 }
149 if (!isAvailable(spi))
150 break;
151 //try to open valid spi port
152 //qDebug("Trying to open %s (%s)",spi.portName().toUtf8().data(),spi.systemLocation().toUtf8().data());
153 if (!openPort(spi, m_baud)) {
154 apxMsgW() << tr("Serial port open failed") << QString("(%1)").arg(spi.portName());
155 break;
156 }
157 info = spi;
158 openPorts.append(info.portName());
159 if (m_devName != info.portName())
160 setUrl(QString("%1:%2").arg(m_devName).arg(info.portName()));
161 setStatus(tr("Connected"));
162 apxMsg() << tr("Serial port connected").append(":") << info.portName();
163 opened();
164 return;
165 }
166 //continue to watch for ports to be available
167 openTimer.start();
168}

170bool DatalinkSerial::openPort(const QSerialPortInfo &spi, uint baud)
171{
172 setStatus(tr("Connecting"));
173
174 dev->setPort(spi);
175 dev->setBaudRate(9600);
176 if (!dev->open(QIODevice::ReadWrite))
177 return false;
178 dev->setBaudRate(static_cast<int>(baud));
179 //lock
180 if (lock) {
181 lock->unlock();
182 delete lock;
183 }
184 lock = new QLockFile(
185 QDir::tempPath() + "/"
186 + QString(QCryptographicHash::hash(QByteArray(spi.systemLocation().toUtf8()),
187 QCryptographicHash::Sha1)
188 .toHex())
189 + ".lock");
190 if (!lock->tryLock()) {
191 closePort();
192 apxMsgW() << "Unable to lock" << spi.systemLocation();
193 return false;
194 }
195 return true;
196}

207 closed();
208 setStatus("");
209 setUrl(m_devName);
210 //if(openTimer && openTimer->isActive())openTimer->stop();

213bool DatalinkSerial::isAvailable(const QSerialPortInfo &spi)
214{
215 if (spi.isNull())
216 return false;
217 if (openPorts.contains(spi.portName()))
218 return false;
219
220 QLockFile tlock(QDir::tempPath() + "/"
221 + QString(QCryptographicHash::hash(QByteArray(spi.systemLocation().toUtf8()),
222 QCryptographicHash::Sha1)
223 .toHex())
224 + ".lock");
225 if (!tlock.tryLock(10))
226 return false;
227 tlock.unlock();
228 return true;
229}

274QByteArray DatalinkSerial::read()
275{
276 if (!dev->isOpen()) {
277 resetDataStream();
278 return {};
279 }
280
281 if (dev->bytesAvailable() <= 0)
282 return {};
283
284 // qDebug() << "RX" << dev->bytesAvailable();
285
286 // read PHY and continue decoding
287 auto rcnt = dev->read(reinterpret_cast<char *>(_rxbuf_raw), sizeof(_rxbuf_raw));
288 if (rcnt < 0) {
289 serialPortError(QSerialPort::ReadError);
290 return {};
291 }
292 if (rcnt <= 0) {
293 qWarning() << "rx empty" << rcnt;
294 return {};
295 }
296
297 // schedule reads for next packet
298 QTimer::singleShot(0, this, &DatalinkSerial::readDataAvailable);
299
300 return QByteArray(reinterpret_cast<char *>(_rxbuf_raw), rcnt);
301}
