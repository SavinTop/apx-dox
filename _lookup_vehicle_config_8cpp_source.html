51bool LookupVehicleConfig::fixItemDataThr(QVariantMap *item)
52{
53 QString time = QDateTime::fromMSecsSinceEpoch(item->value("time").toLongLong())
54 .toString("yyyy MMM dd hh:mm:ss");
55 QString callsign = item->value("callsign").toString();
56 QString title = item->value("title").toString();
57 QString notes = item->value("notes").toString();
58 item->insert("title", time);
59 item->insert("value", title);
60 item->insert("descr", callsign + (notes.isEmpty() ? "" : QString(" - %1").arg(notes)));
61 return true;
62}
