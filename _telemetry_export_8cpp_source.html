59bool TelemetryExport::write(quint64 telemetryID, QString fileName)
60{
61 {
62 DBReqTelemetryMakeStats req(telemetryID);
63 if (!req.execSynchronous())
64 return false;
65 }
66 DBReqTelemetryReadData req(telemetryID);
67 connect(this, &QueueWorker::stopRequested, &req, &DatabaseRequest::discard, Qt::QueuedConnection);
68 if (!req.execSynchronous())
69 return false;
70
71 QVariantMap info = req.info;
72
73 if (req.records.values.size() <= 0) {
74 apxMsgW() << tr("Nothing to export");
75 }
76
77 QString sharedHash;
78 {
79 DBReqTelemetryReadSharedHashId req(telemetryID);
80 if (!req.execSynchronous())
81 return false;
82 sharedHash = req.hash;
83 }
84
85 QFile file(fileName);
86 if (!file.open(QFile::WriteOnly | QFile::Text)) {
87 apxMsgW() << tr("Cannot write file").append(":") << fileName
88 << QString("(%1)").arg(file.errorString());
89 return false;
90 }
91 bool ok = false;
92 QString ftype = QFileInfo(fileName).suffix();
93 if (ftype == "csv") {
94 ok = writeCSV(&file, req);
95 } else {
96 ok = writeXml(&file, req, QFileInfo(fileName).completeBaseName(), sharedHash, info);
97 }
98 file.close();
99 if (ok) {
100 apxMsg() << tr("Exported telemetry").append(':') << QFileInfo(fileName).fileName();
101 } else {
102 file.remove();
103 }
104 return ok;
105}

107bool TelemetryExport::writeXml(QFile *file_p,
108 const DBReqTelemetryReadData &req,
109 const QString &title,
110 const QString &sharedHash,
111 const QVariantMap &info)
112{
113 QXmlStreamWriter stream(file_p);
114 stream.setAutoFormatting(true);
115 stream.setAutoFormattingIndent(2);
116 stream.writeStartDocument();
117 stream.writeStartElement("telemetry");
118 stream.writeAttribute("href", "http://www.uavos.com/");
119 stream.writeAttribute("format", "11");
120
121 stream.writeTextElement("title", title);
122 stream.writeTextElement("timestamp",
123 QDateTime::fromMSecsSinceEpoch(info.value("time").toLongLong())
124 .toString(Qt::RFC2822Date));
125 stream.writeTextElement("exported", QDateTime::currentDateTime().toString(Qt::RFC2822Date));
126 stream.writeTextElement("version", AppBase::version());
127 if (!sharedHash.isEmpty())
128 stream.writeTextElement("sharedHash", sharedHash);
129
130 stream.writeStartElement("user");
131 stream.writeTextElement("machineUID", AppBase::machineUID());
132 stream.writeTextElement("hostname", AppBase::hostname());
133 stream.writeTextElement("username", AppBase::username());
134 stream.writeEndElement();
135
136 //telemetry record info
137 stream.writeStartElement("info");
138 for (auto key : info.keys()) {
139 if (key == "trash")
140 continue;
141 stream.writeTextElement(key, info.value(key).toString());
142 }
143 stream.writeEndElement();
144
145 //fields list
146 QList<quint64> fidList;
147 QStringList fieldNames;
148 for (auto key : req.fieldNames.keys()) {
149 fidList.append(key);
150 fieldNames.append(req.fieldNames.value(key));
151 }
152 stream.writeTextElement("fields", fieldNames.join(','));
153
154 const int iTime = req.records.names.indexOf("time");
155 const int iType = req.records.names.indexOf("type");
156 const int iName = req.records.names.indexOf("name");
157 const int iValue = req.records.names.indexOf("value");
158 const int iUID = req.records.names.indexOf("uid");
159
160 // vehicle configs and missions
161 QSet<QString> configs, missions;
162 for (auto const &r : req.records.values) {
163 const auto type = r.at(iType).toUInt();
164 if (type < 2)
165 continue;
166 auto name = r.at(iName).toString();
167 auto hash = r.at(iUID).toString();
168 if (hash.isEmpty())
169 continue;
170 if (name == "mission") {
171 missions.insert(hash);
172 } else if (name == "nodes") {
173 configs.insert(hash);
174 }
175 }
176 stream.writeStartElement("packages");
177 for (auto const &hash : configs) {
178 DBReqLoadVehicleConfig req(hash);
179 if (!req.execSynchronous()) {
180 apxMsgW() << tr("Can't export config") << hash;
181 continue;
182 }
183 QByteArray data = QJsonDocument::fromVariant(req.config()).toJson(QJsonDocument::Compact);
184
185 stream.writeStartElement("vehicle");
186 stream.writeAttribute("hash", hash);
187 stream.writeCharacters(qCompress(data, 9).toBase64());
188 stream.writeEndElement();
189 }
190 for (auto const &hash : missions) {
191 DBReqMissionsLoad req(hash);
192 if (!req.execSynchronous()) {
193 apxMsgW() << tr("Can't export mission") << hash;
194 continue;
195 }
196 QByteArray data = QJsonDocument::fromVariant(req.mission()).toJson(QJsonDocument::Compact);
197
198 stream.writeStartElement("mission");
199 stream.writeAttribute("hash", hash);
200 stream.writeCharacters(qCompress(data, 9).toBase64());
201 stream.writeEndElement();
202 }
203 stream.writeEndElement();
204
205 //write data
206 stream.setAutoFormattingIndent(0);
207 stream.writeStartElement("data");
208
209 bool ok = true;
210 QStringList values;
211 quint64 valuesTime = 0;
212 values.reserve(fidList.size());
213 int progress_s = 0;
214 const int cnt = req.records.values.size();
215 for (int i = 0; ok && i < cnt; ++i) {
216 const auto &r = req.records.values.at(i);
217 const auto time = r.at(iTime).toULongLong();
218 const auto type = r.at(iType).toUInt();
219 //progress
220 int v_p = i * 100 / cnt;
221 if (progress_s != v_p) {
222 progress_s = v_p;
223 emit progress(fact, v_p);
224 }
225 //record
226 if (type == 0) { //downlink
227 if (values.isEmpty())
228 valuesTime = time;
229 else if (valuesTime != time) {
230 writeDownlink(stream, valuesTime, values);
231 valuesTime = time;
232 values.clear();
233 }
234 int vi = fidList.indexOf(r.at(iName).toULongLong());
235 ok = vi >= 0;
236 if (!ok)
237 break;
238 while (vi >= values.size())
239 values.append("");
240 if (!values.at(vi).isEmpty()) {
241 qWarning() << "duplicate value" << time << fieldNames.at(vi);
242 }
243 values[vi] = r.at(iValue).toString();
244 continue;
245 }
246 if (!values.isEmpty()) {
247 writeDownlink(stream, valuesTime, values);
248 values.clear();
249 }
250 switch (type) {
251 case 1: { //uplink
252 int vi = fidList.indexOf(r.at(iName).toULongLong());
253 ok = vi >= 0;
254 if (!ok)
255 break;
256 writeUplink(stream, time, fieldNames.at(vi), r.at(iValue).toString());
257 } break;
258 case 2:
259 case 3: { //event
260 QString name = r.at(iName).toString();
261 QString uid = r.at(iUID).toString();
262 writeEvent(stream, time, name, r.at(iValue).toString(), uid, type == 3);
263 } break;
264 }
265 }
266 if (!values.isEmpty()) {
267 writeDownlink(stream, valuesTime, values);
268 }
269 stream.writeEndElement(); //data
270 stream.writeEndElement(); //telemetry
271
272 return ok;
273}

275void TelemetryExport::writeDownlink(QXmlStreamWriter &stream,
276 quint64 time,
277 const QStringList &values)
278{
279 stream.writeStartElement("D");
280 stream.writeAttribute("t", QString::number(time));
281 // combine zero values
282 uint n = 0;
283 QStringList st;
284 for (auto const &v : values) {
285 if (v.isEmpty()) {
286 n++;
287 continue;
288 }
289 if (n > 2) {
290 st.append("#" + QString::number(n));
291 n = 0;
292 } else if (n > 0) {
293 while (n--) {
294 st.append(QString());
295 }
296 n = 0;
297 }
298 st.append(v);
299 }
300 stream.writeCharacters(st.join(','));
301 stream.writeEndElement();
302}

332bool TelemetryExport::writeCSV(QFile *file_p, DBReqTelemetryReadData &req)
333{
334 QTextStream stream(file_p);
335
336 //fields list
337 QList<quint64> fidList;
338 QStringList fieldNames;
339 fieldNames << "time";
340 foreach (quint64 fid, req.fieldNames.keys()) {
341 fidList.append(fid);
342 fieldNames.append(req.fieldNames.value(fid));
343 }
344 stream << fieldNames.join(',') << QString("\n");
345
346 //write data
347 QStringList values;
348 for (int i = 0; i < fieldNames.size(); ++i) {
349 values.append("0");
350 }
351
352 int iTime = req.records.names.indexOf("time");
353 int iType = req.records.names.indexOf("type");
354 int iName = req.records.names.indexOf("name");
355 int iValue = req.records.names.indexOf("value");
356
357 quint64 t0 = 0;
358 quint64 t = 0;
359 quint64 t_s = 0;
360 QHash<quint64, double> fvalues;
361 for (int i = 0; i < req.records.values.size(); ++i) {
362 const QVariantList &r = req.records.values.at(i);
363 if (r.isEmpty())
364 continue;
365
366 //time
367 t = r.at(iTime).toULongLong();
368 if (i == 0)
369 t0 = t;
370 t -= t0;
371
372 if (t_s != t) {
373 stream << values.join(',') << QString("\n");
374 t_s = t;
375 }
376
377 quint64 fid = 0;
378
379 switch (r.at(iType).toUInt()) {
380 case 0: {
381 //downlink data
382 fid = r.at(iName).toULongLong();
383 } break;
384 case 1: {
385 //uplink data
386 fid = r.at(iName).toULongLong();
387 } break;
388 }
389
390 if (!fid)
391 continue;
392 int vidx = fidList.indexOf(fid) + 1;
393 if (vidx < 0)
394 continue;
395 values[0] = QString::number(t);
396 values[vidx] = r.at(iValue).toString();
397 }
398 //final data tail at max time
399 if (t_s != t) {
400 stream << values.join(',') << QString("\n");
401 }
402
403 //finish
404 stream.flush();
405 return true;
406}
