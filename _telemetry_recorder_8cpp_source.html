133bool TelemetryRecorder::dbCheckRecord()
134{
135 checkAutoRecord();
136 if (recTelemetryID)
137 return true;
138
139 //register telemetry file record
140 if (!reqNewRecord) {
141 QString title = _vehicle->confTitle();
142 if (!title.isEmpty())
143 confTitle = title;
144 DBReqTelemetryNewRecord *req
145 = new DBReqTelemetryNewRecord(_vehicle->uid(),
146 _vehicle->title(),
147 confTitle,
148 recording(),
149 QDateTime::currentDateTime().toMSecsSinceEpoch());
150 connect(req,
151 &DBReqTelemetryNewRecord::idUpdated,
152 this,
153 &TelemetryRecorder::dbRecordCreated,
154 Qt::QueuedConnection);
155 reqNewRecord = req;
156 reqPendingList.clear();
157
158 // write initial mandala state
159 PBase::Values values;
160 for (auto f : _vehicle->f_mandala->valueFacts()) {
161 if (!f->everReceived())
162 continue;
163 values.insert(f->uid(), f->value());
164 // qDebug() << f->mpath() << f->value();
165 }
166 _values = values;
167 reqPendingList.append(new DBReqTelemetryWriteData(0, 0, _values, false));
168
169 apxConsole() << tr("Telemetry record request");
170 reqNewRecord->exec();
171 }
172
173 if (reqPendingList.size() > 50000)
174 reqPendingList.takeFirst();
175
176 return false;
177}
178void TelemetryRecorder::dbRecordCreated(quint64 telemetryID)
179{
180 if (!reqNewRecord)
181 return;
182 if (!telemetryID)
183 return;
184
185 //qDebug() << telemetryID << reqPendingList.size();
186 auto chash = configHash;
187 recTelemetryID = telemetryID;
188 reqNewRecord = nullptr;
189 configHash.clear(); //force write
190 missionHash.clear(); //force write
191
192 recordMission(false);
193 recordConfig(chash, "init");
194
195 for (auto req : reqPendingList) {
196 req->telemetryID = recTelemetryID;
197 req->exec();
198 }
199 reqPendingList.clear();
200
201 //record shared info
202 QVariantMap info;
203 info["machineUID"] = App::machineUID();
204 info["hostname"] = App::hostname();
205 info["username"] = App::username();
206 DBReqTelemetryWriteSharedInfo *req = new DBReqTelemetryWriteSharedInfo(telemetryID, info);
207 req->exec();
208}

210{
211 for (auto uid : values->keys()) {
212 auto v = values->value(uid);
213 if (_values.value(uid) == v) {
214 values->remove(uid);
215 } else {
216 _values.insert(uid, v);
217 }
218 }

241void TelemetryRecorder::recordTelemetry(PBase::Values values, quint64 timestamp_ms)
242{
243 // always start DB recorder time counter from zero
244 auto t = timestamp_ms;
245
246 if (t < _ts_t0)
247 reset();
248
249 dbCheckRecord();
250
251 if (_reset_timestamp) {
252 _reset_timestamp = false;
253 _ts_t0 = t;
254 if (_tsElapsed.isValid()) {
255 _ts_t1 = _tsElapsed.elapsed();
256 _ts_t1 = (_ts_t1 / 100 + 1) * 100;
257 }
258 }
259 t -= _ts_t0;
260 t += _ts_t1;
261 _ts_t2 = t;
262 _tsElapsed.start();
263
264 // qDebug() << _vehicle->title() << "ts" << t << _ts_t0;
265
266 setTime(t / 1000);
267
268 // record changed values only
269 cleanupValues(&values);
270 if (values.isEmpty())
271 return;
272
273 dbWriteRequest(new DBReqTelemetryWriteData(recTelemetryID, t, values, false));
274}
275void TelemetryRecorder::recordData(PBase::Values values, bool uplink)
276{
277 if (!uplink)
278 cleanupValues(&values);
279 if (values.isEmpty())
280 return;
281
282 dbCheckRecord();
283
284 dbWriteRequest(new DBReqTelemetryWriteData(recTelemetryID, getEventTimestamp(), values, uplink));
285}

355void TelemetryRecorder::recordConfig(QString hash, QString title)
356{
357 if (!dbCheckRecord())
358 return;
359
360 if (hash.isEmpty())
361 return;
362 if (configHash == hash)
363 return;
364 configHash = hash;
365 //qDebug()<<hash<<info;
366 writeEvent("nodes", title, hash, false);
367
368 //check for config title change
369 if (confTitle != title) {
370 confTitle = title;
371 if (recTelemetryID) {
372 QVariantMap info;
373 info.insert("comment", title);
374 DBReqTelemetryWriteInfo *req = new DBReqTelemetryWriteInfo(recTelemetryID, info);
375 req->exec();
376 }

380bool TelemetryRecorder::checkAutoRecord(void)
381{
382 if (_vehicle->streamType() != PVehicle::TELEMETRY)
383 return recording();
384
385 Vehicle::FlightState fs = _vehicle->flightState();
386 if (flightState_s != fs) {
387 flightState_s = fs; //only once
388
389 // stop recording when landed
390 if (fs == Vehicle::FS_LANDED && recording() && !recStopTimer.isActive())
391 recStopTimer.start(2000);
392
393 // start or restart when starts flying
394 if (fs == Vehicle::FS_TAKEOFF) {
395 reset(); //restart
396 setRecording(true);
397 }
398 }
399 return recording();
400}
