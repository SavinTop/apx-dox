347void App::setGlobalProperty(const QString &path, const QJSValue &value)
348{
349 AppEngine *e = _instance->engine();
350 if (!e)
351 return;
352
353 QJSValue v = e->globalObject();
354
355 QStringList list = path.split('.');
356 for (int i = 0; i < list.size(); ++i) {
357 const QString pname = list.at(i);
358 if (i == (list.size() - 1)) {
359 v.setProperty(pname, value);
360 break;
361 }
362 QJSValue vp = v.property(pname);
363 if (vp.isUndefined() || (!vp.isObject())) {
364 vp = e->newObject();
365 v.setProperty(pname, vp);
366 }
367 v = vp;
368 }
369 //e->collectGarbage();
370 if (value.isQObject() || value.isObject()) {
371 e->jsProtectPropertyWrite(path);
372 }
373}

443QFont App::getMonospaceFont()
444{
445 QFont font(QGuiApplication::font());
446 font.setStyleHint(QFont::Monospace);
447 font.setFixedPitch(true);
448 font.setKerning(false);
449
450 do {
451#ifdef Q_OS_MAC
452 font.setFamily("Menlo");
453#else
454 font.setFamily("FreeMono");
455#endif
456 if (isFixedPitch(font))
457 break;
458 font.setStyleHint(QFont::Monospace);
459 if (isFixedPitch(font))
460 break;
461 font.setStyleHint(QFont::TypeWriter);
462 if (isFixedPitch(font))
463 break;
464 font.setFamily("monospace");
465 if (isFixedPitch(font))
466 break;
467 font.setFamily("courier");
468 if (isFixedPitch(font))
469 break;
470 font.setFamily("unexistent");
471 if (isFixedPitch(font))
472 break;
473 return QGuiApplication::font();
474 } while (0);
475 return font;
476}
