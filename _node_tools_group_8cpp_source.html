41Fact *NodeToolsGroup::addCommand(Fact *group,
42 QString name,
43 QString title,
44 xbus::node::usr::cmd_t cmd)
45{
46 name = name.toLower();
47 if (group != node) {
48 // create groups
49 FactList plist;
50 while (group != node) {
51 plist.insert(0, group);
52 group = group->parentFact();
53 }
54 group = this;
55 for (auto i : plist) {
56 NodeToolsGroup *g = qobject_cast<NodeToolsGroup *>(group->child(i->name()));
57 if (g)
58 group = g;
59 else
60 group = new NodeToolsGroup(group, node, i->name(), i->title(), i->descr());
61 }
62 } else
63 group = this;
64
65 Fact *f = new Fact(group, name, title);
66 if (name.contains("reboot") || name.contains("restart") || name.contains("reset"))
67 f->setIcon("reload");
68 else if (name.contains("mute"))
69 f->setIcon("volume-mute");
70 else if (name.contains("start"))
71 f->setIcon("play");
72 else if (name.contains("stop"))
73 f->setIcon("stop");
74 else if (name.contains("erase") || name.contains("clear"))
75 f->setIcon("close-circle");
76 else if (name.contains("conf"))
77 f->setIcon("alert-octagram");
78 else if (group->name() == "scr")
79 f->setIcon("code-braces");
80 else
81 f->setIcon("asterisk");
82 f->setProperty("cmd", cmd);
83 registerOnlineAction(f);
84 //App::jsync(this);
85 return f;
86}
