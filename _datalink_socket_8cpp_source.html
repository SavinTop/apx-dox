66bool DatalinkSocket::isLocalHost(const QHostAddress address)
67{
68 if (address.isLoopback())
69 return true;
70 for (const auto &i : QNetworkInterface::allAddresses())
71 if (address.isEqual(i))
72 return true;
73 return false;
74}
