34TelemetryReader::TelemetryReader(LookupTelemetry *lookup, Fact *parent)
35 : Fact(parent, "reader", "", "", Group, "progress-download")
36 , lookup(lookup)
37 , blockNotesChange(false)
38 , m_totalSize(0)
39 , m_totalTime(0)
40{
41 f_notes = new Fact(parent, "notes", tr("Notes"), tr("Current record notes"), Text);
42 f_notes->setIcon("note-text");
43 connect(f_notes, &Fact::valueChanged, this, &TelemetryReader::notesChanged);
44
45 //actions
46 f_reload = new Fact(this,
47 "reload",
48 tr("Reload"),
49 tr("Invalidate cache"),
50 Action | ShowDisabled,
51 "reload");
52 f_reload->setEnabled(false);
53 connect(f_reload, &Fact::triggered, this, &TelemetryReader::reloadTriggered);
54
55 //info
56 connect(lookup, &LookupTelemetry::recordInfoChanged, this, &TelemetryReader::updateRecordInfo);
57 connect(this, &TelemetryReader::totalTimeChanged, this, &TelemetryReader::updateStatus);
58
59 //load sequence
60 qRegisterMetaType<fieldData_t>("fieldData_t");
61 qRegisterMetaType<fieldNames_t>("fieldNames_t");
62 qRegisterMetaType<times_t>("times_t");
63 qRegisterMetaType<events_t>("events_t");
64 qRegisterMetaType<FactList>("FactList");
65
66 connect(lookup, &LookupTelemetry::recordTriggered, this, &TelemetryReader::load);
67
68 connect(&loadEvent, &DelayedEvent::triggered, this, &TelemetryReader::dbLoadData);
69 loadEvent.setInterval(500);
70
71 connect(this, &Fact::triggered, this, &TelemetryReader::loadCurrent);
72
73 updateRecordInfo();
74}

88void TelemetryReader::updateRecordInfo()
89{
90 QVariantMap info = lookup->recordInfo();
91 setTotalTime(info.value("totalTime").toULongLong());
92 quint64 downlink = info.value("downlink").toULongLong();
93 quint64 uplink = info.value("uplink").toULongLong();
94 quint64 events = info.value("events").toULongLong();
95 setTotalSize(downlink + uplink + events);
96 evtCountMap.clear();
97 foreach (QString s, info.value("evtDetails").toString().split(',')) {
98 if (s.isEmpty())
99 continue;
100 QString key = s.left(s.indexOf('='));
101 QString v = s.mid(s.indexOf('=') + 1);
102 evtCountMap.insert(key, v.toUInt());
103 }
104 if (uplink)
105 evtCountMap.insert("uplink", uplink);
106
107 qint64 t = info.value("time").toLongLong();
108 QString title = t > 0 ? QDateTime::fromMSecsSinceEpoch(t).toString("yyyy MMM dd hh:mm:ss")
109 : tr("Telemetry Data");
110 QString callsign = info.value("callsign").toString();
111 QString comment = info.value("comment").toString();
112 QString notes = info.value("notes").toString();
113 QString stime = AppRoot::timeToString(totalTime() / 1000, true);
114
115 QStringList descr;
116 if (!callsign.isEmpty())
117 descr.append(callsign);
118 if (!comment.isEmpty() && comment != callsign)
119 descr.append(comment);
120
121 setTitle(title);
122 setDescr(descr.join(" | "));
123 blockNotesChange = true;
124 f_notes->setValue(notes);
125 blockNotesChange = false;
126
127 emit statsAvailable();
128}

130void TelemetryReader::load()
131{
132 f_reload->setEnabled(false);
133 quint64 key = lookup->recordId();
134 if (!key)
135 return;
136 setTotalSize(0);
137 setTotalTime(0);
138 setProgress(0);
139 deleteChildren();
140 DBReqTelemetryFindCache *req = new DBReqTelemetryFindCache(key);
141 connect(req,
142 &DBReqTelemetryFindCache::cacheFound,
143 this,
144 &TelemetryReader::dbCacheFound,
145 Qt::QueuedConnection);
146 connect(req,
147 &DBReqTelemetryFindCache::cacheNotFound,
148 this,
149 &TelemetryReader::dbCacheNotFound,
150 Qt::QueuedConnection);
151 req->exec();
152}

241void TelemetryReader::dbResultsDataProc(quint64 telemetryID,
242 quint64 cacheID,
243 fieldData_t fieldData,
244 fieldNames_t fieldNames,
245 times_t times,
246 events_t events,
247 QGeoPath path,
248 Fact *f_events)
249{
250 if (telemetryID != lookup->recordId())
251 return;
252
253 deleteChildren();
254 f_reload->setEnabled(true);
255
256 times.swap(this->times);
257 fieldNames.swap(this->fieldNames);
258 fieldData.swap(this->fieldData);
259 events.swap(this->events);
260 this->geoPath = path;
261
262 changeThread(f_events, thread());
263 f_events->setParentFact(this);
264 for (int i = 0; i < f_events->size(); ++i) {
265 Fact *g = f_events->child(i);
266 for (int j = 0; j < g->size(); ++j) {
267 Fact *f = g->child(j);
268 connect(f, &Fact::triggered, this, [this, f]() { emit recordFactTriggered(f); });
269 }
270 }
271
272 quint64 tMax = 0;
273 if (!this->times.isEmpty())
274 tMax = qRound(this->times.last() * 1000.0);
275 setTotalTime(tMax);
276
277 setProgress(-1);
278 emit dataAvailable(cacheID);
279}
