139bool LookupTelemetry::fixItemDataThr(QVariantMap *item)
140{
141 QString time = QDateTime::fromMSecsSinceEpoch(item->value("time").toLongLong())
142 .toString("yyyy MMM dd hh:mm:ss");
143 QString callsign = item->value("callsign").toString();
144 QString comment = item->value("comment").toString();
145 QString notes = item->value("notes").toString();
146 QString total;
147 quint64 t = item->value("totalTime").toULongLong();
148 if (t > 0)
149 total = AppRoot::timeToString(t / 1000);
150 QStringList descr;
151 if (!comment.isEmpty())
152 descr << comment;
153 if (!notes.isEmpty())
154 descr << notes;
155 QStringList value;
156 if (!callsign.isEmpty())
157 value << callsign;
158 if (!total.isEmpty())
159 value << total;
160
161 item->insert("title", time);
162 item->insert("descr", descr.join(" - "));
163 item->insert("value", value.join(' '));
164 //active current
165 item->insert("active", item->value("key").toULongLong() == recordId());
166 return true;
167}

291void LookupTelemetry::dbResultsNumNext(DatabaseRequest::Records records)
292{
293 if (records.values.isEmpty())
294 return;
295 quint64 key = records.values.first().at(0).toULongLong();
296 if (key != recordId())
297 return;
298 quint64 num = recordNum();
299 for (int i = 0; i < records.values.size(); ++i) {
300 if (records.values.at(i).at(1).toULongLong() == key)
301 break;
302 num--;
303 }

407void LookupTelemetry::dbResultsPrevNext(DatabaseRequest::Records records)
408{
409 if (records.values.isEmpty())
410 return;
411 const QStringList &n = records.names;
412 quint64 cur = recordId();
413 const int cnt = records.values.size() - 1;
414 for (int i = 0; i <= cnt; ++i) {
415 quint64 key = records.values.at(i).at(0).toULongLong();
416 //qDebug()<<i<<key<<cur<<cnt;
417 if (key != cur)
418 continue;
419 if (i == cnt)
420 break;
421 const QVariantList &r = records.values.at(i + 1);
422 setRecordTimestamp(r.at(n.indexOf("time")).toULongLong());
423 jumpToRecord(r.at(0).toULongLong());
424 return;
425 }
