134void KmlGeoPolygon::recalcCoordinates()
135{
136 clearPolygon();
137 auto visibleRegion = m_map->property("visibleRegion").value<QGeoShape>().boundingGeoRectangle();
138 QRect vr(fromCoordinate(visibleRegion.topLeft()), fromCoordinate(visibleRegion.bottomRight()));
139 for (int i = 0; i < m_geoPolygon.size(); i++) {
140 QPoint point = fromCoordinate(m_geoPolygon.coordinateAt(i));
141 if (std::isnan(point.x()) || std::isnan(point.y()))
142 continue;
143
144 m_polygon.append(point);
145 }
146 for (int j = 0; j < m_geoPolygon.holesCount(); j++) {
147 QPolygon polyHole;
148 for (int k = 0; k < m_geoPolygon.hole(j).size(); k++) {
149 QPoint point = fromCoordinate(m_geoPolygon.hole(j)[k].value<QGeoCoordinate>());
150 if (std::isnan(point.x()) || std::isnan(point.y()))
151 continue;
152 polyHole.append(point);
153 }
154 m_holes.append(polyHole);
155 }
156
157 setX(vr.x());
158 setY(vr.y());
159 setWidth(vr.width());
160 setHeight(vr.height());
161}
