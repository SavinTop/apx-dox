96QVariant Fact::data(int col, int role)
97{
98 switch (role) {
99 case ModelDataRole:
100 case FactRole:
101 return QVariant::fromValue(const_cast<Fact *>(this));
102 case NameRole:
103 return name();
104 case ValueRole:
105 return value();
106 case TextRole:
107 return text();
108 case Qt::ForegroundRole:
109 if (col == Fact::FACT_MODEL_COLUMN_NAME) {
110 if (modified())
111 return QColor(Qt::red).lighter();
112 if (!enabled())
113 return QColor(Qt::gray);
114 if (active())
115 return QColor(Qt::green).lighter();
116 if (isZero())
117 return QColor(Qt::gray);
118 return QColor(Qt::white);
119 }
120 if (col == Fact::FACT_MODEL_COLUMN_VALUE) {
121 if (!enabled())
122 return QColor(Qt::darkGray);
123 if (modified())
124 return QColor(Qt::yellow);
125 if (size() && !text().startsWith('[')) //expandable
126 return QColor(Qt::darkGray);
127 if (isZero())
128 return QColor(Qt::gray);
129 //if (isDefault())
130 // return QColor(Qt::blue).lighter(180);
131 return QColor(Qt::cyan).lighter(180);
132 }
133 return QColor(Qt::darkCyan);
134 case Qt::BackgroundRole:
135 return QVariant();
136 case Qt::FontRole: {
137 // QFont font(QFontDatabase::systemFont(QFontDatabase::GeneralFont));
138 QFont font(QGuiApplication::font());
139 if (col == Fact::FACT_MODEL_COLUMN_DESCR)
140 return font;
141 if (treeType() && col == Fact::FACT_MODEL_COLUMN_NAME)
142 font.setBold(true);
143 return font;
144 }
145 case Qt::ToolTipRole:
146 if (col == Fact::FACT_MODEL_COLUMN_NAME) {
147 return toolTip();
148 } else if (col == Fact::FACT_MODEL_COLUMN_VALUE) {
149 if (size()) {
150 QString s = name();
151 for (int i = 0; i < size(); ++i) {
152 Fact *f = child(i);
153 s += QString("\n%1: %2").arg(f->title()).arg(f->text());
154 }
155 return s;
156 } else {
157 return toolTip();
158 }
159 }
160 return data(col, Qt::DisplayRole);
161 }
162
163 //value roles
164 if (role != Qt::DisplayRole && role != Qt::EditRole)
165 return QVariant();
166
167 switch (col) {
168 case Fact::FACT_MODEL_COLUMN_NAME:
169 return title();
170 case Fact::FACT_MODEL_COLUMN_VALUE: {
171 const QString s = text();
172 if (s.isEmpty()) {
173 if (treeType() == Group) {
174 return QVariant();
175 }
176 }
177 if (role == Qt::EditRole)
178 return editorText();
179 /*if (role == Qt::EditRole && enumStrings().size() <= 1) {
180 if (dataType() == Bool)
181 return value().toBool();
182 if (dataType() == Int)
183 return value().toInt();
184 if (dataType() == Float) {
185 bool ok = false;
186 s.toDouble(&ok);
187 if (ok)
188 return s;
189 return value().toDouble();
190 }
191 }*/
192 return s;
193 }
194 case Fact::FACT_MODEL_COLUMN_DESCR:
195 return descr();
196 }
197 return QVariant();
198}

200QString Fact::toolTip() const
201{
202 QStringList st;
203 QString sDataType;
204 if (dataType())
205 sDataType = QMetaEnum::fromType<FactBase::Flag>().valueToKey(static_cast<int>(dataType()));
206 if (!units().isEmpty())
207 sDataType += (sDataType.isEmpty() ? "" : ", ") + units();
208 QString s = name();
209 if (s != title())
210 s = QString("%1 (%2)").arg(s).arg(title());
211
212 if (!sDataType.isEmpty())
213 s = QString("%1 [%2]").arg(s).arg(sDataType);
214
215 st << s;
216
217 if (!descr().isEmpty())
218 st << QString("Descr: %1").arg(descr());
219 st << QString("Path: %1").arg(path());
220 if (!m_enumStrings.isEmpty()) {
221 if (m_enumStrings.size() > 25)
222 st << QString("Values: [%1]").arg(m_enumStrings.size());
223 else
224 st << QString("Values: {%1}").arg(m_enumStrings.join(','));
225 }
226 if (!defaultValue().isNull())
227 st << QString("Default: %1").arg(defaultValue().toString());
228 if (!min().isNull())
229 st << QString("Min: %1").arg(min().toString());
230 if (!max().isNull())
231 st << QString("Max: %1").arg(max().toString());
232 if (increment() > 0)
233 st << QString("Increment: %1").arg(increment());
234 if (precision() >= 0)
235 st << QString("Precision: %1").arg(precision());
236
237 if (!section().isEmpty())
238 st << QString("Section: %1").arg(section());
239
240 if (!opts().isEmpty()) {
241 st << "";
242 st << "[opts]";
243 for (auto k : opts().keys())
244 st << QString("%1: %2").arg(k).arg(opts().value(k).toString());
245 }
246
247 return st.join('\n');
248}

301Fact *Fact::findChild(const QString &factNamePath, bool exactMatch) const
302{
303 FactList slist;
304 bool del = factNamePath.contains('.');
305 for (auto i : facts()) {
306 if (i->name() == factNamePath)
307 return i;
308 if (del && i->path().endsWith(factNamePath))
309 return i;
310 if ((!exactMatch) && i->name().startsWith(factNamePath))
311 slist.append(i);
312 }
313 if (del) {
314 for (auto i : facts()) {
315 i = i->findChild(factNamePath);
316 if (i)
317 return i;
318 }
319 }
320 //qDebug()<<slist.size();
321 if (slist.size() == 1)
322 return slist.first();
323 if (exactMatch && (!del))
324 apxConsoleW() << "Fact not found:" << factNamePath; //<<sender();
325 return nullptr;
326}

339{
340 QString s;
341 for (const FactBase *i = this; i; i = i->parentFact()) {
342 const Fact *f = static_cast<const Fact *>(i);
343 if (i->treeType() == Root) {
344 s.prepend(pathDelimiter);
345 break;
346 }
347 if (i->options() & Section)
348 continue;
349 QString s2 = f->pTitle();

354 }

362Fact *Fact::factByTitlePath(const QString &spath) const
363{
364 int del = spath.indexOf('/');
365 if (del < 0) {
366 //find child fact by title
367 for (int i = 0; i < size(); ++i) {
368 Fact *f = child(i);
369 if (f->pTitle() == spath)
370 return f;
371 }
372 return nullptr;
373 }
374 //ignore leading delimiter
375 if (del == 0)
376 return factByTitlePath(spath.mid(1));
377 //find child fact section by title
378 QString s = spath.left(del);
379 for (int i = 0; i < size(); ++i) {
380 Fact *f = child(i);
381 if (f->pTitle() == s)
382 return f->factByTitlePath(spath.mid(del + 1));
383 }
384 return nullptr;
385}

526QVariant Fact::toVariant()
527{
528 if (treeType() == Action || !visible())
529 return {};
530 if (size() > 0) {
531 if (treeType() != NoFlags && dataType() == Count) {
532 QVariantList a;
533 for (auto i : facts()) {
534 QVariant v = i->toVariant();
535 if (!v.isNull())
536 a.append(v);
537 }
538 if (a.isEmpty())
539 return {};
540 return a;
541 }
542
543 QVariantMap h;
544 for (auto i : facts()) {
545 QVariant v = i->toVariant();
546 if (!v.isNull())
547 h.insert(i->name(), v);
548 }
549 if (h.isEmpty())
550 return {};
551 return h;
552 }
553
554 QVariant v;
555 do {
556 if (treeType() != NoFlags)
557 break;
558 if (dataType() == NoFlags || dataType() == Count || valueText().isEmpty())
559 break;
560
561 if (dataType() == Text || dataType() == Bool)
562 v = valueText();
563 else if (enumStrings().isEmpty())
564 v = value();
565 else
566 v = valueText();
567 } while (0);
568
569 return v;
570}
571void Fact::fromVariant(const QVariant &var)
572{
573 if (var.isNull())
574 return;
575 if (var.canConvert(QMetaType::QVariantMap)) {
576 auto m = var.value<QVariantMap>();
577 for (auto key : m.keys()) {
578 Fact *f = child(key);
579 if (!f) {
580 qWarning() << "missing json fact" << key << path();
581 continue;
582 }
583 auto v = m.value(key);
584 if (v.canConvert(QMetaType::QVariantMap)) {
585 f->fromVariant(v);
586 continue;
587 }
588 if (v.canConvert(QMetaType::QVariantList)) {
589 f->fromVariant(v);
590 continue;
591 }
592 f->setValue(v);
593 }
594 return;
595 }
596 if (var.canConvert(QMetaType::QVariantList)) {
597 // must be implemented in subclasses to create children structure from array
598 return;
599 }
600 if (size() > 0)
601 return;
602 if (treeType() != NoFlags)
603 return;
604 if (dataType() == NoFlags || dataType() == Count)
605 return;
606 setValue(var);
607}

823void Fact::trackProgress()
824{
825 int ncnt = 0, v = 0;
826 if (options() & ProgressTrack) {
827 for (auto const f : facts()) {
828 int np = f->progress();
829 if (np < 0)
830 continue;
831 ncnt++;
832 v += np;
833 }
834 }
835 if (ncnt > 0) {
836 if (m_progress_s < ncnt) {
837 m_progress_s = ncnt;
838 //qDebug()<<"progressCnt"<<p->progress_s<<ncnt<<p->name()<<name();
839 } else if (ncnt < m_progress_s) {
840 v += (m_progress_s - ncnt) * 100;
841 ncnt = m_progress_s;
842 }
843 setProgress(v / ncnt);
844 } else {
