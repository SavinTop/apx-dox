131void Joystick::updateDevice(bool connected)
132{
133 if (connected) {
134 if (!dev) {
135 dev = SDL_JoystickOpen(device_index);
136 if (!dev) {
137 apxMsgW() << tr("Can't open joystick") << title() << SDL_GetError();
138 return;
139 }
140 instanceID = SDL_JoystickInstanceID(dev);
141 }
142 if (value().toBool())
143 setActive(true);
144 } else {
145 if (device_index >= 0) {
146 SDL_JoystickClose(SDL_JoystickOpen(device_index));
147 }
148 dev = nullptr;
149 device_index = -1;
150 setActive(false);
151 }
152}

172void Joystick::updateHat(int i, quint8 v)
173{
174 quint8 m = 1;
175 for (int j = 0; j < 4; ++j) {
176 bool b = v & m;
177 m <<= 1;
178 Fact *f = f_hats->child(i * 4 + j);
179 if (!f)
180 continue;
181 f->setActive(b);
182 if (!b)
183 continue;
184 QString s = f->text().simplified();
185 if (s.isEmpty())
186 return;
187 App::jsexec(s);
188 }

191void Joystick::loadConfig(const QJsonObject &config)
192{
193 //clear
194 for (int i = 0; i < f_axes->size(); ++i)
195 f_axes->child(i)->setValue(QVariant());
196 for (int i = 0; i < f_buttons->size(); ++i)
197 f_buttons->child(i)->setValue(QVariant());
198 for (int i = 0; i < f_hats->size(); ++i)
199 f_hats->child(i)->setValue(QVariant());
200
201 //load
202 f_title->setValue(config["title"]);
203 const QJsonObject &conf = config["config"].toObject();
204 foreach (QJsonValue v, conf["axes"].toArray()) {
205 int i = v["id"].toInt() - 1;
206 JoystickAxis *f = static_cast<JoystickAxis *>(f_axes->child(i));
207 if (f)
208 f->loadConfig(v.toObject());
209 }
210 foreach (QJsonValue v, conf["buttons"].toArray()) {
211 int i = v["id"].toInt() - 1;
212 Fact *f = f_buttons->child(i);
213 if (!f)
214 continue;
215 f->setValue(v["scr"]);
216 }
217 foreach (QJsonValue v, conf["hats"].toArray()) {
218 int i = v["id"].toInt() - 1;
219 Fact *f = f_hats->child(i);
220 if (!f)
221 continue;
222 f->setValue(v["scr"]);
223 }
224}

226QJsonObject Joystick::saveConfig()
227{
228 QJsonObject config;
229 config.insert("title", f_title->text());
230 config.insert("name", devName);
231 config.insert("index", device_index);
232 config.insert("uid", uid);
233 QJsonObject conf;
234 FactList f_list{f_axes, f_buttons, f_hats};
235 for (auto f : f_list) {
236 QJsonArray a;
237 for (int i = 0; i < f->size(); ++i) {
238 QString s = f->child(i)->text().simplified();
239 if (s.isEmpty())
240 continue;
241 QJsonObject v;
242 v.insert("id", i + 1);
243 v.insert("scr", s);
244 if (f == f_axes) {
245 v.insert("hyst", f->child(i)->child(0)->value().toDouble());
246 }
247 a.append(v);
248 }
249 if (!a.empty())
250 conf.insert(f->name(), a);
251 }
252 config.insert("config", conf);
253 return config;
254}
