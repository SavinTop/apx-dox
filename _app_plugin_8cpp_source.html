57void AppPlugin::loadLib()
58{
59 QString fname = fileName;
60 if (fname.endsWith(".bundle")) {
61 fname += "/Contents/MacOS/" + QFileInfo(fname).baseName();
62 }
63 //load lib
64 apxConsole() << tr("Loading").append(":") << name;
65 QCoreApplication::processEvents();
66
67 QSettings sx_blacklist;
68 sx_blacklist.beginGroup("plugins_blacklist");
69
70 if (!checkLib(fname))
71 return;
72
73 QObject *instance = nullptr;
74 PluginInterface *p = nullptr;
75 QLibrary lib(fname);
76 try {
77 if (!lib.load()) {
78 apxMsgW() << "lib-load:" << lib.errorString() << "(" + fname + ")";
79 return;
80 }
81 loader = new QPluginLoader(fname);
82 instance = loader->instance();
83 } catch (...) {
84 apxMsgW() << "Plugin load error" << name << "(" + fname + ")";
85 instance = nullptr;
86 }
87 if (!instance) {
88 //try to load pure qml qrc plugin
89 if (!lib.isLoaded()) {
90 return;
91 }
92 QString qrcFileName = QString(":/%1/%1Plugin.qml").arg(name);
93 if (QFile::exists(qrcFileName)) {
94 fileName = "qrc" + qrcFileName;
95 loadQml();
96 return;
97 }
98 if (loader) {
99 apxMsgW() << "loader:" << loader->errorString() << "(" + fname + ")";
100 }
101 return;
102 }
103 if (sx_blacklist.value(name).toString() == _hash) {
104 apxMsgW() << tr("Plugin blacklisted").append(':') << name << "(" + fname + ")";
105 return;
106 }
107
108 sx_blacklist.setValue(name, _hash);
109 sx_blacklist.sync();
110
111 p = qobject_cast<PluginInterface *>(instance);
112 interface = p;
113
114 if (!interface)
115 return;
116
117 if (!f_enabled->value().toBool())
118 return;
119
120 //check depends
121 depends = p->depends();
122 if (!depends.isEmpty()) {
123 bool ok = false;
124 //apxConsole() << tr("Deps").append(":") << depends;
125 for (auto d : *plugins) {
126 if (!depends.contains(d->name))
127 continue;
128 ok = true;
129 apxConsole() << tr("Depends").append(":") << d->name;
130 d->load();
131 }
132 if (!ok) {
133 apxMsgW() << tr("Dependency not found").append(":") << depends;
134 }
135 apxConsole() << tr("Initializing").append(":") << name;
136 }
137
138 //define section
139 switch (p->flags() & PluginInterface::PluginSectionMask) {
140 default:
141 section = tr("Other");
142 break;
143 case PluginInterface::System:
144 section = tr("System features");
145 break;
146 case PluginInterface::Tool:
147 section = tr("Tools");
148 break;
149 case PluginInterface::Map:
150 section = tr("Mission");
151 break;
152 }
153
154 //initialize lib
155 p->init();
156 QString title = p->title();
157 QString descr = p->descr();
158 switch (p->flags() & PluginInterface::PluginTypeMask) {
159 default:
160 break;
161 case PluginInterface::Feature: {
162 control = p->createControl();
163 Fact *f = qobject_cast<Fact *>(control);
164 if (f) {
165 if (title.isEmpty())
166 title = f->title();
167 if (descr.isEmpty())
168 descr = f->descr();
169 //connect(App::instance(), &App::aboutToQuit, f, &Fact::remove);
170 } else {
171 //connect(App::instance(), &App::aboutToQuit, control, &QObject::deleteLater);
172 }
173 f_enabled->setSection(section);
174 if (!descr.isEmpty())
175 descr.prepend(tr("Tool").append(": "));
176
177 plugins->loadedTool(this);
178 } break;
179 case PluginInterface::Widget: {
180 f_enabled->setSection(tr("Windows"));
181 descr.prepend(tr("Window").append(": "));
182 plugins->loadedWindow(this);
183 } break;
184 }
185 f_enabled->setTitle(title);
186 f_enabled->setDescr(descr);
187
188 sx_blacklist.remove(name);
189 sx_blacklist.sync();
190}

254bool AppPlugin::checkLib(const QString &fname)
255{
256 QFileInfo tool(plugins->check_tool);
257 if (!tool.exists()) {
258 qWarning() << "no tool:" << tool.absoluteFilePath();
259 return true;
260 }
261
262 QSettings sx;
263 sx.beginGroup("plugins_test");
264 QCryptographicHash h(QCryptographicHash::Sha1);
265 QFileInfo fi(fname);
266 h.addData(fname.toUtf8());
267 h.addData(fi.filePath().toUtf8());
268 h.addData(fi.lastModified().toString().toUtf8());
269 h.addData(tool.filePath().toUtf8());
270 h.addData(tool.lastModified().toString().toUtf8());
271 _hash = h.result().toHex().toUpper();
272 if (sx.value(name).toString() == _hash) {
273 // qDebug() << "already checked" << name << hash;
274 return true;
275 }
276
277 sx.remove(name);
278
279 // qDebug() << "checking:" << tool.absoluteFilePath();
280
281 QCoreApplication::processEvents();
282 QCoreApplication::processEvents();
283 QProcess proc;
284 proc.start(tool.absoluteFilePath(), QStringList() << fname);
285 if (!proc.waitForStarted())
286 return false;
287 if (!proc.waitForFinished())
288 return false;
289 if (proc.exitCode() != 0) {
290 _errorString = proc.readAllStandardError();
291 apxMsgW() << "Error loading plugin:" << name;
292 apxMsgW() << _errorString;
293 f_enabled->setTitle(QString("%1 (%2)").arg(f_enabled->title()).arg(tr("error").toUpper()));
294 return false;
295 }
296 sx.setValue(name, _hash);
297 // qDebug() << "checked" << name << hash;
298 return true;
299}
