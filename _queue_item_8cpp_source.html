61bool QueueItem::loadFirmware(QString hw)
62{
63 QString stype = type().toUpper();
64
65 ApxFw *apxfw = AppGcs::apxfw();
66 QString ver = apxfw->value().toString();
67
68 QString s = QString("%1 %2 (%3)").arg(_name).arg(_hw).arg(ver);
69 s = QString("%1 (%2): %3").arg(tr("Firmware upload")).arg(stype).arg(s);
70 AppNotify::instance()->report(s, AppNotify::FromApp | AppNotify::Important);
71
72 _data.clear();
73 _offset = 0;
74
75 QString rel_type = type();
76 if (rel_type == "fw")
77 rel_type = "firmware";
78 else if (rel_type == "ldr")
79 rel_type = "loader";
80 else
81 apxMsgW() << "unknown type:" << type();
82
83 return apxfw->loadFirmware(_name, _hw, rel_type, &_data, &_offset);
84}

86void QueueItem::upload()
87{
88 if (!loadFirmware(_hw)) {
89 finish(false);
90 return;
91 }
92 auto p = protocol();
93 if (!p) {
94 finish(false);
95 return;
96 }
97 bindProperty(p, "progress", true);
98 bindProperty(p, "value", true);
99 p->upgradeFirmware(_uid, _type, _data, _offset);
100 connect(p, &PFirmware::upgradeFinished, this, &QueueItem::upgradeFinished);
101}

103void QueueItem::upgradeFinished(QString uid, bool success)

106 finish(success);
107}
