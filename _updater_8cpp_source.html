27Updater::Updater(Fact *parent)
28 : Fact(parent,
29 QString(PLUGIN_NAME).toLower(),
30 tr("Check for updates"),
31 tr("Update application"),
32 Group)
33{
34 f_auto = new Fact(this,
35 "auto",
36 tr("Auto"),
37 tr("Automatically check for updates"),
38 Bool | PersistentValue);
39 f_auto->setDefaultValue(true);
40
41 f_check = new Fact(this, "update", tr("Check for updates"), title(), Action | Apply, "update");
42
43 //add menu to app
44 Fact *m = new Fact(AppGcs::instance()->f_menu->app, f_check->name());
45 m->setOpt("role", QAction::ApplicationSpecificRole);
46 m->setBinding(f_check);
47
48 initUpdaterImpl();
49
50 if (m_impl) {
51 connect(f_check, &Fact::triggered, this, &Updater::check);
52 connect(f_auto, &Fact::valueChanged, this, &Updater::updateAuto);
53 updateAuto();
54 } else {
55 setEnabled(false);
56 }
57
58 App::jsync(this);
59}

61void Updater::initUpdaterImpl()
62{
63 m_impl = nullptr;
64 if (!App::bundle())
65 return;
66
67#ifdef Q_OS_MAC
68 m_impl = std::make_unique<SparkleAutoUpdater>();
69 m_impl->setFeedURL("https://uavos.github.io/apx-gcs/docs/releases/appcast");
70#endif
71
72#ifdef Q_OS_LINUX
73 m_impl = new AppImageAutoUpdater(this);
74#endif
75}
