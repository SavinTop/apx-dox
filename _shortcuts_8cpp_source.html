75void Shortcuts::updateStats()
76{
77 bool bSz = f_sys->size();
78 f_allonSys->setEnabled(bSz);
79 f_alloffSys->setEnabled(bSz);
80 bSz = f_usr->size();
81 f_allonUsr->setEnabled(bSz);
82 f_alloffUsr->setEnabled(bSz);
83 //disable f_sys shortcuts found in user
84 for (auto i : f_usr->facts()) {
85 Shortcut *item = static_cast<Shortcut *>(i);
86 if (item->_enabled->value().toBool() == false)
87 continue;
88 for (auto j : f_sys->facts()) {
89 Shortcut *fsys = static_cast<Shortcut *>(j);
90 if (fsys->_enabled->value().toBool() == false)
91 continue;
92 if (fsys->_key->text() == item->_key->text()) {
93 apxMsgW() << tr("Duplicate shortcut").append(":") << fsys->_key->text();
94 item->_enabled->setValue(false);
95 }
96 }
97 }

161void Shortcuts::fromVariant(const QVariant &var)
162{
163 f_sys->deleteChildren();
164 f_usr->deleteChildren();
165 QMap<QString, QVariantMap> msys, musr;
166 QStringList lsys, lusr;
167 QFile fsys(AppDirs::res().filePath("templates/shortcuts.json"));
168 if (fsys.open(QFile::ReadOnly | QFile::Text)) {
169 QJsonDocument jsys = QJsonDocument::fromJson(fsys.readAll());
170 fsys.close();
171 for (auto v : jsys["system"].toArray()) {
172 auto key = v.toObject()["key"].toString();
173 lsys.append(key);
174 auto jso = v.toObject().toVariantMap();
175 jso.insert("enb", true); //default
176 msys.insert(key, jso);
177 }
178 }
179 QVariantMap m = var.value<QVariantMap>();
180 for (auto i : m.value("system").value<QVariantList>()) {
181 auto m = i.value<QVariantMap>();
182 if (m.isEmpty())
183 continue;
184 auto key = m.value("key").toString();
185 if (!msys.contains(key))
186 continue;
187 auto sys = msys.value(key);
188 sys.insert("enb", m.value("enb"));
189 msys.insert(key, sys);
190 }
191 for (auto i : m.value("user").value<QVariantList>()) {
192 auto m = i.value<QVariantMap>();
193 if (m.isEmpty())
194 continue;
195 auto key = m.value("key").toString();
196 lusr.append(key);
197 musr.insert(key, m);
198 }
199
200 for (auto key : lsys) {
201 f_add->defaults();
202 f_add->fromVariant(msys.value(key));
203 new Shortcut(f_sys, this, f_add, false);
204 }
205
206 for (auto key : lusr) {
207 f_add->defaults();
208 f_add->fromVariant(musr.value(key));
209 addUserShortcut();
210 }
211 f_add->defaults();
212}
