104void WaypointActions::actionsValueChanged()
105{
106 if (blockActionsValueChanged)
107 return;
108 QMap<QString, QString> map;
109 foreach (const QString &s, this->value().toString().split(',')) {
110 int i = s.indexOf('=');
111 if (i <= 0)
112 continue;
113 map[s.left(i)] = s.mid(i + 1);
114 }
115 foreach (const QString &s, map.keys()) {
116 Fact *f = child(s);
117 if (!f) {
118 qDebug() << "Waypoint action data key not found" << s;
119 continue;
120 }
121 f->setValue(map.value(s));
122 }
123}
