711/*
712 * XPLMCommandRef
713 *
714 * A command ref is an opaque identifier for an X-Plane command. Command
715 * references stay the same for the life of your plugin but not between
716 * executions of X-Plane. Command refs are used to execute commands, create
717 * commands, and create callbacks for particular commands.
718 *
719 * Note that a command is not "owned" by a particular plugin. Since many
720 * plugins may participate in a command's execution, the command does not go
721 * away if the plugin that created it is unloaded.
722 *
723 */
724typedef void * XPLMCommandRef;

726/*
727 * XPLMCommandCallback_f
728 *
729 * A command callback is a function in your plugin that is called when a
730 * command is pressed. Your callback receives the commadn reference for the
731 * particular command, the phase of the command that is executing, and a
732 * reference pointer that you specify when registering the callback.
733 *
734 * Your command handler should return 1 to let processing of the command
735 * continue to other plugins and X-Plane, or 0 to halt processing, potentially
736 * bypassing X-Plane code.
737 *
738 */
739typedef int (* XPLMCommandCallback_f)(
740 XPLMCommandRef inCommand,
741 XPLMCommandPhase inPhase,
742 void * inRefcon);
