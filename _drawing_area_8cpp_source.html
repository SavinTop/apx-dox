90void DrawingArea::resizeArea(int width, int height)
91{
92 int space = 25;
93 dScale = (((width < height) ? width : height) - space * 2) / 2;
94
95 dOffset = ((width < height) ? width : height) / 2;
96
97 if ((width < 50) || (height < 50))
98 return;
99 //this->setFixedSize(width, height);
100 //resize(width,height);
101
102 delete pixMap;
103 pixMap = new QPixmap(this->size());
104 pixMap->fill(Qt::black);
105
106 painter->begin(pixMap);
107
108 QPen wpen(Qt::white, 1, Qt::SolidLine);
109 QPen dgpen(QColor(20, 20, 50), 1, Qt::SolidLine);
110 painter->setPen(dgpen);
111
112 for (int i = -5; i <= 5; i++) {
113 painter->drawLine(DPoint(i / 5.0, -1), DPoint(i / 5.0, 1));
114 }
115
116 for (int i = -5; i <= 5; i++) {
117 painter->drawLine(DPoint(1, i / 5.0), DPoint(-1, i / 5.0));
118 }
119
120 painter->setPen(wpen);
121 painter->drawLine(DPoint(0, -1), DPoint(0, 1));
122 painter->drawLine(DPoint(-1, 0), DPoint(1, 0));
123
124 for (int i = -10; i <= 10; i++) {
125 painter->drawLine(DPoint(i / 10.0, -0.02), DPoint(i / 10.0, 0.02));
126 }
127
128 for (int i = -10; i <= 10; i++) {
129 painter->drawLine(DPoint(0.02, i / 10.0), DPoint(-0.02, i / 10.0));

167void DrawingArea::DrawData()
168{
169 double MaxRadius = 0;
170
171 // Calculate Data
172 double dx = (xMin + xMax) / 2;
173 double dy = (yMin + yMax) / 2;
174
175 double dxR = (xMax - xMin);
176 double dyR = (yMax - yMin);
177
178 double kxR = dxR; //(dxR <= dyR) ? dxR / dyR : 1;
179 double kyR = dyR; //(dyR <= dxR) ? dyR / dxR : 1;
180
181 double bearingRadians = std::atan2(cVector.y(), cVector.x()); // get bearing in radians
182 double bearingDegrees = bearingRadians * (180.0 / M_PI); // convert to degrees
183 bearingDegrees = (bearingDegrees > 0.0 ? bearingDegrees
184 : (360.0 + bearingDegrees)); // correct discontinuity
185
186 bearingRadians = std::atan2(cVector.y() + dy, cVector.x() - dx); // get bearing in radians
187 double bearingDegrees2 = bearingRadians * (180.0 / M_PI); // convert to degrees
188 bearingDegrees2 = (bearingDegrees2 > 0.0 ? bearingDegrees2
189 : (360.0 + bearingDegrees2)); // correct discontinuity
190 double angError = bearingDegrees - bearingDegrees2;
191
192 bX = dx;
193 bY = dy;
194 sX = 1.0 / kxR;
195 sY = 1.0 / kyR;
196
197 QPen pen;
198 pen.setCapStyle(Qt::RoundCap);
199
200 // Draw Points
201 pen.setColor(Qt::green);
202 pen.setWidth(1);
203 painter->setPen(pen);
204 foreach (QPointF cv, cVectorList) {
205 painter->drawPoint(DPoint(cv));
206 double tx = cv.x() - dx;
207 double ty = cv.y() + dy;
208 double r = std::sqrt(tx * tx + ty * ty);
209 MaxRadius = (MaxRadius < r) ? r : MaxRadius;
210 }
211
212 // Draw Ellipses and DX DY point
213 pen.setColor(Qt::blue);
214 pen.setWidth(1);
215 painter->setPen(pen);
216 painter->drawEllipse(DPoint(dx, -dy),
217 static_cast<int>(MaxRadius * dScale),
218 static_cast<int>(MaxRadius * dScale));
219
220 pen.setColor(Qt::gray);
221 pen.setWidth(1);
222 painter->setPen(pen);
223 painter->drawEllipse(DPoint(dx, -dy),
224 static_cast<int>(MaxRadius * dScale * kxR),
225 static_cast<int>(MaxRadius * dScale * kyR));
226
227 pen.setColor(Qt::red);
228 pen.setWidth(7);
229 painter->setPen(pen);
230 painter->drawPoint(DPoint(dx, -dy));
231
232 // Draw Vectors
233 pen.setColor(Qt::cyan);
234 pen.setWidth(1);
235 painter->setPen(pen);
236 painter->drawLine(DPoint(0, 0), DPoint(cVector));
237
238 pen.setColor(QColor(255, 100, 100));
239 pen.setWidth(1);
240 painter->setPen(pen);
241 painter->drawLine(DPoint(dx, -dy), DPoint(cVector));
242
243 pen.setColor(Qt::green);
244 pen.setWidth(7);
245 painter->setPen(pen);
246 painter->drawPoint(DPoint(cVector.rx(), cVector.ry()));
247 painter->setPen(QPen());
248
249 // Draw Text DATA
250 pen.setColor(Qt::yellow);
251 pen.setWidth(2);
252 painter->setPen(pen);
253 painter->drawText(QPointF(10, 10), QString("b%1 = %2").arg(sAxis[0]).arg(dx, 0, 'f', 3));
254 painter->drawText(QPointF(10, 20), QString("b%1 = %2").arg(sAxis[1]).arg(dy, 0, 'f', 3));
255
256 painter->drawText(QPointF(dOffset + 20, 10),
257 QString("s%1 = %2").arg(sAxis[0]).arg(1.0 / kxR, 0, 'f', 2));
258 painter->drawText(QPointF(dOffset + 20, 20),
259 QString("s%1 = %2").arg(sAxis[1]).arg(1.0 / kyR, 0, 'f', 2));
260
261 pen.setColor(Qt::lightGray);
262 pen.setWidth(2);
263 painter->setPen(pen);
264 painter->drawText(QPointF(10, 30), QString("r: %1").arg(MaxRadius, 0, 'f', 2));
265
266 pen.setColor(Qt::cyan);
267 pen.setWidth(2);
268 painter->setPen(pen);
269 painter->drawText(QPointF(10, 40),
270 QString("a: %1%2").arg(bearingDegrees, 0, 'f', 0).arg(QChar(176)));
271
272 pen.setColor(QColor(255, 100, 100));
273 pen.setWidth(2);
274 painter->setPen(pen);
275
276 painter->drawText(QPointF(10, 50),
277 QString("b: %1%2").arg(bearingDegrees2, 0, 'f', 0).arg(QChar(176)));
278
279 // ERROR - COlor changing
280 int maxError = 10;
281 int errcolor = (static_cast<int>(std::abs(angError)) * 255) / maxError;
282
283 errcolor = (errcolor > 230) ? 230 : errcolor;
284 pen.setColor(QColor(errcolor, 120 - errcolor / 2, 255 - errcolor));
285 pen.setWidth(2);
286 painter->setPen(pen);
287 painter->drawText(QPointF(10, 60),
288 QString("e: %1%2").arg(AppRoot::angle(angError), 0, 'f', 0).arg(QChar(176)));
289}
