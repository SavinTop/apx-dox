91void DatalinkStatsCounter::updateTimerTimeout()
92{
93 //data count
94 f_datacnt->setValue(dataToString(dataCnt));
95
96 //rate
97 int t = time.restart();
98 f_datarate->setValue(dataToString(getRate(&dataRate, dataCnt - dataCntT, t)) + "/sec");
99 dataCntT = dataCnt;
100 f_rate->setValue(QString::number((packetCnt - packetCntT) * 1000 / t) + "/sec");
101 packetCntT = packetCnt;
102}

104double DatalinkStatsCounter::getRate(double *v, uint dcnt, uint t)
105{
106 double vt = (double) dcnt / (double) t * 1000.0;
107 *v = (*v) * 0.5 + vt * 0.5;
108 return *v;
109}

111QString DatalinkStatsCounter::dataToString(uint v)
112{
113 if (v < 1024)
114 return QString("%1 bytes").arg(v);
115 else if (v < (1024 * 1024))
116 return QString("%1 KB").arg(v / 1024.0, 0, 'f', 1);
117 else if (v < (1024 * 1024 * 1024))
118 return QString("%1 MB").arg(v / (1024.0 * 1024.0), 0, 'f', 2);
119 else
120 return QString("%1 GB").arg(v / (1024.0 * 1024.0 * 1024.0), 0, 'f', 3);
121}
