48void DatalinkPorts::updateStatus()
49{
50 int cnt = f_list->size();
51 if (cnt <= 0) {
52 setValue("");
53 setActive(false);
54 } else {
55 int ecnt = 0, acnt = 0;
56 for (int i = 0; i < cnt; ++i) {
57 Fact *f = f_list->child(i);
58 if (f->value().toBool())
59 ecnt++;
60 if (f->active())
61 acnt++;
62 }
63 setValue(QString("%1/%2/%3").arg(acnt).arg(ecnt).arg(cnt));
64 setActive(acnt > 0);
65 }
66}

81void DatalinkPorts::load()
82{
83 QFile file(AppDirs::prefs().filePath("ports.json"));
84 if (file.exists() && file.open(QFile::ReadOnly | QFile::Text)) {
85 QJsonDocument json = QJsonDocument::fromJson(file.readAll());
86 file.close();
87 for (auto const v : json.array()) {
88 f_add->clear();
89 f_add->fromVariant(v.toObject().toVariantMap());
90 addPort(new DatalinkPort(this, datalink, f_add));
91 }
92 f_add->defaults();
93 }
94}

117QList<DatalinkPort *> DatalinkPorts::serialPorts() const
118{
119 QList<DatalinkPort *> list;
120 for (auto i : f_list->facts()) {
121 DatalinkPort *port = qobject_cast<DatalinkPort *>(i);
122 if (!port)
123 continue;
124 if (port->f_type->value().toInt() != DatalinkPort::SERIAL)
125 continue;
126 list.append(port);
127 }
128 return list;
129}

131void DatalinkPorts::blockSerialPorts()
132{
133 for (auto port : serialPorts()) {
134 if (!port->f_enable->value().toBool())
135 continue;
136 blockedPorts.append(port);
137 port->f_enable->setValue(false);
138 }
139 if (blockedPorts.isEmpty())
140 return;
141 qDebug() << "Ports blocked" << blockedPorts.size();
142}
143void DatalinkPorts::unblockSerialPorts()
144{
145 if (blockedPorts.isEmpty())
146 return;
147 for (auto port : serialPorts()) {
148 if (!blockedPorts.contains(port))
149 continue;
150 port->f_enable->setValue(true);
151 }
152 qDebug() << "Ports unblocked" << blockedPorts.size();
153 blockedPorts.clear();
154}
