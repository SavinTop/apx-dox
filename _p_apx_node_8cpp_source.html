67void PApxNode::process_downlink(const xbus::pid_s &pid, PStreamReader &stream)
68{
69 mandala::uid_t uid = pid.uid;
70
71 if (uid == mandala::cmd::env::nmt::search::uid) {
72 if (upgrading())
73 return;
74 if (pid.pri != xbus::pri_response)
75 return;
76
77 requestIdent();
78 return;
79 }
80
81 //filter zero sized packets if any
82 if (stream.available() == 0)
83 return;
84
85 // file ops - make them download data from any source
86 if (uid == mandala::cmd::env::nmt::file::uid) {
87 // accept both pid requests and responses
88 size_t spos = stream.pos();
89
90 if (stream.available() <= sizeof(xbus::node::file::op_e))
91 return;
92
93 xbus::node::file::op_e op;
94 stream >> op;
95
96 if (op & xbus::node::file::reply_op_mask)
97 trace()->block("re");
98 trace()->block(QString::number(op & ~xbus::node::file::reply_op_mask));
99
100 const char *s = stream.read_string(16);
101 if (!s)
102 return;
103
104 trace()->block(QString(s));
105 trace()->data(stream.payload());
106
107 auto f = file(s);
108 if (!f)
109 return;
110
111 f->process_downlink(op, stream);
112
113 stream.reset(spos);
114 }
115
116 // field updates from remote gcs
117 if (uid == mandala::cmd::env::nmt::upd::uid) {
118 // qDebug() << stream.available();
119 if (stream.available() < sizeof(xbus::node::conf::fid_t))
120 return;
121
122 auto pos_s = stream.pos();
123
124 xbus::node::conf::fid_t fid;
125 stream >> fid;
126 trace()->block(QString::number(fid >> 8));
127 trace()->block(QString::number(fid & 0xFF));
128 trace()->data(stream.payload());
129
130 if (pid.pri == xbus::pri_response) {
131 // intrercept conf saved response
132 stream.reset(pos_s);
133 if (fid == 0xFFFFFFFF) {
134 emit confSaved();
135 }
136 } else if (pid.pri == xbus::pri_request) {
137 auto fidx = fid >> 8;
138 if (fidx >= _field_types.size())
139 return;
140 auto aidx = fid & 0xFF;
141 auto name = _field_names.at(fidx);
142 auto array = _field_arrays.at(fidx);
143 if (array > 0) {
144 if (aidx >= array)
145 return;
146 name.append(QString("_%1").arg(aidx + 1));
147 }
148 if (name == _script_field)
149 return;
150 auto value = read_param(stream, _field_types.at(fidx));
151 QVariantMap values;
152 values.insert(name, value);
153 emit confUpdated(values);
154 return;
155 } else
156 return;
157 }
158
159 // node messages
160 if (uid == mandala::cmd::env::nmt::msg::uid) {
161 if (stream.available() < (sizeof(xbus::node::msg::type_e) + 1))
162 return;
163
164 xbus::node::msg::type_e t;
165 stream >> t;
166 trace()->block(QString::number(t));
167
168 bool msg_init = false;
169
170 while (stream.available() > 0) {
171 auto s = stream.read_string(stream.available());
172 QString msg(QString(s).trimmed());
173
174 msg.replace("\r", "\n");
175 msg = msg.trimmed();
176
177 auto msg_lines = msg.split('\n', Qt::SkipEmptyParts);
178 trace()->block(msg_lines.join(" | "));
179
180 for (auto line : msg_lines) {
181 line.replace(":", ": ");
182 line = line.simplified();
183 if (line.isEmpty())
184 continue;
185
186 emit messageReceived((msg_type_e) t, line);
187
188 if (line.contains(QString("%1: initialized").arg(title()), Qt::CaseInsensitive))
189 msg_init = true;
190 }
191 }
192
193 if (msg_init && !(_nodes->local() || _nodes->upgrading()))
194 requestIdent();
195
196 return;
197 }
198
199 // check for requests responses
200 if (pid.pri != xbus::pri_response) {
201 // a request from another GCS?
202 return;
203 }
204
205 // check for pending request's response
206 for (auto req : _requests) {
207 if (req->uid() != uid)
208 continue;
209 // there's a request waiting for the packet's uid
210 if (req->check_response(stream)) {
211 delete_request(req);
212 }
213 }
214}

216void PApxNode::schedule_request(PApxNodeRequest *req)
217{
218 mandala::uid_t uid = req->uid();
219 // check duplicates
220 for (auto i : _requests) {
221 if (i->equals(req)) {
222 // the most recent for the uid is the only valid
223 if (uid == mandala::cmd::env::nmt::ident::uid || i->active()) {
224 delete_request(req);
225 return;
226 }
227 qDebug() << "dup" << req->title();
228 delete_request(i);
229 }
230 }
231 _requests.append(req);
232 updateProgress();
233 emit request_scheduled(req);
234}

268void PApxNode::updateFiles(QStringList fnames)
269{
270 if (fnames == _files_map.keys()) {
271 for (auto i : _files_map) {
272 i->reset();
273 }
274 return;
275 }
276
277 for (auto i : _files_map) {
278 i->deleteLater();
279 }
280 _files_map.clear();
281 for (auto i : _requests) {
282 if (i->uid() == mandala::cmd::env::nmt::file::uid)
283 delete_request(i);
284 }
285 for (auto i : fnames) {
286 PApxNodeFile *f = new PApxNodeFile(this, i);
287 _files_map.insert(i, f);
288
289 if (i == "dict") {
290 connect(f, &PApxNodeFile::downloaded, this, &PApxNode::parseDictData);
291 } else if (i == "conf") {
292 connect(f, &PApxNodeFile::downloaded, this, &PApxNode::parseConfData);
293 } else if (i == "script") {
294 connect(f, &PApxNodeFile::downloaded, this, &PApxNode::parseScriptData);
295 connect(f, &PApxNodeFile::uploaded, this, &PApxNode::parseScriptDataUpload);
296 }
297 }

300bool PApxNode::find_field(QString name,
301 xbus::node::conf::fid_t *fid,
302 xbus::node::conf::type_e *type) const
303{
304 xbus::node::conf::fid_t v{};
305 QRegExp re("_(\\d+)$");
306 auto a = re.indexIn(name);
307 if (a > 1) {
308 auto i = re.cap(1).toInt() - 1;
309 if (i < 0) {
310 qWarning() << "array" << name;
311 return false;
312 }
313 name = name.left(a);
314 v = i;
315 }
316 auto i = _field_names.indexOf(name);
317 if (i < 0) {
318 qWarning() << "missing field:" << name;
319 return false;
320 }
321 v |= i << 8;
322 *fid = v;
323 *type = _field_types.at(i);
324 return true;
325}

367void PApxNode::dictCacheLoaded(QVariantMap dict)
368{
369 auto hash = dict.value("hash").toString();
370 if (_dict_hash != hash) {
371 qWarning() << "wrong hash" << _dict_hash << hash;
372 return;
373 }
374 if (dict.isEmpty()) {
375 qWarning() << "no dict data";
376 requestDictDownload();
377 return;
378 }
379 qDebug() << "dict from cache";
380
381 _field_types.clear();
382 _field_names.clear();
383 _field_arrays.clear();
384 _field_units.clear();
385
386 _values.clear();
387 _script_value = {};
388 _script_field.clear();
389
390 for (auto i : dict.value("fields").value<QVariantList>()) {
391 auto field = i.value<QVariantMap>();
392 auto type = field.value("type").toString();
393 xbus::node::conf::type_e type_id{};
394 for (uint8_t t = xbus::node::conf::type_field; t < xbus::node::conf::type_max; ++t) {
395 QString s = xbus::node::conf::type_to_str((xbus::node::conf::type_e) t);
396 if (s != type)
397 continue;
398 type_id = (xbus::node::conf::type_e) t;
399 break;
400 }
401 if (!type_id)
402 continue;
403 _field_types.append(type_id);
404 _field_names.append(field.value("name").toString());
405 _field_arrays.append(field.value("array").toUInt());
406 _field_units.append(field.value("units").toString());
407 }
408
409 emit dictReceived(dict);
410}

419void PApxNode::parseDictData(PApxNode *node,
420 const xbus::node::file::info_s &info,
421 const QByteArray data)
422{
423 Q_UNUSED(node)
424
425 PStreamReader stream(data);
426
427 bool err = true;
428 QVariantList fields;
429 _field_types.clear();
430 _field_names.clear();
431 _field_arrays.clear();
432 _field_units.clear();
433
434 _values.clear();
435 _script_value = {};
436 _script_field.clear();
437
438 do {
439 // check node hash
440 QString hash = hashToText(info.hash);
441 if (hash != _dict_hash) {
442 qWarning() << "node hash error:" << hash << _dict_hash;
443 break;
444 }
445
446 QStringList names;
447 QList<int> groups;
448 QList<int> group_idx;
449
450 while (stream.available() > 4) {
451 QVariantMap field;
452
453 xbus::node::conf::type_e type_id = static_cast<xbus::node::conf::type_e>(
454 stream.read<uint8_t>());
455
456 QString type = xbus::node::conf::type_to_str(type_id);
457 auto array = stream.read<uint8_t>();
458
459 field.insert("type", type);
460
461 if (array > 0)
462 field.insert("array", array);
463
464 uint8_t group = stream.read<uint8_t>();
465
466 QStringList st;
467 QString name, title, units;
468
469 bool is_writable = false;
470
471 switch (type_id) {
472 case xbus::node::conf::group:
473 st = stream.read_strings(2);
474 if (st.isEmpty() || st.at(0).isEmpty())
475 break;
476 //qDebug() << "group" << field.insert(group << st;
477 name = st.at(0);
478 title = st.at(1);
479 group_idx.append(fields.size());
480 break;
481 case xbus::node::conf::command:
482 st = stream.read_strings(2);
483 if (st.isEmpty() || st.at(0).isEmpty())
484 break;
485 name = st.at(0);
486 title = st.at(1);
487 break;
488 default:
489 st = stream.read_strings(2, stream.available());
490 if (st.isEmpty() || st.at(0).isEmpty())
491 break;
492 //qDebug() << "field" << field.insert(type << field.insert(array << field.insert(group << st;
493 name = st.at(0);
494 units = st.at(1);
495 is_writable = true;
496 }
497 if (name.isEmpty())
498 break;
499
500 groups.append(group);
501 names.append(name);
502
503 if (title.isEmpty())
504 title = name;
505 field.insert("title", title);
506
507 if (units.isEmpty() && type_id == xbus::node::conf::option) {
508 units = "off,on";
509 }
510
511 if (!units.isEmpty())
512 field.insert("units", units);
513
514 // guess path prepended with groups
515 QStringList path;
516 path.append(name);
517 for (auto i = group; i > 0;) {
518 i--;
519 if (i < group_idx.size()) {
520 int gidx = group_idx.at(i);
521 path.prepend(names.at(gidx));
522 i = groups.at(gidx);
523 continue;
524 }
525 qWarning() << "missing group:" << type << field.value("array").toInt() << group
526 << st;
527 path.clear(); //mark error
528 break;
529 }
530 name = path.join('.');
531 field.insert("name", name);
532
533 fields.append(field);
534
535 if (is_writable) {
536 _field_types.append(type_id);
537 _field_names.append(name);
538 _field_arrays.append(array);
539 _field_units.append(units);
540 }
541
542 //qDebug() << field << st << stream.available();
543
544 if (stream.available() == 0) {
545 err = false;
546 break;
547 }
548 }
549
550 } while (0);
551
552 if (err) {
553 qWarning() << "dict error" << data.toHex().toUpper();
554 _field_types.clear();
555 _field_names.clear();
556 _field_arrays.clear();
557 _field_units.clear();
558 return;
559 }
560 qDebug() << "dict parsed";
561
562 QVariantMap dict;
563 dict.insert("hash", _dict_hash);
564 dict.insert("cached", false);
565 dict.insert("fields", fields);
566
567 emit dictReceived(dict);
568}

577void PApxNode::parseConfData(PApxNode *node,
578 const xbus::node::file::info_s &info,
579 const QByteArray data)
580{
581 Q_UNUSED(node)
582
583 PStreamReader stream(data);
584
585 _values.clear();
586
587 bool err = true;
588 QVariantMap values;
589 int fidx = 0;
590
591 do {
592 // read offsets
593 QList<size_t> offsets;
594 while (stream.available() >= 2) {
595 uint16_t v;
596 stream >> v;
597 if (!v)
598 break;
599 offsets.append(v);
600 }
601
602 // read Parameters::Data struct
603 size_t pos_s = stream.pos();
604
605 // read and check prepended hash
606 xbus::node::hash_t vhash;
607 stream >> vhash;
608
609 auto hash = hashToText(vhash);
610
611 if (hash != _dict_hash) {
612 qWarning() << "data hash error:" << hash << _dict_hash;
613 break;
614 }
615
616 size_t offset_s = 0;
617 while (stream.available() > 0) {
618 // stream padding with offset
619 size_t d_pos = stream.pos() - pos_s;
620 size_t offset = offsets.at(fidx);
621 size_t d_offset = offset - offset_s;
622 offset_s = offset;
623 if (d_offset > d_pos) {
624 d_pos = d_offset - d_pos;
625 //qDebug() << "padding:" << d_pos;
626 if (stream.available() < d_pos)
627 break;
628 stream.reset(stream.pos() + d_pos);
629 } else if (d_offset < d_pos) {
630 qWarning() << "padding negative:" << d_offset << d_pos << offsets;
631 break;
632 }
633
634 pos_s = stream.pos();
635
636 // read value
637 auto array = _field_arrays.value(fidx);
638 auto type = _field_types.value(fidx);
639 auto units = _field_units.value(fidx);
640 QVariant value;
641
642 if (array > 0) {
643 QVariantList list;
644 for (auto i = 0; i < array; ++i) {
645 QVariant v = read_param(stream, type);
646 if (!v.isValid())
647 break;
648 if (type == xbus::node::conf::option)
649 v = optionToText(v, fidx);
650 else if (type == xbus::node::conf::bind)
651 v = mandalaToString(v.toUInt());
652 list.append(v);
653 }
654 if (list.size() == array)
655 value = list;
656 } else {
657 value = read_param(stream, type);
658 if (type == xbus::node::conf::option)
659 value = optionToText(value, fidx);
660 else if (type == xbus::node::conf::bind)
661 value = mandalaToString(value.toUInt());
662 }
663
664 //qDebug() << v << stream.pos() << stream.available();
665 if (!value.isValid())
666 break;
667
668 QString name = _field_names.value(fidx);
669
670 if (type == xbus::node::conf::script) {
671 _script_value = value.value<xbus::node::conf::script_t>();
672 _script_field = name;
673 value = QVariant();
674 }
675
676 values.insert(name, value);
677
678 fidx++;
679 if (fidx == _field_types.size()) {
680 err = stream.available() > 8 ? true : false;
681 break;
682 }
683 }
684
685 } while (0);
686
687 if (err) {
688 qWarning() << "conf error" << fidx << stream.available() << data.toHex().toUpper();
689 _skip_cache = true;
690 // requestDict();
691 return;
692 }
693
694 // conf file parsed
695 if (_script_field.isEmpty()) {
696 emit confReceived(values);
697 return;
698 }
699
700 // store values and download script
701 _values = values;
702 if (_req_conf)
703 new PApxNodeRequestFileRead(this, "script");
704}
705template<typename T, typename Tout = T>
706static QVariant _read_param(PStreamReader &stream)
707{
708 if (stream.available() < sizeof(T))
709 return QVariant();
710 return QVariant::fromValue(stream.read<T, Tout>());
711}
712template<typename _T>
713static QVariant _read_param_str(PStreamReader &stream)
714{
715 size_t pos_s = stream.pos();
716 const char *s = stream.read_string(sizeof(_T));
717 if (!s)
718 return QVariant();
719 stream.reset(pos_s + sizeof(_T));
720 return QVariant::fromValue(QString(s));
721}

793QVariant PApxNode::optionToText(QVariant value, size_t fidx)
794{

817void PApxNode::parseScriptData(PApxNode *node,
818 const xbus::node::file::info_s &info,
819 const QByteArray data)
820{
821 Q_UNUSED(node)
822
823 //qDebug() << "script data" << info.size << data.size();
824 // check script hash
825 if (info.hash != _script_value) {
826 qWarning() << "script hash error:" << QString::number(info.hash, 16)
827 << QString::number(_script_value, 16);
828 return;
829 }
830
831 if (_script_field.isEmpty()) {
832 qWarning() << "missing script field";
833 return;
834 }
835
836 QVariant value;
837 if (info.size > 0) {
838 PStreamReader stream(data);
839 xbus::script::file_hdr_s hdr{};
840 if (stream.available() < hdr.psize()) {
841 qWarning() << "hdr size" << stream.available();
842 return;
843 }
844 hdr.read(&stream);
845
846 if (stream.available() != (hdr.code_size + hdr.src_size)) {
847 qWarning() << "size" << stream.available() << hdr.code_size << hdr.src_size;
848 return;
849 }
850 const QByteArray code = stream.toByteArray(stream.pos(), hdr.code_size);
851 const QByteArray src = qUncompress(
852 stream.toByteArray(stream.pos() + hdr.code_size, hdr.src_size));
853 if (src.isEmpty() || code.isEmpty()) {
854 qWarning() << "empty" << stream.available() << src.size() << code.size();
855 return;
856 }
857 QString title = QString(QByteArray(hdr.title, sizeof(hdr.title)));
858
859 qDebug() << "script:" << title; // << _script_code.toHex().toUpper();
860
861 QStringList st;
862 st << title;
863 st << qCompress(src, 9).toHex().toUpper();
864 st << qCompress(code, 9).toHex().toUpper();
865 value = st.join(',');
866 }
867
868 _values.insert(_script_field, value);
869 emit confReceived(_values);
870 _values.clear();
871}

889QByteArray PApxNode::pack_script(QVariant value)
890{
891 QStringList st = value.toString().split(',', Qt::KeepEmptyParts);
892
893 if (st.size() != 3)
894 return {};
895
896 QString title = st.at(0);
897 QByteArray src = QByteArray::fromHex(st.at(1).toLocal8Bit());
898 QByteArray code = qUncompress(QByteArray::fromHex(st.at(2).toLocal8Bit()));
899
900 QByteArray data(xbus::script::max_file_size, '\0');
901 PStreamWriter stream(data.data(), data.size());
902
903 xbus::script::file_hdr_s hdr{};
904 hdr.code_size = code.size();
905 hdr.src_size = src.size();
906 strncpy(hdr.title, title.toUtf8(), sizeof(xbus::script::title_t));
907 hdr.write(&stream);
908 stream.append(code);
909 stream.append(src);
910 return stream.toByteArray();
