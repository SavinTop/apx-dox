109void NodeStorage::metaDataLoaded(QVariantMap meta)
110{
111 uint cnt = 0;
112 for (auto name : meta.keys()) {
113 auto f = _node->findChild(name);
114 if (!f)
115 continue;
116
117 cnt++;
118
119 auto m = meta.value(name).value<QVariantMap>();
120 auto descr = m.value("descr").toString();
121
122 descr = descr.left(descr.indexOf('\n'));
123 descr = descr.left(descr.indexOf('.'));
124
125 f->setDescr(descr.simplified());
126
127 NodeField *nf = qobject_cast<NodeField *>(f);
128 if (!nf)
129 continue;
130
131 auto def = m.value("def");
132 auto min = m.value("min");
133 auto max = m.value("max");
134 auto increment = m.value("increment");
135 auto decimal = m.value("decimal");
136
137 if (!def.isNull())
138 nf->setDefaultValue(def);
139 if (!min.isNull())
140 nf->setMin(min);
141 if (!max.isNull())
142 nf->setMax(max);
143 if (!increment.isNull())
144 nf->setIncrement(increment.toDouble());
145 // if (!decimal.isNull())
146 // nf->setPrecision(decimal.toInt());
147 }
148 if (cnt > 0)
149 qDebug() << "meta data loaded:" << cnt;
150}
