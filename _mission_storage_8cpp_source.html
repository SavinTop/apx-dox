93void MissionStorage::loadMission(QString hash)
94{
95 //qDebug()<<"load mission"<<missionID;
96 if (hash.isEmpty())
97 return;
98 if ((!mission->modified()) && mission->missionSize() > 0 && dbHash == hash)
99 return;
100 dbHash = hash;
101
102 DBReqMissionsLoad *req = new DBReqMissionsLoad(hash);
103 connect(req, &DBReqMissionsLoad::loaded, this, &MissionStorage::dbLoaded, Qt::QueuedConnection);
104 req->exec();
105}
106void MissionStorage::dbLoaded(QVariant var)
107{
108 auto m = var.value<QVariantMap>();
109 dbHash = m.value("hash").toString();
110 qDebug() << "mission load:" << dbHash;
111
112 mission->fromVariant(var);
113
114 if (mission->missionSize() > 0) {
115 emit loaded();
116 }
117 QString s(tr("Mission loaded"));
118 QString title = mission->f_title->text();
119 if (title.isEmpty())
120 mission->vehicle->message(s);
121 else
122 mission->vehicle->message(s.append(": ").append(title));
123}
