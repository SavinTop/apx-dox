103void AppRoot::updateProgress(Fact *fact)
104{
105 if (fact && !progressList.contains(fact)) {
106 progressList.append(fact);
107 connect(fact, &QObject::destroyed, this, [this]() { updateProgress(nullptr); });
108 }
109 int total = 0;
110 int cnt = 0;
111 for (int i = 0; i < progressList.size(); ++i) {
112 auto f = progressList.at(i);
113 int v = f ? f->progress() : -1;
114 if (v < 0) {
115 progressList.removeAt(i--);
116 continue;
117 }
118 cnt++;
119 total += v;
120 }
121 int v = cnt > 0 ? total / cnt : -1;
122 if (m_progress == v)
123 return;
124 m_progress = v;
125 emit progressChanged();
126}

156double AppRoot::latFromString(QString s)
157{
158 bool ok;
159 int i;
160 s = s.simplified();
161 if (QString("NS").contains(s.at(0))) {
162 bool bN = s.at(0) == 'N';
163 s = s.remove(0, 1).trimmed();
164 i = s.indexOf(QChar(176));
165 double deg = s.left(i).toDouble(&ok);
166 if (!ok)
167 return 0;
168 s = s.remove(0, i + 1).trimmed();
169 i = s.indexOf('\'');
170 double min = s.left(i).toDouble(&ok);
171 if (!ok)
172 return 0;
173 s = s.remove(0, i + 1).trimmed();
174 i = s.indexOf('\"');
175 double sec = s.left(i).toDouble(&ok);
176 if (!ok)
177 return 0;
178 deg = deg + min / 60.0 + sec / 3600.0;
179 return bN ? deg : -deg;
180 }
181 return s.toDouble();
182}

242quint64 AppRoot::timeFromString(QString s, bool seconds)
243{
244 quint64 t = 0;
245 s = s.trimmed().toLower();
246
247 if (s.contains('d')) {
248 QString ds = s.left(s.indexOf('d')).trimmed();
249 s = s.remove(0, s.indexOf('d') + 1).trimmed();
250 bool ok = false;
251 double dv = ds.toDouble(&ok);
252 if (ok && dv > 0)
253 t += static_cast<quint64>(std::floor(dv * (24.0 * 60.0 * 60.0)));
254 }
255 if (s.contains('h')) {
256 QString ds = s.left(s.indexOf('h')).trimmed();
257 s = s.remove(0, s.indexOf('h') + 1).trimmed();
258 bool ok = false;
259 double dv = ds.toDouble(&ok);
260 if (ok && dv > 0)
261 t += static_cast<quint64>(std::floor(dv * (60.0 * 60.0)));
262 }
263 if (s.contains('m')) {
264 QString ds = s.left(s.indexOf('m')).trimmed();
265 s = s.remove(0, s.indexOf('m') + 1).trimmed();
266 bool ok = false;
267 double dv = ds.toDouble(&ok);
268 if (ok && dv > 0)
269 t += static_cast<quint64>(std::floor(dv * (60.0)));
270 }
271 if (s.contains('s')) {
272 QString ds = s.left(s.indexOf('s')).trimmed();
273 s = s.remove(0, s.indexOf('s') + 1).trimmed();
274 bool ok = false;
275 double dv = ds.toDouble(&ok);
276 if (ok && dv > 0)
277 t += static_cast<quint64>(std::floor(dv));
278 s.clear();
279 }
280 if (s.contains(':')) {
281 int cap = seconds ? 0 : 1;
282 for (auto st = s.split(':'); st.size(); cap++) {
283 auto s = st.takeLast();
284 bool ok = false;
285 double v = s.toDouble(&ok);
286 if (!ok || v <= 0)
287 v = 0;
288 double mult{};
289 switch (cap) {
290 default:
291 t = 0;
292 break;
293 case 0: //seconds
294 mult = 1.;
295 break;
296 case 1: //minutes
297 mult = 60.;
298 break;
299 case 2: //hours
300 mult = 60. * 60.;
301 break;
302 case 3: //days
303 mult = 24. * 60. * 60.;
304 break;
305 }
306 t += static_cast<quint64>(std::floor(v * mult));
307 }
308 s.clear();
309 }
310 if (!s.isEmpty()) {
311 bool ok = false;
312 double v = s.toDouble(&ok);
313 if (ok && v > 0)
314 t += static_cast<quint64>(std::floor(v));
315 return t;
316 }
317 if (!seconds)
318 return t / 60;
319
320 return t;
321}

372QPointF AppRoot::rotate(const QPointF &p, double a)
373{
374 const double psi_r = qDegreesToRadians(a);
375 double cos_theta = std::cos(psi_r);
376 double sin_theta = std::sin(psi_r);
377 return QPointF(p.x() * cos_theta + p.y() * sin_theta, p.y() * cos_theta - p.x() * sin_theta);
378}

404{
