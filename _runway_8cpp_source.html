104void Runway::updateTitle()
105{
106 setValue(
107 f_type->valueText().left(1).toUpper()
108 + QString("%1").arg((int) AppRoot::angle360(round(AppRoot::angle360(heading()) / 10.0) * 10)
109 / 10,
110 2,
111 10,
112 QLatin1Char('0')));
113 QStringList st;
114 st.append(QString::number(num() + 1));
115 st.append(value().toString());
116 st.append(AppRoot::distanceToString(f_approach->value().toInt()));
117 if (!f_hmsl->isZero())
118 st.append("MSL" + f_hmsl->valueText());
119 setTitle(st.join(' '));

122void Runway::updateMissionStartPoint()
123{
124 if (num() != 0)
125 return;
126 QGeoCoordinate p(coordinate());
127 QGeoCoordinate p2 = endPoint();
128 group->mission->setStartPoint(p2);
129 group->mission->setStartHeading(heading());
130 double slen = f_approach->value().toDouble() * 2.0 - p.distanceTo(endPoint());
131 group->mission->setStartLength(slen < 0 ? 0 : slen);
132 group->mission->setCoordinate(p);
133}

151QGeoCoordinate Runway::endPoint() const
152{
153 QGeoCoordinate p(coordinate());
154 double dN = f_dN->value().toDouble();
155 double dE = f_dE->value().toDouble();
156 double azimuth = qRadiansToDegrees(std::atan2(dE, dN));
157 double distance = std::sqrt(std::pow(dN, 2) + std::pow(dE, 2));
158 return p.atDistanceAndAzimuth(distance, azimuth);
159}
160void Runway::setEndPoint(const QGeoCoordinate &v)

166 f_dE->setValue(d * std::sin(a));

168QGeoCoordinate Runway::appPoint() const
169{
170 QGeoCoordinate p(coordinate());
171 double dN = f_dN->value().toDouble();
172 double dE = f_dE->value().toDouble();
173 double azimuth = qRadiansToDegrees(std::atan2(dE, dN)) + 180.0;
174 double distance = f_approach->value().toDouble();
175 return p.atDistanceAndAzimuth(distance, azimuth);
176}
177void Runway::setAppPoint(const QGeoCoordinate &v)
178{
179 QGeoCoordinate p(coordinate());
180 double a = qDegreesToRadians(p.azimuthTo(v));
181 double d = p.distanceTo(v);
182 QPointF ne(d * std::cos(a), d * std::sin(a));
183 ne = AppRoot::rotate(ne, heading() + 180.0);
184 if (std::abs(ne.y()) > (f_approach->value().toDouble() / 2.0)) {
185 //switch turn direction
186 f_type->setValue(ne.y() > 0 ? "left" : "right");
187 }
188 int dist = ne.x();
189 if (dist < 5)
190 dist = 0;
191 else if (dist > 50000)
192 dist = 50000;
193 else if (dist > 500)
194 dist = (dist / 100) * 100;
195 else if (dist > 100)
196 dist = (dist / 10) * 10;
197 f_approach->setValue(dist);
198}
199double Runway::heading() const
200{
201 double dN = f_dN->value().toDouble();
202 double dE = f_dE->value().toDouble();
203 return qRadiansToDegrees(std::atan2(dE, dN));
