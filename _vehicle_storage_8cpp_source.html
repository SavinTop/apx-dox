44void VehicleStorage::saveVehicleConfig()
45{
46 auto nodes = _vehicle->f_nodes;
47
48 if (!nodes->valid())
49 return;
50
51 if (nodes->nodes().isEmpty())
52 return;
53
54 QList<quint64> nconfIDs;
55 for (auto i : nodes->nodes()) {
56 if (!i->valid())
57 return;
58 auto nconfID = i->storage->configID();
59 if (!nconfID)
60 return;
61 nconfIDs.append(nconfID);
62 }
63 auto vuid = _vehicle->uid();
64 auto title = nodes->getConfigTitle();
65
66 auto *req = new DBReqSaveVehicleConfig(vuid, nconfIDs, title);
67 connect(req,
68 &DBReqSaveVehicleConfig::configSaved,
69 this,
70 &VehicleStorage::configSaved,
71 Qt::QueuedConnection);
72 req->exec();
73}
