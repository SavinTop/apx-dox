119void FactListModel::populate(ItemsList *list, Fact *f)
120{
121 bool flt = (this->fact->options() & Fact::FilterModel) && !filter().isEmpty();
122 bool sect = false;
123 for (int i = 0; i < f->size(); ++i) {
124 Fact *item = f->child(i);
125 if (!item)
126 continue;
127 if (flt && item->treeType() != Fact::Group
128 && !item->showThis(QRegExp(filter(), Qt::CaseInsensitive)))
129 continue;
130 if (!sect && !item->section().isEmpty())
131 sect = true;
132 connect(item, &FactBase::destroyed, this, &FactListModel::scheduleSync, Qt::UniqueConnection);
133
134 if ((f->options() & Fact::FlatModel) && (item->options() & Fact::Section)) {
135 connect(item,
136 &Fact::itemInserted,
137 this,
138 &FactListModel::scheduleSync,
139 Qt::UniqueConnection);
140 connect(item,
141 &Fact::itemRemoved,
142 this,
143 &FactListModel::scheduleSync,
144 Qt::UniqueConnection);
145 connect(item, &Fact::itemMoved, this, &FactListModel::scheduleSync, Qt::UniqueConnection);
146 connect(item,
147 &Fact::optionsChanged,
148 this,
149 &FactListModel::scheduleSync,
150 Qt::UniqueConnection);
151 populate(list, item);
152 continue;
153 }
154 if (flt && item->treeType() == Fact::Group) {
155 populate(list, item);
156 } else {
157 list->append(item);
158 }
159 }
160 if (sect && f == this->fact) {
161 //check for gaps in sections to sort
162 QString s = "@/./";
163 QStringList slist;
164 for (auto i : *list) {
165 const QString &si = i->section();
166 if (s == si)
167 continue;
168 s = si;
169 if (!slist.contains(s)) {
170 slist.append(s);
171 continue;
172 }
173 //gap found
174 std::stable_sort(list->begin(), list->end(), [](Fact *a, Fact *b) {
175 return a->section().localeAwareCompare(b->section()) < 0;
176 });
177 break;
178 }
179 }
180}

194void FactListModel::syncModel(const ItemsList &list)
195{
196 bool bLayoutChanged = false;
197 //find deleted
198 for (int i = 0; i < _items.size(); ++i) {
199 if (list.contains(_items.at(i)))
200 continue;
201 beginRemoveRows(QModelIndex(), i, i);
202 _items.removeAt(i);
203 endRemoveRows();
204 bLayoutChanged = true;
205 //qDebug()<<"del"<<i<<this;
206 emit countChanged();
207 i--;
208 }
209 //find inserted and moved
210 for (int i = 0; i < list.size(); ++i) {
211 Fact *src = list.at(i);
212 Fact *dest = _items.value(i, nullptr);
213 if (src == dest)
214 continue;
215 int j = _items.indexOf(src);
216 if (dest && j > 0) {
217 //moved
218 beginMoveRows(QModelIndex(), j, j, QModelIndex(), i);
219 _items.removeAt(j);
220 _items.insert(i, src);
221 endMoveRows();
222 bLayoutChanged = true;
223 //qDebug()<<"mov"<<j<<i<<this;
224 } else {
225 //inserted
226 beginInsertRows(QModelIndex(), i, i);
227 _items.insert(i, src);
228 endInsertRows();
229 bLayoutChanged = true;
230 //qDebug()<<"ins"<<i<<this;
231 emit countChanged();
232 }
233 }
234 if (bLayoutChanged)
235 emit layoutChanged();
236}
