59/*
60 * A widget is a call-back driven screen entity like a push-button, window,
61 * text entry field, etc.
62 *
63 * Use the widget API to create widgets of various classes. You can nest them
64 * into trees of widgets to create complex user interfaces.
65 *
66 */

69/*
70 * XPWidgetID
71 *
72 * A Widget ID is an opaque unique non-zero handle identifying your widget.
73 * Use 0 to specify "no widget". This type is defined as wide enough to hold a
74 * pointer. You receive a widget ID when you create a new widget and then use
75 * that widget ID to further refer to the widget.
76 *
77 */
78typedef void * XPWidgetID;

80/*
81 * XPWidgetPropertyID
82 *
83 * Properties are values attached to instances of your widgets. A property is
84 * identified by a 32-bit ID and its value is the width of a pointer.
85 *
86 * Each widget instance may have a property or not have it. When you set a
87 * property on a widget for the first time, the property is added to the
88 * widget; it then stays there for the life of the widget.
89 *
90 * Some property IDs are predefined by the widget package; you can make up
91 * your own property IDs as well.
92 *
93 */
94enum {
95 /* A window's refcon is an opaque value used by client code to find other data *
96 * based on it. */
97 xpProperty_Refcon = 0
98
99 /* These properties are used by the utlities to implement dragging. */
100 ,xpProperty_Dragging = 1
101
102 ,xpProperty_DragXOff = 2
103
104 ,xpProperty_DragYOff = 3
105
106 /* Is the widget hilited? (For widgets that support this kind of thing.) */
107 ,xpProperty_Hilited = 4
108
109 /* Is there a C++ object attached to this widget? */
110 ,xpProperty_Object = 5
111
112 /* If this property is 1, the widget package will use OpenGL to restrict *
113 * drawing to the Wiget's exposed rectangle. */
114 ,xpProperty_Clip = 6
115
116 /* Is this widget enabled (for those that have a disabled state too)? */
117 ,xpProperty_Enabled = 7
118
119 /* NOTE: Property IDs 1 - 999 are reserved for the widget's library. *
120 * *
121 * NOTE: Property IDs 1000 - 9999 are allocated to the standard widget classes *
122 * provided with the library Properties 1000 - 1099 are for widget class 0, *
123 * 1100 - 1199 for widget class 1, etc. */
124 ,xpProperty_UserStart = 10000

127};

180/*
181 * XPDispatchMode
182 *
183 * The dispatching modes describe how the widgets library sends out messages.
184 * Currently there are three modes:
185 *
186 */
187enum {
188 /* The message will only be sent to the target widget. */
189 xpMode_Direct = 0
190
191 /* The message is sent to the target widget, then up the chain of parents *
192 * until the message is handled or a parentless widget is reached. */
193 ,xpMode_UpChain = 1
194
195 /* The message is sent to the target widget and then all of its children *
196 * recursively depth-first. */
197 ,xpMode_Recursive = 2
198
199 /* The message is snet just to the target, but goes to every callback, even if *
200 * it is handled. */
201 ,xpMode_DirectAllCallbacks = 3
202
203 /* The message is only sent to the very first handler even if it is not *
204 * accepted. (This is really only useful for some internal Widget Lib *
205 * functions. */
206 ,xpMode_Once = 4
207
208
209};
210typedef int XPDispatchMode;

236 *
237 * Widgets receive 32-bit messages indicating what action is to be taken or
238 * notifications of events. The list of messages may be expanded.

241enum {
242 /* No message, should not be sent. */

245 /* The create message is sent once per widget that is created with your widget *
246 * function and once for any widget that has your widget function attached. *
247 * *
248 * Dispatching: Direct *
249 * *
250 * Param 1: 1 if you are being added as a subclass, 0 if the widget is first *

254 /* The destroy message is sent once for each message that is destroyed that *
255 * has your widget function. *
256 * *

258 * *

260 * explicit deletion. */
261 ,xpMsg_Destroy = 2
262
263 /* The paint message is sent to your widget to draw itself. The paint message *
264 * is the bare-bones message; in response you must draw yourself, draw your *
265 * children, set up clipping and culling, check for visibility, etc. If you *
266 * don't want to do all of this, ignore the paint message and a draw message *
267 * (see below) will be sent to you. *
268 * *
269 * Dispatching: Direct */
270 ,xpMsg_Paint = 3
271
272 /* The draw message is sent to your widget when it is time to draw yourself. *
273 * OpenGL will be set up to draw in 2-d global screen coordinates, but you *
274 * should use the XPLM to set up OpenGL state. *
275 * *
276 * Dispatching: Direct */
277 ,xpMsg_Draw = 4
278
279 /* The key press message is sent once per key that is pressed. The first *
280 * parameter is the type of key code (integer or char) and the second is the *
281 * code itself. By handling this event, you consume the key stroke. *
282 * *
283 * Handling this message 'consumes' the keystroke; not handling it passes it *
284 * to your parent widget. *
285 * *
286 * Dispatching: Up Chain *
287 * *
288 * : Param 1: A pointer to an XPKeyState_t structure with the keystroke. */

290
291 /* Keyboard focus is being given to you. By handling this message you accept *
292 * keyboard focus. The first parameter will be one if a child of yours gave up *
293 * focus to you, 0 if someone set focus on you explicitly. *
294 * *
295 * : Handling this message accepts focus; not handling refuses focus. *
296 * *
297 * Dispatching: direct *

299 * Param 1: 1 if you are gaining focus because your child is giving it up, 0 *
300 * if someone is explicitly giving you focus. */
301 ,xpMsg_KeyTakeFocus = 6
302

304 * one if you are losing focus because another widget is taking it, or 0 if *

308 * *
309 * Param 1: 1 if focus is being taken by another widget, 0 if code requested *

311 ,xpMsg_KeyLoseFocus = 7
312
313 /* You receive one mousedown event per click with a mouse-state structure *
314 * pointed to by parameter 1, by accepting this you eat the click, otherwise *
315 * your parent gets it. You will not receive drag and mouse up messages if you *
316 * do not accept the down message. *
317 * *
318 * Handling this message consumes the mouse click, not handling it passes it *
319 * to the next widget. You can act 'transparent' as a window by never handling *
