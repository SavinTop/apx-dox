39QSvgRenderer *SvgImageProvider::loadRenderer(const QString &svgFile)
40{
41 if (svgFile.isNull())
42 return nullptr;
43 QString s = svgFile.trimmed();
44 while (s.contains("//"))
45 s.replace("//", "/");
46 if (s.size() < 5)
47 return nullptr;
48
49 QSvgRenderer *renderer = m_renderers.value(s);
50
51 if (!renderer) {
52 //qDebug()<<svgFile;
53 renderer = new QSvgRenderer(this);
54 connect(renderer, &QSvgRenderer::destroyed, this, [this, renderer]() {
55 m_renderers.remove(m_renderers.key(renderer));
56 });
57
58 // convert path to be relative to base
59 QString fn;
60 if (m_basePath.contains(':') && s.contains(':'))
61 s.remove(0, s.indexOf('/') + 1);
62 fn = m_basePath + "/"
63 + s; //QUrl::fromLocalFile(m_basePath).resolved(svgFile).toLocalFile();
64
65 if (!renderer->isValid()) {
66 renderer->load(fn);
67 }
68
69 if (!renderer->isValid()) {
70 apxConsoleW() << "Failed to load svg file:" << svgFile << fn;
71 delete renderer;
72 return nullptr;
73 }
74
75 m_renderers.insert(s, renderer);
76 }
77
78 return renderer;
79}

95QImage SvgImageProvider::requestImage(const QString &id, QSize *size, const QSize &requestedSize)
96{
97 //if(size->isNull())return QImage(1,1,QImage::Format_Mono);
98 //if(requestedSize.isNull())return QImage(1,1,QImage::Format_Mono);
99
100 QString svgFile = id;
101 QStringList params;
102
103 int sepPos = id.indexOf('?');
104 if (sepPos != -1) {
105 svgFile = id.left(sepPos);
106 params = id.mid(sepPos + 1).split('&');
107 }
108
109 QString element;
110 int hSlicesCount = 0;
111 int hSlice = 0;
112 int vSlicesCount = 0;
113 int vSlice = 0;
114 int border = 0;
115 QString color;
116 for (int i = 0; i < params.size(); ++i) {
117 const QString s = params.at(i);
118 int idel = s.indexOf('=');
119 if (idel <= 0)
120 continue;
121 const QString &sn = s.left(idel);
122 const QString &sv = s.mid(idel + 1);
123 if (sn == "e") {
124 element = sv;
125 } else if (sn == "hslice") {
126 QRegExp rx("(\\d+):(\\d+)");
127 if (rx.indexIn(sv) != -1) {
128 hSlice = rx.cap(1).toInt();
129 hSlicesCount = rx.cap(2).toInt();
130 }
131 } else if (sn == "vslice") {
132 QRegExp rx("(\\d+):(\\d+)");
133 if (rx.indexIn(sv) != -1) {
134 vSlice = rx.cap(1).toInt();
135 vSlicesCount = rx.cap(2).toInt();
136 }
137 } else if (sn == "border") {
138 QRegExp rx("(\\d+)");
139 if (rx.indexIn(sv) != -1) {
140 border = rx.cap(1).toInt();
141 }
142 } else if (sn == "color") {
143 color = sv;
144 }
145 }
146
147 if (size) {
148 *size = QSize();
149 }
150
151 QSvgRenderer *renderer = loadRenderer(svgFile);
152 if (!renderer) {
153 return QImage(1, 1, QImage::Format_Mono);
154 }
155 if (element.isEmpty()) {
156 renderer->deleteLater(); //don't cache
157 }
158
159 qreal xScale = 1.0;
160 qreal yScale = 1.0;
161
162 QSize docSize = renderer->defaultSize();
163
164 if (!requestedSize.isEmpty()) {
165 if (!element.isEmpty()) {
166 QSize rsz(requestedSize);
167 double sz; //=1024;
168 /*if(rsz.width()>sz||rsz.height()>sz)
169 rsz.scale(rsz/2,Qt::KeepAspectRatio);*/
170 sz = 2048;
171 if (rsz.width() > sz || rsz.height() > sz)
172 rsz.scale(QSizeF(sz, sz).toSize(), Qt::KeepAspectRatio);
173 QRectF elementBounds = renderer->boundsOnElement(element);
174 xScale = qreal(rsz.width()) / elementBounds.width();
175 yScale = qreal(rsz.height()) / elementBounds.height();
176 } /*else if (!docSize.isEmpty()) {
177 xScale = qreal(requestedSize.width()) / docSize.width();
178 yScale = qreal(requestedSize.height()) / docSize.height();
179 }*/
180 }
181
182 // keep the aspect ratio
183 xScale = yScale = qMin(xScale, yScale);
184
185 if (!element.isEmpty()) {
186 if (!renderer->elementExists(element)) {
187 apxConsoleW() << "invalid element:" << element << "of" << svgFile;
188 return QImage();
189 }
190
191 QRectF elementBounds = renderer->boundsOnElement(element);
192 int elementWidth = qRound(elementBounds.width() * xScale);
193 int elementHeigh = qRound(elementBounds.height() * yScale);
194 int w = elementWidth;
195 int h = elementHeigh;
196 int x = 0;
197 int y = 0;
198 if (elementWidth <= 0 || elementHeigh <= 0)
199 return QImage(1, 1, QImage::Format_ARGB32_Premultiplied);
200
201 if (hSlicesCount > 1) {
202 x = (w * hSlice) / hSlicesCount;
203 w = (w * (hSlice + 1)) / hSlicesCount - x;
204 }
205
206 if (vSlicesCount > 1) {
207 y = (h * (vSlice)) / vSlicesCount;
208 h = (h * (vSlice + 1)) / vSlicesCount - y;
209 }
210
211 QImage img(w + border * 2, h + border * 2, QImage::Format_ARGB32_Premultiplied);
212 if (!img.isNull()) {
213 img.fill(0);
214 QPainter p(&img);
215 if (p.isActive()) {
216 p.setRenderHints(QPainter::TextAntialiasing | QPainter::Antialiasing
217 | QPainter::SmoothPixmapTransform);
218
219 p.translate(-x + border, -y + border);
220 QRectF rElement(0, 0, elementBounds.width(), elementBounds.height());
221 p.scale(w / rElement.width(), h / rElement.height());
222 renderer->render(&p, element, rElement);
223 }
224 }
225 if (size) {
226 *size = QSize(w, h);
227 }
228 //if(img.size().isNull()) return QImage(1,1,QImage::Format_ARGB32_Premultiplied);
229
230 //img.save("/tmp/img/" + element + params.join('.') + ".png");
231 //qDebug() << img.size();
232 return img;
233 }
234
235 // render the whole svg file
236 int w = qRound(docSize.width() * xScale);
237 int h = qRound(docSize.height() * yScale);
238
239 QImage img(w, h, QImage::Format_ARGB32_Premultiplied);
240 if (!img.isNull()) {
241 img.fill(0);
242 QPainter p(&img);
243 if (p.isActive()) {
244 p.setRenderHints(QPainter::TextAntialiasing | QPainter::Antialiasing
245 | QPainter::SmoothPixmapTransform);
246
247 p.scale(xScale, yScale);
248 renderer->render(&p, QRectF(QPointF(), QSizeF(docSize)));
249 }
250 }
251 if (size) {
252 *size = QSize(w, h);
253 }
254 return img;
255}
