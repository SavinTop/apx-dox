38QIcon MaterialIcon::icon(const QString &name, const QColor &color) const
39{
40 QString c = getChar(name);

42 return QIcon();
43 }
44 QFontIconEngine *engine = new QFontIconEngine;
45 engine->setFontFamily("Material Design Icons");
46 engine->setLetter(c);
47 engine->setBaseColor(color);

51QString MaterialIcon::getChar(const QString &name)

65void MaterialIcon::updateMap()
66{
67 QFile res;
68 res.setFileName(":/icons/material-icons.json");
69 if (res.open(QFile::ReadOnly | QFile::Text)) {
70 QJsonDocument json = QJsonDocument::fromJson(res.readAll());
71 res.close();
72 QJsonObject obj = json.object();
73 for (auto v = obj.constBegin(); v != obj.constEnd(); ++v) {
74 QString s = v.value().toVariant().toString(); //.toString();
75 QString c = 0;
76 if (s.size() == 1) {
77 c = s.at(0);
78 } else if (s.startsWith("\\u")) {
79 std::wstring str = s.toStdWString();
80 s = QString::fromStdWString(str);
81 if (s.size() == 1)
82 c = s.at(0);
83 } else if (s.startsWith("\\", Qt::CaseInsensitive)) {
84 uint code = s.mid(1).toUInt(nullptr, 16);
85 c = QString::fromUcs4(&code, 1);
86 }
87 if (c == '\0') {
88 //qWarning() << v.key() << s << v.value() << s.size();
89 continue;
90 }
91 map[v.key()] = c;
92 }
93 }
94 if (map.isEmpty())
95 map.insert("", QChar('\0'));
96}

101void QFontIconEngine::paint(QPainter *painter,
102 const QRect &rect,
103 QIcon::Mode mode,
104 QIcon::State state)
105{
106 Q_UNUSED(state)
107
108 QFont font = QFont(mFontFamily);
109 int drawSize = rect.height(); //qRound(rect.height() * 0.8);
110 font.setPixelSize(drawSize < 5 ? 5 : drawSize);
111
112 QColor penColor;
113 if (!mBaseColor.isValid())
114 penColor = QApplication::palette("QWidget").color(QPalette::Normal, QPalette::ButtonText);
115 else
116 penColor = mBaseColor;
117
118 if (mode == QIcon::Disabled)
119 penColor = QApplication::palette("QWidget").color(QPalette::Disabled, QPalette::ButtonText);
120
121 if (mode == QIcon::Selected)
122 penColor = QApplication::palette("QWidget").color(QPalette::Active, QPalette::ButtonText);
123
124 painter->save();
125 painter->setPen(QPen(penColor));
126 painter->setFont(font);
127 painter->drawText(rect, Qt::AlignCenter | Qt::AlignVCenter, mLetter);
128
129 painter->restore();
130}
