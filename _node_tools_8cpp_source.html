118Fact *NodeTools::addCommand(Fact *group, QString name, QString title, xbus::node::usr::cmd_t cmd)
119{
120 //qDebug() << node->title() << group << name << title << cmd;
121
122 NodeToolsGroup *g = f_usr;
123 if (group->name() == f_sys->name()) {
124 g = f_sys;
125 group = node;
126 }
127 Fact *f = g->addCommand(group, name, title, cmd);
128 if (!f)
129 return f;
130
131 connect(f, &Fact::triggered, this, [this, f]() { execUsr(f); });
132
133 return f;
134}

142void NodeTools::execUsr(Fact *f)
143{
144 QString msg = tr("User command").append(": ").append(f->title());
145 if (!f->descr().isEmpty())
146 msg.append(QString(" (%1)").arg(f->descr()));
147
148 if (!node->protocol()) {
149 msg.append(QString(" (%1)").arg(tr("ignored")));
150 node->message(msg);
151 return;
152 }
153
154 node->message(msg);
155 node->protocol()->requestUsr(static_cast<quint8>(f->property("cmd").toUInt()), QByteArray());
156}
