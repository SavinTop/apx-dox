55void AppGcs::openFile(AppGcs::FileType type)
56{
57 QString title;
58 QString ftype;
59 QDir defaultDir;
60 switch (type) {
61 default:
62 title = tr("Import");
63 defaultDir = AppDirs::user();
64 break;
65 case TelemetryFile:
66 title = tr("Import telemetry data");
67 ftype = "telemetry";
68 defaultDir = AppDirs::user();
69 break;
70 case ConfigFile:
71 title = tr("Import configuration");
72 ftype = "nodes";
73 defaultDir = AppDirs::configs();
74 break;
75 case FirmwareFile:
76 title = tr("Firmware");
77 ftype = "apxfw";
78 defaultDir = AppDirs::user();
79 break;
80 }
81 QString settingName = QString("SharePath_%1")
82 .arg(QMetaEnum::fromType<FileType>().valueToKey(type));
83 if (!defaultDir.exists())
84 defaultDir.mkpath(".");
85 QDir dir = QDir(QSettings().value(settingName, defaultDir.canonicalPath()).toString());
86 if (!dir.exists())
87 dir = defaultDir;
88 QFileDialog dlg(nullptr, title, dir.canonicalPath());
89 dlg.setAcceptMode(QFileDialog::AcceptOpen);
90 dlg.setFileMode(QFileDialog::ExistingFiles);
91 dlg.setViewMode(QFileDialog::Detail);
92 QStringList filters;
93 filters << tr("All supported types") + " (*.telemetry,*.nodes,*.apxfw)"
94 << tr("Telemetry") + " (*.telemetry)" << tr("Configuration") + " (*.nodes)"
95 << tr("Firmware") + " (*.apxfw)" << tr("Any files") + " (*)";
96 dlg.setNameFilters(filters);
97 dlg.setDefaultSuffix(ftype);
98 if (!ftype.isEmpty())
99 dlg.selectNameFilter("*." + ftype);
100 if (!(dlg.exec() && dlg.selectedFiles().size() >= 1))
101 return;
102 QSettings().setValue(settingName, dlg.directory().absolutePath());
103
104 foreach (QString fname, dlg.selectedFiles()) {
105 emit fileOpenRequest(fname);
106 }
107}
