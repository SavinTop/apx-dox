64void PTrace::end(size_t sz)
65{
66 if (!_enabled)
67 return;
68
69 if (_blocks.isEmpty())
70 return;
71
72 if (_blocks.first() == ">") {
73 // re-arrange uplink nested blocks
74 for (int i = _blocks.lastIndexOf(">"); i > 0; i = _blocks.lastIndexOf(">")) {
75 QStringList tail = _blocks.mid(i);
76 tail.append("+");
77 tail.append(_blocks.mid(1, i - 1));
78 _blocks = tail;
79 }
80 if (sz > 0) {
81 _blocks[0] = QString(">%1").arg(sz);
82 }
83 }
84
85 // qDebug() << _blocks;
86 qInfo(&log) << _blocks;
87 // QMessageLogger(0, 0, 0).info(&log) << _blocks;
88
89 emit packet(_blocks);
90 reset();
91}

118void PTrace::data(QByteArray data)
119{
120 if (!_enabled)
121 return;
122
123 if (data.isEmpty())
124 return;
125
126 QString s;
127
128 if (data.size() > 16) {
129 s = QString("[%1]%2...").arg(data.size()).arg(QString(data.left(16).toHex().toUpper()));
130 } else if (data.size() > 2) {
131 s = QString("[%1]%2").arg(data.size()).arg(QString(data.toHex().toUpper()));
132 } else {
133 s = QString("%1").arg(QString(data.toHex().toUpper()));
134 }
135
136 block(s);
137}
