37Mandala::Mandala(Fact *parent)
38 : Fact(parent,
39 "mandala",
40 "Mandala",
41 tr("Vehicle data tree"),
42 Group | FilterModel | ModifiedGroup,
43 "hexagon-multiple")
44{
45 setMandala(this);
46 qmlRegisterUncreatableType<MandalaFact>("APX.Facts", 1, 0, "MandalaFact", "Reference only");
47
48 connect(this, &Mandala::sendValue, this, &Mandala::recordSendValue);
49
50 Fact *group = this;
51 uint8_t level = 0;
52 QString sect;
53 for (size_t i = 0; i < (sizeof(mandala::meta) / sizeof(*mandala::meta)); ++i) {
54 const mandala::meta_s &d = mandala::meta[i];
55 if (d.group) {
56 //move group to upper level
57 for (; level > d.level; --level)
58 if (level != 2)
59 group = group->parentFact();
60 level = d.level + 1;
61 if (d.level == 1) {
62 sect = d.title;
63 continue;
64 }
65 if (group->child(d.name)) {
66 apxMsgW() << "dup group:" << group->child(d.name)->path(1);
67 }
68 MandalaFact *f = new MandalaFact(this, group, d);
69 _uid_map.insert(f->uid(), f);
70 group = f;
71 if (d.level == 2)
72 group->setSection(sect);
73 continue;
74 }
75 if (group->child(d.name)) {
76 apxMsgW() << "dup fact:" << group->child(d.name)->path(2);
77 }
78 MandalaFact *f = new MandalaFact(this, group, d);
79 _uid_map.insert(f->uid(), f);
80 }
81
82 // fill value facts list
83 for (auto f : _uid_map.values()) {
84 if (f->isSystem() || f->isGroup())
85 continue;
86 _valueFacts.append(f);
87 }
88
89 _total = _valueFacts.size();
90}

109MandalaFact *Mandala::fact(mandala::uid_t uid) const
110{
111 if (uid == 0xFFFF || mandala::is_bundle(uid))
112 return nullptr;
113 MandalaFact *f = _uid_map.value(uid);
114 if (f)
115 return f;
116 apxMsgW() << "Mandala uid not found:" << uid;
117 return nullptr;

120MandalaFact *Mandala::fact(const QString &mpath, bool silent) const
121{
122 MandalaFact *f = nullptr;
123 if (mpath.isEmpty())
124 return f;
125 if (mpath.contains('.')) {
126 f = static_cast<MandalaFact *>(findChild(mpath));
127 }
128 if (!f && !silent) {
129 apxMsgW() << "Mandala fact not found:" << mpath;
130 }
131 return f;
