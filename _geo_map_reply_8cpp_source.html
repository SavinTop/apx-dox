28static const unsigned char pngSignature[] = {0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A, 0x00};
29static const unsigned char jpegSignature[] = {0xFF, 0xD8, 0xFF, 0x00};

32GeoMapReply::GeoMapReply(const QGeoTileSpec &spec, GeoTileFetcher *fetcher)
33 : QGeoTiledMapReply(spec, fetcher)
34{
35 TileLoader *loader = TileLoader::instance();
36 if (!loader) {
37 setError(QGeoTiledMapReply::UnknownError, "no loader");
38 setFinished(true);
39 return;
40 }
41
42 _uid = TileLoader::uid(spec.mapId(), spec.zoom(), spec.x(), spec.y());
43 connect(loader, &TileLoader::tileLoaded, this, &GeoMapReply::tileLoaded);
44 connect(loader, &TileLoader::tileError, this, &GeoMapReply::tileError);
45 loader->loadTile(_uid);
46}

54void GeoMapReply::tileLoaded(quint64 uid, QByteArray data)
55{
56 if (uid != _uid)
57 return;
58 setMapImageData(data);
59 QString format = getImageFormat(data);
60 if (!format.isEmpty()) {
61 setMapImageFormat(format);
62 }
63 setFinished(true);
64}
65void GeoMapReply::tileError(quint64 uid, QString errorString)
66{
67 if (uid != _uid)
68 return;
69 setError(QGeoTiledMapReply::CommunicationError, errorString);
70 setFinished(true);
71}

79QString GeoMapReply::getImageFormat(const QByteArray &image)
80{
81 QString format;
82 if (image.size() > 2) {
83 if (image.startsWith(reinterpret_cast<const char *>(pngSignature)))
84 format = "png";
85 else if (image.startsWith(reinterpret_cast<const char *>(jpegSignature)))
86 format = "jpg";
87 else if (image.startsWith(reinterpret_cast<const char *>(gifSignature)))
88 format = "gif";
89 else {
90 apxConsoleW() << "Unknown image signature";
91 format = "png";
92 }
93 }
94 return format;
95}
