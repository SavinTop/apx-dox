52bool VehicleShare::importRequest(QString format, QString fileName)
53{
54 auto var = parseJsonDocument(loadData(fileName));
55 if (var.isNull())
56 return false;
57
58 _vehicle->storage()->importVehicleConfig(var.value<QVariantMap>());
59
60 if (QFileInfo(fileName).absoluteDir().absolutePath() == _templatesDir.absolutePath()) {
61 // imported
62 return true;
63 }
64
65 _vehicle->fromVariant(var);
66 _imported(fileName);
67 return true;
68}

75void VehicleShare::syncTemplates()
76{
77 auto format = _importFormats.first();
78 QSettings sx;
79 sx.beginGroup("templates_update");
80 auto importedFiles = sx.value(format).toMap();
81
82 bool updated = false;
83 for (auto fi : _templatesDir.entryInfoList()) {
84 auto name = fi.completeBaseName();
85 auto t_res = fi.lastModified().toMSecsSinceEpoch();
86 if (importedFiles.contains(name)) {
87 auto t_imp = importedFiles.value(name).toULongLong();
88 if (t_res <= t_imp)
89 continue;
90 }
91
92 qDebug() << "template update:" << fi.fileName();
93 if (!importRequest(format, fi.absoluteFilePath()))
94 continue;
95
96 importedFiles.insert(name, t_res);
97 updated = true;
98 }
99 if (updated) {
100 sx.setValue(format, importedFiles);
101 }
102}
