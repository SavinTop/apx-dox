24PVehicle::PVehicle(PBase *parent, QString callsign, QString uid, VehicleType type)

54void PVehicle::setStreamType(StreamType type)
55{
56 if (type != OFFLINE) {
57 onlineTimer.start();
58
59 switch (type) {
60 default:
61 break;
62 case TELEMETRY:
63 time_telemetry.start();
64 break;
65 case XPDR:
66 time_xpdr.start();
67 break;
68
69 case NMT:
70 case DATA:
71 if (time_telemetry.elapsed() < 2000 || time_xpdr.elapsed() < 3000)
72 return;
73 break;
74 }
75 }
76 if (m_streamType == type)
77 return;
78
79 m_streamType = type;
80 emit streamTypeChanged();
81}
