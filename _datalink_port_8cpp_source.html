254void DatalinkPort::updateRoutingValue()
255{
256 if (_blockUpdateRoutingValue)
257 return;
258 _blockUpdateRoutingFacts = true;
259 QStringList rx, tx;
260 for (int i = 0; i < f_rx.size(); ++i) {
261 if (f_rx.at(i)->value().toBool())
262 rx.append(f_rx.at(i)->title());
263 }
264 for (int i = 0; i < f_tx.size(); ++i) {
265 if (f_tx.at(i)->value().toBool())
266 tx.append(f_tx.at(i)->title());
267 }
268 f_routing->setValue(QString("%1:%2").arg(rx.join(',')).arg(tx.join(',')));
269 _blockUpdateRoutingFacts = false;
270}
271void DatalinkPort::updateRoutingFacts()
272{
273 if (_blockUpdateRoutingFacts)
274 return;
275 _blockUpdateRoutingValue = true;
276 QString v = f_routing->value().toString();
277 QStringList rx = v.left(v.indexOf(':')).split(',', Qt::SkipEmptyParts);
278 for (int i = 0; i < f_rx.size(); ++i) {
279 f_rx.at(i)->setValue(rx.contains(f_rx.at(i)->title()));
280 }
281 QStringList tx = v.mid(v.indexOf(':') + 1).split(',', Qt::SkipEmptyParts);
282 for (int i = 0; i < f_rx.size(); ++i) {
283 f_tx.at(i)->setValue(tx.contains(f_tx.at(i)->title()));
284 }
285 _blockUpdateRoutingValue = false;
286}
287void DatalinkPort::updateRoutingStatus()
288{
289 QString v = f_routing->value().toString();
290 QString rx = v.left(v.indexOf(':'));
291 QString tx = v.mid(v.indexOf(':') + 1);
292 if (rx.isEmpty())
293 rx = "NO";
294 if (tx.isEmpty())
295 tx = "NO";
296 f_routing->setDescr(QString("RX: %1 / TX: %2").arg(rx).arg(tx));
297}
298void DatalinkPort::updateConnectionNetwork()
299{
300 if (!f_connection)
301 return;
302 const QMetaEnum &m = QMetaEnum::fromType<Datalink::NetworkMask>();
303 QString v = f_routing->value().toString();
304 quint16 rxNetwork = 0;
305 foreach (QString s, v.left(v.indexOf(':')).split(',', Qt::SkipEmptyParts)) {
306 rxNetwork |= m.keyToValue(s.toUtf8());
307 }
308 quint16 txNetwork = 0;
309 foreach (QString s, v.mid(v.indexOf(':') + 1).split(',', Qt::SkipEmptyParts)) {
310 txNetwork |= m.keyToValue(s.toUtf8());
311 }
312 f_connection->setRxNetwork(rxNetwork);
313 f_connection->setTxNetwork(txNetwork);
314}

316void DatalinkPort::syncUrlEnum()
317{
318 QStringList st;
319 switch (f_type->value().toInt()) {
320 case SERIAL: {
321 st << "auto";
322 foreach (QSerialPortInfo spi, QSerialPortInfo::availablePorts()) {
323 if (st.contains(spi.portName()))
324 continue;
325 st.append(spi.portName());
326 }
327 } break;
328 case TCP: {
329 const Fact *fg = static_cast<const Fact *>(sender());
330 for (int i = 0; i < fg->size(); ++i) {
331 const Fact *f = fg->child(i);
332 if (st.contains(f->title()))
333 continue;
334 st.append(f->title());
335 }
336 } break;
337 }
338 f_url->setEnumStrings(st);
339 if (f_url->text().isEmpty() && (!st.isEmpty()))
340 f_url->setValue(st.first());
