68void AppQuickView::saveStateDo()
69{
70 if (visibility() == QWindow::Minimized)
71 return;
72 if (visibility() == QWindow::Hidden)
73 return;
74
75 bool fullscreen = visibility() == QWindow::FullScreen || visibility() == QWindow::Maximized
76 || width() == screen()->availableSize().width()
77 || height() == screen()->availableSize().height();
78
79 QSettings s;
80 s.beginGroup("windows");
81 if (!fullscreen)
82 s.setValue(objectName() + "_geometry", geometry());
83 s.setValue(objectName() + "_maximized", fullscreen);
84}

86void AppQuickView::restoreState()
87{
88 QSettings s;
89 s.beginGroup("windows");
90 setGeometry(s.value(objectName() + "_geometry", geometry()).toRect());
91 if (s.value(objectName() + "_maximized").toBool()) {
92#if defined(Q_OS_MAC)
93 showFullScreen();
94#else
95 showMaximized();
96#endif
97 } else {
98 showNormal();
99 }
100
101 raise();
102}

105{
106 if (event->type() == QEvent::Close) {
