75QGeoPath Waypoint::getPath()
76{
77 QGeoPath p;
78
79 double spd = 0; //QMandala::instance()->current->apcfg.value("spd_cruise").toDouble();
80 /*if(f_speed->value().toUInt()>0)
81 spd=f_speed->value().toUInt();*/
82 if (spd <= 0)
83 spd = 22;
84 double dt = 1.0;
85 double turnR = 0; //QMandala::instance()->current->apcfg.value("turnR").toDouble();
86 if (turnR <= 0)
87 turnR = 100;
88 double turnRate = (360.0 / (2.0 * M_PI)) * spd / turnR;
89 double crs = m_bearing;
90 double distance = 0;
91 MissionItem *prev = prevItem();
92 QGeoCoordinate dest(coordinate());
93 QGeoCoordinate pt;
94 bool wptReached = true;
95 bool wptWarning = false;
96 bool wptLine = false;
97 while (1) {
98 if (!prev) {
99 pt = group->mission->startPoint();
100 if (!pt.isValid()) {
101 crs = 0;
102 //pt=QGeoCoordinate(vm->factById(idx_home_pos|(0<<8))->value().toDouble(),vm->factById(idx_home_pos|(1<<8))->value().toDouble());
103 //p.addCoordinate(dest);
104 p.addCoordinate(dest);
105 //crs=dest.azimuthTo(next->coordinate());
106 break;
107 }
108 p.addCoordinate(pt);
109 crs = group->mission->startHeading();
110 double slen = group->mission->startLength();
111 if (slen > 0) {
112 pt = pt.atDistanceAndAzimuth(slen, crs);
113 p.addCoordinate(pt);
114 distance += slen;
115 }
116 } else {
117 pt = prev->coordinate();
118 if (prev->geoPath().path().size() > 1) {
119 crs = prev->bearing();
120 wptLine = f_type->text().toLower() == "track";
121 } else
122 wptLine = true;
123 }
124 //fly to wpt
125 p.addCoordinate(pt);
126 //int cnt=0;
127 double turnCnt = 0;
128 while (1) {
129 double deltaHdg = AppRoot::angle(pt.azimuthTo(dest) - crs);
130 double deltaDist = pt.distanceTo(dest);
131 double step = dt * spd;
132 if (wptLine || std::abs(deltaHdg) < (dt * 10.0)) {
133 //crs ok (turn finished)
134 step = 10.0e+3 * dt;
135 crs += deltaHdg;
136 deltaHdg = 0;
137 if (deltaDist <= step) {
138 //wpt reached
139 crs += deltaHdg;
140 deltaHdg = 0;
141 distance += deltaDist;
142 pt = dest;
143 p.addCoordinate(dest);
144 wptReached = true;
145 break;
146 }
147 }
148 //propagate position
149 pt = pt.atDistanceAndAzimuth(step, crs);
150 distance += step;
151 deltaHdg = dt * AppRoot::limit(deltaHdg, -turnRate, turnRate);
152 crs += deltaHdg;
153 p.addCoordinate(pt);
154 turnCnt += deltaHdg;
155 if (std::abs(turnCnt) > (360 * 2)) { //(++cnt)>(360/turnRate)){
156 wptReached = false;
157 break;
158 }
159 }
160 if (p.path().size() < 2)
161 p.addCoordinate(dest);
162 //qDebug()<<plist;
163 break;
164 }
165
166 //update properties
167 wptWarning |= distance < turnR * (2.0 * M_PI * 0.8);
168
169 setReachable(wptReached);
170 setWarning(wptWarning);
171
172 setDistance(distance);
173 setTime(distance / spd);
174
175 //end bearing
176 if (p.path().size() == 2 && crs == m_bearing) {
177 crs = p.path().first().azimuthTo(p.path().at(1));
178 }
179 crs = AppRoot::angle(crs);
180 int icrs = (int) (crs / 10) * 10;
181 if (m_bearing != icrs) {
182 m_bearing = icrs;
183 //force next item to be updated
184 MissionItem *next = nextItem();
185 if (next)
186 next->resetPath();
187 }
188 setBearing(crs);
189
190 return p;
191}
