56void Notifications::updateItems()
57{
58 for (int i = 0; i < newItems.size(); ++i) {
59 Fact *fact = newItems.at(i);
60 if (!fact)
61 continue;
62 //check parents
63 bool found = false;
64 for (Fact *f = fact->parentFact(); f; f = f->parentFact()) {
65 if (newItems.contains(f) || items.contains(f)) {
66 found = true;
67 break;
68 }
69 }
70 if (!found)
71 createItem(fact);
72 }
73 newItems.clear();
74}
