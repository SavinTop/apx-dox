31QPointF KmlPolygonsModel::setPolygons(const QList<KmlPolygon> &kmlPolygons)
32{
33 m_allPolygons.clear();
34
35 QPolygonF allPoints;
36 for (auto &p : kmlPolygons) {
37 QPolygonF polygon = toPolygon(p.data);
38 allPoints.append(polygon);
39
40 KmlPolygonExtended kmlPolygonExtended;
41 kmlPolygonExtended.kmlPolygon = p;
42 kmlPolygonExtended.polygon = polygon;
43 m_allPolygons.append(kmlPolygonExtended);
44 }
45
46 auto center = std::accumulate(allPoints.begin(), allPoints.end(), QPointF(0, 0))
47 / allPoints.size();
48
49 beginResetModel();
50 m_viewPolygons.clear();
51 endResetModel();
52 updateViewPolygons();
53
54 return center;
55}

75QVariant KmlPolygonsModel::data(const QModelIndex &index, int role) const
76{
77 QVariant result;
78 int row = index.row();
79 if (row >= 0 && row < m_viewPolygons.size()) {
80 if (role == Polygon) {
81 result = QVariant::fromValue(m_viewPolygons[row].kmlPolygon.data);
82 } else if (role == Color) {
83 result = m_viewPolygons[row].kmlPolygon.color;
84 }
85 }
86 return result;
87}

95void KmlPolygonsModel::updateViewPolygons()
96{
97 for (int i = 0; i < m_viewPolygons.size(); i++) {
98 if (!m_viewPolygons[i].polygon.intersects(m_bb)) {
99 beginRemoveRows(QModelIndex(), i, i);
100 m_viewPolygons.removeAt(i);
101 endRemoveRows();
102 }
103 }
104
105 for (auto &p : m_allPolygons) {
106 if (!m_viewPolygons.contains(p) && p.polygon.intersects(m_bb)) {
107 int row = m_viewPolygons.size();
108 beginInsertRows(QModelIndex(), row, row);
109 m_viewPolygons.append(p);
110 endInsertRows();
111 }
112 }
113}

115QPolygonF KmlPolygonsModel::toPolygon(const QGeoPolygon &geoPolygon)
116{
117 QPolygonF polygon;
118 for (int i = 0; i < geoPolygon.size(); i++)
119 polygon.append(
120 QPointF(geoPolygon.coordinateAt(i).latitude(), geoPolygon.coordinateAt(i).longitude()));
121 return polygon;
122}
