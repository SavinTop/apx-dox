137bool AppDirs::copyPath(QString sourceDir, QString destinationDir)
138{
139 QFileInfo srcInfo(sourceDir);
140 QFileInfo destInfo(destinationDir);
141
142 if (srcInfo.isFile()) {
143 if (destInfo.isDir()) {
144 destInfo = QFileInfo(destInfo.dir().filePath(srcInfo.fileName()));
145 }
146 if (destInfo.exists()) {
147 if (destInfo.lastModified() == srcInfo.lastModified())
148 return false;
149 QFile::remove(destInfo.absoluteFilePath());
150 } else {
151 if (!destInfo.dir().exists()) {
152 destInfo.dir().mkpath(".");
153 }
154 }
155 QFile::copy(srcInfo.absoluteFilePath(), destInfo.absoluteFilePath());
156 return true;
157 }
158
159 bool rv = false;
160
161 QDir originDirectory(sourceDir);
162
163 if (!originDirectory.exists()) {
164 return rv;
165 }
166
167 QDir destinationDirectory(destinationDir);
168
169 /*if(destinationDirectory.exists() && overWriteDirectory)
170 {
171 destinationDirectory.removeRecursively();
172 }*/
173
174 if (!destinationDirectory.exists()) {
175 destinationDirectory.mkpath(".");
176 rv = true;
177 }
178
179 foreach (QString directoryName, originDirectory.entryList(QDir::Dirs | QDir::NoDotAndDotDot)) {
180 QString destinationPath = destinationDir + "/" + directoryName;
181 //destinationDirectory.mkpath(directoryName);
182 copyPath(sourceDir + "/" + directoryName, destinationPath);
183 }
184
185 foreach (QString fileName, originDirectory.entryList(QDir::Files)) {
186 QFileInfo dest(destinationDir + "/" + fileName);
187 QFileInfo src(sourceDir + "/" + fileName);
188 if (dest.exists()) {
189 if (dest.lastModified() == src.lastModified())
190 continue;
191 QFile::remove(dest.absoluteFilePath());
192 }
193 rv = true;
194 QFile::copy(src.absoluteFilePath(), dest.absoluteFilePath());
195 }
196
198 QDir finalDestination(destinationDir);
199 finalDestination.refresh();
200
201 return rv;
202}
