33TelemetryShare::TelemetryShare(Telemetry *telemetry, Fact *parent, Flags flags)
34 : Share(parent,
35 "telemetry",
36 tr("Telemetry data"),
37 QStandardPaths::writableLocation(QStandardPaths::DownloadLocation),
38 flags)
39 , _telemetry(telemetry)
40{
41 QSettings sx;
42 if (sx.contains("DefaultExportPath")) {
43 _defaultDir.setPath(
44 sx.value(QString("ShareExportPath_%1").arg(_exportFormats.first())).toString());
45 }
46
47 _exportFormats << "csv";
48
49 QString sect = tr("Queue");
50 qimp = new QueueJob(this, "qimp", tr("Import queue"), "", new TelemetryImport());
51 qimp->setIcon("import");
52 qimp->setSection(sect);
53 connect(qimp, &Fact::progressChanged, this, &TelemetryShare::updateProgress);
54 connect(qimp, &Fact::valueChanged, this, &TelemetryShare::updateStatus);
55
56 connect(qimp, &QueueJob::finished, this, [this](Fact *, QVariantMap result) {
57 if (result.value("key").toULongLong()) {
58 emit imported("", result.value("title").toString());
59 }
60 });
61 connect(qimp, &QueueJob::jobDone, this, [this](QVariantMap latestResult) {
62 emit importJobDone(latestResult.value("key").toULongLong());
63 });
64
65 qexp = new QueueJob(this, "qexp", tr("Export queue"), "", new TelemetryExport());
66 qexp->setIcon("export");
67 qexp->setSection(sect);
68 connect(qexp, &Fact::progressChanged, this, &TelemetryShare::updateProgress);
69
70 f_stop = new Fact(this, "stop", tr("Stop"), tr("Stop conversion"), Action | Stop);
71 connect(f_stop, &Fact::triggered, qimp, &QueueJob::stop);
72 connect(f_stop, &Fact::triggered, qexp, &QueueJob::stop);
73
74 connect(telemetry->f_lookup,
75 &LookupTelemetry::recordIdChanged,
76 this,
77 &TelemetryShare::updateActions);
78
79 descr_s = descr();
80 updateProgress();
81 updateStatus();
82 updateActions();
83}

85QString TelemetryShare::getDefaultTitle()
86{
87 QVariantMap info = _telemetry->f_lookup->recordInfo();
88 QString fname = QDateTime::fromMSecsSinceEpoch(_telemetry->f_lookup->recordTimestamp())
89 .toString("yyyy_MM_dd_hh_mm_ss_zzz");
90 QString callsign = info.value("callsign").toString();
91 if (!callsign.isEmpty())
92 fname.append("-").append(callsign);
93 return fname;
94}
95bool TelemetryShare::exportRequest(QString format, QString fileName)
96{
97 quint64 key = _telemetry->f_lookup->recordId();
98 if (!key) {
99 apxMsgW() << tr("Missing data in database");
100 return false;
101 }
102 auto fi = QFileInfo(fileName);
103
104 QSettings().setValue(QString("ShareExportPath_%1").arg(_exportFormats.first()),
105 fi.dir().absolutePath());
106
107 //add to queue
108 auto title = fi.completeBaseName();
109 Fact *f = new Fact(nullptr, title, title, fileName);
110 f->setValue(key);
111 f->setParentFact(qexp);
112
113 return true;
114}
115bool TelemetryShare::importRequest(QString format, QString fileName)
116{
117 //add to queue
118 auto title = QFileInfo(fileName).completeBaseName();
119 new Fact(qimp, title, title, fileName);
120 return true;
121}

142void TelemetryShare::updateDescr()
143{
144 QString s;
145 if (qimp->progress() >= 0)
146 s = tr("Importing");
147 else if (qexp->progress() >= 0)
148 s = tr("Exporting");
149 if (s.isEmpty())
150 setDescr(descr_s);
151 else
152 setDescr(s.append(QString("... %1").arg(value().toString())));
153}
