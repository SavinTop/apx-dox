27PNode::PNode(PNodes *parent, QString uid)
28 : PTreeBase(parent, "node#", "node", "", Group)
29 , m_uid(uid)
30{
31 auto p = findParent<PBase>()->firmware();
32 if (p) {
33 connect(p, &PFirmware::upgradeStarted, this, &PNode::upgradeStarted);
34 connect(p, &PFirmware::upgradeFinished, this, &PNode::upgradeFinished);
35
36 _upgradingDoneTimer.setSingleShot(true);
37 _upgradingDoneTimer.setInterval(1000);
38 connect(&_upgradingDoneTimer, &QTimer::timeout, this, [this]() { setUpgrading(false); });
39 } else {
40 qWarning() << "missing PApxFirmware";
41 }
42}

44void PNode::setUpgrading(bool v)
45{
46 _upgradingDoneTimer.stop();

48 return;
49 m_upgrading = v;
50 setActive(v);
51 emit upgradingChanged();
