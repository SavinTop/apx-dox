87void AppLog::add(const QString &categoryName, const QString &fileName, bool silent)
88{
89 QDir dir(QFileInfo(AppDirs::logs().absoluteFilePath(fileName)).absoluteDir());
90 if (!dir.exists())
91 dir.mkpath(".");
92
93 QFile *f = new QFile(AppDirs::logs().absoluteFilePath(fileName), AppLog::_instance);
94 if (!f->open(QFile::WriteOnly | QFile::Truncate)) {
95 apxMsgW() << "Cant write log file" << fileName;
96 f->deleteLater();
97 return;
98 }
99 QTextStream *stream = new QTextStream(f);
100 QMutexLocker lock(&AppLog::_mutex);
101 AppLog::_instance->streams.insert(categoryName, stream);
102 if (silent)
103 AppLog::_instance->silentCategories.append(categoryName);
104}

106void AppLog::message(QtMsgType type, const QMessageLogContext &context, const QString &message)
107{
108 if (!silentCategories.contains(context.category))
109 messageHandlerChain(type, context, message);
110
111 QMutexLocker lock(&_mutex);
112
113 QString msg = qFormatLogMessage(type, context, message);
114 if (!msg.endsWith('\n'))
115 msg.append('\n');
116
117 QTextStream *stream = streams.value(context.category);
118 if (stream) {
119 *stream << qPrintable(msg);
120 stream->flush();
121 } else if (appLogStream) {
122 *appLogStream << qPrintable(msg);
123 appLogStream->flush();
124 }
125 if (silentCategories.contains(context.category))
126 return;
127
128 emit consoleMessage(message);
129
130 if (display(context)) {
131 switch (type) {
132 default:
133 emit infoMessage(message);
134 break;
135 case QtInfoMsg:
136 emit infoMessage(message);
137 break;
138 case QtWarningMsg:
139 emit warningMessage(message);
140 break;
141 }
142 }
143}
