43void DatabaseWorker::run()
44{
45 bool bEmpty = true;
46 uint qcnt = 0;
47 QSqlQuery query(db->sql);
48 query.setForwardOnly(true);
49
50 while (!isInterruptionRequested()) {
51 //wait for requests and update info every 1 sec
52 if (queueSize() == 0) {
53 queueMutex.lockForRead();
54 waitCondition.wait(&queueMutex, 1000);
55 bool empty = queue.empty();
56 queueMutex.unlock();
57 if (empty) {
58 infoUpdate(true);
59 continue;
60 }
61 }
62 //take next request from queue
63 queueMutex.lockForWrite();
64 DatabaseRequest *req = queue.front();
65 queue.pop_front();
66 queueMutex.unlock();
67 rcnt++;
68 infoUpdate(false);
69 if (!req)
70 continue;
71 if (req->discarded()) {
72 req->finish(false);
73 if (!req->isSynchronous)
74 delete req; //->deleteLater();
75 continue;
76 }
77 //process request
78 if (bEmpty) {
79 qcnt = 0;
80 } else {
81 qcnt++;
82 }
83 if (!db->inTransaction)
84 db->transaction(query); //begin if not already
85 bool rv = req->run(query);
86 if (!rv) {
87 apxConsoleW() << "query error:" << query.lastError().text() << query.lastQuery() << req;
88 db->commit(query, true);
89 } else if (req->discarded()) {
90 db->rollback(query);
91 }
92 req->finish(!rv);
93 query.finish();
94
95 if (!req->isSynchronous)
96 delete req; //req->deleteLater();
97
98 queueMutex.lockForRead();
99 bEmpty = queue.empty();
100 queueMutex.unlock();
101
102 if (bEmpty || qcnt > 200) {
103 qcnt = 0;
104 db->commit(query);
105 query.finish();
106 //qDebug()<<"empty";
107 }
108 }
109 QWriteLocker locker(&queueMutex);
110 for (auto req : queue) {
111 req->discard();
112 req->finish(false);
113 if (!req->isSynchronous)
114 delete req; //req->deleteLater();
115 }
116 queue.clear();
117}

142bool DatabaseWorker::infoUpdate(bool force)
143{
144 if (isInterruptionRequested())
145 return false;
146
147 int dt = infoUpdateTime.elapsed();
148 if (force == false && dt < 1000)
149 return false;
150 infoUpdateTime.start();
151
152 //calc rate
153 double vr = rcnt / (dt / 1000.0);
154 rcnt = 0;
155 m_rate = std::round(vr);
156
157 emit infoChanged();
158 return true;
159}
