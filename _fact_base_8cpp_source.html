45void FactBase::bindProperty(Fact *src, QString name, bool oneway)
46{
47 unbindProperties(src, name);
48
49 if (src == this) {
50 qWarning() << "recursive bind:" << path() << name;
51 return;
52 }
53
54 // find two-way binds to filter loops
55 FactPropertyBinding *src_binding = nullptr;
56 for (auto i : src->_property_binds) {
57 if (!i->match(static_cast<Fact *>(this), name))
58 continue;
59 src_binding = i;
60 break;
61 }
62
63 FactPropertyBinding *b = new FactPropertyBinding(static_cast<Fact *>(this),
64 src,
65 name,
66 src_binding);
67 _property_binds.append(b);
68
69 if (oneway)
70 return;
71
72 src->bindProperty(static_cast<Fact *>(this), name, true);
73}

105void FactBase::addChild(Fact *item)
106{
107 //item->Qobject::setParent(this);
108 if (item->treeType() == Action) {
109 if (!m_actions.contains(item)) {
110 m_actions.append(item);
111 emit actionsUpdated();
112 }
113 return;
114 }
115 if (m_facts.contains(item))
116 return;
117 if (property(item->name().toUtf8()).isValid()) {
118 qWarning() << "Property override:" << path() << item->name();
119 }
120 emit itemToBeInserted(m_facts.count(), item);
121 m_facts.append(item);
122 updateChildrenNums();
123 updateSize();
124 emit itemInserted(item);
125}
126void FactBase::removeChild(Fact *item)
127{
128 int i = m_actions.indexOf(item);
129 if (i >= 0) {
130 m_actions.removeAll(item);
131 emit actionsUpdated();
132 }
133 i = m_facts.indexOf(item);
134 if (i < 0)
135 return;
136 emit itemToBeRemoved(i, item);
137 m_facts.removeAt(i);
138 updateChildrenNums();
139 updateSize();
140 emit itemRemoved(item);
141}
142void FactBase::moveChild(Fact *item, int n, bool safeMode)
143{
144 int i = m_facts.indexOf(item);
145 if (i < 0 || i == n)
146 return;
147 if (safeMode) {
148 emit itemToBeRemoved(i, item);
149 m_facts.removeAt(i);
150 emit itemRemoved(item);
151 emit itemToBeInserted(n, item);
152 m_facts.insert(n, item);
153 emit itemInserted(item);
154 updateChildrenNums();
155 } else {
156 if (n > i)
157 n++;
158 emit itemToBeMoved(i, n, item);
159 m_facts.removeAt(i);
160 m_facts.insert(n, item);
161 emit itemMoved(item);
162 }
163 for (int i = 0; i < m_facts.count(); ++i)
164 child(i)->updateNum();
165}

174void FactBase::deleteChildren()
175{
176 if (m_facts.count() <= 0)
177 return;
178 for (int i = 0; i < m_facts.count(); i++) {
179 FactBase *item = child(i);
180 disconnect(this, nullptr, item, nullptr);
181 item->deleteChildren();
182 }
183 //qDebug()<<"removeAll"<<this;
184 while (m_facts.count() > 0) {
185 FactBase *item = child(m_facts.count() - 1);
186 disconnect(this, nullptr, item, nullptr);
187 emit itemToBeRemoved(m_facts.count() - 1, item);
188 m_facts.takeLast();
189 m_size = m_facts.count();
190 item->removed();
191 emit itemRemoved(item);
192 item->deleteLater();
193 }
194 emit sizeChanged();
195}

268QString FactBase::jsname() const
269{
270 QString s = name();
271 if (s.contains('#')) {
272 // array properties are accessible through fact's 'model'
273 return "#";
274 }
275 s = s.simplified()
276 .trimmed()
277 .replace(' ', '_')
278 .replace('.', '_')
279 .replace(':', '_')
280 .replace('/', '_')
281 .replace('\\', '_')
282 .replace('?', '_')
283 .replace('-', '_')
284 .replace('+', '_');
285
286 // check for reserved names
287 const char *r = nullptr, *n = s.toUtf8();
288 if (metaObject()->indexOfProperty(n) >= 0) {
289 r = "property";
290 } else if (metaObject()->indexOfMethod(n) >= 0) {
291 r = "method";
292 } else if (metaObject()->indexOfSignal(n) >= 0) {
293 r = "signal";
294 } else if (metaObject()->indexOfSlot(n) >= 0) {
295 r = "slot";
296 }
297 if (r) {
298 qWarning() << QString("Name is reserved for %1: %2 (%3)").arg(r).arg(s).arg(path());
299 // just a warning by now
300 }
301 return s;
302}
