124void parse_sensors(void)
125{
126 float elapsed = XPLMGetElapsedTime();
127 static float elapsed_s = 0;
128 float dt = elapsed - elapsed_s;
129 elapsed_s = elapsed;
130
131 // AHRS
132 sim_bundle.att_deg[0] = XPLMGetDataf(xp.roll);
133 sim_bundle.att_deg[1] = XPLMGetDataf(xp.pitch);
134 sim_bundle.att_deg[2] = XPLMGetDataf(xp.yaw);
135
136 sim_bundle.gyro_deg[0] = XPLMGetDataf(xp.p);
137 sim_bundle.gyro_deg[1] = XPLMGetDataf(xp.q);
138 sim_bundle.gyro_deg[2] = XPLMGetDataf(xp.r);
139
140 sim_bundle.acc_ned[0] = -XPLMGetDataf(xp.az);
141 sim_bundle.acc_ned[1] = +XPLMGetDataf(xp.ax);
142 sim_bundle.acc_ned[2] = -XPLMGetDataf(xp.ay) - 9.81f;
143
144 sim_bundle.lat_deg = XPLMGetDatad(xp.lat);
145 sim_bundle.lon_deg = XPLMGetDatad(xp.lon);
146 sim_bundle.hmsl = XPLMGetDataf(xp.alt);
147
148 sim_bundle.vel_ned[0] = -XPLMGetDataf(xp.vz);
149 sim_bundle.vel_ned[1] = +XPLMGetDataf(xp.vx);
150 sim_bundle.vel_ned[2] = -XPLMGetDataf(xp.vy);
151
152 // AGL
153 sim_bundle.agl = XPLMGetDataf(xp.agl);
154
155 // RPM
156 static float a[8];
157 XPLMGetDatavf(xp.rpm, a, 0, 1);
158 sim_bundle.rpm = (a[0] * 60.f) / (M_PI * 2.f);
159
160 // Airdata
161 sim_bundle.airspeed = 0.51444f * (float) XPLMGetDataf(xp.airspeed); //knots to mps
162
163 float rho = (float) XPLMGetDataf(xp.rho);
164 float at = (float) XPLMGetDataf(xp.air_temp);
165 sim_bundle.baro_mbar = rho * 287.1f * (at + 273.15f) / 100.0f;
166
167 sim_bundle.slip = XPLMGetDataf(xp.slip);
168}

170static void parse_rx(const void *data, size_t size)
171{
172 XbusStreamReader stream(data, size);
173 if (stream.available() < xbus::pid_s::psize())
174 return;
175 xbus::pid_s pid;
176 pid.read(&stream);
177
178 mandala::uid_t uid = pid.uid;
179 if (!mandala::cmd::env::sim::match(uid))
180 return;
181
182 switch (uid) {
183 case mandala::cmd::env::sim::ctr::uid: {
184 if (pid.pri != xbus::pri_final)
185 break;
186 if (!xpl_channels)
187 break;
188 for (uint8_t i = 0; i < xpl_channels; i++) {
189 float v = stream.read<float>();
190 if (!xpl_ref[i])
191 continue;
192 //printf("set: %s=%f\n",(char*)xpl[i],(pwm_ch[i]));
193 if (xpl_is_array[i]) {
194 XPLMSetDatavf(xpl_ref[i], &v, xpl_array_idx[i], 1);
195 // if (xpl_type[i] & xplmType_FloatArray)
196 // XPLMSetDatavf(xpl_ref[i], &(pwm_ch[i]), xpl_array_idx[i], 1);
197 // else if (xpl_type[i] & xplmType_IntArray) {
198 // int v = pwm_ch[i];
199 // XPLMSetDatavi(xpl_ref[i], &v, xpl_array_idx[i], 1);
200 // }
201 continue;
202 }
203 XPLMSetDataf(xpl_ref[i], v);
204 // if (xpl_type[i] & xplmType_Float)
205 // XPLMSetDataf(xpl_ref[i], pwm_ch[i]);
206 // else if (xpl_type[i] & xplmType_Double)
207 // XPLMSetDatad(xpl_ref[i], pwm_ch[i]);
208 // else if (xpl_type[i] & xplmType_Int)
209 // XPLMSetDatai(xpl_ref[i], pwm_ch[i]);
210 // else
211 // XPLMSetDataf(xpl_ref[i], pwm_ch[i]);
212 }
213 } break;
214 case mandala::cmd::env::sim::cfg::uid: {
215 if (pid.pri != xbus::pri_response)
216 break;
217 for (xpl_channels = 0; xpl_channels < xpl_channels_max; ++xpl_channels) {
218 const char *s = stream.read_string(sizeof(xpl[0]));
219 if (!s)
220 break;
221 strncpy(xpl[xpl_channels], s, sizeof(xpl[0]));
222 }
223 printf("X-Plane controls updated (%u)\n", xpl_channels);
224 memset(&xpl_ref, 0, sizeof(xpl_ref));
225 memset(&xpl_is_array, 0, sizeof(xpl_is_array));
226 memset(&xpl_array_idx, 0, sizeof(xpl_array_idx));
227 if (!xpl_channels)
228 break;
229 for (uint8_t i = 0; i < xpl_channels; ++i) {
230 xbus::node::conf::text_t &s = xpl[i];
231 if (s[0] == 0)
232 continue;
233 char *it1 = strchr(s, '[');
234 if (it1) {
235 *it1++ = 0;
236 char *it2 = strchr(it1, ']');
237 if (!it2)
238 continue;
239 if (*(it2 + 1) != 0)
240 continue;
241 *it2 = 0;
242 uint aidx;
243 if (sscanf(it1, "%u", &aidx) != 1)
244 continue;
245 //if(!(XPLMGetDataRefTypes(sref)&(xplmType_FloatArray|xplmType_IntArray)))continue;
246 xpl_array_idx[i] = aidx;
247 xpl_is_array[i] = true;
248 }
249 XPLMDataRef ref = XPLMFindDataRef(s);
250 if (!ref)
251 continue;
252 xpl_type[i] = XPLMGetDataRefTypes(s);
253 xpl_ref[i] = ref;
254 printf("DataRef: %s\n", s);
255 }
256 } break;
257 }
258}

331static float flightLoopCallback(float inElapsedSinceLastCall,
332 float inElapsedTimeSinceLastFlightLoop,
333 int inCounter,
334 void *inRefcon)
335{
336 static uint ap_link_timeout = 0;
337
338 ap_link_timeout++;
339
340 if (ap_link_timeout > 3) {
341 ap_link_timeout = 0;
342 xpl_channels = 0; // reset channels assignments
343 }
344
345 if (!(enabled && udp.is_connected())) {
346 xpl_channels = 0;
347 ap_link_timeout = 0;
348 return 1.f;
349 }
350
351 do {
352 for (;;) {
353 if (!udp.dataAvailable())
354 break;
355
356 ap_link_timeout = 0;
357
358 // printf("udp.dataAvailable\n");
359
360 size_t rcnt = udp.read(packet_buf, sizeof(packet_buf));
361 if (!rcnt)
362 break;
363
364 // printf("udp.read: %lu\n", rcnt);
365
366 const uint8_t *data = packet_buf;
367 while (rcnt > 0) {
368 auto dcnt = _rx_decoder.decode(data, rcnt);
369
370 switch (_rx_decoder.status()) {
371 case SerialDecoder::PacketAvailable: {
372 data = static_cast<const uint8_t *>(data) + dcnt;
373 parse_rx(_rx_decoder.data(), _rx_decoder.size());
374 break;
375 }
376
377 case SerialDecoder::DataAccepted:
378 case SerialDecoder::DataDropped:
379 rcnt = 0;
380 break;
381 default:
382 break;
383 }
384
385 if (rcnt > dcnt)
386 rcnt -= dcnt;
387 else
388 rcnt = 0;
389 }
390 }
391
392 if (!xpl_channels) {
393 printf("X-Plane controls request\n");
394 //xpl_channels = 1;
395 request_controls();
396 return 2.5f;
397 }
398
399 } while (0);
400
401 parse_sensors();
402 send_bundle();
403
404 return -1.f; //call me next loop
405}
