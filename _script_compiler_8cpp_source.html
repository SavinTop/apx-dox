88bool ScriptCompiler::lookup()
89{
90 if (f_cc->value().toInt() == 0) {
91 if (lookup_llvm())
92 return true;
93 } else {
94 if (lookup_wasi() || lookup_llvm())
95 return true;

99 qWarning() << "compilers not found";
100 return false;

103bool ScriptCompiler::lookup_llvm()
104{
105 QStringList st;
106
107 QString s = f_llvm_path->value().toString();
108 if (QFile::exists(s))
109 st << s;
110
111 st << "/opt/homebrew/opt/llvm/bin/clang"
112 << "/usr/local/bin/clang"
113 << "/usr/bin/clang";
114
115 for (auto &s : st) {
116 if (!QFile::exists(s))
117 continue;
118
119 // check for wasm32 target support
120 qDebug() << "checking" << s;
121 QProcess p;
122 p.start(s, QStringList() << "--print-targets");
123 p.waitForFinished();
124 if (p.exitCode())
125 continue;
126
127 auto out = p.readAllStandardOutput().simplified().split(' ');
128 // qDebug() << out;
129
130 if (!out.contains("wasm32"))
131 continue;
132
133 // cc found
134 setDescr(s);
135 setCompiler(s);
136 return true;
137 }
138
139 return false;
140}

192void ScriptCompiler::update_vscode()
193{
194 // update cc vscode settings
195 do {
196 QFile fsettings(AppDirs::scripts().absoluteFilePath(".vscode/settings.json"));
197
198 if (!fsettings.open(QFile::ReadOnly | QFile::Text))
199 break;
200
201 QJsonDocument json = QJsonDocument::fromJson(fsettings.readAll());
202 fsettings.close();
203 QJsonObject root = json.object();
204 QJsonValueRef ref = root.find("wasm").value();
205 if (ref.isUndefined())
206 break;
207
208 QJsonObject o = ref.toObject();
209 o.insert("cc", m_cc);
210 ref = o;
211 json.setObject(root);
212 fsettings.open(QFile::WriteOnly | QFile::Text | QFile::Truncate);
213 fsettings.write(json.toJson());
214 fsettings.close();
215
216 return;
217 } while (0);
218 qWarning() << "vscode settings error";
219}

221void ScriptCompiler::download()
222{
223 QString fileName = QString("%1.tar.gz").arg(m_sdk);
224
225 if (QFile::exists(m_dir.absoluteFilePath(fileName))) {
226 extract(fileName);
227 return;
228 }
229
230 apxMsg() << tr("Downloading") << title().append("...");
231
232 QUrl url(QString("https://github.com/WebAssembly/wasi-sdk/releases/download/wasi-sdk-%1/%2")
233 .arg(m_version.left(m_version.indexOf('.')))
234 .arg(fileName));
235
236 qDebug() << url;
237
238 net.setRedirectPolicy(QNetworkRequest::NoLessSafeRedirectPolicy);
239
240 QNetworkRequest *request = new QNetworkRequest(url);
241
242 QSslConfiguration ssl = request->sslConfiguration();
243 ssl.setPeerVerifyMode(QSslSocket::VerifyNone);
244 request->setSslConfiguration(ssl);
245
246 request->setRawHeader("Accept", "*/*");
247 request->setRawHeader("User-Agent",
248 QString("%1 (v%2)")
249 .arg(QCoreApplication::applicationName())
250 .arg(App::version())
251 .toUtf8());
252
253 setProgress(0);
254 QNetworkReply *reply = net.get(*request);
255 connect(reply, &QNetworkReply::finished, this, &ScriptCompiler::responseDownload);
256 connect(reply, &QNetworkReply::downloadProgress, this, &ScriptCompiler::downloadProgress);
257}

266void ScriptCompiler::responseDownload()
267{
268 setProgress(-1);
269
270 QNetworkReply *reply = qobject_cast<QNetworkReply *>(sender());
271 if (!reply)
272 return;
273 reply->deleteLater();
274
275 if (reply->error()) {
276 qWarning() << reply->errorString();
277 return;
278 }
279 QByteArray data = reply->readAll();
280 if (data.isEmpty()) {
281 qWarning() << "no data";
282 return;
283 }
284
285 QString s = reply->header(QNetworkRequest::ContentDispositionHeader).toString();
286 s = s.mid(s.lastIndexOf("filename=", Qt::CaseInsensitive));
287 s = s.mid(s.indexOf("=") + 1);
288 s = s.left(s.indexOf(";"));
289 if (!s.startsWith(m_sdk)) {
290 apxMsgW() << title().append(':') << tr("Unknown response") << s;
291 return;
292 }
293 apxMsg() << title().append(':') << tr("Received") << s;
294
295 QFile fzip(m_dir.absoluteFilePath(s));
296 if (!fzip.open(QFile::WriteOnly)) {
297 qWarning() << "can't write" << fzip.fileName();
298 return;
299 }
300 fzip.write(data);
301 fzip.close();
302 qDebug() << "downloaded" << fzip.fileName();
303
304 // extract
305 extract(s);
306}
