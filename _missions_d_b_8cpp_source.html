132bool DBReqMissionsFindSite::run(QSqlQuery &query)
133{
134 if (!(std::isnan(lat) || std::isnan(lon) || lat == 0.0 || lon == 0.0)) {
135 query.prepare("SELECT (((?-lat)*(?-lat)*?)+((?-lon)*(?-lon))) AS dist, * FROM Sites"
136 " WHERE lat!=0 AND lon!=0 AND dist<2"
137 " ORDER BY dist ASC LIMIT 1");
138 query.addBindValue(lat);
139 query.addBindValue(lat);
140 query.addBindValue(std::pow(std::cos(qDegreesToRadians(lat)), 2));
141 query.addBindValue(lon);
142 query.addBindValue(lon);
143 if (!query.exec())
144 return false;
145 if (query.next()) {
146 siteID = query.value("key").toULongLong();
147 site = query.value("title").toString();
148 }
149 }
150 emit siteFound(siteID, site);
151 return true;
152}

192bool DBReqMissionsSave::run(QSqlQuery &query)
193{
194 if (_data.value("rw").value<QVariantList>().isEmpty()) {
195 qWarning() << "missing runways in mission";
196 }
197
198 //find siteID
199 if (!_reqSite.run(query))
200 return false;
201
202 QString title = _data.value("title").toString().simplified();
203 if (_reqSite.siteID) {
204 title = title.remove(_reqSite.site, Qt::CaseInsensitive).simplified();
205 } else {
206 qDebug() << "no site";
207 }
208
209 QVariant siteID = _reqSite.siteID ? _reqSite.siteID : QVariant();
210
211 auto callsign = _data.value("callsign").toString();
212 auto time = _data.value("time").toString();
213
214 //find existing mission by hash
215 QString hash = _data.value("hash").toString();
216 query.prepare("SELECT * FROM Missions WHERE hash=?");
217 query.addBindValue(hash);
218 if (!query.exec())
219 return false;
220 if (query.next()) {
221 _missionID = query.value(0).toULongLong();
222 qDebug() << "mission exists";
223 if (callsign.isEmpty())
224 callsign = query.value("callsign").toString();
225 //update mission access time
226 query.prepare("UPDATE Missions SET time=?, title=?, siteID=?, callsign=? WHERE key=?");
227 query.addBindValue(time);
228 query.addBindValue(title);
229 query.addBindValue(siteID);
230 query.addBindValue(callsign);
231 query.addBindValue(_missionID);
232 if (!query.exec())
233 return false;
234 emit missionHash(hash);
235 return true;
236 }
237
238 //create new mission record
239 if (!db->transaction(query))
240 return false;
241
242 QVariantMap info = filterFields("Missions", _data);
243 info["hash"] = hash;
244 info["time"] = time;
245 info["title"] = title;
246 info["callsign"] = callsign;
247 info["siteID"] = siteID;
248
249 if (!recordInsertQuery(query, info, "Missions"))
250 return false;
251 if (!query.exec())
252 return false;
253 _missionID = query.lastInsertId().toULongLong();
254
255 //write items
256 if (!writeItems(query, _data.value("rw"), "Runways"))
257 return false;
258 if (!writeItems(query, _data.value("wp"), "Waypoints"))
259 return false;
260 if (!writeItems(query, _data.value("tw"), "Taxiways"))
261 return false;
262 if (!writeItems(query, _data.value("pi"), "Pois"))
263 return false;
264
265 if (!db->commit(query))
266 return false;
267
268 emit missionHash(hash);
269 return true;
270}
271bool DBReqMissionsSave::writeItems(QSqlQuery &query, const QVariant &var, QString tableName)
272{
273 DatabaseRequest::Records records;
274 records.names = db->tableFields(tableName);
275 records.names.removeOne("key");
276 quint32 num = 0;
277 for (auto j : var.value<QVariantList>()) {
278 auto obj = j.value<QVariantMap>();
279 QVariantList r;
280 r.append(_missionID);
281 r.append(num++);
282 for (int i = 2; i < records.names.size(); ++i) {
283 auto v = obj.value(records.names.at(i));
284 if (v.canConvert<QVariantMap>()) {
285 QStringList st;
286 auto vobj = v.value<QVariantMap>();
287 for (auto k : vobj.keys()) {
288 st.append(k + "=" + vobj.value(k).toString());
289 }
290 r.append(st.join(','));
291 } else {
292 r.append(v);
293 }
294 }
295 records.values.append(r);
296 }
297 for (int i = 0; i < records.values.size(); ++i) {
298 recordInsertQuery(query, records, i, tableName);
299 if (!query.exec())
300 return false;
301 }
302 return true;
303}

305bool DBReqMissionsLoad::run(QSqlQuery &query)
306{
307 query.prepare(
308 "SELECT * FROM Missions"
309 " LEFT JOIN (SELECT key, title AS site FROM Sites) AS Sites ON Missions.siteID=Sites.key"
310 " WHERE hash=?");
311 query.addBindValue(hash);
312 if (!query.exec())
313 return false;
314 if (!query.next())
315 return false;
316
317 auto missionID = query.value("Missions.key").toULongLong();
318
319 _mission = filterIdValues(queryRecord(query));
320
321 query.prepare("SELECT * FROM Runways WHERE missionID=? ORDER BY num ASC");
322 query.addBindValue(missionID);
323 if (!query.exec())
324 return false;
325 _mission.insert("rw", readItems(query));
326
327 query.prepare("SELECT * FROM Waypoints WHERE missionID=? ORDER BY num ASC");
328 query.addBindValue(missionID);
329 if (!query.exec())
330 return false;
331 _mission.insert("wp", readItems(query));
332
333 query.prepare("SELECT * FROM Taxiways WHERE missionID=? ORDER BY num ASC");
334 query.addBindValue(missionID);
335 if (!query.exec())
336 return false;
337 _mission.insert("tw", readItems(query));
338
339 query.prepare("SELECT * FROM Pois WHERE missionID=? ORDER BY num ASC");
340 query.addBindValue(missionID);
341 if (!query.exec())
342 return false;
343 _mission.insert("pi", readItems(query));
344
345 emit loaded(_mission);
346 return true;
347}
