27Shortcut::Shortcut(Fact *parent, Shortcuts *shortcuts, const Shortcut *sc, bool bUsr)
28 : Fact(parent,
29 sc ? (bUsr ? "usr#" : "sys#") : tr("add"),
30 sc ? "" : tr("Add new shortcut"),
31 sc ? "" : tr("Configure new hotkey"),
32 Group | (sc ? Bool : NoFlags))
33 , container(shortcuts)
34 , _new(sc ? false : true)
35 , bUsr(bUsr)
36{
37 _enabled = new Fact(this, "enb", tr("Enabled"), "", Bool);
38
39 _key = new Fact(this, "key", tr("Key sequence"), "", Text);
40 _key->setOpt("editor", "qrc:/" PLUGIN_NAME "/EditorKey.qml");
41
42 _cmd = new Fact(this, "scr", tr("Java script"), "", Text);
43
44 if (_new) {
45 _save = new Fact(this, "save", tr("Save"), "", Action | Apply | CloseOnTrigger);
46 connect(_save, &Fact::triggered, shortcuts, &Shortcuts::addTriggered);
47 defaults();
48 } else {
49 setSection(bUsr ? shortcuts->f_usr->section() : shortcuts->f_sys->section());
50 copyValuesFrom(sc);
51 _remove = new Fact(this, "remove", tr("Remove"), "", Action | Remove);
52 connect(_remove, &Fact::triggered, this, &Shortcut::deleteFact);
53 connect(_remove, &Fact::triggered, shortcuts, &Shortcuts::save);
54 connect(shortcuts, &Fact::sizeChanged, this, &Shortcut::updateStats);
55
56 connect(bUsr ? shortcuts->f_allonUsr : shortcuts->f_allonSys,
57 &Fact::triggered,
58 this,
59 &Shortcut::enable);
60 connect(bUsr ? shortcuts->f_alloffUsr : shortcuts->f_alloffSys,
61 &Fact::triggered,
62 this,
63 &Shortcut::disable);
64 connect(this,
65 &Shortcut::titleChanged,
66 shortcuts,
67 &Shortcuts::updateStats,
68 Qt::QueuedConnection);
69
70 connect(this, &Fact::valueChanged, this, [this]() { _enabled->setValue(value()); });
71 connect(_enabled, &Fact::valueChanged, this, [this]() { setValue(_enabled->value()); });
72 setValue(_enabled->value());
73 }
74
75 for (int i = 0; i < size(); ++i) {
76 connect(child(i), &Fact::valueChanged, this, &Shortcut::updateStats);
77 if (!_new) {
78 connect(child(i), &Fact::valueChanged, shortcuts, &Shortcuts::save);
79 }
80 }
81 updateStats();
82
83 App::jsync(this);
84}

93void Shortcut::updateStats()
94{
95 if (_new) {
96 _save->setEnabled(!(_key->text().isEmpty() || _cmd->text().isEmpty()));
97 } else {
98 _remove->setVisible(bUsr);
99 setTitle(QString("%1 -> %2").arg(_key->text()).arg(_cmd->text()));
100 _key->setEnabled(bUsr);
101 _cmd->setEnabled(bUsr);
102 }
103}
