295bool DBReqTelemetryUpdateMandala::run(QSqlQuery &query)
296{
297 // called by LOCAL vehicle with 'records' initialized to current mandala fields
298 connect(
299 this,
300 &DBReqTelemetryUpdateMandala::progress,
301 db,
302 [this](int v) { db->setProgress(v); },
303 Qt::QueuedConnection);
304
305 const QStringList &n = records.names;
306 int i_id = n.indexOf("id");
307 int i_name = n.indexOf("name");
308
309 TelemetryDB::FieldsByUID byUID;
310 TelemetryDB::FieldsByName byName;
311 QList<quint64> all_keys;
312
313 //load existing fields
314 query.prepare("SELECT * FROM TelemetryFields");
315 if (!query.exec())
316 return false;
317 Records db_records = queryRecords(query);
318 const QStringList &rn = db_records.names;
319 int i_r_name = rn.indexOf("name");
320
321 bool mod = false;
322 for (auto const &f : records.values) {
323 QString f_name = f.at(i_name).toString();
324 mandala::uid_t uid = f.at(i_id).toUInt();
325
326 quint64 key = 0;
327 bool upd = false;
328 for (auto const &r : db_records.values) {
329 const QString &name = r.at(i_r_name).toString();
330 if (name == f_name) {
331 key = r.at(0).toULongLong();
332 byUID.insert(uid, key);
333 byName.insert(f_name, key);
334 all_keys.append(key);
335 //check entire row match
336 for (int i = 1; i < r.size(); ++i) {
337 if (r.at(i).toString() == f.value(n.indexOf(rn.at(i))).toString()) {
338 continue;
339 }
340 //qWarning() << r.at(i) << f.value(n.indexOf(rn.at(i)));
341 upd = true; //record update needed
342 break;
343 }
344 break; //exsisting record found
345 }
346 }
347 if (key && !upd)
348 continue;
349 if (!mod) {
350 if (!db->transaction(query))
351 return false;
352 mod = true;
353 }
354 if (!key) {
355 apxConsole() << "new telemetry field:" << f_name;
356 query.prepare("INSERT INTO TelemetryFields(name) VALUES(?)");
357 query.addBindValue(f_name);
358 if (!query.exec())
359 return false;
360 key = query.lastInsertId().toULongLong();
361 byUID.insert(uid, key);
362 byName.insert(f_name, key);
363 all_keys.append(key);
364 } else {
365 apxConsole() << "update telemetry field:" << f_name;
366 }
367 //update existing record
368 QStringList nlist = rn;
369 if (!nlist.isEmpty()) {
370 nlist.removeAt(0); //skip key
371 }
372 if (nlist.isEmpty()) {
373 nlist = n;
374 nlist.removeOne("alias");
375 }
376 QString qs = QString("UPDATE TelemetryFields SET %1=? WHERE key=?").arg(nlist.join("=?,"));
377 query.prepare(qs);
378 for (auto s : nlist) {
379 query.addBindValue(f.value(n.indexOf(s)));
380 }
381 query.addBindValue(key);
382 if (!query.exec())
383 return false;
384 }
385 if (mod) {
386 if (!db->commit(query))
387 return false;
388 apxMsgW() << tr("Telemetry DB updated");
389 }
390
391 //check for deprecated records
392 QList<quint64> rmlist;
393 for (auto const &r : db_records.values) {
394 quint64 key = r.at(0).toULongLong();
395 if (all_keys.contains(key))
396 continue;
397 apxConsole() << "remove telemetry field:" << r.at(i_r_name).toString();
398 rmlist.append(key);
399 }
400 if (!rmlist.isEmpty()) {
401 db->disable();
402 emit progress(0);
403 apxMsgW() << tr("Telemetry DB maintenance in progress...");
404 QStringList st;
405 for (auto k : rmlist)
406 st << QString::number(k);
407 if (!query.exec("PRAGMA foreign_keys = OFF"))
408 return false;
409 query.prepare(
410 QString("DELETE FROM TelemetryUplink WHERE fieldID IN (%1)").arg(st.join(',')));
411 if (!query.exec())
412 return false;
413 query.prepare(
414 QString("DELETE FROM TelemetryDownlink WHERE fieldID IN (%1)").arg(st.join(',')));
415 if (!query.exec())
416 return false;
417 query.prepare(QString("DELETE FROM TelemetryFields WHERE key IN (%1)").arg(st.join(',')));
418 if (!query.exec())
419 return false;
420 if (!query.exec("PRAGMA foreign_keys = ON"))
421 return false;
422 emit progress(-1);
423 apxMsgW() << tr("Telemetry DB ready");
424 db->enable();
425 }
426
427 static_cast<TelemetryDB *>(db)->updateFieldsMap(byUID, byName);
428
429 return true;
430}

432bool DBReqTelemetryEmptyTrash::run(QSqlQuery &query)
433{
434 db->disable();
435 emit progress(0);
436 bool rv = false;
437 do {
438 query.prepare("SELECT COUNT(*) FROM Telemetry WHERE trash IS NOT NULL");
439 if (!query.exec())
440 break;
441 if (!query.next())
442 break;
443
444 qint64 cnt = query.value(0).toLongLong();
445 qint64 dcnt = 0;
446 if (cnt > 0) {
447 apxMsg() << tr("Permanently deleting %1 records").arg(cnt).append("...");
448
449 while (dcnt < cnt) {
450 query.prepare(
451 "SELECT key FROM Telemetry WHERE trash IS NOT NULL ORDER BY time DESC LIMIT 1");
452 if (!query.exec())
453 break;
454 if (!query.next())
455 break;
456 quint64 key = query.value(0).toULongLong();
457
458 if (!db->transaction(query))
459 break;
460 query.prepare("DELETE FROM Telemetry WHERE key=?");
461 query.addBindValue(key);
462 if (!query.exec())
463 break;
464 if (!db->commit(query))
465 break;
466
467 dcnt++;
468 emit progress((dcnt * 100 / cnt));
469 if (discarded())
470 break;
471 }
472 }
473 if (dcnt < cnt)
474 apxMsgW() << tr("Telemetry trash not empty") << cnt - dcnt;
475 else {
476 apxMsg() << tr("Telemetry trash is empty");
477 rv = true;
478 }
479 } while (0);
480 emit progress(-1);
481 db->enable();
482 return rv;
483}

560bool DBReqTelemetryRecover::run(QSqlQuery &query)
561{
562 query.prepare("SELECT * FROM Telemetry WHERE hash=?");
563 query.addBindValue(_hash);
564 if (!query.exec())
565 return false;
566 if (!query.next()) {
567 emit unavailable(_hash);
568 return true;
569 }
570 auto telemetryID = query.value(0).toULongLong();
571 auto trash = query.value("trash").toBool();
572 if (trash) {
573 query.prepare("UPDATE Telemetry SET trash=NULL WHERE key=?");
574 query.addBindValue(telemetryID);
575 if (!query.exec())
576 return false;
577 }
578 emit available(telemetryID, _hash);
