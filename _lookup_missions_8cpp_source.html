56bool LookupMissions::fixItemDataThr(QVariantMap *item)
57{
58 QString title = item->value("title").toString();
59 QString cnt = QString::number(item->value("cnt").toUInt());
60 QString callsign = item->value("callsign").toString();
61 QString site = item->value("site").toString();
62 QString runway = item->value("runway").toString();
63 //title=QString("%1 [%2]").arg(title).arg(cnt);
64 if (!site.isEmpty())
65 title.prepend(title.isEmpty() ? "" : " - ").prepend(site);
66 if (!callsign.isEmpty())
67 title.prepend(title.isEmpty() ? "" : " - ").prepend(callsign);
68 if (!runway.isEmpty())
69 title.append(title.isEmpty() ? "" : " - ").append(runway);
70
71 (*item)["title"] = title;
72 QString sdescr = QDateTime::fromMSecsSinceEpoch(item->value("time").toLongLong())
73 .toString("d MMM yyyy hh:mm");
74 if (item->contains("dist") && reqPoint.isValid()) {
75 QGeoCoordinate p(item->value("lat").toDouble(), item->value("lon").toDouble());
76 sdescr.append(QString(" (%1)").arg(AppRoot::distanceToString(reqPoint.distanceTo(p))));
77 }
78 item->insert("descr", sdescr);
79 QString value = QString("%1").arg(
80 AppRoot::distanceToString(item->value("distance").toULongLong()));
81 value.append(" ").append(cnt);
82 item->insert("value", value);
83 return true;
84}

86void LookupMissions::defaultLookup()
87{
88 const QString s = QString("%%%1%%").arg(filter());
89 query("SELECT * FROM Missions"
90 " LEFT JOIN (SELECT key, title AS site FROM Sites) AS Sites ON Missions.siteID=Sites.key"
91 " LEFT JOIN (SELECT missionID, COUNT(*) AS cnt FROM Waypoints GROUP BY missionID) AS "
92 "Waypoints ON Waypoints.missionID=Missions.key"
93 " WHERE"
94 " ( title LIKE ? OR callsign LIKE ? OR site LIKE ? OR runway LIKE ? )"
95 " ORDER BY time DESC",
96 QVariantList() << s << s << s << s);
97 setDescr(tr("Ordered by time"));
98}
99void LookupMissions::dbLookupMissionsByArea(QGeoCoordinate c, QString siteName)
100{
101 const QString s = QString("%%%1%%").arg(filter());
102 reqPoint = c;
103 QVariantList v;
104 v << c.latitude();
105 v << c.latitude();
106 v << std::pow(std::cos(qDegreesToRadians(c.latitude())), 2);
107 v << c.longitude();
108 v << c.longitude();
109 query("SELECT *, ((?-lat)*(?-lat)*?)+((?-lon)*(?-lon)) AS dist FROM Missions"
110 " LEFT JOIN (SELECT key, title AS site FROM Sites) AS Sites ON Missions.siteID=Sites.key"
111 " LEFT JOIN (SELECT missionID, COUNT(*) AS cnt FROM Waypoints GROUP BY missionID) AS "
112 "Waypoints ON Waypoints.missionID=Missions.key"
113 " WHERE dist<200 AND"
114 " ( title LIKE ? OR callsign LIKE ? OR site LIKE ? OR runway LIKE ? )"
115 " ORDER BY dist ASC, time DESC"
116 " LIMIT 200",
117 v << s << s << s << s);
118 QString sdescr = tr("Closest to site");
119 setDescr(siteName.isEmpty() ? sdescr : (sdescr.append(QString(" '%1'").arg(siteName))));
120}
