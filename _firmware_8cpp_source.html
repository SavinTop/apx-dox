99void Firmware::nodeNotify(NodeItem *node)
100{
101 if (node->ident().isEmpty())
102 return;
103
104 auto uid = node->uid();
105
106 // update nodes map
107 QStringList st;
108 st << node->title();
109 st << node->descr();
110 st << node->value().toString();
111 st.removeAll("");
112 QString s = st.join(' ');
113 if (_nodesMap.value(uid) != s) {
114 _nodesMap.insert(uid, s);
115 nodesMapUpdated(_nodesMap);
116 }
117
118 if (queued(f_queue, uid))
119 return;
120 if (queued(f_available, uid))
121 return;
122
123 new QueueItem(f_available,
124 node->uid(),
125 node->title(),
126 node->ident().value("hardware").toString(),
127 QString());
128}

193void Firmware::next()
194{
195 if (f_queue->size() <= 0) {
196 setActive(false);
197 return;
198 }
199
200 //find next node in queue
201 QueueItem *item = nullptr;
202 //loaders first
203 for (int i = 0; item == nullptr && i < f_queue->size(); ++i) {
204 QueueItem *f = static_cast<QueueItem *>(f_queue->child(i));
205 if (f->type() == "ldr")
206 item = f;
207 }
208 //sorted by names
209 QStringList names;
210 names << "servo"
211 << "gps"
212 << "cas"
213 << "nav"
214 << "ifc"
215 << "";
216 for (auto s : names) {
217 for (int i = 0; item == nullptr && i < f_queue->size(); ++i) {
218 QueueItem *f = static_cast<QueueItem *>(f_queue->child(i));
219 if (s.isEmpty() || f->title() == s) {
220 item = f;
221 }
222 }
223 if (item)
224 break;
225 }
226 if (!item)
227 return;
228
229 setActive(true);
230 emit upgradeStarted(item->uid(), item->type());
231
232 connect(item, &QueueItem::finished, this, &Firmware::itemFinished);
233
234 item->move(0);
235 item->start();
236}

238void Firmware::itemFinished(QueueItem *item, bool success)
239{
240 emit upgradeFinished(item->uid(), item->type());
241
242 if (success) {
243 item->deleteFact();
244 } else {
245 f_queue->deleteChildren();
246 }
247
248 if (f_queue->size() == 0) {
249 setActive(false);
250 }
251
252 QTimer::singleShot(100, this, &Firmware::next);
253}
