109void ServosForm::sendVolz(uint cmd, uint id, uint arg)
110{
111 QByteArray pack;
112 pack[0] = cmd;
113 pack[1] = id;
114 pack[2] = (arg >> 8) & 0xFF;
115 pack[3] = arg & 0xFF;
116
117 uint16_t crc = 0xFFFF;
118
119 for (int x = 0; x < 4; x++) {
120 crc = ((pack.at(x) << 8) ^ crc);
121 for (int y = 0; y < 8; y++) {
122 if (crc & 0x8000)
123 crc = (crc << 1) ^ 0x8005;
124 else
125 crc = crc << 1;
126 }
127 }
128 pack[4] = (crc >> 8) & 0xFF;
129 pack[5] = crc & 0xFF;
130
131 auto protocol = Vehicles::instance()->current()->protocol();
132 if (protocol)
133 protocol->data()->sendSerial(static_cast<quint8>(ui->ePortID->value()), pack);
134}

136void ServosForm::sendFutabaAddr(uint servoID, uint newAddr)
137{
138 qInfo() << QString("SBUS ADDR %1-%2: %3").arg(servoID >> 16).arg(servoID & 0xFFFF).arg(newAddr);
139 QByteArray pack;
140 pack[0] = 0xF9;
141 pack[1] = newAddr - 1;
142 pack[2] = servoID >> 16 & 0xFF;
143 pack[3] = servoID >> 8 & 0xFF;
144 pack[4] = servoID & 0xFF;
145 pack[5] = 0x17;
146 pack[6] = 0x72;
147 pack[7] = 0xD0;
148 pack[8] = 0x80;
149 pack[9] = 0x80;
150 pack[10] = 0x80;
151 pack[11] = 0x00;
152 pack[12] = 0x00;
153 pack[13] = 0x34;
154 pack[14] = 0x01;
155 pack[15] = 0x2D;
156 pack[16] = 0x00;
157 //crc 2's complement sum
158 uint crc = 0;
159 for (uint i = 1; i < 16; i++)
160 crc += pack[i];
161 pack[16] = 0x100 - crc;
162 //reverse bits
163 for (int i = 0; i < pack.size(); i++) {
164 uint v = pack[i];
165 uint vx = 0;
166 uint m2 = 0x80;
167 for (uint m = 1; m < 0x0100; m <<= 1) {
168 if (v & m)
169 vx |= m2;
170 m2 >>= 1;
171 }
172 pack[i] = vx;
173 }
174
175 PVehicle *protocol = Vehicles::instance()->current()->protocol();
176 if (!protocol)
177 return;
178 protocol->data()->sendSerial(static_cast<quint8>(ui->ePortID->value()), pack);
179}
