26bool DBReqTelemetryFindCache::run(QSqlQuery &query)
27{
28 //check for invalid cache (telemetry hash is null)
29 query.prepare("SELECT * FROM Telemetry WHERE key=?");
30 query.addBindValue(telemetryID);
31 if (!query.exec())
32 return false;
33 bool invalid;
34 if (query.next())
35 invalid = query.value("hash").toString().isEmpty();
36 else
37 invalid = true;
38
39 //check db invalid cache list
40 if (!invalid) {
41 if (!static_cast<TelemetryDB *>(db)->invalidCacheList().isEmpty()) {
42 invalid = static_cast<TelemetryDB *>(db)->invalidCacheList().contains(telemetryID);
43 }
44 }
45
46 //find existing cache record
47 if (!invalid) {
48 query.prepare("SELECT * FROM TelemetryCache WHERE telemetryID=?");
49 query.addBindValue(telemetryID);
50 if (!query.exec())
51 return false;
52 if (query.next()) {
53 cacheID = query.value(0).toULongLong();
54 }
55 }
56 if (!cacheID) {
57 emit cacheNotFound(telemetryID);
58 return true;
59 }
60 //cache exists
61 query.prepare("UPDATE TelemetryCache SET time=?"
62 " WHERE key=?");
63 query.addBindValue(t);
64 query.addBindValue(cacheID);
65 if (!query.exec())
66 return false;
67 if (discarded())
68 return true;
69 emit cacheFound(telemetryID);
70 return true;
71}

73bool DBReqTelemetryMakeCache::run(QSqlQuery &query)
74{
75 QElapsedTimer t0;
76 t0.start();
77
78 QList<quint64> rmList;
79 //check db invalid cache list
80 if (!static_cast<TelemetryDB *>(db)->invalidCacheList().isEmpty()) {
81 rmList = static_cast<TelemetryDB *>(db)->invalidCacheList();
82 static_cast<TelemetryDB *>(db)->clearInvalidCacheList();
83 }
84 if (forceUpdate)
85 rmList.append(telemetryID);
86
87 for (int i = 0; i < rmList.size(); ++i) {
88 quint64 key = rmList.at(i);
89 if (!db->transaction(query))
90 return false;
91 query.prepare("DELETE FROM TelemetryCache WHERE telemetryID=?");
92 query.addBindValue(key);
93 if (!query.exec())
94 return false;
95 if (!db->commit(query))
96 return false;
97 qDebug() << "cache invalidated for" << key;
98 }
99
100 if (!DBReqTelemetryFindCache::run(query))
101 return false;
102 if (cacheID)
103 return true; //cache found
104
105 //check cache size and remove some if necessary
106 while (1) {
107 if (discarded())
108 return true;
109 query.prepare("SELECT SUM(records) FROM TelemetryCache");
110 if (!query.exec())
111 return false;
112 if (!query.next())
113 return false;
114 quint64 count = query.value(0).toULongLong();
115 if (count < MAX_CACHE_RECORDS)
116 break;
117 query.prepare("SELECT key,MIN(time) FROM TelemetryCache");
118 if (!query.exec())
119 return false;
120 if (!query.next())
121 return false;
122 quint64 key = query.value(0).toULongLong();
123 query.prepare("DELETE FROM TelemetryCache WHERE key=?");
124 query.addBindValue(key);
125 if (!query.exec())
126 return false;
127 qDebug() << "removed" << key;
128 }
129
130 //create new cache record
131 if (!db->transaction(query))
132 return false;
133 bool bCommit = false;
134 while (1) {
135 query.prepare("INSERT INTO TelemetryCache(telemetryID,time) VALUES(?,?)");
136 query.addBindValue(telemetryID);
137 query.addBindValue(t);
138 if (!query.exec())
139 return false;
140 if (discarded())
141 break;

143 //fill data
144 query.prepare("DROP TABLE IF EXISTS tmp");
145 if (!query.exec())
146 return false;
147 query.prepare("CREATE TEMPORARY TABLE tmp AS SELECT time,type,name,value,uid FROM ("
148 " SELECT time,CASE WHEN uplink IS NULL THEN 2 ELSE 3 END AS "

151 " SELECT time,NULL AS type,fieldID AS name,value,NULL AS uid FROM "

153 " UNION ALL"

159 query.addBindValue(telemetryID);

162 return false;
163 if (discarded())
164 break;

167 " SELECT ? AS cacheID,* FROM tmp WHERE true");
168 query.addBindValue(cacheID);
169 if (!query.exec())
170 return false;
171 if (discarded())
172 break;
173 query.prepare("DROP TABLE IF EXISTS tmp");
174 if (!query.exec())
175 return false;
176 if (discarded())
177 break;
178
179 //update records count
180 query.prepare("UPDATE TelemetryCache SET records="
181 " (SELECT COUNT(*) FROM TelemetryCacheData WHERE cacheID=?)"
182 " WHERE key=?");
183 query.addBindValue(cacheID);
184 query.addBindValue(cacheID);
185 if (!query.exec())
186 return false;
187
188 bCommit = true;
189 break;
190 }
191 if (bCommit) {
192 if (!db->commit(query))
193 return false;
194 if (discarded())
195 return true;
196 newCacheID = cacheID;
197 qDebug() << t0.elapsed() << "ms";
198 } else {
199 db->rollback(query);
200 cacheID = 0;
201 }
202 return true;

205bool DBReqTelemetryMakeStats::run(QSqlQuery &query)
206{
207 QElapsedTimer t0;
208 t0.start();
209 query.prepare("SELECT * FROM Telemetry WHERE key=?");
210 query.addBindValue(telemetryID);
211 if (!query.exec())
212 return false;
213 if (query.next()) {
214 //check db invalid cache list
215 stats = queryRecord(query);
216 if (static_cast<TelemetryDB *>(db)->invalidCacheList().contains(telemetryID)) {
217 stats.remove("hash");
218 }
219 } else
220 return false;
221 if (discarded())
222 return true;
223 if (!stats.value("hash").toString().isEmpty()) {
224 //stats exists and valid
225 if (!DBReqTelemetryMakeCache::run(query))
226 return false;
227 emit statsFound(telemetryID, stats);
228 return true;
229 }
230
231 //collect data and update stats record
232 if (discarded())
233 return true;
234
235 //create cache
236 forceUpdate = true;
237 if (!DBReqTelemetryMakeCache::run(query))
238 return false;
239 if (!cacheID) {
240 qWarning() << "missing cache";
241 return true;
242 }
243
244 //data hash
245 query.prepare("SELECT time,type,name,value,uid FROM TelemetryCacheData"
246 " WHERE cacheID=?"
247 " ORDER BY time,type,name");
248 query.addBindValue(cacheID);
249 if (!query.exec())
250 return false;
251 QCryptographicHash h(QCryptographicHash::Sha1);
252 getHash(h, query);
253 QString hash = h.result().toHex().toUpper();
254
255 //downlink cnt
256 query.prepare("SELECT COUNT(*) FROM TelemetryCacheData"
257 " WHERE type IS NULL AND cacheID=?");
258 query.addBindValue(cacheID);
259 if (!query.exec())
260 return false;
261 quint64 downlink = query.next() ? query.value(0).toULongLong() : 0;
262 if (discarded())
263 return true;
264
265 //uplink cnt
266 query.prepare("SELECT COUNT(*) FROM TelemetryCacheData"
267 " WHERE type=1 AND cacheID=?");
268 query.addBindValue(cacheID);
269 if (!query.exec())
270 return false;
271 quint64 uplink = query.next() ? query.value(0).toULongLong() : 0;
272 if (discarded())
273 return true;
274
275 //events cnt
276 quint64 events = 0;
277 QString evtDetails;
278 QMap<QString, quint64> evtCount;
279 query.prepare("SELECT COUNT(*),name FROM TelemetryCacheData"
280 " WHERE (type=2 OR type=3) AND cacheID=?"
281 " GROUP BY name ORDER BY name");
282 query.addBindValue(cacheID);
283 if (!query.exec())
284 return false;
285 while (query.next()) {
286 if (discarded())
287 return true;
288 quint64 ecnt = query.value(0).toULongLong();
289 events += ecnt;
290 if (query.value(0).toULongLong())
291 evtCount[query.value(1).toString()] = ecnt;
292 }
293 QStringList st;
294 foreach (QString s, evtCount.keys()) {
295 st.append(QString("%1=%2").arg(s).arg(evtCount.value(s)));
296 }
297 evtDetails = st.join(',');
298 if (discarded())
299 return true;
300
301 //total time in record
302 /*query.prepare("SELECT MIN(time),MAX(time) FROM TelemetryCacheData"
303 " WHERE cacheID=?"
304 );
305 query.addBindValue(cacheID);
306 if(!query.exec())return false;
307 if(!query.next())return false;
308*/
309 query.prepare("SELECT MIN(time) FROM TelemetryCacheData"
310 " WHERE cacheID=?");
311 query.addBindValue(cacheID);
312 if (!query.exec())
313 return false;
314 if (!query.next())
315 return false;
316 quint64 st0 = query.value(0).toULongLong();
317 query.prepare("SELECT MAX(time) FROM TelemetryCacheData"
318 " WHERE cacheID=?");
319 query.addBindValue(cacheID);
320 if (!query.exec())
321 return false;
322 if (!query.next())
323 return false;
324 quint64 totalTime = query.value(0).toULongLong() - st0;
325
326 //save recalculated values to table
327 bool ok = false;
328 while (1) {
329 if (totalTime != stats.value("totalTime").toULongLong())
330 break;
331 if (downlink != stats.value("downlink").toULongLong())
332 break;
333 if (uplink != stats.value("uplink").toULongLong())
334 break;
335 if (events != stats.value("events").toULongLong())
336 break;
337 if (evtDetails != stats.value("evtDetails").toString())
338 break;
339 if (hash != stats.value("hash").toString())
340 break;
341 ok = true;
342 break;
343 }
344 quint64 key = stats.value("key").toULongLong();
345 stats.clear();
346 stats["totalTime"] = totalTime;
347 stats["downlink"] = downlink;
348 stats["uplink"] = uplink;
349 stats["events"] = events;
350 stats["evtDetails"] = evtDetails;
351 stats["hash"] = hash;
352
353 if (!ok) {
354 if (recordUpdateQuery(query, stats, "Telemetry", "WHERE key=?")) {
355 query.addBindValue(key);
356 if (!query.exec())
357 return false;
358 emit dbModified();
359 }
360 }
361
362 if (discarded())
363 return true;
364
365 qDebug() << t0.elapsed() << "ms";
366 emit statsUpdated(telemetryID, stats);
367 return true;
368}

370bool DBReqTelemetryReadData::run(QSqlQuery &query)
371{
372 //QTime t0;
373 //t0.start();
374 if (discarded())
375 return true;
376 emit progress(telemetryID, 0);
377
378 //get count for progress
379 query.prepare("SELECT * FROM Telemetry WHERE key=?");
380 query.addBindValue(telemetryID);
381 if (!query.exec())
382 return false;
383
384 if (!query.next())
385 return false;
386
387 quint64 count = 0;
388 count += query.value("downlink").toULongLong();
389 count += query.value("uplink").toULongLong();
390 count += query.value("events").toULongLong();
391
392 info = filterIdValues(queryRecord(query));
393
394 //collect fields map
395 query.prepare("SELECT key, name FROM TelemetryFields");
396 if (!query.exec())
397 return false;
398 while (query.next()) {
399 quint64 fid = query.value(0).toULongLong();
400 fieldNames.insert(fid, query.value(1).toString());
401 }
402
403 //data
404 query.prepare("SELECT * FROM TelemetryCache"
405 " WHERE telemetryID=?");
406 query.addBindValue(telemetryID);
407 if (!query.exec())
408 return false;
409 if (!query.next())
410 return false;
411 cacheID = query.value(0).toULongLong();
412
413 const QStringList rows = QStringList() << "time"
414 << "type"
415 << "name"
416 << "value"
417 << "uid";
418 query.prepare(QString("SELECT %1 FROM TelemetryCacheData"
419 " WHERE cacheID=?")
420 .arg(rows.join(',')));
421 query.addBindValue(cacheID);
422 if (!query.exec())
423 return false;
424
425 quint64 row = 0;
426 int vp_s = 0;
427 records.names = rows;
428 QSet<quint64> usedFields;
429 //qDebug()<<t0.elapsed()<<"ms";
430 while (query.next()) {
431 if (discarded())
432 return true;
433 int vp = count > 0 ? row * 100 / count : 0;
434 if (vp_s != vp) {
435 vp_s = vp;
436 emit progress(telemetryID, vp);
437 }
438 row++;
439 if (query.value(1).toUInt() <= 1) {
440 usedFields.insert(query.value(2).toULongLong());
441 }
442 records.values.append(QVariantList() << query.value(0) << query.value(1) << query.value(2)
443 << query.value(3) << query.value(4));
444 }
445 if (discarded())
446 return true;
447
448 for (auto k : fieldNames.keys()) {
449 if (!usedFields.contains(k))
450 fieldNames.remove(k);
451 }
452
453 emit dataLoaded(telemetryID, cacheID, records, fieldNames);
454 emit progress(telemetryID, -1);
455
456 return true;
457}
