57void FirmwareSelect::updateNodeEnums()
58{
59 ApxFw *apxfw = AppGcs::apxfw();
60 if (!apxfw->f_current)
61 return;
62 QStringList st;
63 for (int i = 0; i < apxfw->f_current->size(); ++i) {
64 st.append(apxfw->f_current->child(i)->name());
65 }
66 if (apxfw->f_dev) {
67 for (int i = 0; i < apxfw->f_dev->size(); ++i) {
68 st.append(apxfw->f_dev->child(i)->name());
69 }
70 }
71 st.removeDuplicates();
72 st.sort();
73 f_node->setEnumStrings(st);
74 updateHwEnums();
75}
