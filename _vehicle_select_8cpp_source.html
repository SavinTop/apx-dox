39void VehicleSelect::addVehicle(Vehicle *vehicle)
40{
41 Fact *f = new Fact(this, vehicle->name(), vehicle->title(), vehicle->descr());
42 map.insert(vehicle, f);
43
44 connect(f, &Fact::triggered, this, &VehicleSelect::_factTriggered);
45
46 f->bindProperty(vehicle, "visible", true);
47 f->bindProperty(vehicle, "active", true);
48 f->bindProperty(vehicle, "value", true);
49 f->bindProperty(vehicle, "icon", true);
50
51 setEnabled(true);
52}

54void VehicleSelect::_vehicleRemoved(Vehicle *vehicle)
