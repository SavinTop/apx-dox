31NodeField::NodeField(Fact *parent, NodeItem *node, QVariantMap m, size_t id, NodeField *arrayParent)
32 : Fact(parent)
33 , _node(node)
34 , _type(m.value("type").toString())
35 , _id(id)
36 , _fpath(m.value("name").toString())
37{
38 setName(_fpath.split('.').last());
39 setTitle(m.value("title").toString());
40
41 auto funits = m.value("units").toString();
42 _array = m.value("array").toInt();
43
44 new NodeViewActions(this, node->nodes());
45
46 if (_array && !arrayParent) {
47 // create array sub fields
48 auto st = funits.split(',');
49 if (st.size() != _array || _type == "option")
50 st.clear();
51 for (auto i = 0; i < _array; ++i) {
52 auto item = m;
53 QString s;
54 if (st.isEmpty())
55 s = QString::number(i + 1);
56 else {
57 s = st.at(i);
58 item.remove("units");
59 }
60 item.insert("name", QString("%1_%2").arg(_fpath).arg(s.toLower()));
61 item.insert("title", s);
62 item.insert("array", i);
63 auto *f = new NodeField(this, node, item, id, this);
64 connect(f,
65 &Fact::valueChanged,
66 this,
67 &NodeField::updateArrayStatus,
68 Qt::QueuedConnection);
69 }
70 setTreeType(Group);
71 setOption(ModifiedGroup);
72 updateArrayStatus();
73 return;
74 }
75
76 setOption(ModifiedTrack);
77 setUnits(funits);
78
79 if (_type == "real") {
80 setDataType(Float);
81 setPrecision(6);
82 } else if (_type == "byte") {
83 setMax(255);
84 setMin(0);
85 setDataType(Int);
86 } else if (_type == "word") {
87 setMax(65535);
88 setMin(0);
89 setDataType(Int);
90 } else if (_type == "dword") {
91 setMin(0);
92 setDataType(Int);
93 } else if (_type == "bind") {
94 setDataType(Int);
95 setUnits("mandala");
96 } else if (_type == "option") {
97 setDataType(Enum);
98 setEnumStrings(funits.split(','));
99 setUnits(QString());
100 } else if (_type == "text" || _type == "string") {
101 setDataType(Text);
102 } else if (_type == "script") {
103 setDataType(Text);
104 setUnits("script");
105 _script = new NodeScript(this);
106 } else {
107 qDebug() << "Unknown node field data type" << _type << name() << descr();
108 }

111void NodeField::updateArrayStatus()
112{
113 // called for arrays only
114
115 if (size() == 3 && _type == "real") {
116 QStringList st;
117 for (int i = 0; i < size(); ++i) {
118 st.append(child(i)->valueText());
119 }
120 setValue(QString("(%1)").arg(st.join(',')));
121 }
122
123 int acnt = 0;
124 for (auto i : facts()) {
125 if (i->isZero())
126 continue;
127 auto s = i->valueText();
128 if (s.isEmpty())
129 continue;
130 if (s == "0")
131 continue;
132 acnt++;
133 }
134 if (acnt > 0)
135 setValue(QString("[%1/%2]").arg(acnt).arg(size()));
136 else
137 setValue(QString("[%1]").arg(size()));
138}

147QVariant NodeField::toVariant()
148{
149 if (size() > 0) {

153 list.append(static_cast<NodeField *>(i)->toVariant());

155 return list;

157
158 if (_type == "script")
159 return value();
160
161 if (_type == "real")
162 return QString::number(value().toFloat());
163

166void NodeField::fromVariant(const QVariant &var)
167{
168 if (size() > 0) {
169 //expanded field - i.e. array
170 if (var.canConvert<QVariantList>()) {
171 QVariantList values = var.value<QVariantList>();
172 for (int i = 0; i < values.size(); ++i) {
173 if (i >= size())
174 break;
175 child(i)->fromVariant(values.at(i));
176 }
177 return;
178 }
179 qWarning() << path() << var;
180 return;
181 }
182
183 if (_type == "real") {
184 setValue(QString::number(var.toFloat()));
185 return;
186 }
187
188 setValue(var);
189}
