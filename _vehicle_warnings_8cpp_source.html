62Fact *VehicleWarnings::createItem(const QString &msg, MsgType kind)
63{
64 Fact *fact = nullptr;
65 if (size() > 0) {
66 fact = child(0);
67 if (fact->title() != msg || fact->property("kind").toInt() != kind)
68 fact = nullptr;
69 }
70 if (!fact) {
71 fact = new Fact(this, "item#", msg, "");
72 fact->move(0);
73 if (size() > 100)
74 child(size() - 1)->deleteFact();
75 fact->setValue(1);
76 fact->setProperty("kind", kind);
77 switch (kind) {
78 case INFO:
79 fact->setIcon("information");
80 break;
81 case WARNING:
82 fact->setIcon("alert-circle");
83 break;
84 case ERROR:
85 fact->setIcon("alert-octagon");
86 break;
87 }
88 fact->setDescr(QDateTime::currentDateTime().toString());
89 connect(fact, &Fact::destroyed, this, [=]() {
90 showMap.remove(fact);
91 showList.removeAll(fact);
92 });
93 } else {
94 fact->setValue(fact->value().toUInt() + 1);
95 }
96 foreach (Fact *f, showList) {
97 if (f->title() != fact->title())
98 continue;
99 showList.removeAll(f);
100 showMap.remove(f);
101 break;
102 }
103 emit show(fact->title(), kind);
104 showList.insert(showNum, fact);
105 showMap.insert(fact, 0);
106 showNum = showList.indexOf(fact);
107 showTimer.stop();
108 showTimerTimeout();
109 return fact;
110}

112void VehicleWarnings::showTimerTimeout()
113{
114 if (showList.isEmpty()) {
115 return;
116 }
117 //continuously show all facts
118 if (showNum >= showList.size())
119 showNum = 0;
120 Fact *fact = showList.at(showNum);
121 emit showMore(fact->title(), (MsgType) fact->property("kind").toInt());
122 //qDebug()<<fact->title();
123
124 //next item
125 if (++showMap[fact] >= 3) {
126 showList.removeAll(fact);
127 showMap.remove(fact);
128 } else
129 showNum++;
130 if (showNum >= showList.size())
131 showNum = 0;
132 if (!showList.isEmpty()) {
133 showTimer.start();
134 }
135}
