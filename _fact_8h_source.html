101 template<class T>
102 T *findParent() const
103 {
104 for (Fact *i = const_cast<Fact *>(this); i; i = i->parentFact()) {
105 T *p = qobject_cast<T *>(i);
106 if (p)
107 return p;
108 }
109 return nullptr;
110 }
