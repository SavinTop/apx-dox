62void PApxVehicle::process_downlink(PStreamReader &stream)
63{
64 if (stream.available() < xbus::pid_s::psize()) {
65 qWarning() << "packet" << stream.dump_payload();
66 return;
67 }
68 stream.trim();
69
70 xbus::pid_s pid;
71 pid.read(&stream);
72
73 _papx->trace_pid(pid);
74
75 mandala::uid_t uid = pid.uid;
76
77 if (uid > mandala::uid_max) {
78 qWarning() << "wrong uid" << uid << stream.dump_payload();
79 return;
80 }
81 emit packetReceived(uid);
82
83 if (static_cast<PApxNodes *>(m_nodes)->process_downlink(pid, stream)) {
84 setStreamType(NMT);
85 return;
86 }
87
88 if (static_cast<PApxData *>(m_data)->process_downlink(pid, stream)) {
89 return;
90 }
91
92 if (static_cast<PApxTelemetry *>(m_telemetry)->process_downlink(pid, stream)) {
93 return;
94 }
95}

97void PApxVehicle::send_uplink(QByteArray packet)
98{
99 if (!_squawk) {
100 //local vehicle
101 _papx->send_uplink(packet);
102 return;
103 }
104
105 if (!_papx->addressed(_squawk)) {
106 qWarning() << "not addressed" << PApx::squawkText(_squawk);
107 return;
108 }
109
110 _req.request(mandala::cmd::env::vehicle::uplink::uid);
111
112 _req.write<xbus::vehicle::squawk_t>(_squawk);
113 trace()->block(PApx::squawkText(_squawk));
114 trace()->block(title().append(':'));
115 _req.append(packet);
116
117 _req.send();
118}

120bool PApxVehicle::check_vuid(uint8_t n, uint8_t seq) const
121{
122 // seq=[0,1,2,3]
123 seq &= 3;
124 for (uint8_t i = 0; i < sizeof(xbus::vehicle::uid_t); i += 4)
125 if (_vuid[i + seq] == n)
126 return true;
127 return false;
128}
