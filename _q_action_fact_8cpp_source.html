25QActionFact::QActionFact(Fact *f, const QColor &iconColor)
26 : QAction(f)
27 , fact(f)
28 , iconColor(iconColor)
29{
30 if (f->dataType() == Fact::Apply)
31 setObjectName("greenAction");
32 else if (f->dataType() == Fact::Remove)
33 setObjectName("redAction");
34
35 connect(f, &Fact::titleChanged, this, &QActionFact::updateText);
36 connect(f, &Fact::descrChanged, this, &QActionFact::updateToolTip);
37 connect(f, &Fact::iconChanged, this, &QActionFact::updateIcon);
38 connect(f, &Fact::enabledChanged, this, &QActionFact::updateEnabled);
39 connect(f, &Fact::visibleChanged, this, &QActionFact::updateVisible);
40 connect(f, &Fact::valueChanged, this, &QActionFact::updateChecked);
41
42 connect(this, &QAction::triggered, f, [f]() { f->trigger(); });
43
44 QKeySequence::StandardKey key
45 = f->opts().value("shortcut", QKeySequence::UnknownKey).value<QKeySequence::StandardKey>();
46 if (key != QKeySequence::UnknownKey)
47 setShortcut(key);
48
49 QAction::MenuRole role = f->opts().value("role", QAction::NoRole).value<QAction::MenuRole>();
50 if (role != QAction::NoRole)
51 setMenuRole(role);
52
53 setCheckable(f->dataType() == Fact::Bool);
54 connect(this, &QAction::triggered, this, &QActionFact::actionTriggered);
55
56 updateText();
57 updateToolTip();
58 updateIcon();
59 updateEnabled();
60 updateVisible();
61 updateChecked();
62}
