54QVariant FactDelegateArrayModel::data(const QModelIndex &index, int role) const
55{
56 if (index.column() == 0) {
57 Fact *f = group->child(index.row());
58 if (!f)
59 return QVariant();
60 return f->data(Fact::FACT_MODEL_COLUMN_NAME, role);
61 }
62
63 Fact *f = field(index);
64 if (!f)
65 return QVariant();
66
67 if (role == Qt::ForegroundRole && !f->modified()) {
68 Fact *bind_item = group->child(index.row()) ? group->child(index.row())->child(0) : nullptr;
69 if (bind_item && bind_item->name() == "bind" && bind_item->isZero()) {
70 return QColor(Qt::darkGray);
71 }
72 }
73
74 return f->data(Fact::FACT_MODEL_COLUMN_VALUE, role);
75}

132QVariant FactDelegateArrayModel::headerData(int section, Qt::Orientation orientation, int role) const
133{
134 Q_UNUSED(orientation);
135 if (!group || !group->child(0))
136 return QVariant();
137 if (section == 0)
138 return QVariant("#");
139 Fact *f = group->child(0)->child(section - 1);
140 if (!f)
141 return QVariant();
142
143 if (role == Qt::DisplayRole)
