87void MissionGroup::updateDescr()
88{
89 uint d = distance();
90 uint t = time();
91 if ((d | t) == 0 && (!_descr.isEmpty()))
92 setDescr(_descr);
93 else {
94 QStringList st;
95 st.append(AppRoot::distanceToString(d));
96 st.append(AppRoot::timeToString(t, false));
97 setDescr(st.join(' '));
98 }
99}
