120void CompassFrame::calibrationData(mandala::uid_t uid, QByteArray data)
121{
122 if (uid != mandala::sns::nav::mag::uid)
123 return;
124
125 PStreamReader stream(data);
126 if (stream.available() != (sizeof(uint8_t) + 3 * sizeof(float)))
127 return;
128
129 uint8_t pri;
130 stream >> pri;
131
132 double x, y, z;
133 x = stream.read<float>();
134 y = stream.read<float>();
135 z = stream.read<float>();
136
137 dArea[0]->addData(x, y);
138 dArea[1]->addData(x, z);
139 dArea[2]->addData(z, y);
140 lbInfo.setText(QString("bias {%1, %2, %3}\tscale {%4, %5, %6}")
141 .arg(dArea[0]->bX, 0, 'f', 2)
142 .arg(dArea[0]->bY, 0, 'f', 2)
143 .arg(dArea[1]->bY, 0, 'f', 2)
144 .arg(dArea[0]->sX, 0, 'f', 2)
145 .arg(dArea[0]->sY, 0, 'f', 2)
146 .arg(dArea[1]->sY, 0, 'f', 2));
147}
