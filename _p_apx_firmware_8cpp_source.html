43void PApxFirmware::start()
44{
45 if (_uid.isEmpty())
46 return;
47
48 _node = _nodes->getNode(_uid);
49 emit upgradeStarted(_uid, _name);
50
51 qDebug() << _name << _uid;
52
53 if (_name == "ldr") {
54 rebootFinished();
55 } else {
56 auto req = new PApxNodeRequestReboot(_node, xbus::node::reboot::loader);
57 req->silent = true;
58 connect(req, &PApxNodeRequest::finished, this, &PApxFirmware::rebootFinished);
59 }
60}

62void PApxFirmware::rebootFinished()
63{
64 if (!_node)
65 return;
66
67 qDebug() << _name << _uid;
68
69 auto req = new PApxNodeRequestIdent(_node);
70 req->silent = true;
71 connect(req, &PApxNodeRequest::finished, this, &PApxFirmware::identFinished);
72}
73void PApxFirmware::identFinished()
74{
75 qDebug() << _name << _uid << _node;
76
77 if (!_node)
78 return;
79
80 if (_node->file(_name)) {
81 auto req = new PApxNodeRequestFileWrite(_node, _name, _data, _offset);
82 connect(req, &PApxNodeRequest::finished, this, &PApxFirmware::finish);
83 connect(req, &PApxNodeRequestFile::uploaded, this, &PApxFirmware::fileUploaded);
84 connect(req, &PApxNodeRequestFile::progress, this, &PApxFirmware::fileProgress);
85 } else if (++_retry <= 5) {
86 apxMsg() << tr("Upgrade retry %1/%2...").arg(_retry).arg(5);
87 start();
88 } else {
89 finish();
90 }
91}
