113void SimMods::simRead()
114{
115 while (_udp_sim.hasPendingDatagrams()) {
116 qint64 sz = _udp_sim.pendingDatagramSize();
117 // qDebug() << sz;
118 QNetworkDatagram datagram = _udp_sim.receiveDatagram();
119 if (!datagram.isValid())
120 continue;
121
122 auto rcnt = datagram.data().size();
123
124 const uint8_t *data = (const uint8_t *) datagram.data().constData();
125 while (rcnt > 0) {
126 auto dcnt = _dec.decode(data, rcnt);
127
128 switch (_dec.status()) {
129 case SerialDecoder::PacketAvailable: {
130 data = static_cast<const uint8_t *>(data) + dcnt;
131 parse_sim_rx(_dec.data(), _dec.size());
132 break;
133 }
134
135 case SerialDecoder::DataAccepted:
136 case SerialDecoder::DataDropped:
137 rcnt = 0;
138 break;
139 default:
140 break;
141 }
142
143 if (rcnt > dcnt)
144 rcnt -= dcnt;
145 else
146 rcnt = 0;
147 }
148 }
149}

151void SimMods::parse_sim_rx(const void *data, size_t size)
152{
153 XbusStreamReader stream(data, size);
154 if (stream.available() < xbus::pid_s::psize())
155 return;
156 xbus::pid_s pid;
157 pid.read(&stream);
158
159 mandala::uid_t uid = pid.uid;
160 if (!mandala::cmd::env::sim::match(uid))
161 return;
162
163 // qDebug() << "RX" << size << uid;
164
165 switch (uid) {
166 default:
167 qDebug() << "RX" << size << uid;
168 break;
169 case mandala::cmd::env::sim::sns::uid:
170 if (stream.available() < sizeof(mandala::bundle::sim_s)) {
171 qWarning() << "RX" << size << uid << "too small";
172 return;
173 }
174
175 // read simulated sensors data structure
176 _pid_sim_rx = pid; // save for tx
177 auto d = (mandala::bundle::sim_s *) stream.ptr();
178
179 // apply modifications
180 f_mod->modify(d);
181
182 // check if JSON mods enabled
183 if (!f_ext_enb->value().toBool())
184 break;
185
186 // serialize bundle to variant map
187 QVariantMap map;
188
189 map["att_deg"] = QVariantList{d->att_deg[0], d->att_deg[1], d->att_deg[2]};
190 map["gyro_deg"] = QVariantList{d->gyro_deg[0], d->gyro_deg[1], d->gyro_deg[2]};
191 map["acc_ned"] = QVariantList{d->acc_ned[0], d->acc_ned[1], d->acc_ned[2]};
192 map["lat_deg"] = d->lat_deg;
193 map["lon_deg"] = d->lon_deg;
194 map["hmsl"] = d->hmsl;
195 map["vel_ned"] = QVariantList{d->vel_ned[0], d->vel_ned[1], d->vel_ned[2]};
196 map["agl"] = d->agl;
197 map["rpm"] = d->rpm;
198 map["airspeed"] = d->airspeed;
199 map["baro_mbar"] = d->baro_mbar;
200 map["room_temp"] = d->room_temp;
201 map["air_temp"] = d->air_temp;
202 map["slip"] = d->slip;
203
204 // qDebug() << map;
205
206 // convert to JSON
207 QJsonDocument doc = QJsonDocument::fromVariant(map);
208 QByteArray json = doc.toJson(QJsonDocument::Compact);
209
210 // qDebug() << json;
211
212 // forward JSON to UDP_PORT_MOD_SNS
213 _udp_json.writeDatagram(json, QHostAddress::LocalHost, f_port_jtx->value().toInt());
214
215 // check for JSON RX timeout
216 auto to = f_timeout_jrx->value().toInt();
217 if (to > 0 && _json_rx_time.elapsed() > to * 1000) {
218 // no JSON data received for a while
219 // loopback to APX autopilot app as-is
220 break;
221 }
222
223 return;
224 }
225
226 // forward to APX autopilot app as-is
227 forwardToAP(data, size);
228}

230void SimMods::jsonRead()
231{
232 while (_udp_json.hasPendingDatagrams()) {
233 QNetworkDatagram datagram = _udp_json.receiveDatagram();
234 if (!datagram.isValid())
235 continue;
236
237 _json_rx_time.start();
238
239 // qDebug() << datagram.data();
240
241 QJsonParseError err;
242 QJsonDocument doc = QJsonDocument::fromJson(datagram.data(), &err);
243 if (err.error != QJsonParseError::NoError) {
244 qWarning() << err.errorString();
245 continue;
246 }
247
248 QVariantMap map = doc.toVariant().toMap();
249
250 // qDebug() << map;
251
252 // de-serialize bundle from variant map
253 mandala::bundle::sim_s d;
254
255 auto v = map["att_deg"].toList();
256 d.att_deg[0] = v[0].toDouble();
257 d.att_deg[1] = v[1].toDouble();
258 d.att_deg[2] = v[2].toDouble();
259
260 v = map["gyro_deg"].toList();
261 d.gyro_deg[0] = v[0].toDouble();
262 d.gyro_deg[1] = v[1].toDouble();
263 d.gyro_deg[2] = v[2].toDouble();
264
265 v = map["acc_ned"].toList();
266 d.acc_ned[0] = v[0].toDouble();
267 d.acc_ned[1] = v[1].toDouble();
268 d.acc_ned[2] = v[2].toDouble();
269
270 d.lat_deg = map["lat_deg"].toDouble();
271 d.lon_deg = map["lon_deg"].toDouble();
272 d.hmsl = map["hmsl"].toDouble();
273
274 v = map["vel_ned"].toList();
275 d.vel_ned[0] = v[0].toDouble();
276 d.vel_ned[1] = v[1].toDouble();
277 d.vel_ned[2] = v[2].toDouble();
278
279 d.agl = map["agl"].toDouble();
280 d.rpm = map["rpm"].toDouble();
281 d.airspeed = map["airspeed"].toDouble();
282 d.baro_mbar = map["baro_mbar"].toDouble();
283 d.room_temp = map["room_temp"].toDouble();
284 d.air_temp = map["air_temp"].toDouble();
285 d.slip = map["slip"].toDouble();
286
287 // qDebug() << d;
288
289 // send modified structure to the autopilot
290 XbusStreamWriter stream(_buf_sim_tx, sizeof(_buf_sim_tx));
291 _pid_sim_rx.write(&stream); // saved from rx
292 stream.write(&d, sizeof(d));
293
294 forwardToAP(stream.buffer(), stream.pos());
295 }
