67void AppImageAutoUpdater::checkForUpdatesInBackground()
68{
69 checkInstalled();
70 auto updater = createUpdater(qgetenv("APPIMAGE"), false);
71 if (updater) {
72 auto t = std::thread([this, updater]() {
73 bool updateAvailable = false;
74 updater->checkForChanges(updateAvailable);
75 if (updateAvailable) {
76 setState(UpdateAvailable);
77 trigger();
78 } else
79 setState(NoUpdates);
80 });
81 t.detach();
82 }

105void AppImageAutoUpdater::start(bool keepOldVersion)
106{
107 checkInstalled();
108 auto updater = createUpdater(qgetenv("APPIMAGE"), keepOldVersion);
109 if (updater) {
110 setUpdateProgress(0);
111 setState(Updating);
112 setStatusMessage("");
113 m_stopUpdate = false;
114 updater->start();
115 while (!updater->isDone()) {
116 if (updater->hasError()) {
117 QMessageBox::critical(nullptr,
118 QObject::tr("Error"),
119 QObject::tr("Unknown error during update"),
120 QMessageBox::Ok);
121 setState(UpdateAvailable);
122 return;
123 }
124 if (m_stopUpdate) {
125 try {
126 updater->stop();
127 } catch (std::exception &) {
128 }
129 setState(UpdateAvailable);
130 return;
131 }
132
133 double progress = 0;
134 updater->progress(progress);
135 setUpdateProgress(qRound(progress * 100));
136 QCoreApplication::processEvents();
137 std::string nextMessage;
138 while (updater->nextStatusMessage(nextMessage)) {
139 setStatusMessage(QString::fromStdString(nextMessage));
140 QCoreApplication::processEvents();
141 }
142 }
143
144 //It's workaround for updater, that for some reason keep .zs-old file after update.
145 QFile zsOld(qgetenv("APPIMAGE"));
146 zsOld.setFileName(zsOld.fileName() + ".zs-old");
147 if (zsOld.exists())
148 zsOld.remove();
149
150 std::string newFile;
151 updater->pathToNewFile(newFile);
152 QProcess::startDetached(QString::fromStdString(newFile), QStringList());
153 QCoreApplication::exit(0);
154
155 setState(NoUpdates);
156 }
157}

204QNetworkReply *AppImageAutoUpdater::request(QUrl url)
205{
206 QNetworkRequest *request = new QNetworkRequest(url);
207
208 QSslConfiguration ssl = request->sslConfiguration();
209 ssl.setPeerVerifyMode(QSslSocket::VerifyNone);
210 request->setSslConfiguration(ssl);
211
212 request->setRawHeader("Accept", "*/*");
213 request->setRawHeader("User-Agent",
214 QString("%1 (v%2)")
215 .arg(QCoreApplication::applicationName())
216 .arg(App::version())
217 .toUtf8());
218
219 return net.get(*request);
220}
221void AppImageAutoUpdater::requestReleaseNotes()
222{
223 QUrl url("https://api.github.com/repos/uavos/apx-gcs/releases/latest");
224 QNetworkReply *reply = request(url);
225
226 connect(reply, &QNetworkReply::finished, this, [this, reply]() {
227 reply->deleteLater();
228 if (reply->error()) {
229 qWarning() << reply->errorString();
230 return;
231 }
232 QJsonDocument json(QJsonDocument::fromJson(reply->readAll()));
233 QString sver = json["tag_name"].toString();
234 qDebug() << "latest version:" << sver;
235 if (sver.isEmpty())
236 return;
237 m_latestVersion = sver;
238 emit latestVersionChanged();
239
240 QUrl url(QString("https://uavos.github.io/apx-gcs/docs/releases/%1.md").arg(sver));
241 QNetworkReply *reply2 = request(url);
242 connect(reply2, &QNetworkReply::finished, this, [this, reply2]() {
243 reply2->deleteLater();
244 if (reply2->error()) {
245 qWarning() << reply2->errorString();
246 return;
247 }
248 m_releaseNotes = reply2->readAll();
249 emit releaseNotesChanged();
250 //qDebug() << m_releaseNotes;
251 });
252 });
253}
