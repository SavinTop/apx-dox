140bool JSTreeProxyModel::lessThan2(const QModelIndex &left, const QModelIndex &right) const
141{
142 //only first col sorted
143 if (left.column() != JSTreeModel::JS_MODEL_COLUMN_NAME
144 || right.column() != JSTreeModel::JS_MODEL_COLUMN_NAME)
145 return left.row() < right.row();
146 JSTreeItem *item_left = left.data(JSTreeModel::ModelDataRole).value<JSTreeItem *>();
147 JSTreeItem *item_right = right.data(JSTreeModel::ModelDataRole).value<JSTreeItem *>();
148 if (!(item_left && item_right))
149 return QSortFilterProxyModel::lessThan(left, right);
150 return JSTreeItem::lessThan(item_left, item_right);
151}
