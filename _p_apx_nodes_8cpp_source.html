33PApxNodes::PApxNodes(PApxVehicle *parent)
34 : PNodes(parent)

63bool PApxNodes::process_downlink(const xbus::pid_s &pid, PStreamReader &stream)
64{
65 if (!mandala::cmd::env::nmt::match(pid.uid))
66 return false;
67
68 // if upgrading - forward all to local
69 if (upgrading() && !_local) {
70 auto local = findParent<PApx>()->local();
71 trace()->block("LOCAL");
72 trace()->tree();
73 auto nodes = static_cast<PApxNodes *>(local->nodes());
74 return nodes->process_downlink(pid, stream);
75 }
76
77 if (stream.available() < sizeof(xbus::node::guid_t)) {
78 if (pid.pri != xbus::pri_request)
79 qDebug() << "missing guid" << stream.available();
80 return true;
81 }
82
83 QByteArray uid_ba(sizeof(xbus::node::guid_t), '\0');
84 stream.read(uid_ba.data(), sizeof(xbus::node::guid_t));
85 QString uid(uid_ba.toHex().toUpper());
86
87 // filter broadcast requests
88 if (uid.isEmpty() || uid.count('0') == uid.size())
89 return true;
90
91 trace()->block("GUID");
92
93 PApxNode *node = getNode(uid);
94 if (!node)
95 return true;
96
97 trace()->block(node->title().append(':'));
98 trace()->tree();
99
100 node->process_downlink(pid, stream);
101
102 if (pid.pri == xbus::pri_response)
103 emit node_response(node);
104
105 return true;
106}

108PApxNode *PApxNodes::getNode(QString uid, bool createNew)
109{
110 if (uid.isEmpty() || uid.count('0') == uid.size())
111 return nullptr;
112
113 PApxNode *node = _nodes.value(uid, nullptr);
114 if (node)
115 return node;
116
117 if (!createNew)
118 return nullptr;
119
120 node = new PApxNode(this, uid);
121 _nodes.insert(uid, node);
122 connect(node, &Fact::removed, this, [this, node]() { _nodes.remove(_nodes.key(node)); });
123 connect(node, &PApxNode::request_scheduled, this, &PApxNodes::request_scheduled);
124 connect(node, &PApxNode::request_finished, this, &PApxNodes::request_finished);
125 connect(node, &PApxNode::request_extended, this, &PApxNodes::request_extended);
126
127 emit node_available(node);
128 return node;
129}

137void PApxNodes::request_scheduled(PApxNodeRequest *req)
138{
139 // qDebug() << req->title();
140 if (_requests.contains(req)) {
141 // rescheduled request
142 if (_request != req) // not current
143 return;
144 _retry = PApxNodeRequest::retries;
145 _reqTimeout.stop();
146 _reqNext.start();
147 return;
148 }
149 _requests.append(req);
150 if (_request)
151 return;
152 request_next();
153}
154void PApxNodes::request_finished(PApxNodeRequest *req)
155{
156 // qDebug() << req->title();
157 _requests.removeOne(req);
158 if (_request && _request != req)
159 return;
160 _request = nullptr;
161
162 _reqTimeout.stop();
163 if (_requests.isEmpty()) {
164 return;
165 }
166 request_next();
167}

193void PApxNodes::request_current()
194{
195 if (!_request)
196 return;
197 // qDebug() << _request->title();
198 if (!_request->make_request(_req)) {
199 // qDebug() << "discarded";
200 _request->discard();
201 return;
202 }
203 _req.send();
204 _reqTimeout.stop();
205 _reqTimeout.start(_request->timeout_ms() ? _request->timeout_ms() : 100);
206}

208void PApxNodes::request_timeout()
209{
210 if (!_request)
211 return;
212
213 if (!_request->timeout_ms()) {
214 _request->discard();
215 return;
216 }
217
218 if (!_retry) {
219 if (!_request->silent) {
220 apxMsgW() << tr("NMT request dropped").append(':') << _request->title();
221 }
222
223 // clear all node requests
224 cancel_requests(_request->node());
225 return;
226 }
227
228 _retry--;
229 if (!_request->silent) {
230 apxMsgW() << tr("NMT timeout").append(':') << _request->title()
231 << QString("(%1/%2)")
232 .arg(PApxNodeRequest::retries - _retry)
233 .arg(PApxNodeRequest::retries);
234 }
235
236 _reqNext.start();
237}
238void PApxNodes::cancel_requests(PApxNode *node)
239{
240 //qDebug() << node;
241 if (_request && (!node || _request->node() == node)) {
242 _reqTimeout.stop();
243 _request = nullptr;
244 }
245 for (auto req : _requests) {
246 if (node && req->node() != node)
247 continue;
248 _requests.removeOne(req);
249 req->finished();
250 delete req;
251 }
252 if (_requests.isEmpty()) {
253 return;
254 }
255
256 if (!_request)
257 request_next();
258}
