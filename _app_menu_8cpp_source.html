134void AppMenu::updateVehicleTools()
135{
136 Vehicle *v = Vehicles::instance()->current();
137 if (!v)
138 return;
139 vehicleTools->deleteChildren();
140 for (int i = 0; i < v->size(); ++i) {
141 Fact *f = v->child(i);
142 Fact *a = new Fact(vehicleTools, f->name());
143 a->setBinding(f);
144 }
145 updateMenu(vehicleTools->parentFact());
146}

157 connect(a, &Fact::triggered, f, &Fact::trigger);
158 }

161void AppMenu::createMenuBar()
162{
163 if (_menuBar)
164 delete _menuBar;
165
166 _menuBar = new QMenuBar(nullptr);
167 for (int i = 0; i < size(); ++i) {
168 Fact *g = child(i);
169 Fact *f = g->menu();
170 if (!f)
171 continue;
172 QMenu *menu = _menuBar->addMenu(g->title());
173 f->setOpt("qmenu", QVariant::fromValue(menu));
174 updateMenu(f);
175 QAbstractListModel *model = f->model();
176 if (model) {
177 connect(
178 model,
179 &QAbstractListModel::layoutChanged,
180 this,
181 [this, f]() { updateMenu(f); },
182 Qt::QueuedConnection);
183 }
184 }
