96void AppEngine::jsSync(Fact *fact)
97{
98 if (fact->jsname().isEmpty())
99 return;
100
101 QJSValue v = globalObject();
102 //qDebug() << fact << list;
103 //build tree from root
104 const FactList &list = fact->pathList();
105 for (int i = list.size() - 1; i > 0; --i) {
106 Fact *f = list.at(i);
107 QJSValue vp = v.property(f->jsname());
108 if (vp.isUndefined() || (!vp.isQObject()) || vp.toQObject() != f) {
109 jsSetProperty(v, f->jsname(), jsCreateObject(f));
110 jsProtectPropertyWrite(f->jspath());
111 }
112 v = vp;
113 }
114 //set the property and value
115 jsSync(fact, v);
116}

118QJSValue AppEngine::jsSync(Fact *fact, QJSValue parent) //recursive
119{
120 //qDebug() << fact->path();
121 QJSValue js_fact = jsCreateObject(fact);
122
123 jsSetProperty(parent, fact->jsname(), js_fact);
124 jsProtectPropertyWrite(fact->jspath());
125
126 //sync children
127 for (int i = 0; i < fact->size(); ++i)
128 jsSync(fact->child(i), js_fact);
129 //sync actions
130 if (!fact->actions().isEmpty()) {
131 for (auto i : fact->actions()) {
132 jsSync(i, js_fact);
133 }
134 }
135 return js_fact;
136}

138void AppEngine::jsSetProperty(QJSValue parent, QString name, QJSValue value)
139{
140 if (name.isEmpty() || name.contains('#'))
141 return;
142
143 if (value.isUndefined() || value.isNull()) {
144 // delete property
145 parent.deleteProperty(name);
146 return;
147 }
148
149 QJSValue p = parent.property(name);
150 if (!p.isUndefined()) {
151 // modify already existing property
152 if (p.strictlyEquals(value))
153 return;
154 qWarning() << "Rewriting property:" << name << value.toString() << p.toString()
155 << "for parent" << parent.toString();
156 parent.deleteProperty(name);
157 }
158
159 parent.setProperty(name, value);
160}

162void AppEngine::jsProtectPropertyWrite(const QString path)
163{
164 if (path.isEmpty() || path.contains('#'))
165 return;
166
167 QString scope, name;
168 if (path.contains('.')) {
169 scope = path.left(path.lastIndexOf('.'));
170 name = path.mid(scope.size() + 1);
171 } else {
172 scope = "this";
173 name = path;
174 }
175 QString s = QString("Object.defineProperty(%1,'%2',{enumerable:false,configurable:"
176 "false,writable:false})")
177 .arg(scope)
178 .arg(name);
179 // QString s = QString("%1.__defineSetter__('%2',function(v){print('error: read-only (%3)')})")
180 // .arg(scope)
181 // .arg(name)
182 // .arg(path);
183 s = QString("try{%1}catch(e){}").arg(s);
184 jsexec_queued(s);
185}

240QByteArray AppEngine::jsToArray(QJSValue data)
241{
242 //qDebug()<<portID<<data.toString()<<data.isArray()<<data.toVariant().toByteArray().toHex();
243 QByteArray ba;
244 if (data.isString() || data.toString().contains(',')) {
245 bool ok = false;
246 foreach (QString sv, data.toString().trimmed().toLower().split(',', Qt::SkipEmptyParts)) {
247 uint v;
248 sv = sv.trimmed();
249 if (sv.startsWith("0x"))
250 v = sv.mid(2).toUInt(&ok, 16);
251 else
252 v = sv.toUInt(&ok, 10);
253 if (!ok)
254 break;
255 ba.append(static_cast<char>(v));
256 }
257 if (!ok)
258 return QByteArray();
259 } else if (data.isNumber()) {
260 ba.append(static_cast<char>(data.toInt()));
261 } else
262 return QByteArray();
263 return ba;
264}

278QObject *AppEngine::loadQml(const QString qmlFile, const QVariantMap &opts)
279{
280 QString schk = qmlFile;
281 if (schk.startsWith("qrc:"))
282 schk.remove(0, 3);
283 if (!QFile::exists(schk)) {
284 apxMsgW() << tr("Not found").append(':') << schk;
285 return nullptr;
286 }
287
288 QQmlComponent c(this, qmlFile, QQmlComponent::PreferSynchronous);
289 do {
290 if (c.isError())
291 break;
292
293 QObject *obj = c.beginCreate(rootContext());
294 if (c.isError())
295 break;
296
297 for (auto key : opts.keys()) {
298 QQmlProperty::write(obj, key.toUtf8(), opts.value(key));
299 }
300
301 if (c.isError())
302 break;
303 c.completeCreate();
304 if (c.isError())
305 break;
306
307 return obj;
308 } while (0);
309 apxMsgW() << c.errorString();
310 return nullptr;
311}
