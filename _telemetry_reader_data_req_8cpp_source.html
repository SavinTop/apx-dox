25bool TelemetryReaderDataReq::run(QSqlQuery &query)
26{
27 if (!DBReqTelemetryReadData::run(query))
28 return false;
29 if (discarded())
30 return true;
31
32 int iTime = records.names.indexOf("time");
33 int iType = records.names.indexOf("type");
34 int iName = records.names.indexOf("name");
35 int iValue = records.names.indexOf("value");
36 int iUid = records.names.indexOf("uid");
37
38 QGeoPath path;
39 quint64 fidLat = fieldNames.key("est.pos.lat");
40 quint64 fidLon = fieldNames.key("est.pos.lon");
41 quint64 fidHmsl = fieldNames.key("est.pos.hmsl");
42 QVector<QPointF> *vLat = nullptr;
43 QVector<QPointF> *vLon = nullptr;
44 QVector<QPointF> *vHmsl = nullptr;
45
46 fieldData_t fieldData;
47 times_t times;
48 events_t events;
49
50 times.append(0);
51
52 quint64 t0 = 0;
53 QHash<quint64, double> fvalues;
54 int progress_s = 0;
55
56 f_events = new Fact(nullptr,
57 "events",
58 tr("Events"),
59 tr("Recorded events data"),
60 Fact::Group | Fact::Count);
61 f_events->moveToThread(nullptr);
62
63 for (int i = 0; i < records.values.size(); ++i) {
64 if (discarded())
65 return true;
66
67 const QVariantList &r = records.values.at(i);
68 if (r.isEmpty())
69 continue;
70
71 //progress and abort
72 int vp = i * 100 / records.values.size();
73 if (progress_s != vp) {
74 progress_s = vp;
75 emit progress(telemetryID, vp);
76 }
77
78 //time
79 quint64 t = r.at(iTime).toULongLong();
80 if (i == 0)
81 t0 = t;
82 t -= t0;
83 double tf = t / 1000.0;
84 if (times.last() != tf)
85 times.append(tf);
86
87 quint64 fid = 0;
88
89 switch (r.at(iType).toUInt()) {
90 case 0: {
91 //downlink data
92 fid = r.at(iName).toULongLong();
93 } break;
94 case 1: {
95 //uplink data
96 fid = r.at(iName).toULongLong();
97 event_t e;
98 e.time = t;
99 e.name = "uplink";
100 e.value = fieldNames.value(fid, QString::number(fid));
101 e.uid = r.at(iUid).toString();
102 events.append(e);
103 addEventFact(e.time, e.name, e.value, e.uid);
104 } break;
105 case 2:
106 case 3: {
107 //events
108 event_t e;
109 e.time = t;
110 e.name = r.at(iName).toString();
111 e.value = r.at(iValue).toString();
112 e.uid = r.at(iUid).toString();
113 events.append(e);
114 addEventFact(e.time, e.name, e.value, e.uid);
115 } break;
116 }
117
118 if (!fid)
119 continue;
120 QVector<QPointF> *pts = fieldData.value(fid);
121 if (!pts) {
122 pts = new QVector<QPointF>;
123 pts->reserve(1000000);
124 fieldData.insert(fid, pts);
125 }
126 double v = r.at(iValue).toDouble();
127 if (fvalues.contains(fid) && fvalues.value(fid) == v)
128 continue;
129 fvalues[fid] = v;
130
131 if (pts->size() > 0 && (tf - pts->last().x()) > 0.5) {
132 //extrapolate unchanged value tail-1ms
133 pts->append(QPointF(tf, pts->last().y()));
134 }
135 if (pts->isEmpty() && v != 0.0 && t != 0) {
136 //extrapolate/predict initial value at start
137 pts->append(QPointF(0, 0));
138 pts->append(QPointF(tf, 0));
139 }
140 pts->append(QPointF(tf, v));
141
142 //path update
143 while (fid && (fid == fidLat || fid == fidLon || fid == fidHmsl)) {
144 if (!vLat)
145 vLat = fieldData.value(fidLat);
146 if (!vLon)
147 vLon = fieldData.value(fidLon);
148 if (!vHmsl)
149 vHmsl = fieldData.value(fidHmsl);
150
151 if (!(vLat && vLon && vHmsl))
152 break;
153
154 QGeoCoordinate c(vLat->last().y(), vLon->last().y(), vHmsl->last().y());
155
156 if (!c.isValid())
157 break;
158 if (c.latitude() == 0.0)
159 break;
160 if (c.longitude() == 0.0)
161 break;
162 qreal dist = 0;
163 if (!path.isEmpty()) {
164 QGeoCoordinate c0(path.path().last());
165 if (c0.latitude() == c.latitude())
166 break;
167 if (c0.longitude() == c.longitude())
168 break;
169 dist = c0.distanceTo(c);
170 if (dist < 10.0)
171 break;
172 }
173 path.addCoordinate(c);
174 break;
175 }
176 }
177
178 //final data tail at max time
179 qreal tMax = times.isEmpty() ? 0 : times.last();
180 for (int i = 0; i < fieldData.values().size(); ++i) {
181 QVector<QPointF> *pts = fieldData.values().at(i);
182 if (!pts)
183 continue;
184 if (pts->isEmpty()) {
185 pts->append(QPointF(0, 0));
186 //continue;
187 }
188 if (pts->last().x() >= tMax)
189 continue;
190 pts->append(QPointF(tMax, pts->last().y()));
191 }
192
193 if (discarded())
194 return true;
195
196 emit dataProcessed(telemetryID, cacheID, fieldData, fieldNames, times, events, path, f_events);
197
198 return true;
199}

201void TelemetryReaderDataReq::addEventFact(quint64 time,
202 const QString &name,
203 const QString &value,
204 const QString &uid)
205{
206 Fact *g = f_events->child(name);
207 if (!g) {
208 g = new Fact(f_events, name, "", "", Fact::Group | Fact::Count);
209 g->moveToThread(nullptr);
210 }
211
212 Fact *f = nullptr;
213 if (name == "uplink") {
214 f = g->childByTitle(value);
215 if (!f) {
216 f = new Fact(g, value, value, "");
217 f->moveToThread(nullptr);
218 //qDebug() << name << value;
219 f->setValue(1);
220 } else {
221 f->setValue(f->value().toInt() + 1);
222 }
223 } else if (name == "serial") {
224 f = g->childByTitle(uid);
225 if (!f) {
226 f = new Fact(g, uid, uid, "");
227 f->moveToThread(nullptr);
228 //qDebug() << name << value;
229 f->setValue(1);
230 } else {
231 f->setValue(f->value().toInt() + 1);
232 }
233 } else {
234 QString title;
235 if (name == "xpdr")
236 title = name;
237 else
238 title = value;
239 QString descr = uid;
240 if (title.startsWith('[') && title.contains(']')) {
241 int i = title.indexOf(']');
242 QString s = title.mid(1, i - 1).trimmed();
243 title.remove(0, i + 1);
244 if (!s.isEmpty())
245 descr.prepend(QString("%1/").arg(s));
246 }
247 f = new Fact(g, name + "#", title, descr);
248 f->moveToThread(nullptr);
249 QString stime = QTime(0, 0).addMSecs(time).toString("hh:mm:ss.zzz");
250 f->setValue(stime);
251 }
252
253 if (!f)
254 return;
255 f->setProperty("time", QVariant::fromValue(time));
256}
