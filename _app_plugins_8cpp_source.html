50void AppPlugins::load(const QStringList &names)
51{
52 //collect all available plugins filenames
53 QStringList allFiles;
54 QStringList filters(QStringList() << "*.gcs"
55 << "*.so"
56 << "*.dylib"
57 << "*.bundle"
58 << "*.qml");
59
60 QDir userp = AppDirs::userPlugins();
61 userp.setSorting(QDir::Name | QDir::IgnoreCase);
62 userp.refresh();
63 if (!userp.exists())
64 userp.mkpath(".");
65 QStringList stRep, stRepQml;
66 foreach (QString fileName, userp.entryList(filters, QDir::Files | QDir::Dirs)) {
67 if (fileName.startsWith('-'))
68 continue;
69 QString pname = QString(fileName);
70 if (pname.startsWith("lib"))
71 pname.remove(0, 3);
72 if (!pname.endsWith(".qml")) {
73 pname.truncate(pname.lastIndexOf('.'));
74 stRep.append(pname);
75 } else {
76 stRepQml.append(pname.left(pname.lastIndexOf('.')));
77 }
78 allFiles.append(userp.absoluteFilePath(fileName));
79 }
80 if (!stRep.isEmpty())
81 apxConsole() << QObject::tr("User plugins").append(": ").append(stRep.join(','));
82 if (!stRepQml.isEmpty())
83 apxConsole() << QObject::tr("User QML plugins").append(": ").append(stRepQml.join(','));
84
85 QDir pluginsDir = AppDirs::plugins();
86 pluginsDir.setSorting(QDir::Name | QDir::IgnoreCase);
87 pluginsDir.refresh();
88 //apxDebug()<<pluginsDir;
89 foreach (QString fileName, pluginsDir.entryList(filters, QDir::Files | QDir::Dirs)) {
90 //apxDebug()<<fileName;
91 allFiles.append(pluginsDir.absoluteFilePath(fileName));
92 }
93
94 //parse command line arguments (plugins to load)
95 if (!names.isEmpty()) {
96 QStringList st;
97 foreach (QString pname, names) {
98 foreach (QString fname, allFiles) {
99 QString s = QFileInfo(fname).baseName();
100 if (s.startsWith("lib"))
101 s.remove(0, 3);
102 if (s != pname)
103 continue;
104 st.append(fname);
105 }
106 }
107 allFiles.swap(st);
108 }
109
110 apxConsole() << QObject::tr("Loading plugins").append("...");
111
112 //load and initialize
113 QStringList libFiles, qmlFiles;
114 foreach (QString s, allFiles) {
115 if (s.endsWith(".qml", Qt::CaseInsensitive)) {
116 qmlFiles.append(s);
117 } else {
118 libFiles.append(s);
119 }
120 }
121 //sort by name
122 std::sort(libFiles.begin(), libFiles.end(), [](QString a, QString b) {
123 return QFileInfo(a).baseName().localeAwareCompare(QFileInfo(b).baseName()) < 0;
124 });
125
126 std::sort(qmlFiles.begin(), qmlFiles.end(), [](QString a, QString b) {
127 return QFileInfo(a).baseName().localeAwareCompare(QFileInfo(b).baseName()) < 0;
128 });
129
130 //remove duplicates (system but keep user)
131 fixDuplicates(libFiles, userp.absolutePath());
132 fixDuplicates(qmlFiles, userp.absolutePath());
133
134 loadFiles(libFiles);
135 loadFiles(qmlFiles);
136
137 for (auto p : *this) {
138 if (p->f_enabled->value().toBool()) {
139 p->load();
140 } else {
141 apxConsole() << tr("Loading skipped").append(":") << p->name;
142 }
143 }
144
145 emit loaded();
146 updateStatus();
147}

149void AppPlugins::fixDuplicates(QStringList &list, const QString &userPluginsPath) const
150{
151 foreach (QString p, list) {
152 if (!p.startsWith(userPluginsPath))
153 continue;
154 //p is user plugin
155 QString pn = QFileInfo(p).baseName();
156 foreach (QString ps, list) {
157 if (ps.startsWith(userPluginsPath))
158 continue;
159 //ps is system plugin
160 QString psn = QFileInfo(ps).baseName();
161 if (psn == pn) {
162 list.removeAll(ps);
163 apxConsole() << tr("User plugin override").append(":") << psn;
164 }
165 }
166 }
167}

169void AppPlugins::loadFiles(const QStringList &fileNames)
170{
171 QStringList loadedNames;
172 foreach (QString fileName, fileNames) {
173 QString pname = QFileInfo(fileName).baseName();
174 if (pname.startsWith("lib"))
175 pname.remove(0, 3);
176 if (loadedNames.contains(pname)) {
177 apxMsgW() << tr("Duplicate plugin").append(":") << pname;
178 continue;
179 }
180 //apxConsole()<<tr("Loading")<<pname;
181 loadedNames.append(pname);
182
183 AppPlugin *plugin = new AppPlugin(this, pname, fileName);
184 append(plugin);
185 }
186}
