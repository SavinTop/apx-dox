65bool PApxNodeRequest::make_request(PApxRequest &req)
66{
67 _active = true;
68
69 req.request(_uid);
70 QByteArray src(QByteArray::fromHex(_node->uid().toUtf8()));
71 size_t sz = static_cast<size_t>(src.size());
72 if (sz > sizeof(xbus::node::guid_t)) {
73 sz = sizeof(xbus::node::guid_t);
74 qWarning() << "guid size:" << sz;
75 }
76 xbus::node::guid_t guid;
77 memset(guid, 0, sizeof(guid));
78 memcpy(guid, src.data(), sz);
79 req.write(guid, sizeof(guid));
80 trace()->block(_node->title().append(':'));
81 trace()->tree();
82
83 return request(req);
84}

99bool PApxNodeRequestMod::request(PApxRequest &req)
100{
101 switch (_cmd) {
102 default:
103 qWarning() << "wrong mod request" << _cmd << _adr.toHex().toUpper() << _data;
104 return false;
105 case PNode::sh:
106 _op = xbus::node::mod::sh;
107 _timeout_ms = 0;
108 break;
109 case PNode::ls:
110 _op = xbus::node::mod::ls;
111 break;
112 case PNode::status:
113 _op = xbus::node::mod::status;
114 break;
115 }
116
117 req << _op;
118 trace()->block(QString::number(_op));
119
120 if (!_adr.isEmpty()) {
121 req.append(_adr);
122 trace()->data(_adr);
123 }
124
125 if (!_data.isEmpty()) {
126 for (auto const &s : _data) {
127 req.write_string(s.toUtf8().data());
128 }
129 trace()->blocks(_data);
130 }
131 return true;
132}
133bool PApxNodeRequestMod::response(PStreamReader &stream)
134{
135 if (!_active)
136 return false;
137
138 if (stream.available() < sizeof(xbus::node::mod::op_e)) {
139 qWarning() << "size" << stream.available();
140 return false;
141 }
142 xbus::node::mod::op_e op;
143 stream >> op;
144
145 if (op != _op)
146 return false;
147
148 trace()->block(QString::number(op));
149
150 PNode::mod_cmd_e cmd;
151
152 switch (op) {
153 default:
154 return false;
155 case xbus::node::mod::ls:
156 cmd = PNode::ls;
157 break;
158 case xbus::node::mod::status:
159 cmd = PNode::status;
160 break;
161 }
162 // read module adr
163 QByteArray adr;
164 uint8_t n = 0;
165 while (stream.available() > 0) {
166 stream >> n;
167 adr.append(static_cast<char>(n));
168 if (n == 0xFF)
169 break;
170 }
171 if (n != 0xFF || adr.isEmpty()) {
172 qWarning() << "bad adr";
173 return false;
174 }
175 adr.chop(1);
176 if (adr != _adr) // reply for other request
177 return false;
178 trace()->data(adr);
179
180 auto data = stream.read_strings();
181 trace()->blocks(data);
182
183 _node->modReceived(cmd, adr, data);
184
185 return true;
186}

195bool PApxNodeRequestIdent::response(PStreamReader &stream)
196{
197 //trace()->data(stream.payload());
198 if (stream.available() <= (xbus::node::ident::ident_s::psize() + 3 * 2)) {
199 qWarning() << "size" << stream.available() << xbus::node::ident::ident_s::psize();
200 return false;
201 }
202 xbus::node::ident::ident_s ident;
203 ident.read(&stream);
204
205 trace()->block("IDENT");
206
207 QStringList st = stream.read_strings(3);
208 if (st.isEmpty()) {
209 qWarning() << "no strings";
210 return false;
211 }
212 QString sname = st.at(0);
213 QString sversion = st.at(1);
214 QString shardware = st.at(2);
215
216 trace()->blocks(st);
217
218 QStringList fnames = stream.read_strings(ident.flags.bits.files);
219 for (auto i : fnames) {
220 if (!i.isEmpty())
221 continue;
222 fnames.clear();
223 break;
224 }
225 if (fnames.isEmpty()) {
226 qWarning() << "no files";
227 // return false;
228 }
229 trace()->blocks(fnames);
230
231 if (stream.available() > 0) {
232 qWarning() << "corrupted ident_s";
233 return false;
234 }
235
236 fnames.sort();
237
238 _node->setTitle(sname);
239 _node->updateFiles(fnames);
240
241 // node descr
242 QStringList descr;
243 descr.append(shardware);
244 if (sversion != App::version())
245 descr.append(sversion);
246 if (ident.flags.bits.files == 1 && fnames.contains("fw"))
247 descr.append("LOADER");
248 _node->setDescr(descr.join(' '));
249
250 QVariantMap m;
251 m.insert("uid", _node->uid());
252 m.insert("name", sname);
253 m.insert("version", sversion);
254 m.insert("hardware", shardware);
255 m.insert("hash", _node->hashToText(ident.hash));
256 m.insert("format", ident.format);
257
258 m.insert("reconf", ident.flags.bits.reconf ? true : false);
259 m.insert("busy", ident.flags.bits.busy ? true : false);
260
261 QVariantList array;
262 for (auto i : fnames)
263 array.append(i);
264 m.insert("files", array);
265
266 _node->identReceived(m);
267
268 return true;
269}

271bool PApxNodeRequestUpdate::request(PApxRequest &req)
272{
273 if (_index >= _values.size()) {
274 // all written - request to save
275 _fid = 0xFFFFFFFF;
276 req << _fid;
277 trace()->block("SAVE");
278 return true;
279 }
280
281 auto name = _values.keys().at(_index);
282 xbus::node::conf::type_e type;
283 if (!_node->find_field(name, &_fid, &type)) {
284 qWarning() << "no field:" << name;
285 return false;
286 }
287
288 req << _fid;
289 trace()->block(QString::number(_fid >> 8));
290 trace()->block(QString::number(_fid & 0xFF));
291
292 QVariant value = _values.value(name);
293 if (type == xbus::node::conf::option)
294 value = _node->textToOption(value, _fid >> 8);
295 else if (type == xbus::node::conf::bind)
296 value = _node->stringToMandala(value.toString());
297 _node->write_param(req, type, value);
298 trace()->block(value.toString());
299
300 _index++;
301
302 return true;
303}

364bool PApxNodeRequestFile::response(PStreamReader &stream)
365{
366 if (!_active)
367 return false;
368
369 //qDebug() << "re:" << _op << stream.available();
370 if (stream.available() <= sizeof(xbus::node::file::op_e))
371 return false;
372
373 xbus::node::file::op_e op;
374 stream >> op;
375
376 if (!(op & xbus::node::file::reply_op_mask))
377 return false;
378
379 const char *s = stream.read_string(16);
380 if (!s || QString(s) != _name)
381 return false;
382
383 op = static_cast<xbus::node::file::op_e>(op & ~xbus::node::file::reply_op_mask);
384
385 if (op == xbus::node::file::abort) {
386 qWarning() << "transfer aborted by node";
387 reset();
388 return true;
389 }
390 if (op == xbus::node::file::extend) {
391 if (_op != xbus::node::file::write || !_info.flags.bits.owrite)
392 return false;
393 if (stream.available() < sizeof(xbus::node::file::offset_t))
394 return false;
395 xbus::node::file::offset_t offset;
396 stream >> offset;
397 if (offset != _offset) {
398 qWarning() << "ext offset: " << QString::number(offset, 16)
399 << QString::number(_offset, 16);
400 // response offset mismatch - don't interrupt
401 return false;
402 }
403
404 if (stream.available() != sizeof(xbus::node::file::timeout_t))
405 return false;
406 xbus::node::file::timeout_t time;
407 stream >> time;
408 if (time == 0) {
409 qWarning() << "ext time: " << time;
410 return false;
411 }
412 _node->extend_request(this, time);
413
414 return false;
415 }
416
417 if (op != _op)
418 return false;
419
420 //qDebug() << _name << op << _op;
421 if (op == _op_init) {
422 if (stream.available() != xbus::node::file::info_s::psize())
423 return false;
424 reset();
425 _info.read(&stream);
426 _offset = _info.offset;
427 opened();
428 next();
429 return false;
430 }
431 if (op == xbus::node::file::close) {
432 if (_info.size == _tcnt) {
433 if (_info.size > 0 && _hash != _info.hash && _op_file == xbus::node::file::read) {
434 qWarning() << "transfer error:" << _name;
435 qWarning() << "hash: " << QString::number(_hash, 16)
436 << QString::number(_info.hash, 16);
437 reset();
438 return false;
439 }
440 //qDebug() << "transfer ok:" << _name << QString::number(_hash, 16);
441 if (_op_file == xbus::node::file::read)
442 emit downladed();
443 else
444 emit uploaded();
445 }
446 reset();
447 return true;
448 }
449 if (op != _op_file)
450 return false;
451
452 if (stream.available() <= sizeof(xbus::node::file::offset_t))
453 return false;
454
455 xbus::node::file::offset_t offset;
456 stream >> offset;
457 if (offset != _offset) { //just skip non-sequental
458 qWarning() << "offset:" << QString::number(offset, 16) << QString::number(_offset, 16);
459 return false;
460 }
461
462 return response_file(offset, stream);
463}

477bool PApxNodeRequestFileRead::response_file(xbus::node::file::offset_t offset, PStreamReader &stream)
478{
479 size_t size = stream.available();
480 _offset += size;
481 _tcnt += size;
482
483 if (_tcnt > _info.size) {
484 qWarning() << "overflow:" << _tcnt << _info.size;
485 return true;
486 }
487 _hash = apx::crc32(stream.ptr(), size, _hash);
488
489 size_t v = _info.size > 0 ? _tcnt * 100 / _info.size : 0;
490 _node->setProgress(static_cast<int>(v));
491 emit progress(v);
492
493 next();
494 return false;

502bool PApxNodeRequestFileWrite::request_file(PApxRequest &req)
503{
504 // write data part
505 size_t tail = _info.size - _tcnt;
506 size_t sz = 256;
507 if (sz > tail)
508 sz = tail;
509
510 const QByteArray &ba = _data.mid(static_cast<int>(_tcnt), static_cast<int>(sz));
511 if (static_cast<int>(sz) != ba.size()) {
512 qWarning() << "block size: " << sz << ba.size();
513 return false;
514 }
515 sz = req.write(ba.data(), sz);
516 _hash = apx::crc32(ba.data(), sz);
517 return true;
518}
519bool PApxNodeRequestFileWrite::response_file(xbus::node::file::offset_t offset,
520 PStreamReader &stream)
521{
522 if (stream.available() < sizeof(xbus::node::file::size_t))
523 return false;
524
525 xbus::node::file::size_t size;
526 stream >> size;
527 if (size == 0) {
528 qWarning() << "size: " << size;
529 return false;
530 }
531
532 if (stream.available() < sizeof(xbus::node::hash_t))
533 return false;
534
535 xbus::node::hash_t hash;
536 stream >> hash;
537
538 if (hash != _hash) {
539 qWarning() << "hash: " << QString::number(hash, 16) << QString::number(_hash, 16);
540 return false;
541 }
542
543 if (stream.available() > 0)
544 return false;
545
546 _offset += size;
547 _tcnt += size;
548 if (_tcnt > _info.size)
549 return false;
550
551 size_t v = _data.size() > 0 ? _tcnt * 100 / _data.size() : 0;
552 _node->setProgress(static_cast<int>(v));
553 emit progress(v);
554
555 next();
556 return false;
