35 int pidx_dst = _dst->metaObject()->indexOfProperty(name.toLatin1());
36 if (pidx_dst < 0) {

42 int pidx_src = _src->metaObject()->indexOfProperty(name.toLatin1());
43 if (pidx_src < 0) {
44 qWarning() << "misisng property:" << _src->path() << name;

47 _psrc = _src->metaObject()->property(pidx_src);
48
49 //qDebug() << fact_src->path() << "->" << fact_dst->path() << name;
50 if (!_pdst.isWritable()) {
51 qWarning() << "readonly property:" << _dst->path() << name;
52 _clist.append(connect(_src,
53 _psrc.notifySignal().methodSignature().prepend('2'),
54 _dst,
55 _pdst.notifySignal().methodSignature().prepend('2')));
56 return;
57 }
58
59 connect(_src,
60 _psrc.notifySignal().methodSignature().prepend('2'),
61 this,
62 SLOT(propertyChanged()));
63
64 connect(_src, &QObject::destroyed, this, [this]() { _dst->unbindProperties(_src); });
65
66 if (_src_binding)
67 _src_binding->_src_binding = this;
68
69 propertyChanged();
70}

77void FactPropertyBinding::propertyChanged()
78{
79 //qDebug() << _src->name() << "->" << _dst->name() << _name;
80 if (_blocked) {
81 //qDebug() << "blocked" << _src->name() << "->" << _dst->name() << _name;
82 return;
83 }
84 QVariant v = _psrc.read(_src);
85 if (v == _pdst.read(_dst))
86 return;
87
88 if (_src_binding)
89 _src_binding->block();
90
91 _pdst.write(_dst, v);
92
93 if (_src_binding)
94 _src_binding->unblock();
95}
