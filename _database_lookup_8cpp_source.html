97void DatabaseLookup::reloadQueryResults()
98{
99 //run in thread
100 loadMutex.lock();
101 const QStringList &n = records.names;
102 DatabaseLookupModel::ItemsList list;
103 for (int i = 0; i < records.values.size(); ++i) {
104 const QVariantList &r = records.values.at(i);
105 //create item data
106 QVariantMap m;
107 for (int j = 0; j < n.size(); ++j) {
108 const QString &fn = n.at(j);
109 if (m.contains(fn))
110 continue;
111 m.insert(fn, r.value(j));
112 }
113 if (!fixItemDataThr(&m))
114 break;
115 list.append(m);
116 }
117 loadMutex.unlock();
118 emit _itemsLoaded(list);
119}
