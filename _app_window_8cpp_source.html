200void AppWindow::saveStateDo()
201{
202 if (!w)
203 return;
204 QWindow *window = w->windowHandle();
205 if (!window)
206 return;
207 if (w->isHidden())
208 return;
209 if (w->isMinimized())
210 return;
211 if (w->isMaximized())
212 return;
213 if (window->visibility() == QWindow::Minimized)
214 return;
215 if (window->visibility() == QWindow::Maximized)
216 return;
217 if (window->visibility() == QWindow::Hidden)
218 return;
219
220 bool bFullScreen = w->isFullScreen() || window->visibility() == QWindow::FullScreen;
221 if (bFullScreen)
222 return;
223 if (!bFullScreen) {
224 if (window->width() == window->screen()->availableSize().width()
225 || window->height() == window->screen()->availableSize().height())
226 return;
227 }
228 QSettings s;
229 s.beginGroup(parentFact()->name());
230 s.setValue(name(), w->saveGeometry());
231}
