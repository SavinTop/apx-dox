76DatalinkRemote *DatalinkRemotes::registerHost(QUrl url)
77{
78 DatalinkRemote *c = findRemote(url);
79 if (c)
80 return c;
81 c = new DatalinkRemote(f_servers, datalink, url);
82 connect(c, &DatalinkRemote::activeChanged, this, &DatalinkRemotes::updateStatus);
83 connect(f_alloff, &Fact::triggered, c, [c]() { c->setValue(false); });
84 apxMsg() << QString("#%1: %2").arg(tr("found server")).arg(url.toString());
85 datalink->addConnection(c);
86 return c;
87}
88DatalinkRemote *DatalinkRemotes::findRemote(QUrl url)
89{
90 for (auto i : f_servers->findFacts<DatalinkRemote>()) {
91 if (!i->isEqual(QHostAddress(url.host())))
92 continue;
93 return i;
94 }
95 return nullptr;
96}

98void DatalinkRemotes::discover(void)
99{
100 if (f_discover->value().toBool()) {
101 if (udpDiscover->state() == QAbstractSocket::BoundState)
102 return;
103 if (udpDiscover->bind(QHostAddress::AnyIPv4, UDP_PORT_DISCOVER)) {
104 f_discover->setActive(true);
105 qDebug() << "binded";
106 return;
107 }
108 QTimer::singleShot(5000, this, &DatalinkRemotes::discover);
109 // apxConsoleW() << "udp discover bind retry" << udpDiscover->errorString();
110 } else {
111 f_discover->setActive(false);
112 if (udpDiscover->state() != QAbstractSocket::UnconnectedState) {
113 udpDiscover->close();
114 qDebug() << "closed";
115 }
116 }

118void DatalinkRemotes::discoverRead(void)
119{
120 // service:gcs:tcp://username@hostname:port/[\n<other service>]
121 const QByteArray hdr("service:gcs:");
122 while (udpDiscover->hasPendingDatagrams()) {
123 qint64 sz = udpDiscover->pendingDatagramSize();
124 // qDebug() << sz;
125 if (sz < hdr.size()) {
126 qDebug() << "discarded";
127 udpDiscover->receiveDatagram(0); // discard datagram
128 continue;
129 }
130 QNetworkDatagram datagram = udpDiscover->receiveDatagram();
131 if (!datagram.isValid())
132 continue;
133
134 // qDebug() << QString(datagram.data());
135
136 // if (datalink->f_server->active() && DatalinkTcp::isLocalHost(datagram.senderAddress()))
137 // continue;
138
139 for (auto data : datagram.data().split('\n')) {
140 // qDebug() << data;
141
142 if (!data.startsWith(hdr))
143 continue;
144
145 //register or update host
146 QUrl url(data.mid(hdr.size()), QUrl::StrictMode);
147 if (!url.isValid())
148 continue;
149
150 QString saddr = datagram.senderAddress().toString();
151 url.setHost(saddr.mid(saddr.lastIndexOf(':') + 1));
152 DatalinkRemote *c = registerHost(url);
153 c->updateTimeout();
154 }
155 }

158void DatalinkRemotes::updateStatus()
159{
160 //count connected
161 int cnt = 0;
162 for (int i = 0; i < f_servers->size(); ++i) {
163 Fact *f = f_servers->child(i);
164 if (f->active())
165 cnt++;
166 }
167 setConnectedCount(cnt);
168 f_alloff->setEnabled(cnt > 0);
169 setActive(cnt > 0);
170
171 if (f_servers->size() > 0)
172 setValue(QString("%1/%2").arg(cnt).arg(f_servers->size()));
173 else
174 setValue(QVariant());
175}

177void DatalinkRemotes::connectTriggered()
178{
179 QUrl url = DatalinkRemote::fixUrl(f_url->text());
180 QHostAddress addr = QHostAddress(url.host());
181 if (addr.isNull() || addr.isLoopback())
182 return;
183 DatalinkRemote *c = registerHost(url);
184 c->setValue(true);
185}
