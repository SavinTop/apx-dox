31DatalinkRemote::DatalinkRemote(Fact *parent, Datalink *datalink, QUrl url)
32 : DatalinkTcp(parent,
33 new QTcpSocket(),
34 Datalink::SERVERS | Datalink::LOCAL,
35 Datalink::SERVERS | Datalink::CLIENTS | Datalink::LOCAL)
36 , datalink(datalink)
37 , retry(0)
38{
39 setRemoteUrl(url);
40
41 updateStatsTimer.setSingleShot(true);
42 connect(&updateStatsTimer, &QTimer::timeout, this, &DatalinkRemote::updateStats);
43
44 connect(this, &DatalinkConnection::activatedChanged, this, [this]() {
45 if (activated()) {
46 retry = 0;

49 close();

58
59 updateStats();
60}

74QUrl DatalinkRemote::fixUrl(QUrl url)
75{
76 if (url.scheme().isEmpty()) {
77 QString s = url.toString();
78 url.setUrl(QString("%1://%2").arg("tcp").arg(s));
79 }
80 if (url.port() <= 0) {
81 quint16 v = TCP_PORT_SERVER;
82 if (url.scheme().startsWith("http", Qt::CaseInsensitive))
83 v = 80;
84 url.setPort(v);
85 }
86 return url;
87}

89void DatalinkRemote::updateStats()
90{

93 setDescr(QString("%1 (%2)")
94 .arg(t >= 60 ? tr("No service") : tr("Alive"))
95 .arg(t == 0 ? tr("now")
96 : t >= 60 ? QString("%1 %2").arg(t / 60).arg(tr("min"))
97 : QString("%1 %2").arg(t).arg(tr("sec"))));
98 updateStatsTimer.start(t > 60 ? 60000 : 5000);
99 }

119void DatalinkRemote::open()
120{
121 //check if already present in connections
122 for (int i = 0; i < datalink->connections.size(); ++i) {
123 DatalinkTcp *c = qobject_cast<DatalinkTcp *>(datalink->connections.at(i));
124 if (!c)
125 continue;
126 if (!c->active())
127 continue;
128 if (!c->isEqual(_hostAddress))
129 continue;
130 setActivated(false);
131 apxMsgW() << tr("Connection exists");
132 return;
133 }
134
135 //open
136 retry++;
137 connectToHost(_hostAddress, _hostPort);
138}
