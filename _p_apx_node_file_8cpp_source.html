41void PApxNodeFile::process_downlink(xbus::node::file::op_e op, PStreamReader &stream)
42{
43 // collect and build file content from received read/write parts and emit signal with data
44
45 bool is_response = op & xbus::node::file::reply_op_mask;
46 op = static_cast<xbus::node::file::op_e>(op & ~xbus::node::file::reply_op_mask);
47
48 switch (op) {
49 default:
50 break;
51 case xbus::node::file::info:
52 if (!is_response)
53 return;
54 if (!check_info(stream))
55 break; // info changed
56 return;
57
58 case xbus::node::file::ropen:
59 if (!is_response)
60 return;
61 check_info(stream);
62
63 if (!(_info.flags.bits.readable && _info.flags.bits.oread))
64 break;
65
66 reset();
67 _size = _info.size;
68 _offset = _info.offset;
69 _open_op = op;
70 //qDebug() << "opened read" << _size << _offset;
71 return;
72
73 case xbus::node::file::wopen:
74 if (!is_response)
75 return;
76 check_info(stream);
77
78 if (!(_info.flags.bits.writable && _info.flags.bits.owrite))
79 break;
80
81 reset();
82 _size = 65536 * 1024;
83 _offset = _info.offset;
84 _open_op = op;
85 //qDebug() << "opened write" << _offset;
86 return;
87
88 case xbus::node::file::close:
89 if (!is_response)
90 return;
91 if (_open_op == xbus::node::file::ropen) {
92 if (!check_info(stream))
93 break;
94 } else if (_open_op == xbus::node::file::wopen) {
95 // uploads from another GCS
96 check_info(stream);
97 _size = _data.size();
98 _info.size = _size;
99 _info.hash = _hash;
100 if (_size == 0)
101 break;
102 } else
103 break;
104
105 if (_size == 0 && _info.size == 0 && _data.size() == 0) {
106 qDebug() << "download ok:" << name() << "empty";
107 } else {
108 // check downloaded data
109 if (_hash != _info.hash || _tcnt != _size || _data.size() != _tcnt) {
110 qWarning() << "download error:" << name() << _data.size();
111 qWarning() << "hash: " << QString::number(_hash, 16)
112 << QString::number(_info.hash, 16);
113 break;
114 }
115 qDebug() << "download ok:" << name() << _size << "bytes" << QString::number(_hash, 16);
116 }
117
118 if (_open_op == xbus::node::file::ropen)
119 emit downloaded(_node, _info, _data);
120 else
121 emit uploaded(_node, _info, _data);
122
123 break;
124
125 case xbus::node::file::read:
126 if (!is_response)
127 return;
128 if (!_info.flags.bits.oread)
129 break;
130 if (!_size || _open_op != xbus::node::file::ropen)
131 break;
132 if (!read(stream))
133 break;
134 return;
135
136 case xbus::node::file::write:
137 if (is_response)
138 return;
139 if (!_info.flags.bits.owrite)
140 break;
141 if (!_size || _open_op != xbus::node::file::wopen)
142 break;
143 if (!read(stream))
144 break;
145 return;
146
147 case xbus::node::file::extend:
148 qWarning() << "extend response";
149 return;
150 case xbus::node::file::abort:
151 qWarning() << "abort response";
152 break;
153 }
154 // error
155 reset();
156}

158bool PApxNodeFile::read(PStreamReader &stream)
159{
160 if (stream.available() <= sizeof(xbus::node::file::offset_t))
161 return false;
162
163 xbus::node::file::offset_t offset;
164 stream >> offset;
165 if (offset != _offset) { //just skip non-sequental
166 return true;
167 }
168
169 size_t size = stream.available();
170 _offset += size;
171 _tcnt += size;
172
173 //qDebug() << "rd:" << offset << size;
174
175 if (_tcnt > _size)
176 return false;
177
178 _hash = apx::crc32(stream.ptr(), size, _hash);
179
180 _data.append(stream.payload());
181 return true;
182}

197bool PApxNodeFile::check_info(PStreamReader &stream)
198{
199 if (stream.available() != xbus::node::file::info_s::psize())
200 return false;
201
202 xbus::node::file::info_s info;
203 info.read(&stream);
204 bool chg = true;
205 do {
206 if (info.size != _info.size)
207 break;
208 if (info.time != _info.time)
209 break;
210 if (info.hash != _info.hash)
211 break;
212 if (info.offset != _info.offset)
213 break;
214 if (info.flags.bits.readable != _info.flags.bits.readable)
215 break;
216 if (info.flags.bits.writable != _info.flags.bits.writable)
217 break;
218 chg = false;
219 } while (0);
220 _info = info;
221 if (!chg)
222 return true;
223 {
224 // qDebug() << "info update";
225 QStringList st;
226 st << AppRoot::capacityToString(info.size, 2);
227 st << QString("T:%1").arg(info.time);
228 st << QString("H:%1").arg(info.hash, 8, 16, QChar('0')).toUpper();
229 st << QString("O:%1").arg(info.offset, 8, 16, QChar('0')).toUpper();
230 st << QString("%1%2")
231 .arg(info.flags.bits.readable ? 'R' : ' ')
232 .arg(info.flags.bits.writable ? 'W' : ' ');
233 setDescr(st.join(' '));
234 }
235 return false;
236}
