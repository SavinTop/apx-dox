169{
170 qDeleteAll(events);
171 events.clear();
172 const QwtPlotItemList &items = itemList(QwtPlotItem::Rtti_PlotCurve);
173 for (int i = 0; i < items.size(); ++i) {
174 QwtPlotCurve *curve = static_cast<QwtPlotCurve *>(items.at(i));

189 }
190 }
191
192 resetZoom();
193 setTimeCursor(0);

290void TelemetryPlot::showCurve(const QVariant &itemInfo, bool on, int index)
291{
292 Q_UNUSED(index);
293 QwtPlotItem *item = infoToItem(itemInfo);
294 if (item->isVisible() == on)
295 return;
296 //check if was empty plot
297 bool allWereHidden = true;
298 const QwtPlotItemList &items = itemList(QwtPlotItem::Rtti_PlotCurve);
299 for (int i = 0; i < items.size(); ++i) {
300 QwtPlotCurve *curve = static_cast<QwtPlotCurve *>(items.at(i));
301 if (curve->isVisible()) {
302 allWereHidden = false;
303 break;
304 }
305 }
306 //show or hide item
307 item->setVisible(on);
308
309 QwtPlotCurve *clc_curve = static_cast<QwtPlotCurve *>(item);
310 if (clc_curve && calc_curves.contains(clc_curve) && on) {
311 refreshCalculated(clc_curve);
312 }
313
314 emit itemVisibleChanged(item);
315 //update legend
316 QwtLegendLabel *w = qobject_cast<QwtLegendLabel *>(legend->legendWidget(itemInfo));
317 if (w)
318 w->setChecked(item->isVisible());
319 //update zoom or replot
320 if (allWereHidden)
321 resetZoom();
322 else
323 replot();
324}

326void TelemetryPlot::refreshCalculated(QwtPlotCurve* curve_calc)
327{
328 bool ok;
329 QString exp_calc = calc_curves.value(curve_calc);
330 if (exp_calc.isEmpty())
331 exp_calc = "est.att.yaw-est.calc.bearing";
332 QString exp = QInputDialog::getText(nullptr,
333 tr("Calculated field"),
334 tr("JavaScript expression:"),
335 QLineEdit::Normal,
336 exp_calc,
337 &ok);
338 if (ok == false) {
339 curve_calc->setVisible(false);
340 return;
341 }
342 calc_curves[curve_calc] = exp;
343
344 //fill internal data
345 QVector<QPointF> points;
346 QJSEngine engine;
347
348 const QwtPlotItemList &items = itemList(QwtPlotItem::Rtti_PlotCurve);
349
350 QVector<const QwtSeriesData<QPointF> *> fdata;
351 QVector<QString> fnames;
352 quint64 tcnt = 0;
353 QVector<quint64> fpidx;
354 for (int i = 0; i < items.size(); ++i) {
355 QwtPlotCurve *curve = static_cast<QwtPlotCurve *>(items.at(i));
356 fdata.append(curve->data());
357 fnames.append(curve->title().text());
358 fpidx.append(0);
359 engine.globalObject().setProperty(fnames.last(),
360 curve->data()->size() > 0 ? curve->data()->sample(0).y()
361 : 0);
362 tcnt += curve->data()->size();
363 }
364 quint64 cnt = 0;
365 double vcalc = 0;
366 double tMax = 0;
367 while (tcnt) {
368 setProgress(cnt * 100 / tcnt);
369 //find earliest time value
370 double t = -1;
371 for (int i = 0; i < fpidx.size(); ++i) {
372 const QwtSeriesData<QPointF> *points = fdata.at(i);
373 quint64 didx = fpidx.at(i);
374 if (didx >= points->size())
375 continue;
376 const double &ft = points->sample(didx).x();
377 if (t >= 0 && ft >= t)
378 continue;
379 t = ft;
380 }
381 if (t < 0)
382 break; //done
383 if (tMax < t)
384 tMax = t;
385 engine.globalObject().setProperty("time", t);
386 //update fields with time=t
387 for (int i = 0; i < fpidx.size(); ++i) {
388 const QwtSeriesData<QPointF> *points = fdata.at(i);
389 quint64 didx = fpidx.at(i);
390 if (didx >= points->size())
391 continue;
392 const QPointF &p = points->sample(didx);
393 if (p.x() != t)
394 continue;
395 QString s = fnames.at(i);
396 QJSValue obj = engine.globalObject();
397 while (1) {
398 int i = s.indexOf('.');
399 if (i < 0) {
400 obj.setProperty(s, p.y());
401 break;
402 }
403 const QString &ps = s.left(i);
404 QJSValue v = obj.property(ps);
405 if (!v.isObject()) {
406 v = engine.newObject();
407 obj.setProperty(ps, v);
408 }
409 obj = v;
410 s.remove(0, i + 1);
411 }
412 fpidx[i] = didx + 1;
413 cnt++;
414 }
415 double v = engine.evaluate(exp).toNumber();
416 if (v == vcalc)
417 continue;
418 vcalc = v;
419 if (points.size() > 0 && (t - points.last().x()) > 0.5) {
420 //extrapolate unchanged value tail-1ms
421 points.append(QPointF(t - 0.001, points.last().y()));
422 }
423 points.append(QPointF(t, v));
424 }
425 //final data tail at max time
426 if ((!points.isEmpty()) && points.last().x() < tMax) {
427 points.append(QPointF(tMax, points.last().y()));
428 }
429
430 setProgress(0);
431 //install data
432 curve_calc->setData(new QwtPointSeriesData(points));
433 curve_calc->setVisible(true);
434
435 //resetZoom();
436 replot();
437}

447void TelemetryPlot::resetZoom()
448{
449 double vmax = 0.0, vmin = 0.0, tMax = 0.0;
450 const QwtPlotItemList &items = itemList(QwtPlotItem::Rtti_PlotCurve);
451 for (int i = 0; i < items.size(); ++i) {
452 QwtPlotCurve *curve = static_cast<QwtPlotCurve *>(items.at(i));
453 for (size_t i2 = 0; i2 < curve->data()->size(); ++i2) {
454 const QPointF &p = curve->data()->sample(i2);
455 if (tMax < p.x())
456 tMax = p.x();
457 double v = p.y();
458 if (curve->isVisible() && (!std::isnan(v)) && (!std::isinf(v))) {
459 if (vmax < v)
460 vmax = v;
461 if (vmin > v)
462 vmin = v;
463 }
464 }
465 }
466 //reset zoom
467 double margin = std::abs(vmax - vmin) * 0.1;
468 vmin -= margin;
469 vmax += margin;
470 QRectF r(0, vmin, tMax, vmax - vmin);
471 zoomer->zoom(r);
472 zoomer->setZoomBase(r);
473 //zoomer->zoom(r);
474 //zoomer->setZoomBase(r);
475 //QCoreApplication::processEvents();
476 replot();
477}

570double PlotPicker::sampleValue(const QwtPlotCurve *curve, double t) const
571{
572 if (curve->data()->size() < 50) {
573 for (size_t i = 0; i < curve->data()->size(); ++i) {
574 if (curve->data()->sample(i).x() >= t) {
575 return curve->data()->sample(i).y();
576 }
577 }
578 return 0;
579 }
580 size_t ts = curve->data()->size() / 2;
581 size_t tx = ts;
582 bool bFound = false, bFwd = false;
583 while (1) {
584 const QPointF &p = curve->data()->sample(tx);
585 double vx = p.x();
586 ts >>= 1;
587 if (ts == 0) {
588 if (bFound)
589 return p.y();
590 //if(vx>t && tx>0)ts=1;
591 //else return p.y();
592 ts = 1;
593 }
594 if (vx < t) {
595 tx += ts;
596 if (tx >= curve->data()->size()) {
597 tx = curve->data()->size() - 1;
598 if (ts == 1)
599 bFound = true;
600 }
601 bFwd = true;
602 } else if (vx > t) {
603 if (ts == 1 && bFwd == true)
604 bFound = true;
605 if (tx > ts)
606 tx -= ts;
607 else {
608 tx = 0;
609 if (ts == 1)
610 bFound = true;
611 }
612 bFwd = false;
613 } else
614 return p.y();
615 }

646void PlotMagnifier::rescale(double factor)
647{
648 factor = qAbs(factor);
649 if (factor == 1.0 || factor == 0.0)
650 return;
651 bool doReplot = false;
652 QwtPlot *plt = plot();
653 const bool autoReplot = plt->autoReplot();
654 plt->setAutoReplot(false);
655 for (int axisId = 0; axisId < QwtPlot::axisCnt; axisId++) {
656 const QwtScaleDiv &scaleDiv = plt->axisScaleDiv(axisId);
657 if (isAxisEnabled(axisId)) {
658 // get the range of the axis
659 double min_value = scaleDiv.lowerBound();
660 double max_value = scaleDiv.upperBound();
661 // convert to screen coordinates
662 QwtScaleMap mapper = plt->canvasMap(axisId);
663 double min_pixel = mapper.transform(min_value);
664 double max_pixel = mapper.transform(max_value);
665 // get the mouse position on this axis
666 double center;
667 if (axisId == QwtPlot::yLeft || axisId == QwtPlot::yRight)
668 center = mwPos.y();
669 else
670 center = mwPos.x();
671 // use the mouse as the center when we can,
672 // if its -1, then its a keyboard event. use the window center.
673 if (center == -1.0)
674 center = min_pixel + (max_pixel - min_pixel) / 2;
675 // convert back to real values
676 min_value = mapper.invTransform(center - (center - min_pixel) * factor);
677 max_value = mapper.invTransform(center + (max_pixel - center) * factor);
678 // set the new range
679 plt->setAxisScale(axisId, min_value, max_value);
680 //plt->setAxisScale( axisId, center - width_2, center + width_2 );
681 doReplot = true;
682 }
683 }
684 plt->setAutoReplot(autoReplot);
685 if (doReplot)
686 plt->replot();
687}
