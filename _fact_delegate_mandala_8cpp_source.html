62 connect(tree, &FactTreeView::activated, this, [this](const QModelIndex &index) {

120void FactDelegateMandala::finished()
121{
122 do {
123 QModelIndexList sel = tree->selectionModel()->selectedRows();
124 if (sel.isEmpty())
125 break;
126 QModelIndex idx = proxy->mapToSource(sel.first());
127 MandalaFact *f = qobject_cast<MandalaFact *>(idx.data(Fact::ModelDataRole).value<Fact *>());
128 if (!f)
129 break;
130 if (!f->dataType())
131 break;
132 if (f->isSystem())
133 break;
134 fact->setValue(f->mpath());
135 return;
136 } while (0);
137 fact->setValue(QVariant());
138}

158QVariant FactDelegateMandalaModel::data(const QModelIndex &index, int role) const
159{
160 if (!index.isValid())
161 return QVariant();
162 Fact *f = fact(index);
163 if (!f)
164 return QVariant();
165 if (role == Qt::DisplayRole) {
166 switch (index.column()) {
167 case Fact::FACT_MODEL_COLUMN_NAME:
168 return f->title();
169 case Fact::FACT_MODEL_COLUMN_VALUE:
170 if (f->treeType() != Fact::Group)
171 break;
172 return QVariant();
173 case Fact::FACT_MODEL_COLUMN_DESCR:
174 if (MandalaFact *m = qobject_cast<MandalaFact *>(f))
175 return m->mpath();
176 break;
177 }
178 }
179 return FactTreeModel::data(index, role);
180}
