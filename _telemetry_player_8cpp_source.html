150void TelemetryPlayer::play()
151{
152 if (active())
153 return;
154 if (!telemetry->f_lookup->recordId())
155 return;
156 vehicle->f_select->trigger();
157 setActive(true);
158
159 //reste mandala
160 vehicle->f_mandala->restoreDefaults();
161
162 playTime0 = _time;
163 playTime.start();
164
165 //fill facts map
166 if (factByDBID.isEmpty()) {
167 for (auto f : vehicle->f_mandala->valueFacts()) {
168 factByDBID.insert(Database::instance()->telemetry->field_key(f->uid()), f);
169 }
170 }
171
172 //collect data samples for t0
173 double t = _time / 1000.0;
174 for (auto fieldID : telemetry->f_reader->fieldData.keys()) {
175 Fact *f = factByDBID.value(fieldID);
176 if (!f)
177 continue;
178 f->setValue(sampleValue(fieldID, t));
179 }
180 tNext = _time;
181 dbRequestEvents(tNext);
182}

195 else {
196 f_time->setValue(setTime0);
197 if (bPlaying)
198 play();
199 }

279void TelemetryPlayer::loadConfValue(const QString &sn, QString s)
280{
281 NodeItem *node = vehicle->f_nodes->node(sn);
282 if (!node) {
283 qWarning() << "missing node" << sn;
284 return;
285 }
286 int del = s.indexOf('=');
287 if (del < 0)
288 return;
289 QString spath = s.left(del).trimmed();
290 QString sv = s.mid(del + 1);
291 //qDebug()<<spath<<sv;
292 if (spath.startsWith(node->title()))
293 spath.remove(0, node->title().size() + 1);
294 if (spath.isEmpty())
295 return;
296 node->loadConfigValue(spath, sv);
297}

299void TelemetryPlayer::next()
300{
301 if (!active())
302 return;
303 quint64 t = playTime.elapsed();
304 if (_speed > 0 && _speed != 1.0)
305 t = t * _speed;
306 t += playTime0;
307
308 uint updCnt = 0;
309 if (tNext <= t) {
310 quint64 tNextMin = tNext;
311 //mandala data
312 for (const auto fieldID : telemetry->f_reader->fieldData.keys()) {
313 QVector<QPointF> *pts = telemetry->f_reader->fieldData.value(fieldID);
314 if (!pts)
315 continue;
316 for (int i = dataPosMap.value(fieldID); i < pts->size(); ++i) {
317 const QPointF &p = pts->at(i);
318 quint64 tP = p.x() * 1000.0;
319 if (tP > tNext) {
320 if (tNextMin == tNext || tNextMin > tP)
321 tNextMin = tP;
322 break;
323 }
324 dataPosMap[fieldID] = i;
325 Fact *f = factByDBID.value(fieldID);
326 if (f) {
327 if (f->setValue(p.y()))
328 updCnt++;
329 }
330 }
331 }
332 //events
333 const QStringList &n = events.names;
334 for (; iEventRec < events.values.size(); ++iEventRec) {
335 const QVariantList &r = events.values.at(iEventRec);
336 quint64 tP = r.at(n.indexOf("time")).toULongLong();
337 if (tP > t) {
338 if (tNextMin == tNext || tNextMin > tP)
339 tNextMin = tP;
340 break;
341 }
342 //show event
343 QString sv = r.at(n.indexOf("value")).toString();
344 uint type = r.at(n.indexOf("type")).toUInt();
345 if (type == 1) {
346 //uplink data
347 quint64 fieldID = r.at(n.indexOf("name")).toULongLong();
348 if (!fieldID)
349 continue;
350 MandalaFact *f = static_cast<MandalaFact *>(factByDBID.value(fieldID));
351 if (!f)
352 continue;
353 if (f->setValue(sv))
354 updCnt++;
355 const QString &s = f->mpath();
356 if (s.startsWith("cmd.rc."))
357 continue;
358 if (s.startsWith("cmd.gimbal."))
359 continue;
360 vehicle->message(QString("%1: %2 = %3").arg(">").arg(f->title()).arg(f->text()),
361 AppNotify::Important);
362 continue;
363 }
364 if (type == 2 || type == 3) {
365 const QString &evt = r.at(n.indexOf("name")).toString();
366 const QString &uid = r.at(n.indexOf("uid")).toString();
367 if (evt == "msg") {
368 QString s = sv;
369 QString sub;
370 if (s.startsWith('[')) {
371 s.remove(0, 1);
372 sub = s.left(s.indexOf(']'));
373 s.remove(0, sub.size() + 1);
374 }
375 vehicle->message(QString("<: %1").arg(s),
376 AppNotify::FromVehicle | AppNotify::Important,
377 sub);
378 App::sound(sv);
379 continue;
380 }
381 if (evt == "mission") {
382 vehicle->f_mission->storage->loadMission(uid);
383 } else if (evt == "nodes") {
384 vehicle->storage()->loadVehicleConfig(uid);
385 } else if (evt == "conf") {
386 loadConfValue(uid, sv);
387 QString fn = sv.left(sv.indexOf('='));
388 if (sv.size() > (fn.size() + 32) || sv.contains('\n')) {
389 sv = fn + "=<data>";
390 }
391 } else if (evt == "serial") {
392 qDebug() << evt << sv;
393 continue;
394 }
395 AppNotify::NotifyFlags flags = AppNotify::Important;
396 if (type != 3)
397 flags |= AppNotify::FromVehicle;
398 QString s = QString("%1: %2").arg(type == 3 ? ">" : "<").arg(evt);
399 if (!sv.isEmpty())
400 s.append(QString(" (%1)").arg(sv));
401 vehicle->message(s, flags);
402 continue;
403 }
404 }
405
406 //update states
407 if (_time != t) {
408 _time = t;
409 blockTimeChange = true;
410 f_time->setValue(_time);
411 blockTimeChange = false;
412 }
413
414 if (updCnt) {
415 vehicle->f_mandala->telemetryDecoded();
416 }
417
418 //continue next time event
419 if (tNextMin == tNext) {
420 apxMsg() << tr("Replay finished");
421 stop();
422 } else {
423 tNext = tNextMin;
424 }
425 }
426 if (!active())
427 return;
428 if (tNext > t) {
429 timer.start(tNext - t);
430 } else
431 next();
432}

434double TelemetryPlayer::sampleValue(quint64 fieldID, double t)
435{
436 dataPosMap[fieldID] = 0;
437 QVector<QPointF> *pts = telemetry->f_reader->fieldData.value(fieldID);
438 if (!pts)
439 return 0;
440
441 if (pts->size() < 50) {
442 for (int i = 0; i < pts->size(); ++i) {
443 if (pts->at(i).x() >= t) {
444 dataPosMap[fieldID] = i;
445 return pts->at(i).y();
446 }
447 }
448 return 0;
449 }
450 int ts = pts->size() / 2;
451 int tx = ts;
452 bool bFound = false, bFwd = false;
453 while (1) {
454 const QPointF &p = pts->at(tx);
455 double vx = p.x();
456 ts >>= 1;
457 if (ts == 0) {
458 if (bFound) {
459 dataPosMap[fieldID] = tx;
460 return p.y();
461 }
462 ts = 1;
463 }
464 if (vx < t) {
465 tx += ts;
466 if (tx >= pts->size()) {
467 tx = pts->size() - 1;
468 if (ts == 1)
469 bFound = true;
470 }
471 bFwd = true;
472 } else if (vx > t) {
473 if (ts == 1 && bFwd == true)
474 bFound = true;
475 if (tx > ts)
476 tx -= ts;
477 else {
478 tx = 0;
479 if (ts == 1)
480 bFound = true;
481 }
482 bFwd = false;
483 } else {
484 dataPosMap[fieldID] = tx;
485 return p.y();
486 }
487 }
