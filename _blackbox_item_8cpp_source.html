59void BlackboxItem::updateActions()
60{
61 bool enb = enabled();
62 int p = progress();
63 f_start->setEnabled(enb && p < 0 && totalSize > 0);
64 f_stop->setEnabled(p >= 0);
65
66 f_stats->setProgress(p);
67}
68void BlackboxItem::updateStats()
69{
70 f_stats->setTitle(totalSize ? title().append(": ").append(parentFact()->title()) : tr("Empty"));
71 f_stats->setValue(QString("%1 MB").arg(totalSize / 1024.0 / 1024.0, 0, 'f', 2));
72}

74void BlackboxItem::getStats()

78void BlackboxItem::download()
79{
80 if (reader)
81 return;
82 apxMsg() << tr("Blackbox downloading").append("...");
83 f_stats->setTitle(tr("Downloading").append("..."));
84
85 //create tmp reader
86 reader = new BlackboxReader(this, f_callsign->text().trimmed(), uid);
87}
88
89void BlackboxItem::stop()
90{
91 setProgress(-1);
92 updateStats();
93 if (reader) {
94 reader->deleteLater();
95 reader = nullptr;
