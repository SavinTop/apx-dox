54void JoystickAxis::update(qreal v)
55{
56 v = round(v * 100) / 100.0;
57 if (std::abs(v) < _hyst)
58 v = 0;
59 else if (v > 0.0)
60 v = (v - _hyst) / (1.0 - _hyst);
61 else
62 v = (v + _hyst) / (1.0 - _hyst);
63
64 if (_value == v)
65 return;
66
67 _value = v;
68 setActive(v != 0.0);
69 updateDescr();
70 //setStatusText(QString::number(v, 'f', 2).append(" > "));
71 QString s = text().simplified();
72 if (s.isEmpty())
73 return;
74 s.replace("$", QString("(%1)").arg(v));
75 App::jsexec(s);
76}
