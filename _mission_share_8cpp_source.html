46QString MissionShare::getDefaultTitle()
47{
48 QString s = _mission->site().replace(" ", "");
49 if (!s.isEmpty())
50 s.append("-");
51 s.append(_mission->vehicle->title());
52 QString subj = _mission->f_title->text().simplified();
53 if (!subj.isEmpty())
54 s.append(QString("-%1").arg(subj));
55 else
56 s.append(QString("-%1").arg(_mission->missionSize()));
57 s.replace(' ', '-');
58 return s;
59}
