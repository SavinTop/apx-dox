27AppInstances::AppInstances(QObject *parent)
28 : QObject(parent)
29 , lockFile(QDir::temp().absoluteFilePath(QCoreApplication::applicationName() + ".lock"))
30 , m_client(false)
31{
32 socketName = QCoreApplication::applicationName();
33 lockFile.setStaleLockTime(0);
34 if (lockFile.tryLock(0)) {
35 QLocalServer::removeServer(socketName);
36 localServer.listen(socketName);
37 } else {
38 //try to connect to server
39 QLocalSocket socket;
40 socket.connectToServer(socketName);
41 if (socket.waitForConnected(1000)) {
42 m_client = true;
43 } else {
44 //recover server from crash
45 if (!lockFile.tryLock(0)) {
46 lockFile.removeStaleLockFile();
47 qWarning() << "lock file removed";
48 }
49 if (lockFile.tryLock(0)) {
50 QLocalServer::removeServer(socketName);
51 localServer.listen(socketName);
52 } else {
53 qWarning() << "lock file recovery failed";
54 }
55 }
56 }
57 if (m_client) {
58 apxMsg() << tr("Client application mode");
59 } else if (localServer.isListening()) {
60 connect(&localServer, &QLocalServer::newConnection, this, &AppInstances::newConnection);
61 } else {
62 apxMsg() << tr("Local server error") << localServer.errorString();
63 }
64}

71void AppInstances::newConnection()
72{
73 QLocalSocket *socket = localServer.nextPendingConnection();
74 if (!socket)
75 return;
76
77 // Why doesn't Qt have a blocking stream that takes care of this shait???
78 while (socket->bytesAvailable() < static_cast<int>(sizeof(quint32))
79 && socket->state() == QLocalSocket::ConnectedState) {
80 if (!socket->isValid()) // stale request
81 return;
82 socket->waitForReadyRead(1000);
83 }
84 QDataStream ds(socket);
85 QByteArray uMsg;
86 quint32 remaining;
87 ds >> remaining;
88 uMsg.resize(remaining);
89 int got = 0;
90 char *uMsgBuf = uMsg.data();
91 //qDebug() << "RCV: remaining" << remaining;
92 do {
93 got = ds.readRawData(uMsgBuf, remaining);
94 remaining -= got;
95 uMsgBuf += got;
96 //qDebug() << "RCV: got" << got << "remaining" << remaining;
97 } while (remaining && got >= 0 && socket->waitForReadyRead(2000));
98 //### error check: got<0
99 if (got < 0) {
100 qWarning() << "Message reception failed" << socket->errorString();
101 delete socket;
102 return;
103 }
104 // ### async this
105 QString message = QString::fromUtf8(uMsg.constData(), uMsg.size());
106 socket->write(ack, qstrlen(ack));
107 socket->waitForBytesWritten(1000);
108 emit messageReceived(message); // ##(might take a long time to return)
109}

111bool AppInstances::sendMessage(const QString &message, int timeout, bool block)
112{
113 if (!isClient())
114 return false;
115
116 QLocalSocket socket;
117 bool connOk = false;
118 for (int i = 0; i < 2; i++) {
119 // Try twice, in case the other instance is just starting up
120 socket.connectToServer(socketName);
121 connOk = socket.waitForConnected(timeout / 2);
122 if (connOk || i)
123 break;
124 int ms = 250;
125#if defined(Q_OS_WIN)
126 Sleep(DWORD(ms));
127#else
128 struct timespec ts = {ms / 1000, (ms % 1000) * 1000 * 1000};
129 nanosleep(&ts, NULL);
130#endif
131 }
132 if (!connOk)
133 return false;
134
135 QByteArray uMsg(message.toUtf8());
136 QDataStream ds(&socket);
137 ds.writeBytes(uMsg.constData(), uMsg.size());
138 bool res = socket.waitForBytesWritten(timeout);
139 res &= socket.waitForReadyRead(timeout); // wait for ack
140 res &= (socket.read(qstrlen(ack)) == ack);
141 if (block) // block until peer disconnects
142 socket.waitForDisconnected(-1);
143 return res;
144}
