47bool LookupSites::fixItemDataThr(QVariantMap *item)
48{
49 mutex.lock();
50 QGeoCoordinate rc = reqCenter;
51 double rd = reqDist;
52 mutex.unlock();
53 double dist = 0;
54 QGeoCoordinate c(item->value("lat").toDouble(), item->value("lon").toDouble());
55 if (rc.isValid()) {
56 dist = rc.distanceTo(c);
57 if (dist > rd)
58 return false;
59 item->insert("value", AppRoot::distanceToString(dist));
60 }
61 return true;
62}

64void LookupSites::updateRect()
65{
66 QGeoRectangle r = m_area.boundingGeoRectangle();
67 mutex.lock();
68 reqCenter = r.center();
69 mutex.unlock();
70 if ((!r.isValid()) || r.width() <= 0 || r.height() <= 0)
71 return;
72 double sz = qMax(1.0, qMax(r.width(), r.height()) * 2.0);
73 r.setWidth(sz);
74 r.setHeight(sz);
75 while (!reqRect.isEmpty()) {
76 double d = qMax(reqRect.width() / r.width(), reqRect.height() / r.height());
77 if (d > 8.0)
78 break;
79 if (reqRect.contains(r)) {
80 reloadQueryResults();
81 return;
82 }
83 break;
84 }
85 r.setWidth(sz * 2.0);
86 r.setHeight(sz * 2.0);
87 reqRect = r;
88 mutex.lock();
89 reqDist = r.topLeft().distanceTo(r.bottomRight());
90 mutex.unlock();
91 defaultLookup();
92}

94void LookupSites::defaultLookup()
95{
96 if (reqRect.isEmpty())
97 return;
98 QGeoCoordinate c = reqRect.center();
99 QVariantList v;
100 v << c.latitude();
101 v << c.latitude();
102 v << std::pow(std::cos(qDegreesToRadians(c.latitude())), 2);
103 v << c.longitude();
104 v << c.longitude();
105 query("SELECT *, ((?-lat)*(?-lat)*?)+((?-lon)*(?-lon)) AS dist FROM Sites"
106 " ORDER BY dist ASC LIMIT 100",
107 v);
108 //qDebug()<<reqRect;
109}
