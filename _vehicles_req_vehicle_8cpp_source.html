25bool DBReqSaveVehicleInfo::run(QSqlQuery &query)
26{
27 auto time = _info.value("time").toULongLong();
28 auto callsign = _info.value("callsign").toString();
29 auto vclass = _info.value("class").toString();
30
31 auto vuid = _info.value("uid").toString();
32 if (vuid == QString(vuid.size(), QChar('0')))
33 vuid.clear();
34 if (!vuid.isEmpty()) {
35 if (vclass.isEmpty())
36 vclass = "UAV";
37 }
38
39 //find the latest existing record by UID
40 if (vuid.isEmpty()) {
41 query.prepare("SELECT * FROM Vehicles WHERE uid IS NULL ORDER BY time DESC LIMIT 1");
42 } else {
43 query.prepare("SELECT * FROM Vehicles WHERE uid=? ORDER BY time DESC LIMIT 1");
44 query.addBindValue(vuid);
45 }
46 if (!query.exec())
47 return false;
48
49 while (query.next()) {
50 auto latestTime = query.value("time").toULongLong();
51 //merge ident from previous registrations
52 if (callsign.isEmpty())
53 callsign = query.value("callsign").toString();
54
55 if (vuid.isNull()) {
56 //only one record with null uid (LOCAL)
57 vehicleID = query.value(0).toULongLong();
58 } else {
59 //find exact matching unique record
60 query.prepare("SELECT * FROM Vehicles WHERE uid=? AND callsign=? AND class=?");
61 query.addBindValue(vuid);
62 query.addBindValue(callsign);
63 query.addBindValue(vclass);
64 if (!query.exec())
65 return false;
66 if (!query.next())
67 break; //create new
68 vehicleID = query.value(0).toULongLong();
69 qDebug() << "vehicle exists" << callsign;
70 }
71
72 //same vehicle found - update data if newer
73 if (latestTime >= time)
74 return true;
75
76 query.prepare("UPDATE Vehicles SET time=? WHERE key=?");
77 query.addBindValue(time);
78 query.addBindValue(vehicleID);
79 if (!query.exec())
80 return false;
81 emit foundID(vehicleID);
82 return true;
83 }
84
85 //register new vehicle record
86 qDebug() << "new vehicle" << callsign;
87
88 query.prepare("INSERT INTO Vehicles(uid,callsign,class,time) VALUES(?,?,?,?)");
89 query.addBindValue(vuid);
90 query.addBindValue(callsign);
91 query.addBindValue(vclass);
92 query.addBindValue(time);
93 if (!query.exec())
94 return false;
95 vehicleID = query.lastInsertId().toULongLong();
96 emit foundID(vehicleID);
97 return true;
98}

100bool DBReqSaveVehicleConfig::run(QSqlQuery &query)
101{
102 // seaarch vehicle
103 if (_vuid.isEmpty()) {
104 query.prepare("SELECT * FROM Vehicles WHERE uid IS NULL ORDER BY time DESC LIMIT 1");
105 } else {
106 query.prepare("SELECT * FROM Vehicles WHERE uid=? ORDER BY time DESC LIMIT 1");
107 query.addBindValue(_vuid);
108 }
109 if (!query.exec())
110 return false;
111 if (!query.next()) {
112 qWarning() << "no vehicle" << _vuid;
113 return true;
114 }
115 auto vehicleID = query.value(0).toULongLong();
116
117 //generate hash
118 //add hash from each node config and sort nodes by uid
119 QCryptographicHash h(QCryptographicHash::Sha1);
120 QStringList stKeys;
121 for (auto i : _nconfIDs)
122 stKeys.append(QString::number(i));
123
124 query.prepare("SELECT * FROM NodeConfigs "
125 "INNER JOIN Nodes ON NodeConfigs.nodeID=Nodes.key "
126 "INNER JOIN NodeDicts ON NodeConfigs.dictID=NodeDicts.key "
127 "WHERE NodeConfigs.key IN ("
128 + stKeys.join(',')
129 + ")"
130 "ORDER BY sn");
131 if (!query.exec())
132 return false;
133
134 int kcnt = 0;
135 while (query.next()) {
136 kcnt++;
137 h.addData(query.value("NodeConfigs.hash").toString().toUtf8());
138 h.addData(query.value("Nodes.sn").toString().toUtf8());
139 }
140 if (kcnt <= 0) {
141 qWarning() << "nothing to save";
142 return true;
143 }
144 QString hash = h.result().toHex().toUpper();
145
146 //find latest config by hash
147 query.prepare("SELECT * FROM VehicleConfigs"
148 " LEFT JOIN Vehicles ON VehicleConfigs.vehicleID=Vehicles.key"
149 " WHERE hash=?");
150 query.addBindValue(hash);
151 if (!query.exec())
152 return false;
153 if (query.next()) {
154 // config found
155 auto configID = query.value(0).toULongLong();
156 //uptime time and actual values
157 qDebug() << "vehicle config exists" << query.value("callsign").toString() << _title;
158 if (_time <= query.value("VehicleConfigs.time").toULongLong())
159 return true;
160
161 query.prepare("UPDATE VehicleConfigs SET time=?, vehicleID=?, notes=? WHERE key=?");
162 query.addBindValue(_time);
163 query.addBindValue(vehicleID);
164 query.addBindValue(_notes);
165 query.addBindValue(configID);
166 if (!query.exec())
167 return false;
168 emit dbModified();
169 emit configSaved(hash, _title);
170 return true;
171 }
172
173 //create new vehicle config record
174 db->transaction(query);
175
176 query.prepare("INSERT INTO VehicleConfigs(hash,time,title,vehicleID,notes) VALUES(?,?,?,?,?)");
177 query.addBindValue(hash);
178 query.addBindValue(_time);
179 query.addBindValue(_title);
180 query.addBindValue(vehicleID);
181 query.addBindValue(_notes);
182 if (!query.exec())
183 return false;
184 auto configID = query.lastInsertId().toULongLong();
185
186 //save vehicle config bundle
187 for (auto i : _nconfIDs) {
188 query.prepare("INSERT INTO VehicleConfigData(configID,nconfID) VALUES(?,?)");
189 query.addBindValue(configID);
190 query.addBindValue(i);
191 if (!query.exec())
192 return false;
193 }
194
195 db->commit(query);
196 emit dbModified();
197 qDebug() << "new vehicle config" << _title;
198 emit configSaved(hash, _title);
199 return true;
200}

202bool DBReqLoadVehicleConfig::run(QSqlQuery &query)
203{
204 query.prepare("SELECT * FROM VehicleConfigs WHERE hash=?");
205 query.addBindValue(_hash);
206 if (!query.exec())
207 return false;
208 if (!query.next()) {
209 qWarning() << "missing config" << _hash;
210 return false;
211 }
212 auto configID = query.value(0).toULongLong();
213 auto title = query.value("title").toString();
214 auto notes = query.value("notes").toString();
215 auto time = query.value("time").toULongLong();
216 auto vehicleID = query.value("vehicleID").toString();
217
218 // get config data
219 query.prepare("SELECT * FROM VehicleConfigData"
220 " INNER JOIN NodeConfigs ON VehicleConfigData.nconfID=NodeConfigs.key"
221 " INNER JOIN NodeDicts ON NodeConfigs.dictID=NodeDicts.key"
222 " INNER JOIN Nodes ON NodeConfigs.nodeID=Nodes.key"
223 " WHERE VehicleConfigData.configID=?");
224 query.addBindValue(configID);
225 if (!query.exec())
226 return false;
227
228 // get list of nodes IDs
229 QList<QList<quint64>> list;
230 while (query.next()) {
231 auto nodeID = query.value("nodeID").toULongLong();
232 auto dictID = query.value("dictID").toULongLong();
233 auto nconfID = query.value("nconfID").toULongLong();
234 list.append({nodeID, dictID, nconfID});
235 }
236 if (list.isEmpty()) {
237 qWarning() << "missing vehicle config" << configID;
238 return true;
239 }
240
241 //qDebug()<<list.size();
242 QVariantList nodes;
243 for (auto r : list) {
244 QVariantMap node;
245 {
246 auto req = new DBReqLoadNodeInfo(r.value(0));
247 bool ok = req->run(query);
248 if (ok) {
249 node.insert("info", req->info());
250 }
251 delete req;
252 if (!ok)
253 return false;
254 }
255 {
256 auto req = new DBReqLoadNodeDict(r.value(1));
257 bool ok = req->run(query);
258 if (ok) {
259 node.insert("dict", req->dict());
260 }
261 delete req;
262 if (!ok)
263 return false;
264 }
265 {
266 auto req = new DBReqLoadNodeConfig(r.value(2));
267 bool ok = req->run(query);
268 if (ok) {
269 node.insert("values", req->values());
270 node.insert("time", req->time());
271 }
272 delete req;
273 if (!ok)
274 return false;
275 }
276 nodes.append(node);
277 }
278 _config.insert("nodes", nodes);
279 if (!title.isEmpty())
280 _config.insert("title", title);
281 if (!notes.isEmpty())
282 _config.insert("notes", notes);
283 _config.insert("time", time);
284
285 // get vehicle info
286 query.prepare("SELECT * FROM Vehicles WHERE key=?");
287 query.addBindValue(vehicleID);
288 if (!query.exec())
289 return false;
290 if (!query.next()) {
291 qWarning() << "missing vehicle" << vehicleID;
292 } else {
293 QVariantMap vehicle;
294 vehicle.insert("uid", query.value("uid").toString());
295 vehicle.insert("callsign", query.value("callsign").toString());
296 vehicle.insert("class", query.value("class").toString());
297 vehicle.insert("time", query.value("time").toString());
298 _config.insert("vehicle", filterNullValues(vehicle));
299 }
300
301 emit configLoaded(_config);
302 return true;
303}

305bool DBReqImportVehicleConfig::run(QSqlQuery &query)
306{
307 // import vehicle
308 auto vehicle = _config.value("vehicle").value<QVariantMap>();
309 if (!vehicle.isEmpty()) {
310 auto req = new DBReqSaveVehicleInfo(vehicle);
311 auto ok = req->run(query);
312 delete req;
313 if (!ok)
314 return false;
315 }
316
317 // import nodes
318 QList<quint64> nconfIDs;
319 DatabaseRequest *req = {};
320 auto nodes = _config.value("nodes").value<QVariantList>();
321 for (auto i : nodes) {
322 auto node = i.value<QVariantMap>();
323
324 // node info
325 auto info = node.value("info").value<QVariantMap>();
326 if (info.isEmpty())
327 continue;
328 auto uid = info.value("uid").toString();
329
330 req = new DBReqSaveNodeInfo(info);
331 if (!req->run(query))
332 break;
333 delete req;
334 req = {};
335
336 // node dict
337 auto dict = node.value("dict").value<QVariantMap>();
338 if (dict.isEmpty())
339 continue;
340 auto hash = dict.value("hash").toString();
341
342 req = new DBReqSaveNodeDict(uid, dict);
343 if (!req->run(query))
344 break;
345 delete req;
346 req = {};
347
348 // node config
349 auto values = node.value("values").value<QVariantMap>();
350 if (values.isEmpty())
351 continue;
352 auto time = node.value("time").toULongLong();
353
354 req = new DBReqSaveNodeConfig(uid, hash, values, time);
355 if (!req->run(query))
356 break;
357 nconfIDs.append(static_cast<DBReqSaveNodeConfig *>(req)->configID());
358 delete req;
359 req = {};
360 }
361 if (req) {
362 delete req;
363 qWarning() << "failed to import vehicle config";
364 return true;
365 }
366
367 // vehicle config
368 auto vuid = vehicle.value("uid").toString();
369 auto title = _config.value("title").toString();
370 auto time = _config.value("time").toULongLong();
371
372 if (nconfIDs.isEmpty()) {
373 qWarning() << "missing nodes in import";
374 } else {
375 QString notes("imported");
376
377 req = new DBReqSaveVehicleConfig(vuid, nconfIDs, title, notes, time);
378 bool ok = req->run(query);
379 delete req;
380 if (!ok)
381 return false;
382 }
383
384 return true;
385}
