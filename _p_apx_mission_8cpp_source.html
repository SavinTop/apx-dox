43void PApxMission::updateFiles()
44{
45 auto f = _file();
46 if (!f)
47 return;
48 connect(f, &PApxNodeFile::downloaded, this, &PApxMission::parseMissionData, Qt::UniqueConnection);
49 connect(f, &PApxNodeFile::uploaded, this, &PApxMission::parseMissionData, Qt::UniqueConnection);
50 emit missionAvailable();
51}

68 return;

70 auto req = new PApxNodeRequestFileWrite(f->node(), file_name, _pack(var.value<QVariantMap>()));

74void PApxMission::parseMissionData(PApxNode *_node,

76 const QByteArray data)

78 //qDebug() << "mission data" << info.size << data.size();
79 PStreamReader stream(data);
80 auto m = _unpack(stream);
81 if (!m.isEmpty()) {
82 m.insert("node_uid", _node->uid());
83 }
84 emit missionReceived(m);

96QVariantMap PApxMission::_unpack(PStreamReader &stream)
97{
98 //unpack mission
99 if (stream.available() < xbus::mission::file_hdr_s::psize())
100 return {};
101
102 xbus::mission::file_hdr_s fhdr{};
103 fhdr.read(&stream);
104 QString title(QByteArray(fhdr.title, sizeof(fhdr.title)));
105
106 qDebug() << title << stream.size() << "bytes";
107
108 if (fhdr.size != stream.available()) {
109 qWarning() << "data size" << fhdr.size << stream.available();
110 return {};
111 }
112
113 QVariantList wp, rw, tw, pi;
114
115 int ecnt = 0, wpcnt = 0, rwcnt = 0;
116
117 while (stream.available() > 0) {
118 ecnt++;
119 xbus::mission::hdr_s hdr;
120 hdr.read(&stream);
121
122 switch (hdr.type) {
123 case xbus::mission::STOP:
124 stream.reset(stream.size()); //finish
125 ecnt--;
126 continue;
127 case xbus::mission::WP: {
128 if (stream.available() < xbus::mission::wp_s::psize())
129 break;
130 xbus::mission::wp_s e;
131 e.read(&stream);
132 wpcnt++;
133
134 QVariantMap item;
135 item.insert("lat", static_cast<qreal>(e.lat));
136 item.insert("lon", static_cast<qreal>(e.lon));
137 item.insert("altitude", e.alt);
138 QString type = "direct";
139 if (hdr.option == 1)
140 type = "track";
141 item.insert("type", type);
142
143 wp.append(item);
144 continue;
145 }
146 case xbus::mission::ACT: { //wp actions
147 QVariantMap a;
148 switch (hdr.option) {
149 default:
150 break;
151 case xbus::mission::ACT_SPEED: {
152 xbus::mission::act_speed_s e;
153 e.read(&stream);
154 a.insert("speed", e.speed);
155 break;
156 }
157 case xbus::mission::ACT_PI: {
158 xbus::mission::act_pi_s e;
159 e.read(&stream);
160 a.insert("poi", e.index + 1);
161 break;
162 }
163 case xbus::mission::ACT_SCR: {
164 xbus::mission::act_scr_s e;
165 e.read(&stream);
166 a.insert("script", QString(QByteArray(e.scr, sizeof(e.scr))));
167 break;
168 }
169 case xbus::mission::ACT_SHOT: {
170 xbus::mission::act_shot_s e;
171 e.read(&stream);
172 switch (e.opt) {
173 case 0: //single
174 a.insert("shot", "single");
175 a.insert("dshot", 0);
176 break;
177 case 1: //start
178 a.insert("shot", "start");
179 a.insert("dshot", e.dist);
180 break;
181 case 2: //stop
182 a.insert("shot", "stop");
183 a.insert("dshot", 0);
184 break;
185 }
186 break;
187 }
188 }
189 if (wp.isEmpty()) {
190 qWarning() << "Orphan actions in mission";
191 continue;
192 }
193 // append actions to the last waypoint
194 auto wpt = wp.last().value<QVariantMap>();
195 auto actions = wpt.value("actions").value<QVariantMap>();
196 for (auto i : a.keys())
197 actions.insert(i, a[i]);
198 wpt.insert("actions", actions);
199 wp[wp.size() - 1] = wpt;
200 continue;
201 }
202 case xbus::mission::RW: {
203 if (stream.available() < xbus::mission::rw_s::psize())
204 break;
205 xbus::mission::rw_s e;
206 e.read(&stream);
207 rwcnt++;
208
209 QVariantMap item;
210 item.insert("lat", static_cast<qreal>(e.lat));
211 item.insert("lon", static_cast<qreal>(e.lon));
212 item.insert("hmsl", e.hmsl);
213 item.insert("dN", e.dN);
214 item.insert("dE", e.dE);
215 item.insert("approach", e.approach);
216 QString type = "left";
217 if (hdr.option == 1)
218 type = "right";
219 item.insert("type", type);
220
221 rw.append(item);
222 continue;
223 }
224 case xbus::mission::TW: {
225 if (stream.available() < xbus::mission::tw_s::psize())
226 break;
227 xbus::mission::tw_s e;
228 e.read(&stream);
229
230 QVariantMap item;
231 item.insert("lat", static_cast<qreal>(e.lat));
232 item.insert("lon", static_cast<qreal>(e.lon));
233
234 tw.append(item);
235 continue;
236 }
237 case xbus::mission::PI: {
238 if (stream.available() < xbus::mission::pi_s::psize())
239 break;
240 xbus::mission::pi_s e;
241 e.read(&stream);
242 QVariantMap item;
243 item.insert("lat", static_cast<qreal>(e.lat));
244 item.insert("lon", static_cast<qreal>(e.lon));
245 item.insert("hmsl", e.hmsl);
246 item.insert("radius", e.radius);
247 item.insert("loops", e.loops);
248 item.insert("timeout", e.timeout);
249
250 pi.append(item);
251 continue;
252 }
253 case xbus::mission::EMG:
254 case xbus::mission::DIS: {
255 uint16_t sz = stream.available();
256 size_t pointsCnt = hdr.option;
257 if (sz < xbus::mission::area_s::psize(pointsCnt))
258 break;
259 //TODO implement areas in GCS
260 for (size_t i = 0; i < pointsCnt; ++i) {
261 xbus::mission::area_s p;
262 p.read(&stream);
263 }
264 continue;
265 }
266 }
267 //error in mission
268 return {};
269 }
270
271 if (ecnt <= 0)
272 return {};
273
274 QVariantMap m;
275
276 if (!wp.isEmpty())
277 m.insert("wp", wp);
278 if (!rw.isEmpty())
279 m.insert("rw", rw);
280 if (!tw.isEmpty())
281 m.insert("tw", tw);
282 if (!pi.isEmpty())
283 m.insert("pi", pi);
284
285 if (m.isEmpty())
286 return {};
287
288 m.insert("title", title);
289 return m;
290}

292QByteArray PApxMission::_pack(const QVariantMap &m)
293{
294 QByteArray data(65535, '\0');
295 uint8_t *pdata = reinterpret_cast<uint8_t *>(data.data());
296 XbusStreamWriter stream(pdata, data.size());
297
298 xbus::mission::file_hdr_s fhdr{};
299 fhdr.write(&stream); // will update later
300 size_t pos_s = stream.pos();
301
302 fhdr.off.wp = stream.pos() - pos_s;
303 for (auto i : m.value("wp").value<QVariantList>()) {
304 auto wp = i.value<QVariantMap>();
305 xbus::mission::hdr_s hdr{};
306 hdr.type = xbus::mission::WP;
307 if (wp.value("type").toString().toLower() == "track")
308 hdr.option = 1;
309 hdr.write(&stream);
310 xbus::mission::wp_s e{};
311 e.lat = wp.value("lat").toFloat();
312 e.lon = wp.value("lon").toFloat();
313 e.alt = wp.value("altitude").toUInt();
314 e.write(&stream);
315 fhdr.cnt.wp++;
316
317 auto actions = wp.value("actions").value<QVariantMap>();
318 if (actions.isEmpty())
319 continue;
320
321 if (actions.contains("speed")) {
322 xbus::mission::hdr_s ahdr{};
323 ahdr.type = xbus::mission::ACT;
324 ahdr.option = xbus::mission::ACT_SPEED;
325 ahdr.write(&stream);
326 xbus::mission::act_speed_s a{};
327 a.speed = actions.value("speed").toUInt();
328 a.write(&stream);
329 }
330 if (actions.contains("poi")) {
331 xbus::mission::hdr_s ahdr{};
332 ahdr.type = xbus::mission::ACT;
333 ahdr.option = xbus::mission::ACT_PI;
334 ahdr.write(&stream);
335 xbus::mission::act_pi_s a{};
336 a.index = actions.value("poi").toUInt() - 1;
337 a.write(&stream);
338 }
339 if (actions.contains("script")) {
340 xbus::mission::hdr_s ahdr{};
341 ahdr.type = xbus::mission::ACT;
342 ahdr.option = xbus::mission::ACT_SCR;
343 ahdr.write(&stream);
344 xbus::mission::act_scr_s a{};
345 QByteArray src(actions.value("script").toString().toUtf8());
346 memset(a.scr, 0, sizeof(a.scr));
347 memcpy(a.scr, src.data(), static_cast<size_t>(src.size()));
348 a.scr[sizeof(a.scr) - 1] = 0;
349 a.write(&stream);
350 }
351 if (actions.contains("shot")) {
352 xbus::mission::hdr_s ahdr{};
353 ahdr.type = xbus::mission::ACT;
354 ahdr.option = xbus::mission::ACT_SHOT;
355 ahdr.write(&stream);
356 xbus::mission::act_shot_s a{};
357 uint dshot = actions.value("dshot").toUInt();
358 const QString shot = actions.value("shot").toString();
359 a.opt = 0;
360 a.dist = 0;
361 if (shot == "start") {
362 if (dshot == 0) {
363 apxMsgW() << tr("Auto shot distance is zero");
364 break;
365 }
366 a.opt = 1;
367 if (dshot > ((1 << 12) - 1))
368 dshot = ((1 << 12) - 1);
369 a.dist = static_cast<int16_t>(dshot);
370 } else if (shot == "stop") {
371 a.opt = 2;
372 a.dist = 0;
373 } else if (shot != "single") {
374 apxMsgW() << tr("Unknown shot mode") << shot;
375 }
376 a.write(&stream);
377 }
378 }
379
380 fhdr.off.rw = stream.pos() - pos_s;
381 for (auto i : m.value("rw").value<QVariantList>()) {
382 auto rw = i.value<QVariantMap>();
383 xbus::mission::hdr_s hdr{};
384 hdr.type = xbus::mission::RW;
385 if (rw["type"].toString().toLower() == "right")
386 hdr.option = 1;
387 hdr.write(&stream);
388 xbus::mission::rw_s e{};
389 e.lat = rw.value("lat").toFloat();
390 e.lon = rw.value("lon").toFloat();
391 e.hmsl = rw.value("hmsl").toInt();
392 e.dN = rw.value("dN").toInt();
393 e.dE = rw.value("dE").toInt();
394 e.approach = rw.value("approach").toUInt();
395 e.write(&stream);
396 fhdr.cnt.rw++;
397 //qDebug() << m.details;
398 }
399
400 fhdr.off.tw = stream.pos() - pos_s;
401 for (auto i : m.value("tw").value<QVariantList>()) {
402 auto tw = i.value<QVariantMap>();
403 xbus::mission::hdr_s hdr;
404 hdr.type = xbus::mission::TW;
405 hdr.option = 0;
406 hdr.write(&stream);
407 xbus::mission::tw_s e{};
408 e.lat = tw.value("lat").toFloat();
409 e.lon = tw.value("lon").toFloat();
410 e.write(&stream);
411 fhdr.cnt.tw++;
412 }
413
414 fhdr.off.pi = stream.pos() - pos_s;
415 for (auto i : m.value("pi").value<QVariantList>()) {
416 auto pi = i.value<QVariantMap>();
417 xbus::mission::hdr_s hdr;
418 hdr.type = xbus::mission::PI;
419 hdr.option = 0;
420 hdr.write(&stream);
421 xbus::mission::pi_s e{};
422 e.lat = pi.value("lat").toFloat();
423 e.lon = pi.value("lon").toFloat();
424 e.hmsl = pi.value("hmsl").toInt();
425 e.radius = pi.value("radius").toInt();
426 e.loops = pi.value("loops").toUInt();
427 e.timeout = AppRoot::timeFromString(pi.value("timeout").toString(), false);
428 e.write(&stream);
429 fhdr.cnt.pi++;
430 }
431
432 if (stream.pos() <= xbus::mission::hdr_s::psize()) {
433 qDebug() << "Upload empty mission";
434 }
435
436 xbus::mission::hdr_s hdr{};
437 hdr.type = xbus::mission::STOP;
438 hdr.option = 0;
439 hdr.write(&stream);
440 data.resize(stream.pos());
441
442 //update fhdr
443 fhdr.size = stream.pos() - pos_s;
444
445 strncpy(fhdr.title, m.value("title").toString().toLocal8Bit(), sizeof(fhdr.title));
446
447 stream.reset();
448 fhdr.write(&stream);
449
450 return data;
451}
