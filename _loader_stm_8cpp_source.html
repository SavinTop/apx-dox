75void LoaderStm::next()
76{
77 switch (stage) {
78 default:
79 break;
80
81 case 0: //open port
82 dev->setBaudRate(115200);
83 dev->setParity(QSerialPort::EvenParity);
84 retry = 0;
85 stage++;
86 //fallthru
87 [[clang::fallthrough]];
88 case 1:
89 if (dev->open(QIODevice::ReadWrite)) {
90 apxMsg() << tr("Using port").append(':') << dev->portName();
91 if (continuous)
92 apxMsg() << tr("Continuous mode");
93 stage = 10;
94 timer.start(200);
95 return;
96 }
97 //repeat
98 retry++;
99 if (retry > 10) {
100 apxMsgW() << tr("Can't open port") << dev->portName() << dev->error();
101 break;
102 }
103 timer.start(500);
104 return;
105
106 // port opened
107 case 10: //init loader
108 apxMsg() << tr("Initializing").append("...");
109 setProgress(0);
110 write(QByteArray(1, 0x7F), rx_loader, 100);
111 retry = 0;
112 stage++;
113 return;
114 case 11:
115 if (rx_stage == rx_ok) {
116 qDebug() << "Loader OK";
117 stage = 20;
118 timer.start(100);
119 return;
120 }
121 //repeat
122 if (!continuous)
123 retry++;
124 if (retry > 150) {
125 apxMsgW() << tr("Can't initialize stm loader");
126 break;
127 }
128 write(QByteArray(1, 0x7F), rx_loader, 100);
129 return;
130
131 // in loader
132 case 20:
133 //read chip info
134 write(0x00, rx_info);
135 stage++;
136 return;
137 case 21:
138 if (rx_stage == rx_ok) {
139 stage = 30;
140 timer.start(0);
141 return;
142 }
143 //repeat
144 retry++;
145 if (retry > 5) {
146 apxMsgW() << tr("No response from chip");
147 break;
148 }
149 write(0x00, rx_info);
150 return;
151
152 // erase chip
153 case 30:
154 if (_offset != 0x08000000) {
155 /*apxMsg() << tr("Chip erase skipped");
156 stage = 50;
157 timer.start(0);
158 return;*/
159 apxMsg() << tr("Offset").append(':')
160 << QString("0x%1").arg(static_cast<qulonglong>(_offset), 8, 16, QChar('0'));
161 }
162 apxMsg() << tr("Erasing chip").append("...");
163 write(cmd_erase, rx_ack);
164 stage++;
165 return;
166 case 31:
167 if (rx_stage != rx_ok) {
168 apxMsgW() << tr("Command error");
169 break;
170 }
171 if (cmd_erase == 0x43) { //standard
172 write(0xFF, rx_ack, 60000);
173 } else { //extended
174 write(crc(QByteArray(2, 0xFF)), rx_ack, 120000);
175 }
176 stage++;
177 return;
178 case 32:
179 if (rx_stage != rx_ok) {
180 apxMsgW() << tr("Chip erase error");
181 break;
182 }
183 qDebug() << "Chip erased";
184 stage = 50;
185 timer.start(0);
186 return;
187
188 //write data
189 case 50:
190 apxMsg() << tr("Uploading data").append("...");
191 writeCnt = 0;
192 stage++;
193 //fallthru
194 [[clang::fallthrough]];
195 case 51:
196 write(0x31, rx_ack);
197 stage++;
198 return;
199 case 52:
200 if (rx_stage == rx_ok) {
201 QByteArray ba;
202 quint32 writeAddr = _offset + writeCnt;
203 ba.append(static_cast<char>(writeAddr >> 24));
204 ba.append(static_cast<char>(writeAddr >> 16));
205 ba.append(static_cast<char>(writeAddr >> 8));
206 ba.append(static_cast<char>(writeAddr));
207 write(crc(ba), rx_ack);
208 stage++;
209 return;
210 }
211 apxMsgW() << tr("Write command error");
212 break;
213 case 53:
214 if (rx_stage == rx_ok) {
215 QByteArray ba = _data.mid(writeCnt, 256);
216 writeCnt += ba.size();
217 while ((ba.size() & 3) != 0)
218 ba.append(static_cast<char>(0xFF));
219 ba.prepend(ba.size() - 1);
220 write(crc(ba), rx_ack);
221 setProgress(100 * writeCnt / _data.size());
222 stage++;
223 return;
224 }
225 apxMsgW() << tr("Write address error");
226 break;
227 case 54:
228 if (rx_stage == rx_ok) {
229 //qDebug() << "written" << writeCnt;
230 if (writeCnt >= _data.size()) {
231 stage = 100;
232 timer.start(0);
233 return;
234 }
235 stage = 51;
236 timer.start(0);
237 return;
238 }
239 apxMsgW() << tr("Write data error");
240 break;
241
242 // start program
243 case 100:
244 //apxMsg() << tr("Starting program");
245 write(0x21, rx_ack);
246 stage++;
247 return;
248 case 101:
249 if (rx_stage == rx_ok) {
250 QByteArray ba;
251 ba.append(static_cast<char>(_offset >> 24));
252 ba.append(static_cast<char>(_offset >> 16));
253 ba.append(static_cast<char>(_offset >> 8));
254 ba.append(static_cast<char>(_offset));
255 write(crc(ba), rx_ack);
256 stage++;
257 return;
258 }
259 apxMsgW() << tr("Write program address error");
260 break;
261 case 102:
262 if (rx_stage == rx_ok) {
263 success = true;
264 } else {
265 apxMsgW() << tr("Start program error");
266 }
267 break;
268 }
269
270 // retry forever
271 if (stage >= 20 && continuous) {
272 if (!success)
273 qWarning() << "retry" << stage << rx_stage << time.elapsed();
274 else {
275 apxMsg() << tr("Success");
276 apxMsg() << tr("Next chip").append("...");
277 }
278 stage = 10;
279 timer.start(200);
280 return;
281 }
282
283 //exit
284 qDebug() << stage << rx_stage << time.elapsed();
285 deleteLater();
286}

288void LoaderStm::readData()
289{
290 rxData.append(dev->readAll());
291 if (rxData.isEmpty())
292 return;
293 switch (rx_stage) {
294 default:
295 break;
296 case rx_ack:
297 if (rxData == QByteArray(rxData.size(), rxData.at(0))) {
298 quint8 v = rxData.at(0);
299 if (v == SLDR_ACK) {
300 rx_stage = rx_ok;
301 break;
302 }
303 apxMsgW() << tr("Wrong response from chip");
304 qDebug() << "wrong response" << rxData.toHex().toUpper();
305 timer.start(0);
306 }
307 break;
308 case rx_loader:
309 if (rxData == QByteArray(rxData.size(), rxData.at(0))) {
310 quint8 v = rxData.at(0);
311 if (v == SLDR_ACK || v == SLDR_NACK) {
312 rx_stage = rx_ok;
313 break;
314 }
315 }
316 break;
317 case rx_info: {
318 quint8 v = rxData.at(0);
319 if (v != SLDR_ACK) {
320 timer.start(0);
321 break;
322 }
323 rxData.remove(0, 1);
324 rx_stage = rx_info_data;
325 } //fallthru
326 [[clang::fallthrough]];
327 case rx_info_data: {
328 if (rxData.size() < 2)
329 return;
330 int sz = static_cast<quint8>(rxData.at(0)) + 3;
331 if (rxData.size() < sz)
332 return;
333 quint8 vAck = rxData.at(rxData.size() - 1);
334 if (rxData.size() != sz || vAck != SLDR_ACK) {
335 qDebug() << "wrong response" << rxData.toHex().toUpper();
336 timer.start(0);
337 break;
338 }
339 rxData.remove(0, 1);
340 rx_stage = rx_ok;
341 quint8 ver = rxData.at(0);
342 apxMsg() << QString("STM Bootloader v%1.%2").arg(ver >> 4).arg(ver & 0x0F);
343 cmd_erase = rxData.at(7);
344 } break;
345 }
346
347 if (rx_stage == rx_ok) {
348 retry = 0;
349 timer.start(0);
350 }
351 rxData.clear();
352}

356 time.start();

364{
365 QByteArray ba;

371QByteArray LoaderStm::crc(QByteArray data) const
372{
373 quint8 crc = 0;
374 for (int i = 0; i < data.size(); ++i) {
375 crc ^= static_cast<quint8>(data.at(i));
376 }
377 data.append(static_cast<char>(crc));
378 return data;
